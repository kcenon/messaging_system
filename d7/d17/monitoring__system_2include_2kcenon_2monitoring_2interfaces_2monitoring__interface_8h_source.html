161struct monitoring_config {
162 std::size_t history_size = 1000;
163 std::chrono::milliseconds collection_interval{1000};
164 bool enable_compression = false;
165 bool enable_persistence = false;
166 std::size_t max_collectors = 100;
167 std::size_t buffer_size = 10000;
168
173 result_void validate() const {
174 if (history_size == 0) {
175 return result_void(monitoring_error_code::invalid_capacity,
176 "History size must be greater than 0");
177 }
178 if (collection_interval.count() < 10) {
179 return result_void(monitoring_error_code::invalid_interval,
180 "Collection interval must be at least 10ms");
181 }
182 if (buffer_size < history_size) {
183 return result_void(monitoring_error_code::invalid_capacity,
184 "Buffer size must be at least as large as history size");
185 }
186 return result_void::success();
187 }
188};
