42// Lock-free SPSC ring buffer for high-performance logging
43template<typename T, size_t Size>
44class lockfree_ring_buffer {
45private:
46 static_assert((Size & (Size - 1)) == 0, "Size must be power of 2");
47 static constexpr size_t mask = Size - 1;
48
49 std::array<T, Size> buffer_;
50 std::atomic<size_t> head_{0};
51 std::atomic<size_t> tail_{0};
52
53public:
54 bool push(T&& item) {
55 const size_t current_tail = tail_.load(std::memory_order_relaxed);
56 const size_t next_tail = (current_tail + 1) & mask;
57
58 if (next_tail == head_.load(std::memory_order_acquire)) {
59 return false; // Buffer is full
60 }
61
62 buffer_[current_tail] = std::move(item);
63 tail_.store(next_tail, std::memory_order_release);
64 return true;
65 }
66
67 bool pop(T& item) {
68 const size_t current_head = head_.load(std::memory_order_relaxed);
69
70 if (current_head == tail_.load(std::memory_order_acquire)) {
71 return false; // Buffer is empty
72 }
73
74 item = std::move(buffer_[current_head]);
75 head_.store((current_head + 1) & mask, std::memory_order_release);
76 return true;
77 }
78
79 size_t size() const {
80 const size_t tail = tail_.load(std::memory_order_acquire);
81 const size_t head = head_.load(std::memory_order_acquire);
82 return (tail - head) & mask;
83 }
84
85 bool empty() const {
86 return head_.load(std::memory_order_acquire) == tail_.load(std::memory_order_acquire);
87 }
88
89 static constexpr size_t capacity() { return Size; }
90};

103class log_collector::impl {
104public:
105 explicit impl(std::size_t buffer_size)
106 : buffer_size_(buffer_size)
107 , running_(false)
108 , dropped_messages_(0)
109 , processed_messages_(0) {
110 // Ensure buffer size is power of 2 for lock-free ring buffer
111 size_t power_of_2_size = 1;
112 while (power_of_2_size < buffer_size) {
113 power_of_2_size <<= 1;
114 }
115 actual_buffer_size_ = power_of_2_size;
116 }
117
118 ~impl() {
119 stop();
120 }
121
122 bool enqueue(thread_module::log_level level,
123 const std::string& message,
124 const std::string& file,
125 int line,
126 const std::string& function,
127 const std::chrono::system_clock::time_point& timestamp) {
128
129 // Use lock-free ring buffer for high-performance enqueueing
130 log_entry entry{level, message, file, line, function, timestamp};
131
132 // Try to push to lock-free buffer first
133 if (ring_buffer_.push(std::move(entry))) {
134 // Notify worker thread using atomic flag
135 has_work_.store(true, std::memory_order_release);
136 return true;
137 }
138
139 // If lock-free buffer is full, increment dropped count
140 dropped_messages_.fetch_add(1, std::memory_order_relaxed);
141 return false;
142 }
143
144 void add_writer(base_writer* writer) {
145 std::lock_guard<std::mutex> lock(writers_mutex_);
146 writers_.push_back(writer);
147 }
148
149 void clear_writers() {
150 std::lock_guard<std::mutex> lock(writers_mutex_);
151 writers_.clear();
152 }
153
154 void start() {
155 if (!running_.exchange(true)) {
156 worker_thread_ = std::thread(&impl::process_loop, this);
157 }
158 }
159
160 void stop() {
161 if (running_.exchange(false)) {
162 // Signal that work is available to wake up the thread
163 has_work_.store(true, std::memory_order_release);
164 if (worker_thread_.joinable()) {
165 worker_thread_.join();
166 }
167 }
168 }
169
170 void flush() {
171 flush_remaining();
172 }

174 std::pair<size_t, size_t> get_queue_metrics() const {
175 return {ring_buffer_.size(), ring_buffer_.capacity()};

178 // New method to get performance metrics
179 struct performance_stats {
180 uint64_t processed_messages;
181 uint64_t dropped_messages;
182 size_t current_queue_size;
183 size_t queue_capacity;
184 };
185
186 performance_stats get_performance_stats() const {
187 return {
188 processed_messages_.load(std::memory_order_relaxed),
189 dropped_messages_.load(std::memory_order_relaxed),
190 ring_buffer_.size(),
191 ring_buffer_.capacity()
192 };
193 }
194
195private:
196 void process_loop() {
197 std::vector<log_entry> batch;
198 batch.reserve(256); // Pre-allocate for better performance
199
200 while (running_.load(std::memory_order_acquire)) {
201 // Check for work using atomic flag
202 if (!has_work_.load(std::memory_order_acquire)) {
203 // Brief sleep to avoid busy-waiting
204 std::this_thread::sleep_for(std::chrono::microseconds(100));
205 continue;
206 }
207
208 // Process entries from lock-free ring buffer
209 batch.clear();
210 log_entry entry;
211
212 // Batch process up to 256 entries for efficiency
213 while (batch.size() < 256 && ring_buffer_.pop(entry)) {
214 batch.push_back(std::move(entry));
215 }
216
217 // Reset work flag if ring buffer is empty
218 if (ring_buffer_.empty()) {
219 has_work_.store(false, std::memory_order_release);
220 }
221
222 // Write batch to all writers
223 if (!batch.empty()) {
224 write_batch_to_all(batch);
225 processed_messages_.fetch_add(batch.size(), std::memory_order_relaxed);
226 }
227 }
228
229 // Process any remaining entries
230 flush_remaining();
231 }
232
233 void write_to_all(const log_entry& entry) {
234 std::lock_guard<std::mutex> lock(writers_mutex_);
235 for (auto* writer : writers_) {
236 // Ignore write failures for now
237 writer->write(entry.level, entry.message, entry.file,
238 entry.line, entry.function, entry.timestamp);
239 }
240 }
241
242 void write_batch_to_all(const std::vector<log_entry>& batch) {
243 std::lock_guard<std::mutex> lock(writers_mutex_);
244 for (auto* writer : writers_) {
245 for (const auto& entry : batch) {
246 // Ignore write failures for now
247 writer->write(entry.level, entry.message, entry.file,
248 entry.line, entry.function, entry.timestamp);
249 }
250 }
251 }
252
253 void flush_remaining() {
254 // Process all remaining entries
255 log_entry entry;
256 while (ring_buffer_.pop(entry)) {
257 write_to_all(entry);
258 processed_messages_.fetch_add(1, std::memory_order_relaxed);
259 }
260
261 // Flush all writers
262 std::lock_guard<std::mutex> writer_lock(writers_mutex_);
263 for (auto* writer : writers_) {
264 writer->flush();
265 }
266 }
267
268private:
269 std::size_t buffer_size_;
270 std::size_t actual_buffer_size_;
271 std::atomic<bool> running_;
272 std::thread worker_thread_;
273
274 // Lock-free ring buffer for high-performance logging
275 lockfree_ring_buffer<log_entry, 16384> ring_buffer_; // 16K entries
276 std::atomic<bool> has_work_{false};
277
278 // Performance metrics
279 std::atomic<uint64_t> dropped_messages_;
280 std::atomic<uint64_t> processed_messages_;
281
282 // Writers management (still needs mutex for dynamic changes)
283 std::vector<base_writer*> writers_;
284 std::mutex writers_mutex_;
285};
