50class log_collector::impl {
51public:
52 explicit impl(std::size_t buffer_size)
53 : buffer_size_(buffer_size)
54 , running_(false) {
55 }
56
57 ~impl() {
58 stop();
59 }
60
61#ifdef USE_THREAD_SYSTEM_INTEGRATION
62 bool enqueue(kcenon::thread::log_level level,
63#else
64 bool enqueue(logger_system::log_level level,
65#endif
66 const std::string& message,
67 const std::string& file,
68 int line,
69 const std::string& function,
70 const std::chrono::system_clock::time_point& timestamp) {
71 {
72 std::unique_lock<std::mutex> lock(queue_mutex_);
73
74 // Check if queue is full
75 if (queue_.size() >= buffer_size_) {
76 // Drop the message
77 return false;
78 }
79
80 // Create log_entry with optional source location
81#ifdef USE_THREAD_SYSTEM_INTEGRATION
82 // Convert kcenon::thread::log_level to logger_system::log_level
83 logger_system::log_level logger_level;
84 switch (level) {
85 case kcenon::thread::log_level::trace: logger_level = logger_system::log_level::trace; break;
86 case kcenon::thread::log_level::debug: logger_level = logger_system::log_level::debug; break;
87 case kcenon::thread::log_level::info: logger_level = logger_system::log_level::info; break;
88 case kcenon::thread::log_level::warning: logger_level = logger_system::log_level::warn; break;
89 case kcenon::thread::log_level::error: logger_level = logger_system::log_level::error; break;
90 case kcenon::thread::log_level::critical: logger_level = logger_system::log_level::fatal; break;
91 default: logger_level = logger_system::log_level::info; break;
92 }
93#else
94 // In standalone mode, no conversion needed
95 logger_system::log_level logger_level = level;
96#endif
97 log_entry entry(logger_level, message, timestamp);
98 if (!file.empty() || line != 0 || !function.empty()) {
99 entry.location = source_location{file, line, function};
100 }
101 queue_.push(std::move(entry));
102 }
103
104 queue_cv_.notify_one();
105 return true;
106 }
107
108 void add_writer(base_writer* writer) {
109 std::lock_guard<std::mutex> lock(writers_mutex_);
110 writers_.push_back(writer);
111 }
112
113 void clear_writers() {
114 std::lock_guard<std::mutex> lock(writers_mutex_);
115 writers_.clear();
116 }
117
118 void start() {
119 if (!running_.exchange(true)) {
120 worker_thread_ = std::thread(&impl::process_loop, this);
121 }
122 }
123
124 void stop() {
125 if (running_.exchange(false)) {
126 queue_cv_.notify_all();
127 if (worker_thread_.joinable()) {
128 worker_thread_.join();
129 }
130 }
131 }
132
133 void flush() {
134 // Process all remaining entries
135 std::unique_lock<std::mutex> lock(queue_mutex_);
136 while (!queue_.empty()) {
137 auto entry = std::move(queue_.front());
138 queue_.pop();
139 lock.unlock();
140
141 write_to_all(entry);
142
143 lock.lock();
144 }
145
146 // Flush all writers
147 std::lock_guard<std::mutex> writer_lock(writers_mutex_);
148 for (auto* writer : writers_) {
149 writer->flush();
150 }
151 }
152
153 std::pair<size_t, size_t> get_queue_metrics() const {
154 std::lock_guard<std::mutex> lock(queue_mutex_);
155 return {queue_.size(), buffer_size_};
156 }
157
158private:
159 void process_loop() {
160 while (running_.load()) {
161 std::unique_lock<std::mutex> lock(queue_mutex_);
162
163 // Wait for entries or shutdown
164 queue_cv_.wait(lock, [this] {
165 return !queue_.empty() || !running_.load();
166 });
167
168 // Process batch of entries
169 std::vector<log_entry> batch;
170 while (!queue_.empty() && batch.size() < 100) {
171 batch.push_back(std::move(queue_.front()));
172 queue_.pop();
173 }
174 lock.unlock();
175
176 // Write batch to all writers
177 for (const auto& entry : batch) {
178 write_to_all(entry);
179 }
180 }
181
182 // Process any remaining entries
183 flush();
184 }
185
186 void write_to_all(const log_entry& entry) {
187 std::lock_guard<std::mutex> lock(writers_mutex_);
188 for (auto* writer : writers_) {
189 // Ignore write failures for now
190 std::string file = entry.location ? entry.location->file.to_string() : "";
191 int line = entry.location ? entry.location->line : 0;
192 std::string function = entry.location ? entry.location->function.to_string() : "";
193
194 writer->write(entry.level, entry.message.to_string(), file,
195 line, function, entry.timestamp);
196 }
197 }
198
199private:
200 std::size_t buffer_size_;
201 std::atomic<bool> running_;
202 std::thread worker_thread_;
203
204 std::queue<log_entry> queue_;
205 mutable std::mutex queue_mutex_;
206 std::condition_variable queue_cv_;
207
208 std::vector<base_writer*> writers_;
209 std::mutex writers_mutex_;
210};

222kcenon::thread::log_level convert_log_level(logger_system::log_level level) {
223 switch (level) {
224 case logger_system::log_level::trace: return kcenon::thread::log_level::trace;
225 case logger_system::log_level::debug: return kcenon::thread::log_level::debug;
226 case logger_system::log_level::info: return kcenon::thread::log_level::info;
227 case logger_system::log_level::warn: return kcenon::thread::log_level::warning;
228 case logger_system::log_level::error: return kcenon::thread::log_level::error;
229 case logger_system::log_level::fatal: return kcenon::thread::log_level::critical;
230 case logger_system::log_level::off: return kcenon::thread::log_level::critical; // fallback
231 default: return kcenon::thread::log_level::info;
232 }
233}
