52class service_container {
53public:
57 enum class lifetime {
58 singleton, // Single instance shared across all requests
59 transient // New instance for each request
60 };
61
68 template<typename Interface, typename Implementation>
69 void register_singleton(std::shared_ptr<Implementation> instance) {
70 static_assert(std::is_base_of_v<Interface, Implementation>,
71 "Implementation must derive from Interface");
72
73 std::lock_guard<std::mutex> lock(mutex_);
74 services_[std::type_index(typeid(Interface))] =
75 service_entry{lifetime::singleton, instance, nullptr};
76 }
77
84 template<typename Interface>
85 void register_factory(std::function<std::shared_ptr<Interface>()> factory,
86 lifetime lt = lifetime::transient) {
87 std::lock_guard<std::mutex> lock(mutex_);
88 services_[std::type_index(typeid(Interface))] =
89 service_entry{lt, nullptr,
90 [factory]() -> std::shared_ptr<void> {
91 return factory();
92 }};
93 }
94
101 template<typename Interface, typename Implementation, typename... Args>
102 void register_transient() {
103 static_assert(std::is_base_of_v<Interface, Implementation>,
104 "Implementation must derive from Interface");
105
106 register_factory<Interface>(
107 []() { return std::make_shared<Implementation>(); },
108 lifetime::transient);
109 }
110
116 template<typename Interface>
117 std::shared_ptr<Interface> resolve() {
118 std::lock_guard<std::mutex> lock(mutex_);
119
120 auto it = services_.find(std::type_index(typeid(Interface)));
121 if (it == services_.end()) {
122 return nullptr;
123 }
124
125 auto& entry = it->second;
126
127 // For singletons with existing instance
128 if (entry.lifetime_scope == lifetime::singleton && entry.instance) {
129 return std::static_pointer_cast<Interface>(entry.instance);
130 }
131
132 // Create new instance using factory
133 if (entry.factory) {
134 auto instance = entry.factory();
135
136 // Store singleton instances
137 if (entry.lifetime_scope == lifetime::singleton) {
138 entry.instance = instance;
139 }
140
141 return std::static_pointer_cast<Interface>(instance);
142 }
143
144 return nullptr;
145 }
146
152 template<typename Interface>
153 bool is_registered() const {
154 std::lock_guard<std::mutex> lock(mutex_);
155 return services_.find(std::type_index(typeid(Interface))) != services_.end();
156 }
157
161 void clear() {
162 std::lock_guard<std::mutex> lock(mutex_);
163 services_.clear();
164 }
165
170 static service_container& global() {
171 static service_container instance;
172 return instance;
173 }
174
175private:
176 struct service_entry {
177 lifetime lifetime_scope;
178 mutable std::shared_ptr<void> instance;
179 std::function<std::shared_ptr<void>()> factory;
180 };
181
182 mutable std::mutex mutex_;
183 std::unordered_map<std::type_index, service_entry> services_;
184};
