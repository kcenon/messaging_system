93 hazard_pointer_manager::hazard_pointer_manager(size_t max_threads, size_t pointers_per_thread)
94 : max_threads_(max_threads)
95 , pointers_per_thread_(pointers_per_thread)
96 {
97 // Pre-allocate hazard records for all anticipated threads
98 for (size_t i = 0; i < max_threads; ++i) {
99 auto* record = new HazardRecord(pointers_per_thread);
100
101 // Add record to the global list using lock-free insertion
102 auto* head = head_record_.load(std::memory_order_acquire);
103 do {
104 record->next.store(head, std::memory_order_relaxed);
105 } while (!head_record_.compare_exchange_weak(head, record,
106 std::memory_order_release,
107 std::memory_order_acquire));
108 }
109 }

127 while (record) {
128 auto* next = record->next.load(std::memory_order_acquire);
129 delete record;
130 record = next;
131 }

250 auto hazard_pointer_manager::scan_and_reclaim() -> void
251 {
252 // Collect all currently active hazard pointers from all threads
253 auto hazards = collect_hazard_pointers();
254 std::sort(hazards.begin(), hazards.end());
255
256 // Process the thread-local retired list
257 std::vector<RetiredNode> new_retired_list;
258
259 for (auto& retired : retired_list_) {
260 if (std::binary_search(hazards.begin(), hazards.end(), retired.ptr)) {
261 // Memory is still protected by a hazard pointer, defer reclamation
262 new_retired_list.push_back(std::move(retired));
263 } else {
264 // No hazard pointer protects this memory, safe to reclaim
265 retired.deleter(retired.ptr);
266 total_reclaimed_.fetch_add(1, std::memory_order_relaxed);
267 }
268 }
269
270 // Update retired list and scan timestamp
271 retired_list_ = std::move(new_retired_list);
272 last_scan_ = std::chrono::steady_clock::now();
273 }

275 auto hazard_pointer_manager::get_statistics() const -> statistics
276 {
277 statistics stats;
278
279 // Count active hazard pointers
280 size_t active_count = 0;
281 auto* record = head_record_.load(std::memory_order_acquire);
282 while (record) {
283 if (record->owner.load(std::memory_order_acquire) != std::thread::id{}) {
284 for (const auto& hazard : record->hazards) {
285 if (hazard.load(std::memory_order_acquire) != nullptr) {
286 ++active_count;
287 }
288 }
289 }
290 record = record->next.load(std::memory_order_acquire);
291 }
292
293 stats.active_hazard_pointers = active_count;
294 stats.retired_list_size = retired_list_.size();
295 stats.total_reclaimed = total_reclaimed_.load(std::memory_order_relaxed);
296 stats.total_retired = total_retired_.load(std::memory_order_relaxed);
297
298 return stats;
299 }

318 auto hazard_pointer_manager::acquire_record() -> HazardRecord*
319 {
320 // Try to reuse existing thread-local record if still owned
321 if (local_record_ && local_record_->owner.load(std::memory_order_acquire) == std::this_thread::get_id()) {
322 return local_record_;
323 }
324
325 // Search for an unused record in the global list
326 auto* record = head_record_.load(std::memory_order_acquire);
327 while (record) {
328 std::thread::id expected{};
329 if (record->owner.compare_exchange_strong(expected, std::this_thread::get_id(),
330 std::memory_order_acquire,
331 std::memory_order_relaxed)) {
332 // Successfully claimed the record
333 local_record_ = record;
334 return record;
335 }
336 record = record->next.load(std::memory_order_acquire);
337 }
338
339 // No free records available - system is at capacity
340 throw std::runtime_error("No free hazard records available");
341 }

343 auto hazard_pointer_manager::release_record(HazardRecord* record) -> void
344 {
345 if (record) {
346 // Clear all hazard pointers
347 for (auto& hazard : record->hazards) {
348 hazard.store(nullptr, std::memory_order_release);
349 }
350
351 // Release ownership
352 record->owner.store(std::thread::id{}, std::memory_order_release);
353
354 if (local_record_ == record) {
355 local_record_ = nullptr;
356 }
357 }

360 auto hazard_pointer_manager::acquire_slot() -> std::atomic<void*>*
361 {
362 auto* record = acquire_record();
363
364 // Find an empty slot
365 for (auto& hazard : record->hazards) {
366 void* expected = nullptr;
367 if (hazard.compare_exchange_strong(expected, reinterpret_cast<void*>(1),
368 std::memory_order_acquire,
369 std::memory_order_relaxed)) {
370 hazard.store(nullptr, std::memory_order_release);
371 return &hazard;
372 }
373 }
374
375 // No free slots in record
376 throw std::runtime_error("No free hazard pointer slots");
377 }

397 auto hazard_pointer_manager::collect_hazard_pointers() const -> std::vector<void*>
398 {
399 std::vector<void*> hazards;
400
401 auto* record = head_record_.load(std::memory_order_acquire);
402 while (record) {
403 if (record->owner.load(std::memory_order_acquire) != std::thread::id{}) {
404 for (const auto& hazard : record->hazards) {
405 void* ptr = hazard.load(std::memory_order_acquire);
406 if (ptr && ptr != reinterpret_cast<void*>(1)) {
407 hazards.push_back(ptr);
408 }
409 }
410 }
411 record = record->next.load(std::memory_order_acquire);
412 }
413
414 return hazards;
415 }

417 auto hazard_pointer_manager::should_scan() const -> bool
418 {
419 // Scan if retired list is large
420 if (retired_list_.size() >= RETIRED_THRESHOLD) {
421 return true;
422 }
423
424 // Scan if enough time has passed
425 auto now = std::chrono::steady_clock::now();
426 if (now - last_scan_ >= SCAN_INTERVAL) {
427 return true;
428 }
429
430 return false;
