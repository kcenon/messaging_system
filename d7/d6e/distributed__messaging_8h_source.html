178 while (heartbeat_running_) {

181 std::unique_lock<std::mutex> lock(heartbeat_mutex_);

186 void check_node_health() {
187 std::shared_lock<std::shared_mutex> lock(nodes_mutex_);
188 std::vector<std::string> failed_nodes;
189
190 for (const auto& [id, node] : nodes_) {
191 if (!node->is_alive() && node->status == node_status::active) {
192 node->status = node_status::failed;
193 failed_nodes.push_back(id);
194 }
195 }
196
197 lock.unlock();
198
199 // Notify about failed nodes
200 for (const auto& node_id : failed_nodes) {
201 auto node = get_node(node_id);
202 if (node && node_failed_callback_) {
203 node_failed_callback_(*node);
204 }

343 // Message distribution
344 bool distribute_message(const core::message& message) {
345 auto active_nodes = membership_->get_active_nodes();
346 if (active_nodes.empty()) {
347 return false; // No available nodes
348 }
349
350 // Select primary node for the message
351 auto* primary_node = load_balancer_->select_node(active_nodes, message);
352 if (!primary_node) {
353 return false;
354 }
355
356 // Forward to primary node
357 if (message_forwarder_) {
358 message_forwarder_(message, primary_node->node_id);
359 messages_distributed_++;
360 primary_node->message_count++;
361 }
362
363 // Handle replication
364 if (enable_replication_ && replication_factor_ > 1) {
365 replicate_message(message, primary_node, active_nodes);
366 }
367
368 return true;
369 }
