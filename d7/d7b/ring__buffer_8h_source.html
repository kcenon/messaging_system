53 explicit ring_buffer(std::size_t capacity)
54 : buffer_(capacity)

56 , head_(0)

59 }

66 bool push(const T& value) {
67 std::lock_guard<std::mutex> lock(mutex_);
68
69 if (size_ >= capacity_) {
70 // Overwrite oldest element
71 tail_ = (tail_ + 1) % capacity_;
72 } else {
73 size_++;
74 }
75
76 buffer_[head_] = value;
77 head_ = (head_ + 1) % capacity_;
78
79 return true;
80 }
81
87 std::vector<T> get_recent(std::size_t count) const {
88 std::lock_guard<std::mutex> lock(mutex_);
89
90 count = std::min(count, size_);
91 std::vector<T> result;
92 result.reserve(count);
93
94 // Start from the most recent element
95 std::size_t index = (head_ + capacity_ - 1) % capacity_;
96
97 for (std::size_t i = 0; i < count; ++i) {
98 result.push_back(buffer_[index]);
99 index = (index + capacity_ - 1) % capacity_;
100 }
101
102 return result;
103 }
104
109 std::vector<T> get_all() const {
110 std::lock_guard<std::mutex> lock(mutex_);
111
112 std::vector<T> result;
113 result.reserve(size_);
114
115 if (size_ == 0) {
116 return result;
117 }
118
119 std::size_t index = tail_;
120 for (std::size_t i = 0; i < size_; ++i) {
121 result.push_back(buffer_[index]);
122 index = (index + 1) % capacity_;
123 }
124
125 return result;
126 }

131 void clear() {
132 std::lock_guard<std::mutex> lock(mutex_);
133 head_ = 0;
134 tail_ = 0;
135 size_ = 0;

162 }

168 bool full() const {
169 std::lock_guard<std::mutex> lock(mutex_);

171 }
