65// Helper function to create thread pool
66auto create_default(const uint16_t& worker_counts)
67 -> std::tuple<std::shared_ptr<kcenon::thread::thread_pool>, std::optional<std::string>>
68{
69 std::shared_ptr<kcenon::thread::thread_pool> pool;
70 try {
71 pool = std::make_shared<kcenon::thread::thread_pool>();
72 } catch (const std::bad_alloc& e) {
73 return { nullptr, std::string(e.what()) };
74 }
75
76 std::vector<std::unique_ptr<kcenon::thread::thread_worker>> workers;
77 workers.reserve(worker_counts);
78 for (uint16_t i = 0; i < worker_counts; ++i) {
79 workers.push_back(std::make_unique<kcenon::thread::thread_worker>());
80 }
81
82 auto enqueue_result = pool->enqueue_batch(std::move(workers));
83 if (enqueue_result.has_value()) {
84 return { nullptr, formatter::format("cannot enqueue to workers: {}",
85 enqueue_result.value()) };
86 }
87
88 return { pool, std::nullopt };
89}

92template<typename Type>
93auto create_priority_default(const uint16_t& worker_counts)
94 -> std::tuple<std::shared_ptr<typed_kcenon::thread::typed_thread_pool_t<Type>>, std::optional<std::string>>
95{
96 std::shared_ptr<typed_kcenon::thread::typed_thread_pool_t<Type>> pool;
97 try {
98 pool = std::make_shared<typed_kcenon::thread::typed_thread_pool_t<Type>>();
99 } catch (const std::bad_alloc& e) {
100 return { nullptr, std::string(e.what()) };
101 }
102
103 std::vector<std::unique_ptr<typed_kcenon::thread::typed_thread_worker_t<Type>>> workers;
104 workers.reserve(worker_counts);
105 for (uint16_t i = 0; i < worker_counts; ++i) {
106 workers.push_back(std::make_unique<typed_kcenon::thread::typed_thread_worker_t<Type>>(std::vector<Type>{}, true));
107 }

113 }
114
115 return { pool, std::nullopt };

122// Job complexity levels
123enum class JobComplexity {
124 Empty, // No operation
125 Trivial, // Simple arithmetic
126 Light, // Light computation
127 Medium, // Medium computation
128 Heavy, // Heavy computation
129 VeryHeavy, // Very heavy computation
130 Mixed // Random mix
131};

133// Job memory patterns
134enum class MemoryPattern {
135 None, // No memory allocation
136 Small, // Small allocations (<1KB)
137 Medium, // Medium allocations (1KB-100KB)
138 Large, // Large allocations (100KB-1MB)
139 VeryLarge, // Very large allocations (>1MB)
140 Random // Random sizes

143// Helper function to execute jobs with a specific complexity
144static void execute_job_with_complexity(JobComplexity complexity) {
145 switch (complexity) {
146 case JobComplexity::Empty:
147 // No operation
148 break;
149
150 case JobComplexity::Trivial:
151 {
152 volatile int x = 42;
153 x = x * 2 + 1;
154 }
155 break;
156
157 case JobComplexity::Light:
158 {
159 volatile double sum = 0;
160 for (int i = 0; i < 100; ++i) {
161 sum += std::sqrt(i);
162 }
163 }
164 break;
165
166 case JobComplexity::Medium:
167 {
168 volatile double sum = 0;
169 for (int i = 0; i < 1000; ++i) {
170 sum += std::sin(i) * std::cos(i);
171 }
172 }
173 break;
174
175 case JobComplexity::Heavy:
176 {
177 volatile double sum = 0;
178 for (int i = 0; i < 10000; ++i) {
179 sum += std::pow(std::sin(i), 2) + std::pow(std::cos(i), 2);
180 }
181 }
182 break;
183
184 case JobComplexity::VeryHeavy:
185 {
186 volatile double sum = 0;
187 for (int i = 0; i < 100000; ++i) {
188 sum += std::log(std::abs(std::sin(i)) + 1) * std::exp(-i/10000.0);
189 }
190 }
191 break;
192
193 case JobComplexity::Mixed:
194 {
195 static std::random_device rd;
196 static std::mt19937 gen(rd());
197 static std::uniform_int_distribution<> dis(0, 4);
198
199 JobComplexity random_complexity = static_cast<JobComplexity>(dis(gen));
200 execute_job_with_complexity(random_complexity);
201 }
202 break;
203 }
204}

206// Helper function to allocate memory based on pattern
207static std::unique_ptr<char[]> allocate_with_pattern(MemoryPattern pattern) {
208 static std::random_device rd;
209 static std::mt19937 gen(rd());
210
211 size_t size = 0;
212
213 switch (pattern) {
214 case MemoryPattern::None:
215 return nullptr;
216
217 case MemoryPattern::Small:
218 size = std::uniform_int_distribution<>(100, 1024)(gen);
219 break;
220
221 case MemoryPattern::Medium:
222 size = std::uniform_int_distribution<>(1024, 102400)(gen);
223 break;
224
225 case MemoryPattern::Large:
226 size = std::uniform_int_distribution<>(102400, 1048576)(gen);
227 break;
228
229 case MemoryPattern::VeryLarge:
230 size = std::uniform_int_distribution<>(1048576, 10485760)(gen);
231 break;
232
233 case MemoryPattern::Random:
234 size = std::uniform_int_distribution<>(100, 10485760)(gen);
235 break;
236 }
237
238 if (size > 0) {
239 auto buffer = std::make_unique<char[]>(size);
240 // Touch memory to ensure allocation
241 for (size_t i = 0; i < size; i += 4096) {
242 buffer[i] = static_cast<char>(i & 0xFF);
243 }
244 return buffer;
245 }
246
247 return nullptr;
248}

253static void BM_JobComplexity(benchmark::State& state) {
254 const JobComplexity complexity = static_cast<JobComplexity>(state.range(0));
255 const size_t num_jobs = state.range(1);
256 const size_t worker_count = std::thread::hardware_concurrency();
257
258 for (auto _ : state) {
259 auto [pool, error] = create_default(worker_count);
260 if (error.has_value()) {
261 state.SkipWithError("Failed to create thread pool");
262 return;
263 }
264
265 pool->start();
266
267 std::atomic<size_t> completed{0};
268
269 for (size_t i = 0; i < num_jobs; ++i) {
270 pool->enqueue(std::make_unique<callback_job>([complexity, &completed]() -> result_void {
271 execute_job_with_complexity(complexity);
272 completed.fetch_add(1);
273 return result_void();
274 }));
275 }
276
277 pool->stop();
278
279 benchmark::DoNotOptimize(completed.load());
280 }
281
282 state.SetItemsProcessed(state.iterations() * num_jobs);
283 state.counters["complexity"] = static_cast<int>(complexity);
284}

416 }
417
418 pool->stop();
419 }

422 state.counters["batch_size"] = batch_size;

487static void BM_JobDependencies(benchmark::State& state) {
488 const size_t chain_length = state.range(0);
489 const size_t num_chains = state.range(1);
490 const size_t worker_count = std::thread::hardware_concurrency();
491
492 for (auto _ : state) {
493 auto [pool, error] = create_default(worker_count);
494 if (error.has_value()) {
495 state.SkipWithError("Failed to create thread pool");
496 return;
497 }
498
499 pool->start();
500
501 std::atomic<size_t> completed{0};
502
503 for (size_t chain = 0; chain < num_chains; ++chain) {
504 // Create chain of dependent jobs
505 std::vector<std::promise<void>> promises(chain_length);
506 std::vector<std::future<void>> futures;
507
508 for (auto& promise : promises) {
509 futures.push_back(promise.get_future());
510 }
511
512 for (size_t i = 0; i < chain_length; ++i) {
513 pool->enqueue(std::make_unique<callback_job>([i, &futures, &promises, &completed, chain_length]() -> result_void {
514 // Wait for previous job in chain
515 if (i > 0) {
516 futures[i-1].get();
517 }
518
519 execute_job_with_complexity(JobComplexity::Light);
520 completed.fetch_add(1);
521
522 // Signal completion
523 if (i < chain_length) {
524 promises[i].set_value();
525 }
526 return result_void();
527 }));
528 }
529 }
530
531 // Wait for all jobs to complete
532 while (completed.load() < num_chains * chain_length) {
533 std::this_thread::sleep_for(std::chrono::microseconds(100));
534 }
535
536 pool->stop();
537 }
538
539 state.SetItemsProcessed(state.iterations() * num_chains * chain_length);
540 state.counters["chain_length"] = chain_length;
541 state.counters["num_chains"] = num_chains;
542}

554static void BM_PriorityImpact(benchmark::State& state) {
555 using Priority = int;
556 constexpr Priority Critical = 1;
557 constexpr Priority High = 10;
558 constexpr Priority Normal = 50;
559 constexpr Priority Low = 100;
560 constexpr Priority Background = 1000;
561
562 const size_t jobs_per_priority = state.range(0);
563 const size_t worker_count = std::thread::hardware_concurrency();
564
565 for (auto _ : state) {
566 auto [pool, error] = create_priority_default<Priority>(worker_count);
567 if (error.has_value()) {
568 state.SkipWithError("Failed to create typed thread pool");
569 return;
570 }
571
572 pool->start();
573
574 std::map<Priority, std::atomic<size_t>> completed;
575 for (auto p : {Critical, High, Normal, Low, Background}) {
576 completed[p] = 0;
577 }
578
579 // Submit jobs with different priorities
580 for (size_t i = 0; i < jobs_per_priority; ++i) {
581 for (auto priority : {Critical, High, Normal, Low, Background}) {
582 pool->enqueue(std::make_unique<typed_kcenon::thread::callback_typed_job_t<Priority>>([&completed, priority]() -> result_void {
583 execute_job_with_complexity(JobComplexity::Light);
584 completed[priority].fetch_add(1);
585 return result_void();
586 }, priority));
587 }
588 }
589
590 pool->stop();
591
592 // Report completion counts by priority
593 state.counters["critical"] = completed[Critical].load();
594 state.counters["high"] = completed[High].load();
595 state.counters["normal"] = completed[Normal].load();
596 state.counters["low"] = completed[Low].load();
597 state.counters["background"] = completed[Background].load();
598 }
599
600 state.SetItemsProcessed(state.iterations() * jobs_per_priority * 5);
601}

609static void BM_MixedWorkload(benchmark::State& state) {
610 const double cpu_light_pct = state.range(0);
611 const double cpu_heavy_pct = state.range(1);
612 const double io_pct = state.range(2);
613 const double memory_pct = 100.0 - cpu_light_pct - cpu_heavy_pct - io_pct;
614
615 const size_t total_jobs = 10000;
616 const size_t worker_count = std::thread::hardware_concurrency();
617
618 auto generate_job = [=]() -> std::function<void()> {
619 static std::random_device rd;
620 static std::mt19937 gen(rd());
621 static std::uniform_real_distribution<> dis(0.0, 100.0);
622
623 double roll = dis(gen);
624
625 if (roll < cpu_light_pct) {
626 return [] { execute_job_with_complexity(JobComplexity::Light); };
627 } else if (roll < cpu_light_pct + cpu_heavy_pct) {
628 return [] { execute_job_with_complexity(JobComplexity::Heavy); };
629 } else if (roll < cpu_light_pct + cpu_heavy_pct + io_pct) {
630 return [] { std::this_thread::sleep_for(std::chrono::milliseconds(5)); };
631 } else {
632 return [] {
633 auto buffer = allocate_with_pattern(MemoryPattern::Medium);
634 execute_job_with_complexity(JobComplexity::Light);
635 };
636 }
637 };
638
639 for (auto _ : state) {
640 auto [pool, error] = create_default(worker_count);
641 if (error.has_value()) {
642 state.SkipWithError("Failed to create thread pool");
643 return;
644 }
645
646 pool->start();
647
648 for (size_t i = 0; i < total_jobs; ++i) {
649 auto job = generate_job();
650 pool->enqueue(std::make_unique<callback_job>([job]() -> result_void {
651 job();
652 return result_void();
653 }));
654 }
655
656 pool->stop();
657 }
658
659 state.SetItemsProcessed(state.iterations() * total_jobs);
660 state.counters["cpu_light%"] = cpu_light_pct;
661 state.counters["cpu_heavy%"] = cpu_heavy_pct;
662 state.counters["io%"] = io_pct;
663 state.counters["memory%"] = memory_pct;
664}

678static void BM_SustainedThroughput(benchmark::State& state) {
679 const int duration_seconds = state.range(0);
680 const size_t worker_count = std::thread::hardware_concurrency();
681
682 for (auto _ : state) {
683 auto [pool, error] = create_default(worker_count);
684 if (error.has_value()) {
685 state.SkipWithError("Failed to create thread pool");
686 return;
687 }
688
689 pool->start();
690
691 std::atomic<size_t> jobs_submitted{0};
692 std::atomic<size_t> jobs_completed{0};
693 std::atomic<bool> running{true};
694
695 // Job submission thread
696 std::thread submitter([&pool, &jobs_submitted, &running, &jobs_completed]() {
697 while (running.load()) {
698 pool->enqueue(std::make_unique<callback_job>([&jobs_completed]() -> result_void {
699 execute_job_with_complexity(JobComplexity::Medium);
700 jobs_completed.fetch_add(1);
701 return result_void();
702 }));
703 jobs_submitted.fetch_add(1);
704
705 // Small delay to prevent overwhelming
706 if (jobs_submitted.load() % 1000 == 0) {
707 std::this_thread::sleep_for(std::chrono::microseconds(10));
708 }
709 }
710 });
711
712 // Run for specified duration
713 std::this_thread::sleep_for(std::chrono::seconds(duration_seconds));
714
715 running = false;
716 submitter.join();
717 pool->stop();
718
719 state.counters["jobs_completed"] = jobs_completed.load();
720 state.counters["jobs_per_second"] = jobs_completed.load() / duration_seconds;
721 }
722
723 state.SetIterationTime(duration_seconds);
724}

726 ->Arg(5) // 5 second test
727 ->Arg(10) // 10 second test
728 ->UseManualTime()
729 ->Unit(benchmark::kSecond);
