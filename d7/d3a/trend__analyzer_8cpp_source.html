16trend_analyzer::trend_result
17trend_analyzer::analyze_trend(const std::vector<trend_point>& points) {
18 trend_result result{};
19
20 if (points.size() < 2) {
21 return result;
22 }
23
24 // ì‹œê°„ì„ ì´ˆ ë‹¨ìœ„ë¡œ ë³€í™˜
25 std::vector<double> x_values, y_values;
26 auto start_time = points.front().timestamp;
27
28 for (const auto& point : points) {
29 auto duration = point.timestamp - start_time;
30 x_values.push_back(
31 std::chrono::duration_cast<std::chrono::seconds>(duration).count());
32 y_values.push_back(point.value);
33 }
34
35 // í‰ê·  ê³„ì‚°
36 double x_mean = std::accumulate(x_values.begin(), x_values.end(), 0.0) / x_values.size();
37 double y_mean = std::accumulate(y_values.begin(), y_values.end(), 0.0) / y_values.size();
38 result.mean = y_mean;
39
40 // ì„ í˜• íšŒê·€
41 double numerator = 0.0, denominator = 0.0;
42 for (size_t i = 0; i < x_values.size(); ++i) {
43 numerator += (x_values[i] - x_mean) * (y_values[i] - y_mean);
44 denominator += (x_values[i] - x_mean) * (x_values[i] - x_mean);
45 }
46
47 if (denominator != 0) {
48 result.slope = numerator / denominator;
49 result.intercept = y_mean - result.slope * x_mean;
50 }
51
52 // í‘œì¤€í¸ì°¨ ë° R-squared ê³„ì‚°
53 double ss_tot = 0.0, ss_res = 0.0, variance = 0.0;
54 for (size_t i = 0; i < y_values.size(); ++i) {
55 double y_pred = result.slope * x_values[i] + result.intercept;
56 ss_tot += (y_values[i] - y_mean) * (y_values[i] - y_mean);
57 ss_res += (y_values[i] - y_pred) * (y_values[i] - y_pred);
58 variance += (y_values[i] - y_mean) * (y_values[i] - y_mean);
59 }
60
61 result.std_deviation = std::sqrt(variance / y_values.size());
62 result.r_squared = (ss_tot != 0) ? (1.0 - ss_res / ss_tot) : 0.0;
63
64 // íŠ¸ë Œë“œ ë°©í–¥ ë° ê°•ë„
65 double slope_per_minute = result.slope * 60; // ë¶„ë‹¹ ë³€í™”ìœ¨
66 if (std::abs(slope_per_minute) < 0.01 * result.mean) {
67 result.trend_direction = "stable";
68 } else if (slope_per_minute > 0) {
69 result.trend_direction = "increasing";
70 } else {
71 result.trend_direction = "decreasing";
72 }
73
74 result.trend_strength = std::min(100.0, std::abs(result.r_squared) * 100.0);
75
76 return result;
77}

124std::vector<trend_analyzer::anomaly_result>
125trend_analyzer::detect_anomalies(const std::vector<trend_point>& points, double threshold) {
126 std::vector<anomaly_result> anomalies;
127
128 if (points.size() < 10) { // ìµœì†Œ ë°ì´í„° í•„ìš”
129 return anomalies;
130 }
131
132 // í‰ê·  ë° í‘œì¤€í¸ì°¨ ê³„ì‚°
133 double mean = 0.0;
134 for (const auto& point : points) {
135 mean += point.value;
136 }
137 mean /= points.size();
138
139 double variance = 0.0;
140 for (const auto& point : points) {
141 variance += (point.value - mean) * (point.value - mean);
142 }
143 double std_dev = std::sqrt(variance / points.size());
144
145 // Z-score ê¸°ë°˜ ì´ìƒ íƒì§€
146 for (const auto& point : points) {
147 double z_score = (std_dev != 0) ? (point.value - mean) / std_dev : 0;
148
149 if (std::abs(z_score) > threshold) {
150 anomaly_result anomaly;
151 anomaly.timestamp = point.timestamp;
152 anomaly.value = point.value;
153 anomaly.expected_value = mean;
154 anomaly.deviation = z_score;
155
156 if (std::abs(z_score) > 5) {
157 anomaly.severity = "severe";
158 } else if (std::abs(z_score) > 4) {
159 anomaly.severity = "moderate";
160 } else {
161 anomaly.severity = "minor";
162 }
163
164 std::stringstream desc;
165 desc << "Value " << point.value << " deviates "
166 << std::abs(z_score) << " standard deviations from mean";
167 anomaly.description = desc.str();
168
169 anomalies.push_back(anomaly);
170 }
171 }
172
173 return anomalies;
174}

176std::vector<trend_analyzer::prediction_result>
177trend_analyzer::predict_linear(
178 const std::vector<trend_point>& points,
179 duration future_duration,
180 std::size_t num_predictions) {
181
182 std::vector<prediction_result> predictions;
183
184 if (points.size() < 3) {
185 return predictions;
186 }

188 // íŠ¸ë Œë“œ ë¶„ì„

191 // ì˜ˆì¸¡ ì‹ ë¢°êµ¬ê°„ ê³„ì‚°ì„ ìœ„í•œ í‘œì¤€ ì˜¤ì°¨

193
194 // ì‹œì‘ ì‹œê°„ê³¼ ê°„ê²© ê³„ì‚°
195 auto last_time = points.back().timestamp;
196 auto time_step = future_duration / num_predictions;
197 auto start_time = points.front().timestamp;
198
199 for (size_t i = 1; i <= num_predictions; ++i) {
200 prediction_result pred;
201 pred.timestamp = last_time + (time_step * i);
202
203 // ì‹œê°„ì„ ì´ˆ ë‹¨ìœ„ë¡œ ë³€í™˜
204 auto duration_seconds = std::chrono::duration_cast<std::chrono::seconds>(
205 pred.timestamp - start_time).count();
206
207 // ì˜ˆì¸¡ê°’ ê³„ì‚°
208 pred.predicted_value = trend.slope * duration_seconds + trend.intercept;
209
210 // 95% ì‹ ë¢°êµ¬ê°„ (ì•½ 2 í‘œì¤€ì˜¤ì°¨)
211 double margin = 1.96 * se * std::sqrt(1.0 + 1.0/points.size() +
212 std::pow(duration_seconds - duration_seconds/2, 2) /
213 (points.size() * trend.std_deviation * trend.std_deviation));
214
215 pred.confidence_lower = pred.predicted_value - margin;
216 pred.confidence_upper = pred.predicted_value + margin;
217 pred.confidence_level = 0.95;
218
219 predictions.push_back(pred);
220 }

222 return predictions;

225std::size_t trend_analyzer::detect_seasonality(
226 const std::vector<trend_point>& points,
227 std::size_t period_hint) {
228
229 if (points.size() < 20) { // ìµœì†Œ ë°ì´í„° í•„ìš”
230 return 0;
231 }
232
233 // ìê¸°ìƒê´€ í•¨ìˆ˜ ê³„ì‚°
234 std::vector<double> values;
235 for (const auto& point : points) {
236 values.push_back(point.value);
237 }
238
239 double mean = std::accumulate(values.begin(), values.end(), 0.0) / values.size();
240
241 // í›„ë³´ ì£¼ê¸° ë²”ìœ„
242 size_t min_period = period_hint > 0 ? period_hint / 2 : 2;
243 size_t max_period = period_hint > 0 ? period_hint * 2 : values.size() / 3;
244
245 double max_correlation = 0.0;
246 size_t best_period = 0;
247
248 for (size_t lag = min_period; lag <= max_period && lag < values.size(); ++lag) {
249 double correlation = 0.0;
250 double variance1 = 0.0, variance2 = 0.0;
251
252 for (size_t i = 0; i < values.size() - lag; ++i) {
253 double diff1 = values[i] - mean;
254 double diff2 = values[i + lag] - mean;
255 correlation += diff1 * diff2;
256 variance1 += diff1 * diff1;
257 variance2 += diff2 * diff2;
258 }
259
260 if (variance1 > 0 && variance2 > 0) {
261 correlation /= std::sqrt(variance1 * variance2);
262
263 if (correlation > max_correlation && correlation > 0.5) {
264 max_correlation = correlation;
265 best_period = lag;
266 }
267 }
268 }
269
270 return best_period;
271}

273std::vector<std::size_t> trend_analyzer::detect_change_points(
274 const std::vector<trend_point>& points,
275 double sensitivity) {
276
277 std::vector<std::size_t> change_points;
278
279 if (points.size() < 10) {
280 return change_points;
281 }
282
283 // CUSUM (Cumulative Sum) ì•Œê³ ë¦¬ì¦˜
284 std::vector<double> values;
285 for (const auto& point : points) {
286 values.push_back(point.value);
287 }
288
289 double mean = std::accumulate(values.begin(), values.end(), 0.0) / values.size();
290 double std_dev = 0.0;
291 for (double val : values) {
292 std_dev += (val - mean) * (val - mean);
293 }
294 std_dev = std::sqrt(std_dev / values.size());
295
296 double threshold = (3.0 - 2.0 * sensitivity) * std_dev; // ë¯¼ê°ë„ì— ë”°ë¥¸ ì„ê³„ê°’
297
298 double cusum_pos = 0.0, cusum_neg = 0.0;
299 for (size_t i = 1; i < values.size(); ++i) {
300 double diff = values[i] - values[i-1];
301
302 cusum_pos = std::max(0.0, cusum_pos + diff - threshold/2);
303 cusum_neg = std::max(0.0, cusum_neg - diff - threshold/2);
304
305 if (cusum_pos > threshold || cusum_neg > threshold) {
306 change_points.push_back(i);
307 cusum_pos = 0.0;
308 cusum_neg = 0.0;
309 }
310 }
311
312 return change_points;
313}

315double trend_analyzer::calculate_correlation(
316 const std::vector<trend_point>& series1,
317 const std::vector<trend_point>& series2) {
318
319 if (series1.size() != series2.size() || series1.empty()) {
320 return 0.0;
321 }
322
323 double mean1 = 0.0, mean2 = 0.0;
324 for (size_t i = 0; i < series1.size(); ++i) {
325 mean1 += series1[i].value;
326 mean2 += series2[i].value;
327 }
328 mean1 /= series1.size();
329 mean2 /= series2.size();
330
331 double covariance = 0.0, variance1 = 0.0, variance2 = 0.0;
332 for (size_t i = 0; i < series1.size(); ++i) {
333 double diff1 = series1[i].value - mean1;
334 double diff2 = series2[i].value - mean2;
335 covariance += diff1 * diff2;
336 variance1 += diff1 * diff1;
337 variance2 += diff2 * diff2;
338 }
339
340 if (variance1 == 0 || variance2 == 0) {
341 return 0.0;
342 }
343
344 return covariance / std::sqrt(variance1 * variance2);
345}
