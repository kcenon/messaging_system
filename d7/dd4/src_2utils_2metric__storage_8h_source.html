39struct metric_storage_config {
40 size_t ring_buffer_capacity = 8192; // Ring buffer size per metric
41 size_t max_metrics = 10000; // Maximum number of different metrics
42 std::chrono::seconds retention_period{3600}; // Data retention period
43 std::chrono::milliseconds flush_interval{1000}; // Flush interval for persistence
44 bool enable_compression = true; // Enable data compression
45 bool enable_background_processing = true; // Enable background processing
46 size_t batch_size = 256; // Batch size for processing
47
51 result_void validate() const {
52 if (ring_buffer_capacity == 0 || (ring_buffer_capacity & (ring_buffer_capacity - 1)) != 0) {
53 return result_void(monitoring_error_code::invalid_configuration,
54 "Ring buffer capacity must be a power of 2");
55 }
56
57 if (max_metrics == 0) {
58 return result_void(monitoring_error_code::invalid_configuration,
59 "Max metrics must be positive");
60 }
61
62 if (retention_period.count() <= 0) {
63 return result_void(monitoring_error_code::invalid_configuration,
64 "Retention period must be positive");
65 }
66
67 if (flush_interval.count() <= 0) {
68 return result_void(monitoring_error_code::invalid_configuration,
69 "Flush interval must be positive");
70 }
71
72 return result_void::success();
73 }
74};

80struct metric_storage_stats {
81 std::atomic<size_t> total_metrics_stored{0};
82 std::atomic<size_t> total_metrics_dropped{0};
83 std::atomic<size_t> active_metric_series{0};
84 std::atomic<size_t> memory_usage_bytes{0};
85 std::atomic<size_t> compression_saves_bytes{0};
86 std::atomic<size_t> background_flushes{0};
87 std::atomic<size_t> storage_errors{0};
88 std::chrono::system_clock::time_point creation_time;
89
90 metric_storage_stats() : creation_time(std::chrono::system_clock::now()) {}
91
95 double get_storage_efficiency() const {
96 auto stored = total_metrics_stored.load();
97 auto dropped = total_metrics_dropped.load();
98 auto total = stored + dropped;
99 return total > 0 ? (static_cast<double>(stored) / total) * 100.0 : 100.0;
100 }
101
105 double get_avg_memory_per_metric() const {
106 auto active = active_metric_series.load();
107 auto memory = memory_usage_bytes.load();
108 return active > 0 ? static_cast<double>(memory) / active : 0.0;
109 }

118 struct metric_series_entry {
119 std::unique_ptr<ring_buffer<compact_metric_value>> buffer;
120 std::unique_ptr<time_series> series;
121 std::chrono::system_clock::time_point last_access;
122 metric_metadata metadata;
123
124 metric_series_entry(metric_metadata meta, const metric_storage_config& config)
125 : last_access(std::chrono::system_clock::now()), metadata(meta) {
126
127 ring_buffer_config ring_config;
128 ring_config.capacity = config.ring_buffer_capacity;
129 ring_config.overwrite_old = true;
130 ring_config.batch_size = config.batch_size;
131
132 buffer = std::make_unique<ring_buffer<compact_metric_value>>(ring_config);
133
134 time_series_config ts_config;
135 ts_config.retention_period = config.retention_period;
136 ts_config.enable_compression = config.enable_compression;
137
138 series = std::make_unique<time_series>("metric_" + std::to_string(meta.name_hash), ts_config);
139 }
140 };
141
142 mutable std::mutex storage_mutex_;
143 std::unordered_map<uint32_t, std::unique_ptr<metric_series_entry>> metric_series_;
144 metric_storage_config config_;
145 metric_storage_stats stats_;
146
147 // Background processing
148 std::atomic<bool> running_{false};
149 std::thread background_thread_;
150 std::condition_variable background_cv_;
151 mutable std::mutex background_mutex_;
152
156 metric_series_entry* get_or_create_series(const metric_metadata& metadata) {
157 std::lock_guard<std::mutex> lock(storage_mutex_);
158
159 auto it = metric_series_.find(metadata.name_hash);
160 if (it != metric_series_.end()) {
161 it->second->last_access = std::chrono::system_clock::now();
162 return it->second.get();
163 }
164
165 // Check if we've reached the maximum number of metrics
166 if (metric_series_.size() >= config_.max_metrics) {
167 stats_.storage_errors.fetch_add(1, std::memory_order_relaxed);
168 return nullptr; // Storage full
169 }
170
171 // Create new series
172 auto entry = std::make_unique<metric_series_entry>(metadata, config_);
173 auto* ptr = entry.get();
174
175 metric_series_[metadata.name_hash] = std::move(entry);
176 stats_.active_metric_series.store(metric_series_.size(), std::memory_order_relaxed);
177
178 return ptr;
179 }
180
184 void background_processing_loop() {
185 while (running_.load(std::memory_order_acquire)) {
186 std::unique_lock<std::mutex> lock(background_mutex_);
187
188 // Wait for flush interval or shutdown
189 if (background_cv_.wait_for(lock, config_.flush_interval,
190 [this] { return !running_.load(std::memory_order_acquire); })) {
191 break; // Shutdown requested
192 }
193
194 lock.unlock();
195
196 // Perform background tasks
197 flush_ring_buffers_to_series();
198 cleanup_old_series();
199 update_memory_usage();
200
201 stats_.background_flushes.fetch_add(1, std::memory_order_relaxed);
202 }
203 }
204
208 void flush_ring_buffers_to_series() {
209 std::vector<uint32_t> series_to_flush;
210
211 {
212 std::lock_guard<std::mutex> lock(storage_mutex_);
213 series_to_flush.reserve(metric_series_.size());
214
215 for (const auto& [hash, entry] : metric_series_) {
216 if (!entry->buffer->empty()) {
217 series_to_flush.push_back(hash);
218 }
219 }
220 }
221
222 // Process each series outside the lock
223 for (uint32_t hash : series_to_flush) {
224 std::lock_guard<std::mutex> lock(storage_mutex_);
225 auto it = metric_series_.find(hash);
226 if (it == metric_series_.end()) continue;
227
228 auto& entry = it->second;
229 std::vector<compact_metric_value> batch;
230
231 // Read batch from ring buffer
232 size_t read_count = entry->buffer->read_batch(batch, config_.batch_size);
233
234 if (read_count > 0) {
235 // Convert and add to time series
236 std::vector<time_point_data> time_points;
237 time_points.reserve(read_count);
238
239 for (const auto& metric : batch) {
240 time_points.emplace_back(
241 metric.get_timestamp(),
242 metric.as_double()
243 );
244 }
245
246 entry->series->add_points(time_points);
247 }
248 }
249 }
250
254 void cleanup_old_series() {
255 auto cutoff = std::chrono::system_clock::now() - config_.retention_period;
256 std::vector<uint32_t> to_remove;
257
258 {
259 std::lock_guard<std::mutex> lock(storage_mutex_);
260
261 for (const auto& [hash, entry] : metric_series_) {
262 if (entry->last_access < cutoff && entry->buffer->empty()) {
263 to_remove.push_back(hash);
264 }
265 }
266
267 for (uint32_t hash : to_remove) {
268 metric_series_.erase(hash);
269 }
270
271 stats_.active_metric_series.store(metric_series_.size(), std::memory_order_relaxed);
272 }
273 }
274
278 void update_memory_usage() {
279 size_t total_memory = 0;
280
281 {
282 std::lock_guard<std::mutex> lock(storage_mutex_);
283
284 for (const auto& [hash, entry] : metric_series_) {
285 total_memory += entry->series->memory_footprint();
286 total_memory += sizeof(metric_series_entry);
287 // Ring buffer memory is fixed per series
288 total_memory += entry->buffer->capacity() * sizeof(compact_metric_value);
289 }
290 }
291
292 stats_.memory_usage_bytes.store(total_memory, std::memory_order_relaxed);
293 }

330 result_void store_metric(const std::string& name,
331 double value,
332 metric_type type = metric_type::gauge,
333 std::chrono::system_clock::time_point timestamp =
334 std::chrono::system_clock::now()) {
335
336 auto metadata = create_metric_metadata(name, type);
337 compact_metric_value metric(metadata, value);
338 metric.timestamp_us = std::chrono::duration_cast<std::chrono::microseconds>(
339 timestamp.time_since_epoch()).count();
340
341 auto* series = get_or_create_series(metadata);
342 if (!series) {
343 stats_.total_metrics_dropped.fetch_add(1, std::memory_order_relaxed);
344 return result_void(monitoring_error_code::storage_full,
345 "Storage capacity exceeded");
346 }
347
348 auto result = series->buffer->write(std::move(metric));
349 if (result) {
350 stats_.total_metrics_stored.fetch_add(1, std::memory_order_relaxed);
351 } else {
352 stats_.total_metrics_dropped.fetch_add(1, std::memory_order_relaxed);
353 }
354
355 return result;
356 }
357
361 size_t store_metrics_batch(const metric_batch& batch) {
362 size_t stored_count = 0;
363
364 for (const auto& metric : batch.metrics) {
365 auto* series = get_or_create_series(metric.metadata);
366 if (series) {
367 auto result = series->buffer->write(compact_metric_value(metric));
368 if (result) {
369 ++stored_count;
370 stats_.total_metrics_stored.fetch_add(1, std::memory_order_relaxed);
371 } else {
372 stats_.total_metrics_dropped.fetch_add(1, std::memory_order_relaxed);
373 }
374 } else {
375 stats_.total_metrics_dropped.fetch_add(1, std::memory_order_relaxed);
376 }
377 }
378
379 return stored_count;
380 }

385 result<aggregation_result> query_metric(const std::string& name,
386 const time_series_query& query) const {
387 auto hash = hash_metric_name(name);
388
389 std::lock_guard<std::mutex> lock(storage_mutex_);
390 auto it = metric_series_.find(hash);
391
392 if (it == metric_series_.end()) {
393 return make_error<aggregation_result>(monitoring_error_code::collector_not_found,
394 "Metric not found: " + name);
395 }
396
397 return it->second->series->query(query);

467 void shutdown() {
