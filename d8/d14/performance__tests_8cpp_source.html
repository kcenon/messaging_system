51 struct Statistics {
52 double mean;
53 double median;
54 double min;
55 double max;
56 double std_dev;
57 };
58
59 Statistics calculate_stats(const std::vector<double>& data) {
60 Statistics stats{};
61 if (data.empty()) return stats;
62
63 auto sorted_data = data;
64 std::sort(sorted_data.begin(), sorted_data.end());
65
66 stats.min = sorted_data.front();
67 stats.max = sorted_data.back();
68 stats.median = sorted_data[sorted_data.size() / 2];
69
70 stats.mean = std::accumulate(data.begin(), data.end(), 0.0) / data.size();
71
72 double variance = 0.0;
73 for (double value : data) {
74 variance += (value - stats.mean) * (value - stats.mean);
75 }
76 stats.std_dev = std::sqrt(variance / data.size());
77
78 return stats;
79 }
