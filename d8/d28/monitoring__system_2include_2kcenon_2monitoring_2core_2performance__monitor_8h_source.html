51 static std::chrono::nanoseconds calculate_percentile(
52 const std::vector<std::chrono::nanoseconds>& sorted_durations,
53 double percentile) {
54
55 if (sorted_durations.empty()) {
56 return std::chrono::nanoseconds::zero();
57 }
58
59 size_t index = static_cast<size_t>(
60 (percentile / 100.0) * (sorted_durations.size() - 1)
61 );
62 return sorted_durations[index];
63 }
64
68 void update_statistics(const std::vector<std::chrono::nanoseconds>& durations) {
69 if (durations.empty()) return;
70
71 auto sorted = durations;
72 std::sort(sorted.begin(), sorted.end());
73
74 min_duration = sorted.front();
75 max_duration = sorted.back();
76 median_duration = calculate_percentile(sorted, 50.0);
77 p95_duration = calculate_percentile(sorted, 95.0);
78 p99_duration = calculate_percentile(sorted, 99.0);
79
80 total_duration = std::accumulate(
81 sorted.begin(), sorted.end(),
82 std::chrono::nanoseconds::zero()
83 );
84
85 mean_duration = total_duration / sorted.size();
86 }

418 template<typename Func>
419 monitoring_system::result<performance_metrics> run(
420 const std::string& operation_name,
421 Func&& func
422 ) {
423 // Warmup
424 for (std::uint32_t i = 0; i < warmup_iterations_; ++i) {
425 func();
426 }
427
428 // Actual benchmark
429 for (std::uint32_t i = 0; i < iterations_; ++i) {
430 auto start = std::chrono::high_resolution_clock::now();
431
432 try {
433 func();
434 } catch (...) {
435 // Record error
436 auto end = std::chrono::high_resolution_clock::now();
437 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
438 end - start
439 );
440 profiler_.record_sample(operation_name, duration, false);
441 continue;
442 }
443
444 auto end = std::chrono::high_resolution_clock::now();
445 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
446 end - start
447 );
448 profiler_.record_sample(operation_name, duration, true);
449 }
450
451 return profiler_.get_metrics(operation_name);
452 }
