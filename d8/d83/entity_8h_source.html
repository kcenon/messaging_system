54 // C++20 concepts for type safety
55 template<typename T>
56 concept Entity = requires(T t) {
57 typename T::primary_key_type;
58 { t.table_name() } -> std::convertible_to<std::string>;
59 { t.get_metadata() } -> std::same_as<const entity_metadata&>;
60 };

62 template<typename T>
63 concept FieldType = std::is_same_v<T, int32_t> ||
64 std::is_same_v<T, int64_t> ||
65 std::is_same_v<T, double> ||
66 std::is_same_v<T, std::string> ||
67 std::is_same_v<T, bool> ||
68 std::is_same_v<T, std::chrono::system_clock::time_point>;

73 // Field constraint types
74 enum class field_constraint {
75 none = 0,
76 primary_key = 1,
77 not_null = 2,
78 unique = 4,
79 auto_increment = 8,
80 index = 16,
81 foreign_key = 32,
82 default_now = 64
83 };
84
85 inline field_constraint operator|(field_constraint a, field_constraint b) {
86 return static_cast<field_constraint>(static_cast<int>(a) | static_cast<int>(b));
87 }
88
89 inline bool has_constraint(field_constraint constraints, field_constraint check) {
90 return (static_cast<int>(constraints) & static_cast<int>(check)) != 0;
91 }
