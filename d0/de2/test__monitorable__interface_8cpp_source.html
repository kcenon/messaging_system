476TEST_F(MonitorableInterfaceTest, ThreadSafetyMonitorableComponent) {
477 test_monitorable_component component("thread_test");
478
479 const int thread_count = 10;
480 const int operations_per_thread = 1000;
481 std::vector<std::thread> threads;
482
483 // Launch threads that perform operations
484 for (int i = 0; i < thread_count; ++i) {
485 threads.emplace_back([&component]() {
486 for (int j = 0; j < operations_per_thread; ++j) {
487 component.perform_operation();
488
489 // Also get monitoring data periodically
490 if (j % 100 == 0) {
491 auto result = component.get_monitoring_data();
492 EXPECT_TRUE(result);
493 }
494 }
495 });
496 }
497
498 // Wait for all threads
499 for (auto& t : threads) {
500 t.join();
501 }
502
503 // Verify final count
504 EXPECT_EQ(component.get_operation_count(), thread_count * operations_per_thread);
505
506 // Get final monitoring data
507 auto result = component.get_monitoring_data();
508 ASSERT_TRUE(result);
509
510 auto op_count = result.value().get_metric("operation_count");
511 ASSERT_TRUE(op_count.has_value());
512 EXPECT_DOUBLE_EQ(op_count.value(),
513 static_cast<double>(thread_count * operations_per_thread));
514}
