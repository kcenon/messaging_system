56 bool enqueue(const message& msg) {
57 std::lock_guard<std::mutex> lock(mutex_);
58 if (size_locked() >= max_size_) {
59 return false; // Queue full
60 }
61
62 if (enable_priority_) {
63 priority_queue_.push(queued_message{msg, next_sequence_++});
64 } else {
65 queue_.push(msg);
66 }
67 condition_.notify_one();
68 return true;
69 }

71 bool dequeue(message& msg, std::chrono::milliseconds timeout) {
72 std::unique_lock<std::mutex> lock(mutex_);
73 while (!shutdown_) {
74 if (!condition_.wait_for(lock, timeout, [this] { return has_items_locked() || shutdown_; })) {
75 return false;
76 }
77
78 if (!has_items_locked()) {
79 continue;
80 }

87 continue;
88 }
89 } else {

91 queue_.pop();
92 }
93 return true;
94 }
95 return false;

115 struct priority_compare {
116 bool operator()(const queued_message& lhs, const queued_message& rhs) const {
117 auto lhs_priority = static_cast<int>(lhs.msg.metadata.priority);
118 auto rhs_priority = static_cast<int>(rhs.msg.metadata.priority);
119
120 if (lhs_priority == rhs_priority) {
121 // Earlier sequence should be processed first
122 return lhs.sequence > rhs.sequence;
123 }
124
125 // Higher numeric priority value should be processed first
126 return lhs_priority < rhs_priority;
127 }
128 };

130 bool has_items_locked() const {
131 return enable_priority_ ? !priority_queue_.empty() : !queue_.empty();
132 }

135 return enable_priority_ ? priority_queue_.size() : queue_.size();
136 }

138 bool maybe_delay_for_higher_priority(std::unique_lock<std::mutex>& lock, message& current_msg) {
139 if (!enable_priority_ || priority_reorder_window_.count() == 0) {
140 return false;
141 }
142
143 const int current_priority = static_cast<int>(current_msg.metadata.priority);
144 if (current_priority > static_cast<int>(message_priority::low)) {
145 return false;
146 }
147
148 const auto deadline = std::chrono::steady_clock::now() + priority_reorder_window_;
149 while (!shutdown_) {
150 if (!priority_queue_.empty()) {
151 int next_priority = static_cast<int>(priority_queue_.top().msg.metadata.priority);
152 if (next_priority > current_priority) {
153 priority_queue_.push(queued_message{current_msg, next_sequence_++});
154 condition_.notify_one();
155 return true;
156 }
157 }
158
159 if (std::chrono::steady_clock::now() >= deadline) {
160 return false;
161 }
162
163 if (condition_.wait_until(lock, deadline) == std::cv_status::timeout) {
164 return false;
165 }
166 }
167
168 return false;
169 }

176 const size_t max_size_;

179 bool shutdown_ = false;

188 void dispatch(const message& msg) {

191 auto handlers = router_->get_handlers(msg.payload.topic);
192 if (ordered_dispatch_) {
193 std::lock_guard<std::mutex> lock(dispatch_mutex_);
194 invoke_handlers(handlers, msg);
195 } else {
196 invoke_handlers(handlers, msg);
197 }
198 }

203 try {
204 handler(msg);
205 } catch (const std::exception& e) {
206 // Swallow and continue to maintain message flow
207 }
208 }
209 }
