175 metrics_snapshot get_process_snapshot(const process_identifier& process_id) const {
176 std::shared_lock lock(data_mutex_);
177
178 auto it = process_data_.find(process_id);
179 if (it == process_data_.end()) {
180 return metrics_snapshot{};
181 }
182
183 std::shared_lock proc_lock(it->second->mutex);
184
185 metrics_snapshot snapshot;
186 snapshot.capture_time = std::chrono::steady_clock::now();
187 snapshot.system = it->second->system_metrics;
188
189 // Thread pool Î©îÌä∏Î¶≠ ÏßëÍ≥Ñ
190 thread_pool_metrics aggregated_pool{};
191 for (const auto& [pool_id, pool_metrics] : it->second->pool_metrics) {
192 aggregated_pool.worker_threads += pool_metrics.worker_threads;
193 aggregated_pool.idle_threads += pool_metrics.idle_threads;
194 aggregated_pool.jobs_completed += pool_metrics.jobs_completed;
195 aggregated_pool.jobs_pending += pool_metrics.jobs_pending;
196 aggregated_pool.jobs_failed += pool_metrics.jobs_failed;
197 aggregated_pool.total_execution_time_ns += pool_metrics.total_execution_time_ns;
198
199 if (pool_metrics.jobs_completed > 0) {
200 aggregated_pool.average_latency_ns =
201 (aggregated_pool.average_latency_ns + pool_metrics.average_latency_ns) / 2;
202 }
203 }
204 snapshot.thread_pool = aggregated_pool;
205
206 // Worker Î©îÌä∏Î¶≠ ÏßëÍ≥Ñ
207 worker_metrics aggregated_worker{};
208 for (const auto& [worker_id, worker] : it->second->worker_metrics) {
209 aggregated_worker.jobs_processed += worker.jobs_processed;
210 aggregated_worker.total_processing_time_ns += worker.total_processing_time_ns;
211 aggregated_worker.idle_time_ns += worker.idle_time_ns;
212 }
213 snapshot.worker = aggregated_worker;
214
215 return snapshot;
216 }

219 std::unordered_map<std::string, double> compare_process_performance(
220 const std::vector<process_identifier>& process_ids) const {
221
222 std::shared_lock lock(data_mutex_);
223 std::unordered_map<std::string, double> results;
224
225 for (const auto& proc_id : process_ids) {

227 if (it == process_data_.end()) continue;

229 std::shared_lock proc_lock(it->second->mutex);
230
231 // CPU Ìö®Ïú®ÏÑ± Ï†êÏàò
232 double cpu_efficiency = calculate_cpu_efficiency(it->second->system_metrics);
233 results[proc_id.process_name + "_cpu_efficiency"] = cpu_efficiency;
234
235 // Î©îÎ™®Î¶¨ Ìö®Ïú®ÏÑ± Ï†êÏàò
236 double memory_efficiency = calculate_memory_efficiency(it->second->system_metrics);
237 results[proc_id.process_name + "_memory_efficiency"] = memory_efficiency;
238
239 // Ï≤òÎ¶¨Îüâ Ï†êÏàò
240 double throughput_score = calculate_throughput_score(it->second.get());
241 results[proc_id.process_name + "_throughput"] = throughput_score;
242 }
243
244 return results;

373 // Ìö®Ïú®ÏÑ± Í≥ÑÏÇ∞ Ìï®ÏàòÎì§
374 double calculate_cpu_efficiency(const system_metrics& metrics) const {
375 if (metrics.active_threads == 0) return 0.0;
376 return (100.0 - metrics.cpu_usage_percent) / metrics.active_threads;
377 }
378
379 double calculate_memory_efficiency(const system_metrics& metrics) const {
380 if (metrics.memory_usage_bytes == 0) return 100.0;
381 const uint64_t mb = 1024 * 1024;
382 return 100.0 / (1.0 + std::log10(metrics.memory_usage_bytes / mb));
383 }
384
385 double calculate_throughput_score(const process_monitoring_data* proc_data) const {
386 double total_jobs = 0;
387 double total_latency = 0;
388
389 for (const auto& [pool_id, pool_metrics] : proc_data->pool_metrics) {
390 total_jobs += pool_metrics.jobs_completed;
391 if (pool_metrics.jobs_completed > 0) {
392 total_latency += pool_metrics.average_latency_ns;
393 }
394 }
395
396 if (total_jobs == 0) return 0.0;
397 if (total_latency == 0) return total_jobs;
398
399 return total_jobs / (total_latency / 1000000.0); // jobs per millisecond
400 }

402 // ÏàòÏßë Î£®ÌîÑ
403 void collection_loop() {
404 while (is_active_.load()) {
405 auto start_time = std::chrono::steady_clock::now();
406
407 // Ïä§ÎÉÖÏÉ∑ ÏàòÏßë
408 collect_snapshots();
409
410 // Îã§Ïùå ÏàòÏßëÍπåÏßÄ ÎåÄÍ∏∞
411 auto elapsed = std::chrono::steady_clock::now() - start_time;
412 auto sleep_time = std::chrono::milliseconds(collection_interval_ms_) - elapsed;
413
414 if (sleep_time.count() > 0) {
415 std::this_thread::sleep_for(sleep_time);
416 }
417 }
418 }
419
420 void collect_snapshots() {
421 std::shared_lock lock(data_mutex_);
422
423 for (auto& [proc_id, proc_data] : process_data_) {
424 if (!proc_data->monitoring_enabled) continue;
425
426 std::shared_lock proc_lock(proc_data->mutex);
427
428 // ÌîÑÎ°úÏÑ∏Ïä§Î≥Ñ Ïä§ÎÉÖÏÉ∑ ÏÉùÏÑ±
429 metrics_snapshot snapshot;
430 snapshot.capture_time = std::chrono::steady_clock::now();
431 snapshot.system = proc_data->system_metrics;
432
433 // Thread pool Î©îÌä∏Î¶≠ ÏßëÍ≥Ñ
434 thread_pool_metrics aggregated_pool{};
435 for (const auto& [pool_id, pool_metrics] : proc_data->pool_metrics) {
436 aggregated_pool.worker_threads += pool_metrics.worker_threads;
437 aggregated_pool.idle_threads += pool_metrics.idle_threads;
438 aggregated_pool.jobs_completed += pool_metrics.jobs_completed;
439 aggregated_pool.jobs_pending += pool_metrics.jobs_pending;
440 aggregated_pool.jobs_failed += pool_metrics.jobs_failed;
441 }
442 snapshot.thread_pool = aggregated_pool;
443
444 // Worker Î©îÌä∏Î¶≠ ÏßëÍ≥Ñ
445 worker_metrics aggregated_worker{};
446 for (const auto& [worker_id, worker] : proc_data->worker_metrics) {
447 aggregated_worker.jobs_processed += worker.jobs_processed;
448 aggregated_worker.total_processing_time_ns += worker.total_processing_time_ns;
449 }
450 snapshot.worker = aggregated_worker;
451
452 // ÌûàÏä§ÌÜ†Î¶¨Ïóê Ï∂îÍ∞Ä
453 proc_lock.unlock();
454 std::unique_lock write_lock(proc_data->mutex);
455 proc_data->history.push(snapshot);
456
457 // Ï†ÑÏ≤¥ ÌûàÏä§ÌÜ†Î¶¨ÏóêÎèÑ Ï∂îÍ∞Ä
458 global_history_.push(snapshot);
459 }
460 }
