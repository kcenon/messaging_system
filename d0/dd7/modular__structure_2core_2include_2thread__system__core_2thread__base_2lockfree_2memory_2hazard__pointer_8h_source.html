106 template<typename T>
107 [[nodiscard]] auto protect(const std::atomic<T*>& atomic_ptr) -> T*
108 {
109 T* ptr;
110 do {
111 ptr = atomic_ptr.load(std::memory_order_acquire);
112 hp_slot_->store(ptr, std::memory_order_release);
113 // Double-check to ensure the pointer hasn't changed
114 } while (ptr != atomic_ptr.load(std::memory_order_acquire));
115
116 return ptr;
117 }
118
122 auto clear() -> void
123 {
124 if (hp_slot_) {
125 hp_slot_->store(nullptr, std::memory_order_release);
126 }
127 }

133 [[nodiscard]] auto is_valid() const -> bool { return hp_slot_ != nullptr; }
