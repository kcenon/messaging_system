220 GTEST_SKIP() << "UTF-8 round-trip conversion differs on Windows but ASCII preserved";
221 return;
222 } else {
223 // If ASCII is corrupted, fail the test
224 FAIL() << "Round-trip conversion corrupted ASCII characters";
225 }
226 }
227#endif
