76 std::string timestamp;

78 std::string message;
79 std::string file;
80 std::string function;

84 LogEntry parse_v1_log_line(const std::string& line) {
85 LogEntry entry;
86 std::regex v1_pattern(R"(\[([^\]]+)\] \[([^\]]+)\] (.+))");
87 std::smatch matches;
88
89 if (std::regex_match(line, matches, v1_pattern)) {
90 entry.timestamp = matches[1];
91 entry.level = matches[2];
92 entry.message = matches[3];
93 }
94
95 return entry;
96 }
97
98 LogEntry parse_v2_log_line(const std::string& line) {
99 LogEntry entry;
100 std::regex v2_pattern(R"(\[([^\]]+)\] \[([^\]]+)\] \[([^:]+):(\d+)\] \[([^\]]+)\] (.+))");
101 std::smatch matches;
102
103 if (std::regex_match(line, matches, v2_pattern)) {
104 entry.timestamp = matches[1];
105 entry.level = matches[2];
106 entry.file = matches[3];
107 entry.line = std::stoi(matches[4]);
108 entry.function = matches[5];
109 entry.message = matches[6];
110 }
111
112 return entry;
113 }
114};
