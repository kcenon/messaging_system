44 auto prepare_data_async(const std::vector<uint8_t>& input_data,
45 const pipeline& pl,
46 bool use_compress,
47 bool use_encrypt)
48 -> std::future<std::vector<uint8_t>>
49 {
50 return std::async(std::launch::async, [input_data = input_data, &pl, use_compress, use_encrypt]() {
51 std::vector<uint8_t> processed_data = input_data;
52
53 if constexpr (std::is_invocable_r_v<std::vector<uint8_t>, decltype(pl.compress), const std::vector<uint8_t>&>) {
54 if (use_compress) {
55 processed_data = pl.compress(processed_data);
56 }
57 }
58
59 if constexpr (std::is_invocable_r_v<std::vector<uint8_t>, decltype(pl.encrypt), const std::vector<uint8_t>&>) {
60 if (use_encrypt) {
61 processed_data = pl.encrypt(processed_data);
62 }
63 }
64
65 return processed_data;
66 });
67 }

71 auto async_send_with_pipeline_co(std::shared_ptr<tcp_socket> sock,
72 std::vector<uint8_t> data,
73 const pipeline& pl,
74 bool use_compress,
75 bool use_encrypt)
76 -> asio::awaitable<std::error_code>
77 {
78 // Process data with pipeline (compress/encrypt as needed)
79 auto future_processed = prepare_data_async(data, pl, use_compress, use_encrypt);
80 auto processed_data = future_processed.get(); // This blocks, but we're in a coroutine
81
82 // Using structured binding directly with co_await
83 auto [ec, bytes_transferred] = co_await asio::async_write(
84 sock->socket(),
85 asio::buffer(processed_data),
86 asio::experimental::as_tuple(asio::use_awaitable));
87
88 if (ec) {
89 NETWORK_LOG_ERROR("[send_coroutine] Error sending data: " + ec.message());
90 }
91
92 co_return ec;
93 }

97 auto async_send_with_pipeline_no_co(std::shared_ptr<tcp_socket> sock,
98 std::vector<uint8_t> data,
99 const pipeline& pl,
100 bool use_compress,
101 bool use_encrypt)
102 -> std::future<std::error_code>
103 {
104 // Create a promise for the final result
105 auto promise = std::make_shared<std::promise<std::error_code>>();
106 auto future_result = promise->get_future();
107
108 // Process data in a separate thread
109 auto future_processed = prepare_data_async(std::move(data), pl, use_compress, use_encrypt);
110
111 // When processing is done, send the data
112 std::thread([sock, promise, future = std::move(future_processed)]() mutable {
113 try {
114 auto processed_data = future.get();
115
116 // Perform async write and capture result in the promise
117 sock->async_send(processed_data,
118 [promise](std::error_code ec, std::size_t /*bytes_transferred*/) {
119 promise->set_value(ec);
120 });
121 }
122 catch (const std::exception& e) {
123 NETWORK_LOG_ERROR("[send_coroutine] Exception processing data: "
124 + std::string(e.what()));
125 promise->set_value(std::make_error_code(std::errc::io_error));
126 }
127 }).detach();
128
129 return future_result;
130 }
