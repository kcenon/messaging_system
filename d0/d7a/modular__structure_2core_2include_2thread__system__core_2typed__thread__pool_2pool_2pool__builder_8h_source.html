142 [[nodiscard]] std::shared_ptr<typed_thread_pool_t<job_type>> build() {
143 if (!validate()) {
144 throw std::invalid_argument("Invalid thread pool configuration");
145 }
146
147 auto pool = std::make_shared<typed_thread_pool_t<job_type>>(title_);
148
149 // Create workers with specified job types
150 for (size_t i = 0; i < worker_count_; ++i) {
151 auto worker = std::make_unique<typed_thread_worker_t<job_type>>(
152 job_types_, use_time_tag_
153 );
154 pool->enqueue(std::move(worker));
155 }
156
157 if (auto_start_) {
158 auto result = pool->start();
159 if (result.has_error()) {
160 throw std::runtime_error("Failed to start thread pool: " +
161 result.get_error().message());
162 }
163 }
164
165 return pool;
166 }

172 typed_thread_pool_builder& reset() {
173 title_ = "typed_thread_pool";
174 worker_count_ = config::default_worker_count;
175 queue_size_ = config::default_queue_size;
176 job_types_.clear();
177 use_time_tag_ = true;
178 auto_start_ = false;
179 return *this;
180 }
181
182 private:
183 std::string title_ = "typed_thread_pool";
184 size_t worker_count_ = config::default_worker_count;
185 size_t queue_size_ = config::default_queue_size;
186 std::vector<job_type> job_types_;
187 bool use_time_tag_ = true;
188 bool auto_start_ = false;
