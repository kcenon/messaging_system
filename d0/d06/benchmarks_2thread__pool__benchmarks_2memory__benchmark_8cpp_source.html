59// Helper function to create thread pool
60auto create_default(const uint16_t& worker_counts)
61 -> std::tuple<std::shared_ptr<kcenon::thread::thread_pool>, std::optional<std::string>>
62{
63 std::shared_ptr<kcenon::thread::thread_pool> pool;
64 try {
65 pool = std::make_shared<kcenon::thread::thread_pool>();
66 } catch (const std::bad_alloc& e) {
67 return { nullptr, std::string(e.what()) };
68 }
69
70 std::optional<std::string> error_message = std::nullopt;
71 std::vector<std::unique_ptr<kcenon::thread::thread_worker>> workers;
72 workers.reserve(worker_counts);
73 for (uint16_t i = 0; i < worker_counts; ++i) {
74 workers.push_back(std::make_unique<kcenon::thread::thread_worker>());
75 }
76
77 error_message = pool->enqueue_batch(std::move(workers));
78 if (error_message.has_value()) {
79 return { nullptr, formatter::format("cannot enqueue to workers: {}",
80 error_message.value_or("unknown error")) };
81 }
82
83 return { pool, std::nullopt };
84}

97class MemoryMonitor {
98public:
99 struct MemoryStats {
100 size_t virtual_size; // Virtual memory size in bytes
101 size_t resident_size; // Resident set size in bytes
102 size_t peak_size; // Peak memory usage
103 };
104
105 static MemoryStats get_current_memory() {
106 MemoryStats stats = {0, 0, 0};
107
108#ifdef _WIN32
109 PROCESS_MEMORY_COUNTERS_EX pmc;
110 if (GetProcessMemoryInfo(GetCurrentProcess(), (PROCESS_MEMORY_COUNTERS*)&pmc, sizeof(pmc))) {
111 stats.virtual_size = pmc.PrivateUsage;
112 stats.resident_size = pmc.WorkingSetSize;
113 stats.peak_size = pmc.PeakWorkingSetSize;
114 }
115#elif defined(__APPLE__)
116 struct mach_task_basic_info info;
117 mach_msg_type_number_t size = MACH_TASK_BASIC_INFO_COUNT;
118 if (task_info(mach_task_self(), MACH_TASK_BASIC_INFO, (task_info_t)&info, &size) == KERN_SUCCESS) {
119 stats.virtual_size = info.virtual_size;
120 stats.resident_size = info.resident_size;
121 stats.peak_size = info.resident_size_max;
122 }
123#else
124 // Linux
125 FILE* file = fopen("/proc/self/status", "r");
126 if (file) {
127 char line[128];
128 while (fgets(line, 128, file)) {
129 if (strncmp(line, "VmSize:", 7) == 0) {
130 stats.virtual_size = parse_kb(line) * 1024;
131 } else if (strncmp(line, "VmRSS:", 6) == 0) {
132 stats.resident_size = parse_kb(line) * 1024;
133 } else if (strncmp(line, "VmPeak:", 7) == 0) {
134 stats.peak_size = parse_kb(line) * 1024;
135 }
136 }
137 fclose(file);
138 }
139#endif
140 return stats;
141 }
142
143private:
144 static size_t parse_kb(const char* line) {
145 size_t value = 0;
146 const char* p = line;
147 while (*p && !isdigit(*p)) p++;
148 if (*p) {
149 value = atol(p);
150 }
151 return value;
152 }
153};

158static void BM_BaseMemory(benchmark::State& state) {
159 for (auto _ : state) {
160 auto stats = MemoryMonitor::get_current_memory();
161 benchmark::DoNotOptimize(stats);
162 }
163
164 auto final_stats = MemoryMonitor::get_current_memory();
165 state.counters["virtual_MB"] = final_stats.virtual_size / 1024.0 / 1024.0;
166 state.counters["resident_MB"] = final_stats.resident_size / 1024.0 / 1024.0;
167 state.counters["peak_MB"] = final_stats.peak_size / 1024.0 / 1024.0;
168}

174static void BM_ThreadPoolMemory(benchmark::State& state) {
175 const size_t worker_count = state.range(0);
176
177 for (auto _ : state) {
178 auto before = MemoryMonitor::get_current_memory();
179
180 auto [pool, error] = create_default(worker_count);
181 if (error.has_value()) {
182 state.SkipWithError("Failed to create thread pool");
183 return;
184 }
185
186 pool->start();
187 std::this_thread::sleep_for(std::chrono::milliseconds(100));
188
189 auto after = MemoryMonitor::get_current_memory();
190
191 size_t memory_increase = after.resident_size - before.resident_size;
192
193 pool->stop();
194
195 state.counters["total_MB"] = memory_increase / 1024.0 / 1024.0;
196 state.counters["per_worker_KB"] = static_cast<double>(memory_increase) / worker_count / 1024.0;
197 }
198
199 state.counters["workers"] = worker_count;
200}

250static void BM_JobQueueMemory(benchmark::State& state) {
251 const size_t job_count = state.range(0);
252
253 auto [pool, error] = create_default(4);
254 if (error.has_value()) {
255 state.SkipWithError("Failed to create thread pool");
256 return;
257 }
258
259 pool->start();
260
261 for (auto _ : state) {
262 auto before = MemoryMonitor::get_current_memory();
263
264 // Submit jobs that will queue up
265 for (size_t i = 0; i < job_count; ++i) {
266 pool->enqueue(std::make_unique<callback_job>([]() -> result_void {
267 std::this_thread::sleep_for(std::chrono::seconds(10));
268 return result_void();
269 }));
270 }
271
272 std::this_thread::sleep_for(std::chrono::milliseconds(100));
273 auto after = MemoryMonitor::get_current_memory();
274
275 size_t memory_increase = after.resident_size - before.resident_size;
276
277 state.counters["total_MB"] = memory_increase / 1024.0 / 1024.0;
278 state.counters["per_job_bytes"] = static_cast<double>(memory_increase) / job_count;
279
280 // Clear the queue
281 pool->stop();
282 pool->start();
283 }
284
285 pool->stop();
286 state.counters["jobs"] = job_count;
287}

289 ->Arg(1000)
290 ->Arg(10000)
291 ->Arg(50000)
292 ->Arg(100000)
293 ->Unit(benchmark::kMillisecond);

300static void BM_MemoryAllocationPattern(benchmark::State& state) {
301 const size_t allocation_size = state.range(0);
302 const size_t num_allocations = state.range(1);
303
304 for (auto _ : state) {
305 auto before = MemoryMonitor::get_current_memory();
306
307 std::vector<std::unique_ptr<char[]>> allocations;
308 allocations.reserve(num_allocations);
309
310 for (size_t i = 0; i < num_allocations; ++i) {
311 allocations.push_back(std::make_unique<char[]>(allocation_size));
312 // Touch memory to ensure it's actually allocated
313 std::memset(allocations.back().get(), i & 0xFF, allocation_size);
314 }
315
316 auto after = MemoryMonitor::get_current_memory();
317
318 size_t memory_increase = after.resident_size - before.resident_size;
319 state.counters["total_MB"] = memory_increase / 1024.0 / 1024.0;
320 state.counters["efficiency"] = static_cast<double>(allocation_size * num_allocations) / memory_increase;
321 }
322
323 state.counters["alloc_size"] = allocation_size;
324 state.counters["num_allocs"] = num_allocations;
325}
