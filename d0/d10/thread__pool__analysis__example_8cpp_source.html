42 static process_thread_pool_metrics generate_metrics(
43 const thread_pool_identifier& pool_id,
44 workload_type type,
45 std::size_t worker_count,
46 int time_offset) {
47
48 process_thread_pool_metrics metrics;
49 metrics.pool_id = pool_id;
50 metrics.worker_threads = worker_count;
51
52 std::random_device rd;
53 std::mt19937 gen(rd());
54
55 switch (type) {
56 case workload_type::cpu_intensive:
57 metrics.idle_threads = gen() % 2;
58 metrics.jobs_completed = 1000 + (gen() % 500);
59 metrics.jobs_pending = gen() % 10;
60 metrics.average_latency_ns = 1000000 + (gen() % 500000); // 1-1.5ms
61 break;
62
63 case workload_type::io_bound:
64 metrics.idle_threads = worker_count / 2 + (gen() % (worker_count / 2));
65 metrics.jobs_completed = 100 + (gen() % 50);
66 metrics.jobs_pending = 50 + (gen() % 100);
67 metrics.average_latency_ns = 50000000 + (gen() % 50000000); // 50-100ms
68 break;
69
70 case workload_type::balanced:
71 metrics.idle_threads = worker_count / 4;
72 metrics.jobs_completed = 500 + (gen() % 200);
73 metrics.jobs_pending = 20 + (gen() % 30);
74 metrics.average_latency_ns = 5000000 + (gen() % 5000000); // 5-10ms
75 break;
76
77 case workload_type::bursty:
78 if (time_offset % 10 < 3) { // 30% burst
79 metrics.idle_threads = 0;
80 metrics.jobs_completed = 2000 + (gen() % 1000);
81 metrics.jobs_pending = 500 + (gen() % 500);
82 metrics.average_latency_ns = 2000000; // 2ms
83 } else {
84 metrics.idle_threads = worker_count - 1;
85 metrics.jobs_completed = 10 + (gen() % 10);
86 metrics.jobs_pending = 0;
87 metrics.average_latency_ns = 1000000; // 1ms
88 }
89 break;
90 }
91
92 metrics.jobs_failed = metrics.jobs_completed / 100; // 1% Ïã§Ìå®Ïú®
93 metrics.total_execution_time_ns = metrics.jobs_completed * metrics.average_latency_ns;
94
95 // ÏõåÏª§ Î∂ÄÌïò Î∂ÑÏÇ∞ ÏÉùÏÑ±
96 metrics.worker_load_distribution.resize(worker_count);
97 if (type == workload_type::cpu_intensive || type == workload_type::balanced) {
98 // Í∑†Îì± Î∂ÑÏÇ∞
99 auto base_load = metrics.jobs_completed / worker_count;
100 for (auto& load : metrics.worker_load_distribution) {
101 load = base_load + (gen() % (base_load / 10 + 1));
102 }
103 } else {
104 // Î∂àÍ∑†Îì± Î∂ÑÏÇ∞
105 for (size_t i = 0; i < worker_count; ++i) {
106 if (i < worker_count / 3) {
107 metrics.worker_load_distribution[i] =
108 (metrics.jobs_completed * 2) / worker_count;
109 } else {
110 metrics.worker_load_distribution[i] =
111 metrics.jobs_completed / (worker_count * 2);
112 }
113 }
114 }
115
116 return metrics;
117 }
