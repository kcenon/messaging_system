97 [[nodiscard]] double contention_ratio() const noexcept {
98 const auto ops = operations_count.load(std::memory_order_relaxed);
99 const auto contention = contention_count.load(std::memory_order_relaxed);
100 return ops > 0 ? static_cast<double>(contention) / ops : 0.0;
101 }
102
107 [[nodiscard]] double cas_success_ratio() const noexcept {
108 const auto successful = successful_cas_count.load(std::memory_order_relaxed);
109 const auto failed = failed_cas_count.load(std::memory_order_relaxed);
110 const auto total = successful + failed;
111 return total > 0 ? static_cast<double>(successful) / total : 0.0;
112 }
113
117 void reset() noexcept {
118 operations_count.store(0, std::memory_order_relaxed);
119 contention_count.store(0, std::memory_order_relaxed);
120 adaptation_count.store(0, std::memory_order_relaxed);
121 successful_cas_count.store(0, std::memory_order_relaxed);
122 failed_cas_count.store(0, std::memory_order_relaxed);
123 backoff_time_ns.store(0, std::memory_order_relaxed);
124 }
125 };
