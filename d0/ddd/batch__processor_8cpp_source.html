52void batch_processor::stop(bool flush_remaining) {
53 if (!running_.exchange(false)) {
54 return; // Already stopped
55 }
56
57 should_stop_ = true;
58
59 if (processing_thread_.joinable()) {
60 processing_thread_.join();
61 }
62
63 if (flush_remaining) {
64 // Process any remaining entries
65 batch_entry entry;
66 std::vector<batch_entry> final_batch;
67 final_batch.reserve(queue_size);
68
69 while (queue_->dequeue(entry)) {
70 final_batch.push_back(std::move(entry));
71 }
72
73 if (!final_batch.empty()) {
74 process_batch(final_batch);
75 }
76
77 if (writer_) {
78 writer_->flush();
79 }
80 }
81}

114void batch_processor::process_loop() {
115 std::vector<batch_entry> current_batch;
116 auto last_flush_time = std::chrono::steady_clock::now();
117 auto last_adjustment_time = std::chrono::steady_clock::now();
118
119 while (!should_stop_.load(std::memory_order_relaxed)) {
120 const auto batch_size = current_batch_size_.load(std::memory_order_relaxed);
121 const auto wait_time = current_wait_time_.load(std::memory_order_relaxed);
122
123 current_batch.clear();
124 current_batch.reserve(batch_size);
125
126 const auto deadline = std::chrono::steady_clock::now() + wait_time;
127 const auto entries_collected = collect_entries(current_batch, batch_size, deadline);
128
129 if (entries_collected > 0) {
130 const auto process_start = std::chrono::steady_clock::now();
131 const auto processed = process_batch(current_batch);
132 const auto process_end = std::chrono::steady_clock::now();
133
134 const auto processing_time = process_end - process_start;
135 const bool flushed_by_size = (entries_collected >= batch_size);
136 const bool flushed_by_time = should_flush_by_time(last_flush_time);
137
138 std::string flush_reason;
139 if (flushed_by_size) {
140 flush_reason = "size";
141 stats_.flush_by_size.fetch_add(1, std::memory_order_relaxed);
142 } else if (flushed_by_time) {
143 flush_reason = "time";
144 stats_.flush_by_time.fetch_add(1, std::memory_order_relaxed);
145 } else {
146 flush_reason = "partial";
147 }
148
149 update_stats(processed, processing_time, flush_reason);
150 last_flush_time = process_end;
151
152 // Handle back-pressure
153 if (config_.enable_back_pressure && !handle_back_pressure()) {
154 continue;
155 }
156
157 // Dynamic batch size adjustment
158 if (config_.enable_dynamic_sizing) {
159 const auto now = std::chrono::steady_clock::now();
160 if (now - last_adjustment_time > std::chrono::seconds(5)) {
161 adjust_batch_size();
162 last_adjustment_time = now;
163 }
164 }
165 } else {
166 // No entries collected, short sleep to avoid busy waiting
167 std::this_thread::sleep_for(std::chrono::microseconds(100));
168 }
169 }
170}

172size_t batch_processor::collect_entries(std::vector<batch_entry>& batch,
173 size_t max_entries,
174 std::chrono::steady_clock::time_point deadline) {
175 size_t collected = 0;
176 batch_entry entry;
177
178 while (collected < max_entries && std::chrono::steady_clock::now() < deadline) {
179 if (queue_->dequeue(entry)) {
180 batch.push_back(std::move(entry));
181 ++collected;
182 } else {
183 // Queue is empty, short wait
184 std::this_thread::sleep_for(std::chrono::microseconds(10));
185 }
186 }
187
188 // Try to collect more entries if we have time left and space
189 while (collected < max_entries && queue_->dequeue(entry)) {
190 batch.push_back(std::move(entry));
191 ++collected;
192 }
193
194 return collected;
195}

197size_t batch_processor::process_batch(std::vector<batch_entry>& batch) {
198 if (batch.empty() || !writer_) {
199 return 0;
200 }
201
202 size_t processed = 0;
203 for (const auto& entry : batch) {
204 if (auto result = writer_->write(entry.level, entry.message, entry.file,
205 entry.line, entry.function, entry.timestamp)) {
206 ++processed;
207 }
208 }
209
210 // Flush after batch processing
211 writer_->flush();
212
213 stats_.total_batches.fetch_add(1, std::memory_order_relaxed);
214 stats_.total_entries.fetch_add(processed, std::memory_order_relaxed);
215
216 return processed;
217}

219void batch_processor::adjust_batch_size() {
220 const auto current_size = current_batch_size_.load(std::memory_order_relaxed);
221 const auto queue_size = get_queue_size();
222 const auto recent_time = recent_processing_time_ms_.load(std::memory_order_relaxed);
223
224 size_t new_size = current_size;
225
226 // Increase batch size if queue is building up and processing is fast
227 if (queue_size > current_size * 2 && recent_time < 10.0) {
228 new_size = std::min(config_.max_batch_size,
229 static_cast<size_t>(current_size * config_.size_increase_factor));
230 }
231 // Decrease batch size if processing is slow or queue is small
232 else if (recent_time > 100.0 || queue_size < current_size / 4) {
233 new_size = std::max(config_.min_batch_size,
234 static_cast<size_t>(current_size * config_.size_decrease_factor));
235 }
236
237 if (new_size != current_size) {
238 current_batch_size_.store(new_size, std::memory_order_relaxed);
239 stats_.dynamic_size_adjustments.fetch_add(1, std::memory_order_relaxed);
240 }
241}

243bool batch_processor::handle_back_pressure() {
244 const auto queue_size = get_queue_size();
245
246 if (queue_size > config_.back_pressure_threshold) {
247 stats_.back_pressure_events.fetch_add(1, std::memory_order_relaxed);
248
249 // Apply back-pressure delay
250 std::this_thread::sleep_for(config_.back_pressure_delay);
251
252 return queue_size < queue_size * 1.5; // Continue if queue isn't growing too fast
253 }
254
255 return true;
256}

258bool batch_processor::should_flush_by_time(std::chrono::steady_clock::time_point last_flush_time) const {
259 const auto now = std::chrono::steady_clock::now();
260 const auto elapsed = now - last_flush_time;
261 const auto current_wait = current_wait_time_.load(std::memory_order_relaxed);

266void batch_processor::update_stats(size_t batch_size,
267 std::chrono::nanoseconds processing_time,
268 const std::string& flush_reason) {
269 (void)batch_size; // Suppress unused parameter warning
270 (void)flush_reason; // Suppress unused parameter warning
271 const double processing_time_ms =
272 std::chrono::duration_cast<std::chrono::microseconds>(processing_time).count() / 1000.0;
273
274 // Update recent processing time (exponential moving average)
275 const double alpha = 0.1;
276 const double current_time = recent_processing_time_ms_.load(std::memory_order_relaxed);
277 const double new_time = alpha * processing_time_ms + (1.0 - alpha) * current_time;
278 recent_processing_time_ms_.store(new_time, std::memory_order_relaxed);
279
280 // Update average batch size
281 const auto total_batches = stats_.total_batches.load(std::memory_order_relaxed);
282 if (total_batches > 0) {
283 const auto total_entries = stats_.total_entries.load(std::memory_order_relaxed);
284 const double avg_size = static_cast<double>(total_entries) / total_batches;
285 stats_.average_batch_size.store(avg_size, std::memory_order_relaxed);
286 }
287
288 // Update average processing time
289 stats_.average_processing_time_ms.store(
290 recent_processing_time_ms_.load(std::memory_order_relaxed),
291 std::memory_order_relaxed);
292
293 // Update recent queue size for adjustment algorithm
294 recent_queue_size_.store(get_queue_size(), std::memory_order_relaxed);
295}
