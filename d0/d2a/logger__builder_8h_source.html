531 logger_builder& detect_environment() {
532 const char* env = std::getenv("LOG_ENV");
533 const char* level = std::getenv("LOG_LEVEL");
534
535 if (env) {
536 std::string env_str(env);
537 if (env_str == "production") {
538 apply_template(kcenon::logger::configuration_template::production);
539 } else if (env_str == "debug" || env_str == "development") {
540 apply_template(kcenon::logger::configuration_template::debug);
541 }
542 }
543
544 if (level) {
545 std::string level_str(level);
546 if (level_str == "trace") config_.min_level = kcenon::thread::log_level::trace;
547 else if (level_str == "debug") config_.min_level = kcenon::thread::log_level::debug;
548 else if (level_str == "info") config_.min_level = kcenon::thread::log_level::info;
549 else if (level_str == "warn") config_.min_level = kcenon::thread::log_level::warning;
550 else if (level_str == "error") config_.min_level = kcenon::thread::log_level::error;
551 else if (level_str == "fatal") config_.min_level = kcenon::thread::log_level::error; // Note: fatal mapped to error
552 }

554 return *this;

662 result<std::unique_ptr<logger>> build() {
663 // Apply all strategies first
664 for (const auto& strategy : strategies_) {
665 if (auto can_apply = strategy->can_apply(config_); !can_apply) {
666 // Log warning but continue with other strategies
667 continue;
668 }
669
670 if (auto result = strategy->apply(config_); !result) {
671#ifdef USE_THREAD_SYSTEM
672 auto error_code = static_cast<int>(result.get_error().code());
673 auto logger_code = static_cast<logger_error_code>(error_code - 10000);
674 return make_error<std::unique_ptr<logger>>(
675 logger_code,
676 "Strategy application failed: " + result.get_error().message()
677 );
678#else
679 return make_logger_error<std::unique_ptr<logger>>(
680 result.error_code(),
681 "Strategy application failed: " + result.error_message()
682 );
683#endif
684 }
685 }
686
687 // Validate configuration
688 if (auto validation = config_.validate(); !validation) {
689#ifdef USE_THREAD_SYSTEM
690 // Extract error code - we need to reverse the offset we applied
691 auto error_code = static_cast<int>(validation.get_error().code());
692 auto logger_code = static_cast<logger_error_code>(error_code - 10000);
693 return make_error<std::unique_ptr<logger>>(
694 logger_code,
695 "Configuration validation failed: " + validation.get_error().message()
696 );
697#else
698 return make_logger_error<std::unique_ptr<logger>>(
699 validation.error_code(),
700 "Configuration validation failed: " + validation.error_message()
701 );
702#endif
703 }
704
705 // Validate writer count
706 if (!writers_.empty() && writers_.size() > config_.max_writers) {
707 return make_logger_error<std::unique_ptr<logger>>(
708 logger_error_code::invalid_configuration,
709 "Number of writers exceeds max_writers configuration"
710 );
711 }
712
713 // Create logger with validated configuration
714 auto logger_instance = std::make_unique<logger>(config_.async, config_.buffer_size);
715
716 // Apply configuration settings
717 logger_instance->set_min_level(config_.min_level);
718
719 if (config_.enable_metrics) {
720 logger_instance->enable_metrics_collection(true);
721 }
722
723 // Add writers (apply batch writing if enabled)
724 for (auto& [name, writer] : writers_) {
725 if (writer) {
726 writer->set_use_color(config_.enable_color_output);
727
728 // Wrap with batch writer if enabled and async mode
729 if (config_.enable_batch_writing && config_.async) {
730 batch_writer::config batch_cfg;
731 batch_cfg.max_batch_size = config_.batch_size;
732 batch_cfg.flush_interval = config_.flush_interval;
733
734 auto batch_wrapped = std::make_unique<batch_writer>(
735 std::move(writer), batch_cfg
736 );
737 logger_instance->add_writer(name, std::move(batch_wrapped));
738 } else {
739 logger_instance->add_writer(name, std::move(writer));
740 }
741 }
742 }
743
744 // Add filters
745 if (!filters_.empty()) {
746 // Create composite filter if multiple filters
747 if (filters_.size() == 1) {
748 logger_instance->set_filter(std::move(filters_[0]));
749 } else {
750 auto composite = std::make_unique<composite_filter>(composite_filter::logic_type::AND);
751 for (auto& filter : filters_) {
752 composite->add_filter(std::move(filter));
753 }
754 logger_instance->set_filter(std::move(composite));
755 }
756 }
757
758 // Start logger if async
759 if (config_.async) {
760 logger_instance->start();
761 }
762
763 // Store configuration in logger (if we add a config getter)
764 built_config_ = config_;
765
766 return std::move(logger_instance);
767 }
