83auto create_default(const uint16_t& top_priority_workers,
84 const uint16_t& middle_priority_workers,
85 const uint16_t& bottom_priority_workers)
86 -> std::tuple<std::shared_ptr<typed_thread_pool_t<test_priority>>,
87 std::optional<std::string>>
88{
89 std::shared_ptr<typed_thread_pool_t<test_priority>> pool;
90
91 try
92 {
93 pool = std::make_shared<typed_thread_pool_t<test_priority>>();
94 }
95 catch (const std::bad_alloc& e)
96 {
97 return { nullptr, std::string(e.what()) };
98 }
99
100 std::optional<std::string> error_message = std::nullopt;
101
102 std::vector<std::unique_ptr<typed_thread_worker_t<test_priority>>> workers;
103 workers.reserve(top_priority_workers + middle_priority_workers + bottom_priority_workers);
104
105 for (uint16_t i = 0; i < top_priority_workers; ++i)
106 {
107 workers.push_back(std::make_unique<typed_thread_worker_t<test_priority>>(
108 std::vector<test_priority>{ test_priority::Top }, "top priority worker"));
109 }
110 for (uint16_t i = 0; i < middle_priority_workers; ++i)
111 {
112 workers.push_back(std::make_unique<typed_thread_worker_t<test_priority>>(
113 std::vector<test_priority>{ test_priority::Middle }, "middle priority worker"));
114 }
115 for (uint16_t i = 0; i < bottom_priority_workers; ++i)
116 {
117 workers.push_back(std::make_unique<typed_thread_worker_t<test_priority>>(
118 std::vector<test_priority>{ test_priority::Bottom }, "bottom priority worker"));
119 }
120
121 auto enqueue_result = pool->enqueue_batch(std::move(workers));
122 if (enqueue_result.has_error())
123 {
124 return { nullptr, formatter::format("cannot enqueue to workers: {}",
125 enqueue_result.get_error().message()) };
126 }
127
128 return { pool, std::nullopt };
129}

131auto store_job(std::shared_ptr<typed_thread_pool_t<test_priority>> thread_pool)
132 -> std::optional<std::string>
133{
134 int target = 0;
135
136 std::vector<std::unique_ptr<typed_job_t<test_priority>>> jobs;
137 jobs.reserve(test_line_count_);
138
139 for (auto index = 0; index < test_line_count_; ++index)
140 {
141 target = index % 3;
142 jobs.push_back(std::make_unique<callback_typed_job_t<test_priority>>(
143 [target](void) -> result_void
144 {
145 log_module::write_debug("Hello, World!: {} priority", target);
146 return {};
147 },
148 static_cast<test_priority>(target)));
149 }
150
151 auto enqueue_result = thread_pool->enqueue_batch(std::move(jobs));
152 if (enqueue_result.has_error())
153 {
154 return formatter::format("error enqueuing jobs: {}",
155 enqueue_result.get_error().message());
156 }
157
158 log_module::write_sequence("enqueued jobs: {}", test_line_count_);
159
160 return std::nullopt;
161}

163auto main() -> int
164{
165 auto error_message = initialize_logger();
166 if (error_message.has_value())
167 {
168 std::cerr << formatter::format("error starting logger: {}\n",
169 error_message.value_or("unknown error"));
170 return 0;
171 }
172
173 std::shared_ptr<typed_thread_pool_t<test_priority>> thread_pool = nullptr;
174 std::tie(thread_pool, error_message)
175 = create_default(top_priority_workers_, middle_priority_workers_, bottom_priority_workers_);
176 if (error_message.has_value())
177 {
178 log_module::write_error("error creating thread pool: {}",
179 error_message.value_or("unknown error"));
180
181 return 0;
182 }
183
184 log_module::write_information("created {}", thread_pool->to_string());
185
186 error_message = store_job(thread_pool);
187 if (error_message.has_value())
188 {
189 log_module::write_error("error storing job: {}", error_message.value_or("unknown error"));
190
191 thread_pool.reset();
192
193 return 0;
194 }
195
196 auto start_result = thread_pool->start();
197 if (start_result.has_error())
198 {
199 log_module::write_error("error starting thread pool: {}",
200 start_result.get_error().message());
201
202 thread_pool.reset();
203
204 return 0;
205 }
206
207 log_module::write_information("started {}", thread_pool->to_string());
208
209 thread_pool->stop();
210
211 log_module::write_information("stopped {}", thread_pool->to_string());
212
213 thread_pool.reset();
214
215 log_module::stop();
216
217 return 0;
218}
