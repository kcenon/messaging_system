38enum class buffering_strategy_type : uint8_t {
39 immediate, // Immediate processing (no buffering)
40 fixed_size, // Fixed size buffer with configurable behavior
41 time_based, // Time-based batching
42 adaptive, // Adaptive buffering based on load
43 priority_based, // Priority-based buffering
44 compression_based // Compression-aware buffering
45};
46
51enum class buffer_overflow_policy : uint8_t {
52 drop_oldest, // Drop oldest items (FIFO)
53 drop_newest, // Drop newest items
54 drop_lowest_priority, // Drop lowest priority items
55 compress, // Compress buffer contents
56 flush_immediately, // Force immediate flush
57 block_until_space // Block until space is available
58};
59
64enum class buffer_flush_trigger : uint8_t {
65 size_threshold, // Flush when size threshold reached
66 time_interval, // Flush at regular time intervals
67 priority_threshold, // Flush when priority threshold reached
68 memory_pressure, // Flush when memory pressure detected
69 manual, // Manual flush only
70 adaptive // Adaptive flushing based on conditions
71};

77struct buffering_config {
78 buffering_strategy_type strategy = buffering_strategy_type::fixed_size;
79 buffer_overflow_policy overflow_policy = buffer_overflow_policy::drop_oldest;
80 buffer_flush_trigger flush_trigger = buffer_flush_trigger::size_threshold;
81
82 // Size-based configuration
83 size_t max_buffer_size = 1024;
84 size_t flush_threshold_size = 512;
85
86 // Time-based configuration
87 std::chrono::milliseconds flush_interval{1000};
88 std::chrono::milliseconds max_age{5000};
89
90 // Priority-based configuration
91 uint8_t min_priority = 0;
92 uint8_t max_priority = 255;
93 uint8_t flush_priority_threshold = 128;
94
95 // Adaptive configuration
96 double load_factor_threshold = 0.8;
97 std::chrono::milliseconds adaptive_check_interval{100};
98
99 // Compression configuration
100 double compression_ratio_threshold = 0.5;
101 bool enable_compression = false;
102
103 // Performance tuning
104 size_t batch_size = 64;
105 bool enable_background_flushing = true;
106 std::chrono::milliseconds background_flush_interval{500};
107
111 result_void validate() const {
112 if (max_buffer_size == 0) {
113 return result_void(monitoring_error_code::invalid_configuration,
114 "Max buffer size must be positive");
115 }
116
117 if (flush_threshold_size > max_buffer_size) {
118 return result_void(monitoring_error_code::invalid_configuration,
119 "Flush threshold cannot exceed max buffer size");
120 }
121
122 if (flush_interval.count() <= 0) {
123 return result_void(monitoring_error_code::invalid_configuration,
124 "Flush interval must be positive");
125 }
126
127 if (max_age.count() <= 0) {
128 return result_void(monitoring_error_code::invalid_configuration,
129 "Max age must be positive");
130 }
131
132 if (min_priority > max_priority) {
133 return result_void(monitoring_error_code::invalid_configuration,
134 "Min priority cannot exceed max priority");
135 }
136
137 if (load_factor_threshold <= 0.0 || load_factor_threshold > 1.0) {
138 return result_void(monitoring_error_code::invalid_configuration,
139 "Load factor threshold must be between 0 and 1");
140 }
141
142 if (compression_ratio_threshold <= 0.0 || compression_ratio_threshold > 1.0) {
143 return result_void(monitoring_error_code::invalid_configuration,
144 "Compression ratio threshold must be between 0 and 1");
145 }
146
147 return result_void::success();

155struct buffered_metric {
156 compact_metric_value metric;
157 std::chrono::system_clock::time_point timestamp;
158 uint8_t priority;
159 size_t sequence_number;
160
161 buffered_metric() : priority(128), sequence_number(0) {
162 timestamp = std::chrono::system_clock::now();
163 }
164
165 buffered_metric(compact_metric_value m, uint8_t p = 128)
166 : metric(std::move(m)), priority(p), sequence_number(0) {
167 timestamp = std::chrono::system_clock::now();
168 }
169
173 bool is_expired(std::chrono::milliseconds max_age) const {
174 auto now = std::chrono::system_clock::now();
175 auto age = std::chrono::duration_cast<std::chrono::milliseconds>(now - timestamp);
176 return age > max_age;
177 }
178
182 std::chrono::milliseconds get_age() const {
183 auto now = std::chrono::system_clock::now();
184 return std::chrono::duration_cast<std::chrono::milliseconds>(now - timestamp);
185 }

192struct buffer_statistics {
193 std::atomic<size_t> total_items_buffered{0};
194 std::atomic<size_t> total_items_flushed{0};
195 std::atomic<size_t> items_dropped_overflow{0};
196 std::atomic<size_t> items_dropped_expired{0};
197 std::atomic<size_t> total_flushes{0};
198 std::atomic<size_t> forced_flushes{0};
199 std::atomic<size_t> compression_operations{0};
200 std::atomic<size_t> bytes_saved_compression{0};
201
202 std::chrono::system_clock::time_point creation_time;
203
204 buffer_statistics() : creation_time(std::chrono::system_clock::now()) {}
205
206 // Copy constructor for result<buffer_statistics>
207 buffer_statistics(const buffer_statistics& other)
208 : total_items_buffered(other.total_items_buffered.load())
209 , total_items_flushed(other.total_items_flushed.load())
210 , items_dropped_overflow(other.items_dropped_overflow.load())
211 , items_dropped_expired(other.items_dropped_expired.load())
212 , total_flushes(other.total_flushes.load())
213 , forced_flushes(other.forced_flushes.load())
214 , compression_operations(other.compression_operations.load())
215 , bytes_saved_compression(other.bytes_saved_compression.load())
216 , creation_time(other.creation_time) {}
217
221 double get_efficiency() const {
222 auto buffered = total_items_buffered.load();
223 auto dropped = items_dropped_overflow.load() + items_dropped_expired.load();
224 auto total = buffered + dropped;
225 return total > 0 ? (static_cast<double>(buffered) / total) * 100.0 : 100.0;
226 }
227
231 double get_avg_items_per_flush() const {
232 auto flushes = total_flushes.load();
233 auto flushed = total_items_flushed.load();
234 return flushes > 0 ? static_cast<double>(flushed) / flushes : 0.0;
235 }
236
240 double get_compression_ratio() const {
241 auto operations = compression_operations.load();
242 auto saved = bytes_saved_compression.load();
243 return operations > 0 ? static_cast<double>(saved) / operations : 0.0;
244 }
245};

348class fixed_size_strategy : public buffer_strategy_interface {
349private:
350 buffering_config config_;
351 std::vector<buffered_metric> buffer_;
352 mutable buffer_statistics stats_;
353 mutable std::mutex mutex_;
354 std::atomic<size_t> sequence_counter_{0};
355
359 void handle_overflow() {
360 if (buffer_.size() < config_.max_buffer_size) {
361 return;
362 }
363
364 switch (config_.overflow_policy) {
365 case buffer_overflow_policy::drop_oldest:
366 if (!buffer_.empty()) {
367 buffer_.erase(buffer_.begin());
368 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
369 }
370 break;
371
372 case buffer_overflow_policy::drop_newest:
373 // Don't add the new item (it will be dropped by caller)
374 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
375 break;
376
377 case buffer_overflow_policy::drop_lowest_priority: {
378 auto min_it = std::min_element(buffer_.begin(), buffer_.end(),
379 [](const buffered_metric& a, const buffered_metric& b) {
380 return a.priority < b.priority;
381 });
382 if (min_it != buffer_.end()) {
383 buffer_.erase(min_it);
384 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
385 }
386 break;
387 }
388
389 case buffer_overflow_policy::flush_immediately:
390 // Force flush will be handled by should_flush()
391 break;
392
393 default:
394 // For other policies, drop oldest as fallback
395 if (!buffer_.empty()) {
396 buffer_.erase(buffer_.begin());
397 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
398 }
399 break;
400 }
401 }
402
406 void remove_expired_items() {
407 size_t removed = 0;
408 auto it = std::remove_if(buffer_.begin(), buffer_.end(),
409 [this, &removed](const buffered_metric& item) {
410 if (item.is_expired(config_.max_age)) {
411 ++removed;
412 return true;
413 }
414 return false;
415 });
416
417 buffer_.erase(it, buffer_.end());
418 stats_.items_dropped_expired.fetch_add(removed, std::memory_order_relaxed);
419 }
420
421public:
422 explicit fixed_size_strategy(const buffering_config& config)
423 : config_(config) {
424 config_.strategy = buffering_strategy_type::fixed_size;
425 buffer_.reserve(config_.max_buffer_size);
426 }
427
428 result_void add_metric(buffered_metric&& metric) override {
429 std::lock_guard<std::mutex> lock(mutex_);
430
431 // Remove expired items first
432 remove_expired_items();
433
434 // Handle overflow before adding
435 if (buffer_.size() >= config_.max_buffer_size) {
436 if (config_.overflow_policy == buffer_overflow_policy::drop_newest) {
437 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
438 return result_void::success(); // Drop the new item
439 }
440 handle_overflow();
441 }
442
443 // Set sequence number
444 metric.sequence_number = sequence_counter_.fetch_add(1, std::memory_order_relaxed);
445
446 // Add to buffer
447 buffer_.emplace_back(std::move(metric));
448 stats_.total_items_buffered.fetch_add(1, std::memory_order_relaxed);
449
450 return result_void::success();
451 }
452
453 result<std::vector<buffered_metric>> flush() override {
454 std::lock_guard<std::mutex> lock(mutex_);
455
456 if (buffer_.empty()) {
457 return make_success(std::vector<buffered_metric>{});
458 }
459
460 std::vector<buffered_metric> flushed_items;
461 flushed_items.reserve(buffer_.size());
462
463 // Move all items to flushed_items
464 for (auto& item : buffer_) {
465 flushed_items.emplace_back(std::move(item));
466 }
467
468 stats_.total_items_flushed.fetch_add(flushed_items.size(), std::memory_order_relaxed);
469 stats_.total_flushes.fetch_add(1, std::memory_order_relaxed);
470
471 buffer_.clear();
472
473 return make_success(std::move(flushed_items));
474 }
475
476 bool should_flush() const override {
477 std::lock_guard<std::mutex> lock(mutex_);
478
479 switch (config_.flush_trigger) {
480 case buffer_flush_trigger::size_threshold:
481 return buffer_.size() >= config_.flush_threshold_size;
482
483 case buffer_flush_trigger::manual:
484 return false;
485
486 default:
487 return buffer_.size() >= config_.flush_threshold_size;
488 }
489 }
490
491 size_t size() const override {
492 std::lock_guard<std::mutex> lock(mutex_);
493 return buffer_.size();
494 }
495
496 const buffer_statistics& get_statistics() const override {
497 return stats_;
498 }
499
500 void clear() override {
501 std::lock_guard<std::mutex> lock(mutex_);
502 buffer_.clear();
503 }
504
505 const buffering_config& get_config() const override {
506 return config_;
507 }
508};

514class time_based_strategy : public buffer_strategy_interface {
515private:
516 buffering_config config_;
517 std::vector<buffered_metric> buffer_;
518 mutable buffer_statistics stats_;
519 mutable std::mutex mutex_;

531 std::lock_guard<std::mutex> lock(mutex_);
532

534 metric.sequence_number = sequence_counter_.fetch_add(1, std::memory_order_relaxed);
535
536 // Check if buffer is full
537 if (buffer_.size() >= config_.max_buffer_size) {
538 // Force flush if buffer is full
539 stats_.forced_flushes.fetch_add(1, std::memory_order_relaxed);
540 }
541
542 buffer_.emplace_back(std::move(metric));
543 stats_.total_items_buffered.fetch_add(1, std::memory_order_relaxed);
544
545 return result_void::success();
546 }
547
548 result<std::vector<buffered_metric>> flush() override {
549 std::lock_guard<std::mutex> lock(mutex_);
550
551 if (buffer_.empty()) {
552 return make_success(std::vector<buffered_metric>{});
553 }
554
555 std::vector<buffered_metric> flushed_items;
556 flushed_items.reserve(buffer_.size());
557
558 // Move all items to flushed_items
559 for (auto& item : buffer_) {
560 flushed_items.emplace_back(std::move(item));
561 }
562
563 stats_.total_items_flushed.fetch_add(flushed_items.size(), std::memory_order_relaxed);
564 stats_.total_flushes.fetch_add(1, std::memory_order_relaxed);
565
566 buffer_.clear();
567 last_flush_time_ = std::chrono::system_clock::now();
568
569 return make_success(std::move(flushed_items));
570 }
571
572 bool should_flush() const override {
573 std::lock_guard<std::mutex> lock(mutex_);
574
575 if (buffer_.empty()) {
576 return false;
577 }
578
579 auto now = std::chrono::system_clock::now();
580 auto time_since_flush = std::chrono::duration_cast<std::chrono::milliseconds>(
581 now - last_flush_time_);
582
583 // Flush if time interval reached or buffer is full
584 return time_since_flush >= config_.flush_interval ||
585 buffer_.size() >= config_.max_buffer_size;
586 }
587
588 size_t size() const override {
589 std::lock_guard<std::mutex> lock(mutex_);
590 return buffer_.size();
591 }
592
593 const buffer_statistics& get_statistics() const override {
594 return stats_;
595 }
596
597 void clear() override {
598 std::lock_guard<std::mutex> lock(mutex_);
599 buffer_.clear();
600 last_flush_time_ = std::chrono::system_clock::now();
601 }
602
603 const buffering_config& get_config() const override {
604 return config_;
605 }

612class priority_based_strategy : public buffer_strategy_interface {
613private:
614 buffering_config config_;
615 std::vector<buffered_metric> buffer_;
616 mutable buffer_statistics stats_;
617 mutable std::mutex mutex_;
618 std::atomic<size_t> sequence_counter_{0};
619
623 void sort_by_priority() {
624 std::sort(buffer_.begin(), buffer_.end(),
625 [](const buffered_metric& a, const buffered_metric& b) {
626 if (a.priority != b.priority) {
627 return a.priority > b.priority; // Higher priority first
628 }
629 return a.sequence_number < b.sequence_number; // FIFO for same priority
630 });
631 }
632
633public:
634 explicit priority_based_strategy(const buffering_config& config)
635 : config_(config) {
636 config_.strategy = buffering_strategy_type::priority_based;
637 buffer_.reserve(config_.max_buffer_size);
638 }
639
640 result_void add_metric(buffered_metric&& metric) override {
641 std::lock_guard<std::mutex> lock(mutex_);
642
643 // Set sequence number
644 metric.sequence_number = sequence_counter_.fetch_add(1, std::memory_order_relaxed);
645
646 // Handle overflow by dropping lowest priority items
647 if (buffer_.size() >= config_.max_buffer_size) {
648 sort_by_priority();
649
650 // Remove lowest priority item
651 if (!buffer_.empty()) {
652 buffer_.pop_back();
653 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
654 }
655 }
656
657 buffer_.emplace_back(std::move(metric));
658 stats_.total_items_buffered.fetch_add(1, std::memory_order_relaxed);
659
660 return result_void::success();
661 }
662
663 result<std::vector<buffered_metric>> flush() override {
664 std::lock_guard<std::mutex> lock(mutex_);
665
666 if (buffer_.empty()) {
667 return make_success(std::vector<buffered_metric>{});
668 }
669
670 // Sort by priority before flushing
671 sort_by_priority();
672
673 std::vector<buffered_metric> flushed_items;
674 flushed_items.reserve(buffer_.size());
675
676 // Move all items to flushed_items
677 for (auto& item : buffer_) {
678 flushed_items.emplace_back(std::move(item));
679 }
680
681 stats_.total_items_flushed.fetch_add(flushed_items.size(), std::memory_order_relaxed);
682 stats_.total_flushes.fetch_add(1, std::memory_order_relaxed);
683
684 buffer_.clear();
685
686 return make_success(std::move(flushed_items));
687 }
688
689 bool should_flush() const override {
690 std::lock_guard<std::mutex> lock(mutex_);
691
692 if (buffer_.empty()) {
693 return false;
694 }
695
696 // Check if we have high-priority items
697 for (const auto& item : buffer_) {
698 if (item.priority >= config_.flush_priority_threshold) {
699 return true;
700 }
701 }
702
703 // Also flush if buffer is getting full
704 return buffer_.size() >= config_.flush_threshold_size;
705 }
706
707 size_t size() const override {
708 std::lock_guard<std::mutex> lock(mutex_);
709 return buffer_.size();
710 }
711
712 const buffer_statistics& get_statistics() const override {
713 return stats_;
714 }
715
716 void clear() override {
717 std::lock_guard<std::mutex> lock(mutex_);
718 buffer_.clear();
719 }
720
721 const buffering_config& get_config() const override {
722 return config_;
723 }

730class adaptive_strategy : public buffer_strategy_interface {
731private:
732 buffering_config config_;
733 std::vector<buffered_metric> buffer_;
734 mutable buffer_statistics stats_;
735 mutable std::mutex mutex_;
736 std::atomic<size_t> sequence_counter_{0};
737 std::chrono::system_clock::time_point last_adaptation_;
738 double current_load_factor_ = 0.0;
739
743 double calculate_load_factor() const {
744 // Simple load factor based on buffer utilization
745 double buffer_utilization = static_cast<double>(buffer_.size()) / config_.max_buffer_size;
746
747 // Add time pressure (how long since last flush)
748 auto now = std::chrono::system_clock::now();
749 auto time_pressure = std::chrono::duration_cast<std::chrono::milliseconds>(
750 now - last_adaptation_).count() / static_cast<double>(config_.flush_interval.count());
751
752 return std::min(1.0, buffer_utilization + time_pressure * 0.1);
753 }
754
758 void adapt_to_load() {
759 current_load_factor_ = calculate_load_factor();
760 last_adaptation_ = std::chrono::system_clock::now();
761
762 // Adjust flush threshold based on load
763 if (current_load_factor_ > config_.load_factor_threshold) {
764 // High load - reduce flush threshold for more frequent flushing
765 config_.flush_threshold_size = std::max(
766 config_.batch_size,
767 static_cast<size_t>(config_.max_buffer_size * 0.3)
768 );
769 } else {
770 // Normal load - use standard threshold
771 config_.flush_threshold_size = config_.max_buffer_size / 2;
772 }
773 }
774
775public:
776 explicit adaptive_strategy(const buffering_config& config)
777 : config_(config), last_adaptation_(std::chrono::system_clock::now()) {
778 config_.strategy = buffering_strategy_type::adaptive;
779 buffer_.reserve(config_.max_buffer_size);
780 }
781
782 result_void add_metric(buffered_metric&& metric) override {
783 std::lock_guard<std::mutex> lock(mutex_);
784
785 // Periodic adaptation
786 auto now = std::chrono::system_clock::now();
787 if (now - last_adaptation_ >= config_.adaptive_check_interval) {
788 adapt_to_load();
789 }
790
791 // Set sequence number
792 metric.sequence_number = sequence_counter_.fetch_add(1, std::memory_order_relaxed);
793
794 // Handle overflow adaptively
795 if (buffer_.size() >= config_.max_buffer_size) {
796 if (current_load_factor_ > config_.load_factor_threshold) {
797 // High load - drop oldest to make space
798 if (!buffer_.empty()) {
799 buffer_.erase(buffer_.begin());
800 stats_.items_dropped_overflow.fetch_add(1, std::memory_order_relaxed);
801 }
802 } else {
803 // Normal load - trigger immediate flush
804 stats_.forced_flushes.fetch_add(1, std::memory_order_relaxed);
805 }
806 }
807
808 buffer_.emplace_back(std::move(metric));
809 stats_.total_items_buffered.fetch_add(1, std::memory_order_relaxed);
810
811 return result_void::success();
812 }
813
814 result<std::vector<buffered_metric>> flush() override {
815 std::lock_guard<std::mutex> lock(mutex_);
816
817 if (buffer_.empty()) {
818 return make_success(std::vector<buffered_metric>{});
819 }
820
821 std::vector<buffered_metric> flushed_items;
822 flushed_items.reserve(buffer_.size());
823
824 // Move all items to flushed_items
825 for (auto& item : buffer_) {
826 flushed_items.emplace_back(std::move(item));
827 }
828
829 stats_.total_items_flushed.fetch_add(flushed_items.size(), std::memory_order_relaxed);
830 stats_.total_flushes.fetch_add(1, std::memory_order_relaxed);
831
832 buffer_.clear();
833
834 return make_success(std::move(flushed_items));
835 }
836
837 bool should_flush() const override {
838 std::lock_guard<std::mutex> lock(mutex_);
839
840 if (buffer_.empty()) {
841 return false;
842 }
843
844 // Flush based on current load and adaptive threshold
845 return buffer_.size() >= config_.flush_threshold_size ||
846 current_load_factor_ > config_.load_factor_threshold;
847 }
848
849 size_t size() const override {
850 std::lock_guard<std::mutex> lock(mutex_);
851 return buffer_.size();
852 }
853
854 const buffer_statistics& get_statistics() const override {
855 return stats_;
856 }
857
858 void clear() override {
859 std::lock_guard<std::mutex> lock(mutex_);
860 buffer_.clear();
861 }
862
863 const buffering_config& get_config() const override {
864 return config_;
865 }

871inline std::unique_ptr<buffer_strategy_interface> create_buffering_strategy(
872 const buffering_config& config) {
873
874 auto validation = config.validate();
875 if (!validation) {
876 throw std::invalid_argument("Invalid buffering configuration: " +
877 validation.get_error().message);
878 }
879
880 switch (config.strategy) {
881 case buffering_strategy_type::immediate:
882 return std::make_unique<immediate_strategy>(config);
883
884 case buffering_strategy_type::fixed_size:
885 return std::make_unique<fixed_size_strategy>(config);
886
887 case buffering_strategy_type::time_based:
888 return std::make_unique<time_based_strategy>(config);
889
890 case buffering_strategy_type::priority_based:
891 return std::make_unique<priority_based_strategy>(config);
892
893 case buffering_strategy_type::adaptive:
894 return std::make_unique<adaptive_strategy>(config);
895
896 default:
897 return std::make_unique<fixed_size_strategy>(config);
898 }
899}
