207 // External system manager
208 class external_system_manager {
209 private:
210 std::unordered_map<std::string, std::unique_ptr<external_system_adapter>> adapters_;
211 mutable std::mutex adapters_mutex_;
212
213 public:
214 external_system_manager() = default;

217 // Adapter management
218 void register_adapter(const std::string& name, std::unique_ptr<external_system_adapter> adapter) {
219 std::lock_guard<std::mutex> lock(adapters_mutex_);
220 adapters_[name] = std::move(adapter);
221 }
222
223 external_system_adapter* get_adapter(const std::string& name) const {
224 std::lock_guard<std::mutex> lock(adapters_mutex_);
225 auto it = adapters_.find(name);
226 return it != adapters_.end() ? it->second.get() : nullptr;
227 }
228
229 std::vector<std::string> get_adapter_names() const {
230 std::lock_guard<std::mutex> lock(adapters_mutex_);
231 std::vector<std::string> names;
232 for (const auto& [name, adapter] : adapters_) {
233 names.push_back(name);
234 }
235 return names;
236 }
237
238 // Connection management
239 bool connect_all() {
240 std::lock_guard<std::mutex> lock(adapters_mutex_);
241 bool all_connected = true;
242 for (const auto& [name, adapter] : adapters_) {
243 if (!adapter->connect()) {
244 all_connected = false;
245 }
246 }
247 return all_connected;
248 }
249
250 void disconnect_all() {
251 std::lock_guard<std::mutex> lock(adapters_mutex_);
252 for (const auto& [name, adapter] : adapters_) {
253 adapter->disconnect();
254 }
255 }
256
257 // Health monitoring
258 std::unordered_map<std::string, bool> get_health_status() const {
259 std::lock_guard<std::mutex> lock(adapters_mutex_);
260 std::unordered_map<std::string, bool> status;
261 for (const auto& [name, adapter] : adapters_) {
262 status[name] = adapter->is_healthy();
263 }
264 return status;
265 }
266
267 std::unordered_map<std::string, external_connection_info> get_connection_info() const {
268 std::lock_guard<std::mutex> lock(adapters_mutex_);
269 std::unordered_map<std::string, external_connection_info> info;
270 for (const auto& [name, adapter] : adapters_) {
271 info[name] = adapter->get_connection_info();
272 }
273 return info;
274 }
275
276 // Message forwarding
277 bool forward_message(const std::string& adapter_name, const core::message& msg) {
278 auto adapter = get_adapter(adapter_name);
279 return adapter ? adapter->forward_message(msg) : false;
280 }
281
282 // Broadcast message to all connected adapters
283 size_t broadcast_message(const core::message& msg) {
284 std::lock_guard<std::mutex> lock(adapters_mutex_);
285 size_t sent_count = 0;
286 for (const auto& [name, adapter] : adapters_) {
287 if (adapter->is_connected() && adapter->forward_message(msg)) {
288 sent_count++;
289 }
290 }
291 return sent_count;
292 }
293 };
