75 // Resolve services
76 template<typename T>
77 std::shared_ptr<T> resolve(const std::string& name) {
78 std::lock_guard<std::recursive_mutex> lock(mutex_);
79
80 auto it = registrations_.find(name);
81 if (it == registrations_.end()) {
82 return nullptr;
83 }
84
85 auto& registration = it->second;
86
87 // Return existing instance for singletons
88 if (registration.is_singleton && registration.instance) {
89 return std::static_pointer_cast<T>(registration.instance);
90 }
91
92 // Create new instance
93 auto instance = std::static_pointer_cast<T>(registration.factory());
94
95 // Store for singletons
96 if (registration.is_singleton) {
97 registration.instance = instance;
98 }
99
100 return instance;
101 }
