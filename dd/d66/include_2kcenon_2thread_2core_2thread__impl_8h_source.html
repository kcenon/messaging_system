74 template<typename F>
75 void start_thread(F&& func) {
76 #ifdef USE_STD_JTHREAD
77 stop_source_ = std::make_unique<stop_source_type>();
78 thread_ = std::make_unique<thread_type>([func = std::forward<F>(func), this](std::stop_token st) {
79 func(st);
80 });
81 #else
82 stop_requested_ = std::make_unique<stop_token_type>(false);
83 thread_ = std::make_unique<thread_type>([func = std::forward<F>(func), this]() {
84 func(*stop_requested_);
85 });
86 #endif
87 }

92 void request_stop() {
93 #ifdef USE_STD_JTHREAD
94 if (stop_source_) {
95 stop_source_->request_stop();
96 }
97 #else
98 if (stop_requested_) {
99 stop_requested_->store(true);
100 }
101 #endif
102 }
103
107 [[nodiscard]] bool stop_requested() const {
108 #ifdef USE_STD_JTHREAD
109 return stop_source_ && stop_source_->stop_requested();
110 #else
111 return stop_requested_ && stop_requested_->load();
112 #endif
113 }
114
118 void join() {
119 if (thread_ && thread_->joinable()) {
120 thread_->join();
121 }
122 }

127 void detach() {
128 if (thread_ && thread_->joinable()) {
129 thread_->detach();
130 }
131 }

136 [[nodiscard]] bool joinable() const {

138 }
