355 void add_component(std::shared_ptr<monitorable_interface> component) {
356 if (component) {
357 components_.push_back(component);
358 }
359 }
360
366 bool remove_component(const std::string& id) {
367 auto it = std::remove_if(components_.begin(), components_.end(),
368 [&id](const auto& comp) {
369 return comp->get_monitoring_id() == id;
370 });
371
372 if (it != components_.end()) {
373 components_.erase(it, components_.end());
374 return true;
375 }
376 return false;
377 }
378
383 result<monitoring_data> collect_all() const {
384 monitoring_data aggregated(aggregator_id_);
385
386 for (const auto& component : components_) {
387 if (!component->is_monitoring_enabled()) {
388 continue;
389 }
390
391 auto result = component->get_monitoring_data();
392 if (result) {
393 aggregated.merge(result.value(), component->get_monitoring_id());
394 } else {
395 // Log error but continue with other components
396 aggregated.add_tag(
397 component->get_monitoring_id() + ".error",
398 result.get_error().message
399 );
400 }
401 }
402
403 // Add aggregator metadata
404 aggregated.add_metric("aggregator.component_count",
405 static_cast<double>(components_.size()));
406 aggregated.add_metric("aggregator.total_metrics",
407 static_cast<double>(aggregated.metric_count()));
408
409 return make_success(std::move(aggregated));
410 }
411
417 std::shared_ptr<monitorable_interface> get_component(const std::string& id) const {
418 auto it = std::find_if(components_.begin(), components_.end(),
419 [&id](const auto& comp) {
420 return comp->get_monitoring_id() == id;
421 });
422
423 return (it != components_.end()) ? *it : nullptr;
424 }
425
430 std::vector<std::string> get_component_ids() const {
431 std::vector<std::string> ids;
432 ids.reserve(components_.size());
433
434 for (const auto& component : components_) {
435 ids.push_back(component->get_monitoring_id());
436 }
437
438 return ids;
439 }
