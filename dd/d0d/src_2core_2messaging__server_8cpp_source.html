49 auto messaging_server::start_server(unsigned short port) -> void
50 {
51 // If already running, do nothing
52 if (is_running_.load())
53 {
54 return;
55 }
56 is_running_.store(true);
57
58 // Create io_context and acceptor
59 io_context_ = std::make_unique<asio::io_context>();
60 acceptor_ = std::make_unique<tcp::acceptor>(
61 *io_context_, tcp::endpoint(tcp::v4(), port));
62
63 // Prepare promise/future for wait_for_stop()
64 stop_promise_.emplace();
65 stop_future_ = stop_promise_->get_future();
66
67 // Begin accepting connections
68 do_accept();
69
70 // Start thread to run the io_context
71 server_thread_ = std::make_unique<std::thread>(
72 [this]()
73 {
74 try
75 {
76 io_context_->run();
77 }
78 catch (...)
79 {
80 // Optionally handle any uncaught exceptions
81 }
82 });
83
84 NETWORK_LOG_INFO("[messaging_server] Started listening on port " + std::to_string(port));
85 }

87 auto messaging_server::stop_server() -> void
88 {
89 if (!is_running_.load())
90 {
91 return;
92 }
93 is_running_.store(false);
94
95 // Close the acceptor
96 if (acceptor_ && acceptor_->is_open())
97 {
98 asio::error_code ec;
99 acceptor_->close(ec);
100 }
101
102 // Stop all active sessions
103 for (auto& sess : sessions_)
104 {
105 if (sess)
106 {
107 sess->stop_session();
108 }
109 }
110 sessions_.clear();
111
112 // Stop io_context
113 if (io_context_)
114 {
115 io_context_->stop();
116 }
117
118 // Join the thread
119 if (server_thread_ && server_thread_->joinable())
120 {
121 server_thread_->join();
122 }
123
124 // Signal the promise for wait_for_stop()
125 if (stop_promise_.has_value())
126 {
127 stop_promise_->set_value();
128 stop_promise_.reset();
129 }
130
131 NETWORK_LOG_INFO("[messaging_server] Stopped.");
132 }

142 auto messaging_server::do_accept() -> void
143 {
144 auto self = shared_from_this();
145 acceptor_->async_accept(
146 [this, self](std::error_code ec, tcp::socket sock)
147 { on_accept(ec, std::move(sock)); });
148 }

150 auto messaging_server::on_accept(std::error_code ec, tcp::socket socket)
151 -> void
152 {
153 if (!is_running_.load())
154 {
155 return;
156 }
157 if (ec)
158 {
159 NETWORK_LOG_ERROR("[messaging_server] Accept error: " + ec.message());
160 return;
161 }
162
163 // Create a new messaging_session
164 auto new_session = std::make_shared<network_system::session::messaging_session>(
165 std::move(socket), server_id_);
166
167 // Track it in our sessions_ vector
168 sessions_.push_back(new_session);
169
170 // Start the session
171 new_session->start_session();
172
173 // Accept next connection
174 do_accept();
175 }
