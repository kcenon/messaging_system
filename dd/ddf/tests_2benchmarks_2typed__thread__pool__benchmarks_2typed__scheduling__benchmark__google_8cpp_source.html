34// Job execution record for analysis
35struct JobExecutionRecord {
36 size_t job_id;
37 job_types priority;
38 high_resolution_clock::time_point submit_time;
39 high_resolution_clock::time_point start_time;
40 high_resolution_clock::time_point complete_time;
41
42 auto queue_latency_ms() const {
43 return duration_cast<milliseconds>(start_time - submit_time).count();
44 }
45
46 auto total_latency_ms() const {
47 return duration_cast<milliseconds>(complete_time - submit_time).count();
48 }
49};

55// Helper function to submit test job
56static void submit_test_job(std::shared_ptr<typed_thread_pool> pool,
57 size_t job_id,
58 job_types priority,
59 milliseconds work_duration) {
60 auto submit_time = high_resolution_clock::now();
61
62 auto job = std::make_unique<typed_job_t<job_types>>(
63 priority,
64 [job_id, priority, submit_time, work_duration]() -> result_void {
65 auto start_time = high_resolution_clock::now();
66
67 // Simulate work
68 auto work_end = start_time + work_duration;
69 while (high_resolution_clock::now() < work_end) {
70 volatile int sum = 0;
71 for (int i = 0; i < 1000; ++i) {
72 sum += i;
73 }
74 }
75
76 auto complete_time = high_resolution_clock::now();
77
78 // Record execution
79 JobExecutionRecord record;
80 record.job_id = job_id;
81 record.priority = priority;
82 record.submit_time = submit_time;
83 record.start_time = start_time;
84 record.complete_time = complete_time;
85
86 {
87 std::lock_guard<std::mutex> lock(g_records_mutex);
88 g_execution_records.push_back(record);
89 }
90
91 return {};
92 }
93 );
94
95 pool->enqueue(std::move(job));
96}

98// Helper function to setup priority pool
99static std::shared_ptr<typed_thread_pool> setup_priority_pool(size_t num_workers = 4) {
100 auto pool = std::make_shared<typed_thread_pool>();
101
102 // Add workers with different priority responsibilities
103 for (size_t i = 0; i < num_workers; ++i) {
104 std::vector<job_types> responsibilities;
105 if (i < num_workers / 2) {
106 // High-priority workers
107 responsibilities = {job_types::Critical, job_types::RealTime};
108 } else {
109 // General workers
110 responsibilities = {job_types::RealTime, job_types::Batch, job_types::Background};
111 }
112
113 auto worker = std::make_unique<typed_thread_worker_t<job_types>>(
114 pool, responsibilities);
115 pool->enqueue(std::move(worker));
116 }
117
118 auto result = pool->start();
119 if (result.has_error()) {
120 throw std::runtime_error("Failed to start priority pool");
121 }
122
123 return pool;
124}

129static void BM_BasicPriorityOrdering(benchmark::State& state) {
130 const size_t jobs_per_priority = state.range(0);
131
132 for (auto _ : state) {
133 g_execution_records.clear();
134
135 auto pool = setup_priority_pool();
136
137 std::atomic<size_t> job_counter{0};
138 std::atomic<size_t> completed{0};
139
140 // Submit jobs in reverse priority order
141 std::vector<job_types> types = {
142 job_types::Background,
143 job_types::Batch,
144 job_types::RealTime,
145 job_types::Critical
146 };
147
148 for (auto priority : types) {
149 for (size_t i = 0; i < jobs_per_priority; ++i) {
150 submit_test_job(pool, job_counter.fetch_add(1), priority,
151 milliseconds(10));
152 }
153 }
154
155 // Wait for completion
156 while (completed.load() < types.size() * jobs_per_priority) {
157 std::this_thread::sleep_for(milliseconds(1));
158 std::lock_guard<std::mutex> lock(g_records_mutex);
159 completed = g_execution_records.size();
160 }
161
162 pool->stop();
163
164 // Analyze ordering
165 std::lock_guard<std::mutex> lock(g_records_mutex);
166 auto sorted_records = g_execution_records;
167 std::sort(sorted_records.begin(), sorted_records.end(),
168 [](const auto& a, const auto& b) { return a.start_time < b.start_time; });
169
170 // Calculate priority ordering score
171 int correct_orderings = 0;
172 int total_comparisons = 0;
173
174 for (size_t i = 0; i < sorted_records.size(); ++i) {
175 for (size_t j = i + 1; j < sorted_records.size(); ++j) {
176 if (sorted_records[i].priority >= sorted_records[j].priority) {
177 correct_orderings++;
178 }
179 total_comparisons++;
180 }
181 }
182
183 double ordering_score = (total_comparisons > 0) ?
184 (correct_orderings * 100.0 / total_comparisons) : 0.0;
185 state.counters["ordering_score%"] = ordering_score;
186 }
187
188 state.SetItemsProcessed(state.iterations() * jobs_per_priority * 4);
189}
190BENCHMARK(BM_BasicPriorityOrdering)
191 ->Arg(25)
192 ->Arg(50)
193 ->Arg(100)
194 ->Unit(benchmark::kMillisecond);

199static void BM_PriorityFairness(benchmark::State& state) {
200 const size_t total_jobs = state.range(0);
201
202 for (auto _ : state) {
203 g_execution_records.clear();
204
205 auto pool = setup_priority_pool();
206
207 std::atomic<size_t> job_counter{0};
208 std::atomic<size_t> completed{0};
209
210 // Submit jobs with mixed priorities
211 std::random_device rd;
212 std::mt19937 gen(rd());
213 std::discrete_distribution<> priority_dist({10, 30, 40, 20}); // Background, Batch, RealTime, Critical
214
215 auto types = std::vector<job_types>{
216 job_types::Background, job_types::Batch,
217 job_types::RealTime, job_types::Critical
218 };
219
220 for (size_t i = 0; i < total_jobs; ++i) {
221 auto priority = types[priority_dist(gen)];
222 submit_test_job(pool, job_counter.fetch_add(1), priority,
223 milliseconds(5));
224 }
225
226 // Wait for completion
227 while (completed.load() < total_jobs) {
228 std::this_thread::sleep_for(milliseconds(1));
229 std::lock_guard<std::mutex> lock(g_records_mutex);
230 completed = g_execution_records.size();
231 }
232
233 pool->stop();
234
235 // Analyze fairness
236 std::lock_guard<std::mutex> lock(g_records_mutex);
237 std::map<job_types, std::vector<double>> latencies_by_priority;
238
239 for (const auto& record : g_execution_records) {
240 latencies_by_priority[record.priority].push_back(record.total_latency_ms());
241 }
242
243 for (const auto& [priority, latencies] : latencies_by_priority) {
244 if (latencies.empty()) continue;
245
246 double avg_latency = std::accumulate(latencies.begin(), latencies.end(), 0.0) / latencies.size();
247 auto sorted_latencies = latencies;
248 std::sort(sorted_latencies.begin(), sorted_latencies.end());
249 double p95_latency = sorted_latencies[static_cast<size_t>(0.95 * (sorted_latencies.size() - 1))];
250
251 std::string priority_str;
252 switch (priority) {
253 case job_types::Background: priority_str = "background"; break;
254 case job_types::Batch: priority_str = "batch"; break;
255 case job_types::RealTime: priority_str = "realtime"; break;
256 case job_types::Critical: priority_str = "critical"; break;
257 }
258
259 state.counters[priority_str + "_avg_ms"] = avg_latency;
260 state.counters[priority_str + "_p95_ms"] = p95_latency;
261 state.counters[priority_str + "_count"] = latencies.size();
262 }
263 }
264
265 state.SetItemsProcessed(state.iterations() * total_jobs);
266}
267BENCHMARK(BM_PriorityFairness)
268 ->Arg(500)
269 ->Arg(1000)
270 ->Unit(benchmark::kMillisecond);

275static void BM_PriorityInversion(benchmark::State& state) {
276 const size_t low_priority_jobs = state.range(0);
277 const size_t high_priority_jobs = state.range(1);
278
279 for (auto _ : state) {
280 g_execution_records.clear();
281
282 auto pool = setup_priority_pool();
283
284 std::atomic<size_t> job_counter{0};
285 std::atomic<size_t> completed{0};
286
287 // Fill queue with low priority jobs
288 for (size_t i = 0; i < low_priority_jobs; ++i) {
289 submit_test_job(pool, job_counter.fetch_add(1), job_types::Background,
290 milliseconds(100)); // Long work
291 }
292
293 // Wait a bit for some to start processing
294 std::this_thread::sleep_for(milliseconds(50));
295
296 // Submit high priority jobs
297 for (size_t i = 0; i < high_priority_jobs; ++i) {
298 submit_test_job(pool, job_counter.fetch_add(1), job_types::Critical,
299 milliseconds(10)); // Quick work
300 }
301
302 // Wait for completion
303 const size_t total_jobs = low_priority_jobs + high_priority_jobs;
304 while (completed.load() < total_jobs) {
305 std::this_thread::sleep_for(milliseconds(1));
306 std::lock_guard<std::mutex> lock(g_records_mutex);
307 completed = g_execution_records.size();
308 }
309
310 pool->stop();
311
312 // Analyze priority inversion
313 std::lock_guard<std::mutex> lock(g_records_mutex);
314 std::vector<double> critical_latencies;
315 std::vector<double> background_latencies;
316
317 for (const auto& record : g_execution_records) {
318 if (record.priority == job_types::Critical) {
319 critical_latencies.push_back(record.total_latency_ms());
320 } else if (record.priority == job_types::Background) {
321 background_latencies.push_back(record.total_latency_ms());
322 }
323 }
324
325 if (!critical_latencies.empty()) {
326 double avg_critical = std::accumulate(critical_latencies.begin(),
327 critical_latencies.end(), 0.0) / critical_latencies.size();
328 state.counters["critical_avg_ms"] = avg_critical;
329 }
330
331 if (!background_latencies.empty()) {
332 double avg_background = std::accumulate(background_latencies.begin(),
333 background_latencies.end(), 0.0) / background_latencies.size();
334 state.counters["background_avg_ms"] = avg_background;
335 }
336 }
337
338 state.SetItemsProcessed(state.iterations() * (low_priority_jobs + high_priority_jobs));
339}
340BENCHMARK(BM_PriorityInversion)
341 ->Args({50, 10})
342 ->Args({100, 20})
343 ->Unit(benchmark::kMillisecond);

348static void BM_StarvationResistance(benchmark::State& state) {
349 const size_t high_priority_stream = state.range(0);
350 const size_t low_priority_jobs = state.range(1);
351
352 for (auto _ : state) {
353 g_execution_records.clear();
354
355 auto pool = setup_priority_pool();
356
357 std::atomic<size_t> job_counter{0};
358 std::atomic<size_t> completed{0};
359 std::atomic<bool> stop_stream{false};
360
361 // Continuous high-priority job stream
362 std::thread high_priority_thread([&]() -> result_void {
363 for (size_t i = 0; i < high_priority_stream && !stop_stream.load(); ++i) {
364 submit_test_job(pool, job_counter.fetch_add(1), job_types::RealTime,
365 milliseconds(5));
366 std::this_thread::sleep_for(milliseconds(8));
367 }
368 });
369
370 // Low-priority jobs that shouldn't be starved
371 std::thread low_priority_thread([&]() -> result_void {
372 for (size_t i = 0; i < low_priority_jobs; ++i) {
373 submit_test_job(pool, job_counter.fetch_add(1), job_types::Background,
374 milliseconds(20));
375 std::this_thread::sleep_for(milliseconds(100));
376 }
377 });
378
379 high_priority_thread.join();
380 low_priority_thread.join();
381
382 // Wait for completion
383 const size_t total_jobs = high_priority_stream + low_priority_jobs;
384 while (completed.load() < total_jobs) {
385 std::this_thread::sleep_for(milliseconds(10));
386 std::lock_guard<std::mutex> lock(g_records_mutex);
387 completed = g_execution_records.size();
388 if (completed.load() >= total_jobs * 0.95) {
389 // Allow early exit if most jobs complete
390 stop_stream = true;
391 break;
392 }
393 }
394
395 pool->stop();
396
397 // Check if low priority jobs were executed
398 std::lock_guard<std::mutex> lock(g_records_mutex);
399 size_t low_priority_completed = 0;
400 double max_low_priority_latency = 0.0;
401
402 for (const auto& record : g_execution_records) {
403 if (record.priority == job_types::Background) {
404 low_priority_completed++;
405 max_low_priority_latency = std::max(max_low_priority_latency,
406 record.total_latency_ms());
407 }
408 }
409
410 state.counters["low_priority_completed"] = low_priority_completed;
411 state.counters["low_priority_completion%"] = (low_priority_completed * 100.0) / low_priority_jobs;
412 state.counters["max_low_priority_latency_ms"] = max_low_priority_latency;
413 }
414
415 state.SetItemsProcessed(state.iterations() * (high_priority_stream + low_priority_jobs));
416}
417BENCHMARK(BM_StarvationResistance)
418 ->Args({500, 50})
419 ->Unit(benchmark::kMillisecond)
420 ->UseRealTime();

425static void BM_MixedPriorityWorkload(benchmark::State& state) {
426 const size_t duration_ms = state.range(0);
427
428 for (auto _ : state) {
429 g_execution_records.clear();
430
431 auto pool = setup_priority_pool(8);
432
433 std::atomic<size_t> job_counter{0};
434 std::atomic<size_t> completed{0};
435 std::atomic<bool> stop_generators{false};
436
437 std::vector<std::thread> load_generators;
438
439 // High-frequency low priority
440 load_generators.emplace_back([&]() -> result_void {
441 while (!stop_generators.load()) {
442 submit_test_job(pool, job_counter.fetch_add(1), job_types::Background,
443 milliseconds(20));
444 std::this_thread::sleep_for(milliseconds(10));
445 }
446 });
447
448 // Medium-frequency normal priority
449 load_generators.emplace_back([&]() -> result_void {
450 while (!stop_generators.load()) {
451 submit_test_job(pool, job_counter.fetch_add(1), job_types::Batch,
452 milliseconds(30));
453 std::this_thread::sleep_for(milliseconds(25));
454 }
455 });
456
457 // Low-frequency high priority
458 load_generators.emplace_back([&]() -> result_void {
459 while (!stop_generators.load()) {
460 submit_test_job(pool, job_counter.fetch_add(1), job_types::RealTime,
461 milliseconds(15));
462 std::this_thread::sleep_for(milliseconds(50));
463 }
464 });
465
466 // Burst critical priority
467 load_generators.emplace_back([&]() -> result_void {
468 std::this_thread::sleep_for(milliseconds(duration_ms / 2));
469 for (int i = 0; i < 20 && !stop_generators.load(); ++i) {
470 submit_test_job(pool, job_counter.fetch_add(1), job_types::Critical,
471 milliseconds(5));
472 }
473 });
474
475 // Run for specified duration
476 std::this_thread::sleep_for(milliseconds(duration_ms));
477 stop_generators = true;
478
479 for (auto& thread : load_generators) {
480 thread.join();
481 }
482
483 // Wait for remaining jobs
484 std::this_thread::sleep_for(milliseconds(100));
485 pool->stop();
486
487 // Analyze performance by priority
488 std::lock_guard<std::mutex> lock(g_records_mutex);
489 std::map<job_types, std::pair<double, size_t>> priority_stats;
490
491 for (const auto& record : g_execution_records) {
492 auto& stats = priority_stats[record.priority];
493 stats.first += record.total_latency_ms();
494 stats.second++;
495 }
496
497 for (auto& [priority, stats] : priority_stats) {
498 if (stats.second > 0) {
499 stats.first /= stats.second; // Calculate average
500
501 std::string priority_str;
502 switch (priority) {
503 case job_types::Background: priority_str = "background"; break;
504 case job_types::Batch: priority_str = "batch"; break;
505 case job_types::RealTime: priority_str = "realtime"; break;
506 case job_types::Critical: priority_str = "critical"; break;
507 }
508
509 state.counters[priority_str + "_jobs"] = stats.second;
510 state.counters[priority_str + "_avg_latency_ms"] = stats.first;
511 }
512 }
513
514 state.counters["total_jobs"] = g_execution_records.size();
515 }
516
517 state.SetIterationTime(duration_ms / 1000.0);
518}

520 ->Arg(5000) // 5 second test
521 ->UseManualTime()
522 ->Unit(benchmark::kSecond);
