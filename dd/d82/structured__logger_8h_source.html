123 static std::string format(const structured_log_entry& entry) {
124 // Simple JSON formatting - in real implementation would use a proper JSON library
125 std::string json = "{";
126 json += "\"level\":\"" + level_to_string(entry.level) + "\",";
127 json += "\"message\":\"" + entry.message + "\",";
128 json += "\"timestamp\":\"" + std::to_string(std::chrono::duration_cast<std::chrono::milliseconds>(
129 entry.timestamp.time_since_epoch()).count()) + "\",";
130 json += "\"fields\":{";
131
132 bool first = true;
133 for (const auto& [key, value] : entry.fields) {
134 if (!first) json += ",";
135 json += "\"" + key + "\":";
136
137 std::visit([&json](const auto& v) {
138 using T = std::decay_t<decltype(v)>;
139 if constexpr (std::is_same_v<T, std::string>) {
140 json += "\"" + v + "\"";
141 } else if constexpr (std::is_same_v<T, bool>) {
142 json += v ? "true" : "false";
143 } else {
144 json += std::to_string(v);
145 }
146 }, value);
147
148 first = false;
149 }
150
151 json += "}}";
152 return json;
153 }
