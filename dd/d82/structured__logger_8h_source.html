192 // Format log entry based on output format
193 std::string format_entry(const log_entry& entry) const {
194 switch (format_) {
195 case output_format::json:
196 return format_json(entry);
197 case output_format::logfmt:
198 return format_logfmt(entry);
199 case output_format::plain:
200 return format_plain(entry);
201 default:
202 return format_json(entry);
203 }

206 // Format as JSON (simple implementation without nlohmann/json)
207 std::string format_json(const log_entry& entry) const {
208 std::ostringstream oss;
209 oss << "{";
210
211 // Add timestamp
212 auto time_t = std::chrono::system_clock::to_time_t(entry.timestamp_);
213 oss << "\"" << standard_fields::TIMESTAMP << "\":\"";
214 oss << std::put_time(std::gmtime(&time_t), "%Y-%m-%dT%H:%M:%SZ") << "\",";
215
216 // Add standard fields
217 oss << "\"" << standard_fields::LEVEL << "\":\"" << level_to_string(entry.level_) << "\",";
218 oss << "\"" << standard_fields::MESSAGE << "\":\"" << escape_json(entry.message_) << "\",";
219 oss << "\"" << standard_fields::THREAD_ID << "\":\""
220 << std::hash<std::thread::id>{}(std::this_thread::get_id()) << "\"";
221
222 // Add global context
223 for (const auto& [key, value] : global_context_) {
224 oss << ",\"" << key << "\":" << any_to_json_string(value);
225 }
226
227 // Add entry context
228 if (!entry.context_.empty()) {
229 oss << ",\"context\":{";
230 bool first = true;
231 for (const auto& [key, value] : entry.context_) {
232 if (!first) oss << ",";
233 oss << "\"" << key << "\":" << any_to_json_string(value);
234 first = false;
235 }
236 oss << "}";
237 }
238
239 // Add fields
240 for (const auto& [key, value] : entry.fields_) {
241 oss << ",\"" << key << "\":" << any_to_json_string(value);
242 }
243
244 oss << "}";
245 return oss.str();
246 }
247
248 // Format as logfmt
249 std::string format_logfmt(const log_entry& entry) const {
250 std::ostringstream oss;
251
252 // Add timestamp
253 auto time_t = std::chrono::system_clock::to_time_t(entry.timestamp_);
254 oss << "timestamp=\"" << std::put_time(std::gmtime(&time_t), "%Y-%m-%dT%H:%M:%SZ") << "\" ";
255
256 // Add standard fields
257 oss << "level=" << level_to_string(entry.level_) << " ";
258 oss << "message=\"" << escape_string(entry.message_) << "\" ";
259 oss << "thread_id=" << std::hash<std::thread::id>{}(std::this_thread::get_id()) << " ";
260
261 // Add all fields
262 auto add_fields = [&oss, this](const auto& fields) {
263 for (const auto& [key, value] : fields) {
264 oss << key << "=" << format_value_logfmt(value) << " ";
265 }
266 };
267
268 add_fields(global_context_);
269 add_fields(entry.context_);
270 add_fields(entry.fields_);
271
272 return oss.str();
273 }

275 // Format as plain text
276 std::string format_plain(const log_entry& entry) const {
277 std::ostringstream oss;
278
279 // Traditional format with structured data appended
280 oss << entry.message_;
281
282 // Add fields if present
283 if (!entry.fields_.empty()) {
284 oss << " [";

289 first = false;
290 }
291 oss << "]";
292 }
293
294 return oss.str();

297 // Convert std::any to JSON string value
298 std::string any_to_json_string(const std::any& value) const {
299 if (value.type() == typeid(int)) {
300 return std::to_string(std::any_cast<int>(value));
301 } else if (value.type() == typeid(long)) {
302 return std::to_string(std::any_cast<long>(value));
303 } else if (value.type() == typeid(uint64_t)) {
304 return std::to_string(std::any_cast<uint64_t>(value));
305 } else if (value.type() == typeid(double)) {
306 return std::to_string(std::any_cast<double>(value));
307 } else if (value.type() == typeid(float)) {
308 return std::to_string(std::any_cast<float>(value));
309 } else if (value.type() == typeid(bool)) {
310 return std::any_cast<bool>(value) ? "true" : "false";
311 } else if (value.type() == typeid(std::string)) {
312 return "\"" + escape_json(std::any_cast<std::string>(value)) + "\"";
313 } else if (value.type() == typeid(const char*)) {
314 return "\"" + escape_json(std::any_cast<const char*>(value)) + "\"";
315 } else {
316 return "\"unknown\"";
317 }
318 }

320 // Escape string for JSON
321 std::string escape_json(const std::string& str) const {
322 std::string escaped;
323 for (char c : str) {
324 if (c == '"') escaped += "\\\"";
325 else if (c == '\\') escaped += "\\\\";
326 else if (c == '\n') escaped += "\\n";
327 else if (c == '\r') escaped += "\\r";
328 else if (c == '\t') escaped += "\\t";
329 else if (c == '\b') escaped += "\\b";
330 else if (c == '\f') escaped += "\\f";
331 else escaped += c;
332 }
333 return escaped;
334 }

336 // Format value for logfmt
337 std::string format_value_logfmt(const std::any& value) const {
338 if (value.type() == typeid(std::string)) {
339 return "\"" + escape_string(std::any_cast<std::string>(value)) + "\"";
340 } else if (value.type() == typeid(const char*)) {
341 return "\"" + escape_string(std::any_cast<const char*>(value)) + "\"";
342 } else {
343 // For numeric values, use plain format
344 return format_value_plain(value);
345 }
346 }

348 // Format value for plain text
349 std::string format_value_plain(const std::any& value) const {
350 if (value.type() == typeid(int)) {
351 return std::to_string(std::any_cast<int>(value));
352 } else if (value.type() == typeid(long)) {
353 return std::to_string(std::any_cast<long>(value));
354 } else if (value.type() == typeid(uint64_t)) {
355 return std::to_string(std::any_cast<uint64_t>(value));
356 } else if (value.type() == typeid(double)) {
357 return std::to_string(std::any_cast<double>(value));
358 } else if (value.type() == typeid(float)) {
359 return std::to_string(std::any_cast<float>(value));
360 } else if (value.type() == typeid(bool)) {
361 return std::any_cast<bool>(value) ? "true" : "false";
362 } else if (value.type() == typeid(std::string)) {
363 return std::any_cast<std::string>(value);
364 } else if (value.type() == typeid(const char*)) {
365 return std::any_cast<const char*>(value);
366 } else {
367 return "unknown";
368 }
369 }

373 std::string escaped;
374 for (char c : str) {
375 if (c == '"') escaped += "\\\"";

377 else if (c == '\n') escaped += "\\n";
378 else if (c == '\r') escaped += "\\r";
379 else if (c == '\t') escaped += "\\t";
380 else escaped += c;
381 }
382 return escaped;
