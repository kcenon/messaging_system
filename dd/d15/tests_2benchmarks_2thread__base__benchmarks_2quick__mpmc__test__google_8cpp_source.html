21// Helper function for producer-consumer test
22template<typename QueueType>
23static void run_producer_consumer_test(QueueType& queue,
24 size_t num_producers,
25 size_t num_consumers,
26 size_t ops_per_thread) {
27 std::atomic<size_t> total_produced{0};
28 std::atomic<size_t> total_consumed{0};
29
30 std::vector<std::thread> producers;
31 std::vector<std::thread> consumers;
32
33 // Start producers
34 for (size_t i = 0; i < num_producers; ++i) {
35 producers.emplace_back([&queue, &total_produced, ops_per_thread]() {
36 for (size_t j = 0; j < ops_per_thread; ++j) {
37 auto job = std::make_unique<callback_job>([]() -> result_void {
38 // Simulate minimal work
39 volatile int x = 0;
40 for (int i = 0; i < 10; ++i) x = x + 1;
41 return result_void();
42 });
43
44 while (!queue.enqueue(std::move(job))) {
45 std::this_thread::yield();
46 }
47 total_produced.fetch_add(1);
48 }
49 });
50 }
51
52 // Start consumers
53 for (size_t i = 0; i < num_consumers; ++i) {
54 consumers.emplace_back([&queue, &total_consumed, &total_produced, num_producers, ops_per_thread, num_consumers]() {
55 size_t consumed = 0;
56 size_t total_target = num_producers * ops_per_thread;
57
58 while (consumed < total_target / num_consumers ||
59 total_consumed.load() < total_target) {
60 auto result = queue.dequeue();
61 if (result.has_value()) {
62 auto work_result = result.value()->do_work();
63 (void)work_result;
64 consumed++;
65 total_consumed.fetch_add(1);
66 } else if (total_produced.load() >= total_target &&
67 total_consumed.load() >= total_target) {
68 break;
69 }
70 }
71 });
72 }
73
74 // Wait for all threads
75 for (auto& t : producers) t.join();
76 for (auto& t : consumers) t.join();
77}
