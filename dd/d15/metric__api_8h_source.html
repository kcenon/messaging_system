329 JsonBuilder& add(const std::string& key, const std::string& value) {

336 JsonBuilder& add(const std::string& key, double value) {

338 ss_ << "\"" << key << "\":" << value;

341 }
342
343 JsonBuilder& add(const std::string& key, int value) {
344 if (!first_) ss_ << ",";
345 ss_ << "\"" << key << "\":" << value;
346 first_ = false;
347 return *this;
348 }
349
350 JsonBuilder& add(const std::string& key, bool value) {
351 if (!first_) ss_ << ",";
352 ss_ << "\"" << key << "\":" << (value ? "true" : "false");
353 first_ = false;
354 return *this;
355 }
356
357 JsonBuilder& add_object(const std::string& key, const std::string& json_object) {
358 if (!first_) ss_ << ",";
359 ss_ << "\"" << key << "\":" << json_object;
360 first_ = false;
361 return *this;
362 }
363
364 JsonBuilder& add_array(const std::string& key, const std::vector<std::string>& items) {
365 if (!first_) ss_ << ",";
366 ss_ << "\"" << key << "\":[";
367 for (size_t i = 0; i < items.size(); ++i) {
368 if (i > 0) ss_ << ",";
369 ss_ << "\"" << escape_string(items[i]) << "\"";
370 }
371 ss_ << "]";
372 first_ = false;
373 return *this;
374 }
375
376 std::string build() const {
377 return ss_.str() + "}";
378 }
379
380private:
381 mutable std::stringstream ss_;
382 bool first_ = true;
383
384 std::string escape_string(const std::string& str) const {
385 std::string result;
386 for (char c : str) {
387 switch (c) {
388 case '"': result += "\\\""; break;
389 case '\\': result += "\\\\"; break;
390 case '\b': result += "\\b"; break;
391 case '\f': result += "\\f"; break;
392 case '\n': result += "\\n"; break;
393 case '\r': result += "\\r"; break;
394 case '\t': result += "\\t"; break;
395 default: result += c; break;
396 }
397 }
398 return result;
399 }
