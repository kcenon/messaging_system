149private:
150 void collection_loop() {
151 while (active_.load()) {
152 std::unique_lock<std::mutex> lock(cv_mutex_);
153
154 auto interval = std::chrono::milliseconds(collection_interval_ms_.load());
155 cv_.wait_for(lock, interval, [this] { return !active_.load(); });
156
157 if (active_.load()) {
158 collect_metrics();
159 }
160 }
161 }
162
163 void collect_metrics() {
164 metrics_snapshot snapshot;
165
166 // Get current metrics
167 {
168 std::lock_guard<std::mutex> lock(current_mutex_);
169 snapshot = current_snapshot_;
170 }
171
172 // Run custom collectors
173 {
174 std::lock_guard<std::mutex> lock(collectors_mutex_);
175 for (auto& collector : collectors_) {
176 try {
177 collector->collect(snapshot);
178 } catch (...) {
179 collector_errors_.fetch_add(1);
180 }
181 }
182 }
183
184 // Store in history
185 if (!history_.push(snapshot)) {
186 dropped_snapshots_.fetch_add(1);
187 }
188
189 total_collections_.fetch_add(1);
190 }
191
192private:
193 ring_buffer<metrics_snapshot> history_;
