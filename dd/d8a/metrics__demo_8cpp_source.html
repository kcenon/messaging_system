17void generate_logs(logger* log, int thread_id, int count) {
18 std::random_device rd;
19 std::mt19937 gen(rd());
20 std::uniform_int_distribution<> level_dist(0, 5);
21 std::uniform_int_distribution<> message_size_dist(10, 200);
22
23 for (int i = 0; i < count; ++i) {
24 auto level = static_cast<thread_module::log_level>(level_dist(gen));
25
26 // Generate message of random size
27 std::string message = "Thread " + std::to_string(thread_id) + " - Message " + std::to_string(i);
28 message.append(message_size_dist(gen), 'x');
29
30 log->log(level, message);
31
32 std::this_thread::sleep_for(std::chrono::milliseconds(10));
33 }
34}
