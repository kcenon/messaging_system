18void generate_logs(logger* log, int thread_id, int count) {
19 std::random_device rd;
20 std::mt19937 gen(rd());
21 std::uniform_int_distribution<> level_dist(0, 5);
22 std::uniform_int_distribution<> message_size_dist(10, 200);
23
24 for (int i = 0; i < count; ++i) {
25 auto level = static_cast<thread_module::log_level>(level_dist(gen));
26
27 // Generate message of random size
28 std::string message = "Thread " + std::to_string(thread_id) + " - Message " + std::to_string(i);
29 message.append(message_size_dist(gen), 'x');
30
31 log->log(level, message);
32
33 std::this_thread::sleep_for(std::chrono::milliseconds(10));
34 }
35}

98int main() {

101
102 // Add console writer

108 // Start logger

110
111 std::cout << "Starting logger metrics demo..." << std::endl;
112 std::cout << "Generating logs from multiple threads..." << std::endl;
113
114 // Launch multiple threads to generate logs
115 std::vector<std::thread> threads;
116 for (int i = 0; i < 4; ++i) {
117 threads.emplace_back(generate_logs, logger.get(), i, 100);
118 }
119
120 // Wait for threads to complete
121 for (auto& t : threads) {
122 t.join();
123 }
124
125 // Wait a bit for processing
