132 bool set(const std::string& path, const config_value& value) {
133 // Validate if validator exists
134 if (auto it = validators_.find(path); it != validators_.end()) {
135 auto result = it->second(path, value);
136 if (!result.is_valid) {
137 return false;
138 }
139 }
140
141 config_value old_value;
142 bool had_value = false;
143
144 {
145 std::lock_guard<std::mutex> lock(mutex_);
146 if (auto it = config_.find(path); it != config_.end()) {
147 old_value = it->second;
148 had_value = true;
149 }
150 config_[path] = value;
151 }
152
153 // Notify callbacks
154 notify_change(path, value);
155
156 // Publish event if value changed
157 if (event_bus_ && (!had_value || !values_equal(old_value, value))) {
158 event_bus_->publish(config_changed_event(path, old_value, value));
159 }
160
161 return true;
162 }

342 void parse_config_line(const std::string& line) {
343 if (line.empty() || line[0] == '#' || line[0] == ';') {
344 return; // Skip comments and empty lines
345 }
346
347 auto pos = line.find('=');
348 if (pos != std::string::npos) {
349 std::string key = line.substr(0, pos);
350 std::string value_str = line.substr(pos + 1);
351
352 // Trim whitespace
353 key.erase(0, key.find_first_not_of(" \t"));
354 key.erase(key.find_last_not_of(" \t") + 1);
355 value_str.erase(0, value_str.find_first_not_of(" \t"));
356 value_str.erase(value_str.find_last_not_of(" \t") + 1);
357
358 // Parse value type
359 config_value value;
360 if (value_str == "true" || value_str == "false") {
361 value = (value_str == "true");
362 } else if (std::all_of(value_str.begin(), value_str.end(), ::isdigit)) {
363 value = std::stoi(value_str);
364 } else {
365 try {
366 value = std::stod(value_str);
367 } catch (...) {
368 value = value_str; // Default to string
369 }
370 }
371
372 set(key, value);
373 }
374 }

379 std::string value_to_string(const config_value& value) const {
380 return std::visit([](const auto& v) -> std::string {
381 using T = std::decay_t<decltype(v)>;
382 if constexpr (std::is_same_v<T, bool>) {
383 return v ? "true" : "false";
384 } else if constexpr (std::is_same_v<T, std::string>) {
385 return v;
386 } else if constexpr (std::is_arithmetic_v<T>) {
387 return std::to_string(v);
388 } else {
389 return ""; // Complex types need custom serialization
390 }
391 }, value);
392 }
