265 void checkThresholds(const device_config& config, const device_telemetry& telemetry) {
266 alert::severity severity = alert::INFO;
267 std::string message;
268 bool create_alert = false;
269
270 if (telemetry.value < config.min_threshold) {
271 double deviation = config.min_threshold - telemetry.value;
272 double deviation_percent = (deviation / config.min_threshold) * 100;
273
274 if (deviation_percent > 20) {
275 severity = alert::CRITICAL;
276 message = "Value critically below minimum threshold";
277 } else if (deviation_percent > 10) {
278 severity = alert::WARNING;
279 message = "Value below minimum threshold";
280 }
281 create_alert = true;
282
283 } else if (telemetry.value > config.max_threshold) {
284 double deviation = telemetry.value - config.max_threshold;
285 double deviation_percent = (deviation / config.max_threshold) * 100;
286
287 if (deviation_percent > 20) {
288 severity = alert::CRITICAL;
289 message = "Value critically above maximum threshold";
290 } else if (deviation_percent > 10) {
291 severity = alert::WARNING;
292 message = "Value above maximum threshold";
293 }
294 create_alert = true;
295 }
296
297 // Special case for motion detectors - always alert on motion
298 if (config.type == device_type::MOTION_DETECTOR && telemetry.value > 0.5) {
299 severity = alert::INFO;
300 message = "Motion detected at " + config.location;
301 create_alert = true;
302 }
303
304 if (create_alert) {
305 createAlert(config.device_id, severity, message,
306 config.max_threshold, telemetry.value);
307 }

428 void processAnalytics(const device_telemetry& telemetry) {
429 // Calculate running averages, detect anomalies, etc.
430 // This would connect to a real analytics engine in production
431
432 // Simple anomaly detection using z-score
433 static std::map<std::string, std::vector<double>> history;
434 auto& device_history = history[telemetry.device_id];
435
436 device_history.push_back(telemetry.value);
437 if (device_history.size() > 100) {
438 device_history.erase(device_history.begin());
439
440 // Calculate mean and standard deviation
441 double sum = 0, sum_sq = 0;
442 for (double val : device_history) {
443 sum += val;
444 sum_sq += val * val;
445 }
446
447 double mean = sum / device_history.size();
448 double variance = (sum_sq / device_history.size()) - (mean * mean);
449 double stddev = std::sqrt(variance);
450
451 // Check for anomaly (z-score > 3)
452 if (stddev > 0) {
453 double z_score = std::abs((telemetry.value - mean) / stddev);
454 if (z_score > 3) {
455 createAlert(telemetry.device_id, alert::WARNING,
456 "Anomaly detected - unusual value pattern",
457 mean + 3 * stddev, telemetry.value);
458 }
459 }
460 }
