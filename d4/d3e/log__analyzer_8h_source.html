132 double get_error_rate(const std::chrono::minutes& window = std::chrono::minutes(60)) const {
133 auto now = std::chrono::system_clock::now();
134 auto start_time = now - window;
135
136 size_t total_in_window = 0;
137 size_t errors_in_window = 0;
138
139 for (const auto& entry : entries_) {
140 if (entry.timestamp >= start_time) {
141 total_in_window++;
142 if (entry.level == logger_system::log_level::error ||
143 entry.level == logger_system::log_level::fatal) {
144 errors_in_window++;
145 }
146 }
147 }
148
149 return total_in_window > 0 ?
150 static_cast<double>(errors_in_window) / total_in_window : 0.0;
151 }

156 std::string generate_summary_report() {
157 const auto& stats = get_stats();
158
159 std::string report = "=== Log Analysis Summary ===\n";

168 auto duration = std::chrono::duration_cast<std::chrono::minutes>(

170 report += "Time Range: " + std::to_string(duration.count()) + " minutes\n";
171 }
172

176private:
177 void update_stats() {
178 cached_stats_ = analysis_stats{};
179 cached_stats_.total_entries = entries_.size();
180
181 if (entries_.empty()) {
182 return;
183 }
184
185 // Initialize timestamps
186 cached_stats_.earliest_timestamp = entries_[0].timestamp;
187 cached_stats_.latest_timestamp = entries_[0].timestamp;
188
189 // Count levels and update timestamps
190 for (const auto& entry : entries_) {
191 cached_stats_.level_counts[entry.level]++;
192
193 if (entry.timestamp < cached_stats_.earliest_timestamp) {
194 cached_stats_.earliest_timestamp = entry.timestamp;
195 }
196 if (entry.timestamp > cached_stats_.latest_timestamp) {
197 cached_stats_.latest_timestamp = entry.timestamp;
198 }
199 }
200 }

202 std::string level_to_string(logger_system::log_level level) const {
203 switch (level) {

205 case logger_system::log_level::debug: return "DEBUG";

208 case logger_system::log_level::error: return "ERROR";

210 case logger_system::log_level::off: return "OFF";
211 default: return "UNKNOWN";
