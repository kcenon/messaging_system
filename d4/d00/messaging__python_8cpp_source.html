13// Helper function to convert Python dict to message_payload
14core::message_payload dict_to_payload(const std::string& topic, const py::dict& data) {
15 core::message_payload payload;
16 payload.topic = topic;
17
18 for (auto item : data) {
19 std::string key = py::str(item.first);
20 py::handle value_handle = item.second;
21
22 if (py::isinstance<py::str>(value_handle)) {
23 payload.data[key] = py::cast<std::string>(value_handle);
24 } else if (py::isinstance<py::int_>(value_handle)) {
25 payload.data[key] = py::cast<int64_t>(value_handle);
26 } else if (py::isinstance<py::float_>(value_handle)) {
27 payload.data[key] = py::cast<double>(value_handle);
28 } else if (py::isinstance<py::bool_>(value_handle)) {
29 payload.data[key] = py::cast<bool>(value_handle);
30 } else if (py::isinstance<py::bytes>(value_handle)) {
31 std::string bytes_str = py::cast<std::string>(value_handle);
32 std::vector<uint8_t> bytes_vec(bytes_str.begin(), bytes_str.end());
33 payload.data[key] = bytes_vec;
34 }
35 }
36
37 return payload;
38}

40// Helper function to convert message_payload to Python dict
41py::dict payload_to_dict(const core::message_payload& payload) {
42 py::dict result;
43 result["topic"] = payload.topic;
44
45 py::dict data_dict;
46 for (const auto& [key, value] : payload.data) {
47 if (std::holds_alternative<std::string>(value)) {
48 data_dict[key.c_str()] = std::get<std::string>(value);
49 } else if (std::holds_alternative<int64_t>(value)) {
50 data_dict[key.c_str()] = std::get<int64_t>(value);
51 } else if (std::holds_alternative<double>(value)) {
52 data_dict[key.c_str()] = std::get<double>(value);
53 } else if (std::holds_alternative<bool>(value)) {
54 data_dict[key.c_str()] = std::get<bool>(value);
55 } else if (std::holds_alternative<std::vector<uint8_t>>(value)) {
56 const auto& bytes_vec = std::get<std::vector<uint8_t>>(value);
57 std::string bytes_str(bytes_vec.begin(), bytes_vec.end());
58 data_dict[key.c_str()] = py::bytes(bytes_str);
59 }
60 }
61 result["data"] = data_dict;
62
63 if (!payload.binary_data.empty()) {
64 std::string binary_str(payload.binary_data.begin(), payload.binary_data.end());
65 result["binary_data"] = py::bytes(binary_str);
66 }
67
68 return result;
69}
