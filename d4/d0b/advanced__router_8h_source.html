100 // Composite filter (AND/OR operations)
101 class composite_filter : public message_filter {
102 public:
103 enum class operation { AND, OR };
104
105 private:
106 std::vector<std::unique_ptr<message_filter>> filters_;
107 operation op_;
108
109 public:
110 explicit composite_filter(operation op) : op_(op) {}
111
112 void add_filter(std::unique_ptr<message_filter> filter) {
113 filters_.push_back(std::move(filter));
114 }
115
116 bool matches(const core::message& msg) const override {
117 if (filters_.empty()) return true;
118
119 if (op_ == operation::AND) {
120 for (const auto& filter : filters_) {
121 if (!filter->matches(msg)) return false;
122 }
123 return true;
124 } else { // OR
125 for (const auto& filter : filters_) {
126 if (filter->matches(msg)) return true;
127 }
128 return false;
129 }
130 }
131
132 std::string describe() const override {
133 std::string result = "(";
134 std::string op_str = (op_ == operation::AND) ? " AND " : " OR ";
135
136 for (size_t i = 0; i < filters_.size(); ++i) {
137 if (i > 0) result += op_str;
138 result += filters_[i]->describe();
139 }
140 result += ")";
141 return result;
142 }
143
144 std::unique_ptr<message_filter> clone() const override {
145 auto cloned = std::make_unique<composite_filter>(op_);
146 for (const auto& filter : filters_) {
147 cloned->add_filter(filter->clone());
148 }
149 return cloned;
150 }
151 };

287 // Message routing
288 void route_message(const core::message& msg) {
289 total_messages_processed_++;
290 bool routed = false;
291
292 std::shared_lock<std::shared_mutex> lock(rules_mutex_);
293 for (auto& rule : rules_) {
294 if (rule->matches(msg)) {
295 rule->route_message(msg);
296 routed = true;
297 total_messages_routed_++;
298 }
299 }
300
301 // Use default handler if no rules matched
302 if (!routed && default_handler_) {
303 default_handler_(msg);
304 }
305 }

308 struct routing_statistics {
309 uint64_t total_messages_processed;
310 uint64_t total_messages_routed;
311 uint64_t unrouted_messages;
312 size_t active_rules;
313 size_t total_destinations;

315 std::vector<std::pair<std::string, uint64_t>> destination_stats;
316 };
317
318 routing_statistics get_statistics() const {
319 routing_statistics stats;
320 stats.total_messages_processed = total_messages_processed_;
321 stats.total_messages_routed = total_messages_routed_;
322 stats.unrouted_messages = stats.total_messages_processed - stats.total_messages_routed;
323 stats.total_destinations = destinations_.size();
324
325 std::shared_lock<std::shared_mutex> lock(rules_mutex_);
326 stats.active_rules = 0;
327
328 for (const auto& rule : rules_) {
329 if (rule->is_enabled()) stats.active_rules++;
330 stats.rule_stats.emplace_back(rule->get_id(), rule->get_match_count());
331 }
332
333 for (const auto& [name, dest] : destinations_) {
334 stats.destination_stats.emplace_back(name, dest->message_count.load());
335 }
336
337 return stats;
338 }

340 void reset_statistics() {
341 total_messages_processed_ = 0;
342 total_messages_routed_ = 0;
343
344 std::shared_lock<std::shared_mutex> lock(rules_mutex_);
345 for (const auto& [name, dest] : destinations_) {
346 dest->message_count = 0;
347 }
