47// Example 1: Basic single producer, single consumer
48void basic_spsc_example()
49{
50 write_information("[Example 1] Basic SPSC Pattern");
51
52 lockfree_job_queue queue;
53 std::atomic<int> counter{0};
54
55 // Producer thread
56 std::thread producer([&queue, &counter]() {
57 for (int i = 0; i < 10; ++i) {
58 auto job = std::make_unique<callback_job>([&counter, i]() -> result_void {
59 counter.fetch_add(1);
60 write_information("Processed job {}", i);
61 return result_void(); // Success
62 });
63
64 auto result = queue.enqueue(std::move(job));
65 if (!result) {
66 write_error(
67 "Failed to enqueue job {}: {}", i, result.get_error().message());
68 }
69
70 std::this_thread::sleep_for(10ms);
71 }
72 write_information("Producer finished");
73 });
74
75 // Consumer thread
76 std::thread consumer([&queue, &counter]() {
77 int consumed = 0;
78 while (consumed < 10) {
79 auto result = queue.dequeue();
80 if (result.has_value()) {
81 auto& job = result.value();
82 auto work_result = job->do_work();
83 if (!work_result) {
84 // Job executed successfully
85 consumed++;
86 } else {
87 write_error("Job failed: {}", work_result.get_error().message());
88 }
89 } else {
90 std::this_thread::sleep_for(5ms);
91 }
92 }
93 write_information("Consumer finished");
94 });
95
96 producer.join();
97 consumer.join();
98
99 write_information(
100 "Total jobs processed: {}", counter.load());
101}

103// Example 2: Multiple producers, multiple consumers
104void mpmc_example()
105{
106 write_information("\n[Example 2] MPMC Pattern");
107
108 lockfree_job_queue queue;
109 std::atomic<int> produced{0};
110 std::atomic<int> consumed{0};
111 const int num_producers = 3;
112 const int num_consumers = 2;
113 const int jobs_per_producer = 20;
114
115 std::vector<std::thread> producers;
116 std::vector<std::thread> consumers;
117
118 // Start producers
119 for (int p = 0; p < num_producers; ++p) {
120 producers.emplace_back([&queue, &produced, p, jobs_per_producer]() {
121 std::random_device rd;
122 std::mt19937 gen(rd());
123 std::uniform_int_distribution<> delay_dist(1, 10);
124
125 for (int i = 0; i < jobs_per_producer; ++i) {
126 auto job = std::make_unique<callback_job>(
127 [p, i]() -> result_void {
128 write_information("Job from producer {} #{}", p, i);
129 return result_void();
130 });
131
132 // Retry on failure (high contention scenario)
133 while (true) {
134 auto result = queue.enqueue(std::move(job));
135 if (result) {
136 produced.fetch_add(1);
137 break;
138 }
139 std::this_thread::yield();
140 }
141
142 std::this_thread::sleep_for(std::chrono::milliseconds(delay_dist(gen)));
143 }
144
145 write_information(
146 "Producer {} finished", p);
147 });
148 }
149
150 // Start consumers
151 for (int c = 0; c < num_consumers; ++c) {
152 consumers.emplace_back([&queue, &consumed, c, num_producers, jobs_per_producer]() {
153 const int total_jobs = num_producers * jobs_per_producer;
154
155 while (consumed.load() < total_jobs) {
156 auto result = queue.dequeue();
157 if (result.has_value()) {
158 auto& job = result.value();
159 auto work_result = job->do_work();
160 if (!work_result) {
161 // Job executed successfully
162 consumed.fetch_add(1);
163 } else {
164 write_error("Consumer {} job failed: {}", c, work_result.get_error().message());
165 }
166 } else {
167 std::this_thread::sleep_for(1ms);
168 }
169 }
170
171 write_information(
172 "Consumer {} finished", c);
173 });
174 }
175
176 // Wait for all threads
177 for (auto& t : producers) t.join();
178 for (auto& t : consumers) t.join();
179
180 write_information(
181 "Total produced: {}, consumed: {}",
182 produced.load(), consumed.load());
183}

185// Example 3: Batch operations
186void batch_operations_example()
187{
188 write_information("\n[Example 3] Batch Operations");
189
190 lockfree_job_queue queue;
191 std::atomic<int> processed{0};
192
193 // Batch enqueue
194 std::vector<std::unique_ptr<job>> batch;
195 for (int i = 0; i < 50; ++i) {
196 batch.push_back(std::make_unique<callback_job>(
197 [&processed, i]() -> result_void {
198 processed.fetch_add(1);
199 write_information("Batch job {}", i);
200 return result_void();
201 }));
202 }
203
204 write_information(
205 "Enqueueing {} jobs in batch", batch.size());
206
207 auto enqueue_result = queue.enqueue_batch(std::move(batch));
208 if (!enqueue_result) {
209 write_error(
210 "Batch enqueue failed: {}", enqueue_result.get_error().message());
211 return;
212 }
213
214 // Batch dequeue
215 auto dequeued = queue.dequeue_batch();
216 write_information(
217 "Dequeued {} jobs in batch", dequeued.size());
218
219 // Process all dequeued jobs
220 for (auto& job : dequeued) {
221 auto result = job->do_work();
222 if (!result) {
223 // Job executed successfully
224 } else {
225 write_error("Batch job failed: {}", result.get_error().message());
226 }
227 }
228
229 write_information(
230 "Total processed: {}", processed.load());
231}

233// Example 4: Performance measurement
234void performance_example()
235{
236 write_information("\n[Example 4] Performance Measurement");
237
238 lockfree_job_queue queue;
239 const int num_operations = 100000;
240
241 // Measure enqueue performance
242 auto start = std::chrono::high_resolution_clock::now();
243
244 for (int i = 0; i < num_operations; ++i) {
245 auto job = std::make_unique<callback_job>([]() -> result_void {
246 return result_void();
247 });
248
249 while (!queue.enqueue(std::move(job))) {
250 std::this_thread::yield();
251 }
252 }
253
254 auto enqueue_time = std::chrono::high_resolution_clock::now() - start;
255
256 // Measure dequeue performance
257 start = std::chrono::high_resolution_clock::now();
258
259 for (int i = 0; i < num_operations; ++i) {
260 while (true) {
261 auto result = queue.dequeue();
262 if (result.has_value()) {
263 break;
264 }
265 std::this_thread::yield();
266 }
267 }
268
269 auto dequeue_time = std::chrono::high_resolution_clock::now() - start;
270
271 // Get statistics
272 auto stats = queue.get_statistics();
273
274 write_information(
275 "Enqueue performance: {} ops in {} ms = {} ops/sec",
276 num_operations,
277 std::chrono::duration_cast<std::chrono::milliseconds>(enqueue_time).count(),
278 num_operations * 1000.0 / std::chrono::duration_cast<std::chrono::milliseconds>(enqueue_time).count());
279
280 write_information(
281 "Dequeue performance: {} ops in {} ms = {} ops/sec",
282 num_operations,
283 std::chrono::duration_cast<std::chrono::milliseconds>(dequeue_time).count(),
284 num_operations * 1000.0 / std::chrono::duration_cast<std::chrono::milliseconds>(dequeue_time).count());
285
286 write_information(
287 "Queue statistics:\n"
288 " Enqueued: {}\n"
289 " Dequeued: {}\n"
290 " Retries: {}\n"
291 " Average enqueue latency: {} ns\n"
292 " Average dequeue latency: {} ns",
293 stats.enqueue_count,
294 stats.dequeue_count,
295 stats.retry_count,
296 stats.get_average_enqueue_latency_ns(),
297 stats.get_average_dequeue_latency_ns());
298}
