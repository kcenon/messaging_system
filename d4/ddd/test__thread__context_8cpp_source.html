394TEST_F(ThreadContextTest, ThreadIsolation) {
395 const int thread_count = 10;
396 std::vector<std::thread> threads;
397 std::vector<std::future<std::string>> futures;
398
399 // Create threads with their own contexts
400 for (int i = 0; i < thread_count; ++i) {
401 std::promise<std::string> promise;
402 futures.push_back(promise.get_future());
403
404 threads.emplace_back([i, p = std::move(promise)]() mutable {
405 // Each thread creates its own context
406 auto req_id = "thread-" + std::to_string(i);
407 thread_context::create(req_id);
408 thread_context::current()->user_id = "user-" + std::to_string(i);
409
410 // Do some work
411 std::this_thread::sleep_for(std::chrono::milliseconds(10));
412
413 // Verify context is still correct
414 auto* ctx = thread_context::current();
415 if (ctx && ctx->request_id == req_id &&
416 ctx->user_id == "user-" + std::to_string(i)) {
417 p.set_value(req_id);
418 } else {
419 p.set_value("error");
420 }
421 });
422 }
423
424 // Collect results
425 std::set<std::string> results;
426 for (auto& future : futures) {
427 results.insert(future.get());
428 }
429
430 // Wait for threads
431 for (auto& t : threads) {
432 t.join();
433 }
434
435 // Each thread should have maintained its own context
436 EXPECT_EQ(results.size(), thread_count);
437 for (int i = 0; i < thread_count; ++i) {
438 auto expected = "thread-" + std::to_string(i);
439 EXPECT_NE(results.find(expected), results.end());
440 }
441}
