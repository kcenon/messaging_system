71std::vector<uint8_t> encrypted_writer::generate_key(size_t size) {
72 std::random_device rd;
73 std::mt19937 gen(rd());
74 std::uniform_int_distribution<> dis(0, 255);
75
76 std::vector<uint8_t> key(size);
77 for (size_t i = 0; i < size; ++i) {
78 key[i] = static_cast<uint8_t>(dis(gen));
79 }
80
81 return key;
82}

130std::string encrypted_writer::encrypt_data(const std::string& plaintext) {
131 // DEMO IMPLEMENTATION - NOT CRYPTOGRAPHICALLY SECURE!
132 // In production, use OpenSSL, libsodium, or similar
133
134 // Different behavior based on encryption type
135 if (type_ == encryption_type::none) {
136 return plaintext; // No encryption
137 }
138
139 // Convert string to bytes
140 std::vector<uint8_t> data(plaintext.begin(), plaintext.end());
141
142 // Add simple padding
143 size_t pad_len = 16 - (data.size() % 16);
144 data.resize(data.size() + pad_len, static_cast<uint8_t>(pad_len));
145
146 // Update IV with counter
147 counter_++;
148 for (size_t i = 0; i < 8 && i < iv_.size(); ++i) {
149 iv_[i] = (counter_ >> (i * 8)) & 0xFF;
150 }
151
152 // Prepend IV to data
153 std::vector<uint8_t> result = iv_;
154 result.insert(result.end(), data.begin(), data.end());
155
156 // XOR encrypt (DEMO ONLY) - for both aes_256_cbc and chacha20_poly1305 in demo
157 xor_encrypt(result, key_);
158
159 return std::string(result.begin(), result.end());
160}

162std::string encrypted_writer::decrypt_data(const std::string& ciphertext) {
163 // DEMO IMPLEMENTATION - matches encrypt_data
164 std::vector<uint8_t> data(ciphertext.begin(), ciphertext.end());
165
166 // XOR decrypt
167 xor_encrypt(data, key_);
168
169 // Extract IV (first 16 bytes)
170 if (data.size() < 16) {
171 return "";
172 }
173
174 // Remove IV and padding
175 data.erase(data.begin(), data.begin() + 16);
176 if (!data.empty()) {
177 uint8_t pad_len = data.back();
178 if (pad_len <= 16 && pad_len <= data.size()) {
179 data.resize(data.size() - pad_len);
180 }
181 }
182
183 return std::string(data.begin(), data.end());
184}
