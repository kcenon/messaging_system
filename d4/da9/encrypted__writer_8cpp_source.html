66std::vector<uint8_t> encrypted_writer::generate_key(size_t size) {
67 std::random_device rd;
68 std::mt19937 gen(rd());
69 std::uniform_int_distribution<> dis(0, 255);
70
71 std::vector<uint8_t> key(size);
72 for (size_t i = 0; i < size; ++i) {
73 key[i] = static_cast<uint8_t>(dis(gen));

125std::string encrypted_writer::encrypt_data(const std::string& plaintext) {
126 // DEMO IMPLEMENTATION - NOT CRYPTOGRAPHICALLY SECURE!
127 // In production, use OpenSSL, libsodium, or similar
128
129 // Convert string to bytes
130 std::vector<uint8_t> data(plaintext.begin(), plaintext.end());
131
132 // Add simple padding
133 size_t pad_len = 16 - (data.size() % 16);
134 data.resize(data.size() + pad_len, static_cast<uint8_t>(pad_len));
135
136 // Update IV with counter
137 counter_++;
138 for (size_t i = 0; i < 8 && i < iv_.size(); ++i) {
139 iv_[i] = (counter_ >> (i * 8)) & 0xFF;
140 }
141
142 // Prepend IV to data
143 std::vector<uint8_t> result = iv_;
144 result.insert(result.end(), data.begin(), data.end());
145
146 // XOR encrypt (DEMO ONLY)
147 xor_encrypt(result, key_);
148
149 return std::string(result.begin(), result.end());
150}

152std::string encrypted_writer::decrypt_data(const std::string& ciphertext) {
153 // DEMO IMPLEMENTATION - matches encrypt_data
154 std::vector<uint8_t> data(ciphertext.begin(), ciphertext.end());
155
156 // XOR decrypt
157 xor_encrypt(data, key_);
158
159 // Extract IV (first 16 bytes)
160 if (data.size() < 16) {
161 return "";
162 }
163
164 // Remove IV and padding
165 data.erase(data.begin(), data.begin() + 16);
166 if (!data.empty()) {
167 uint8_t pad_len = data.back();
168 if (pad_len <= 16 && pad_len <= data.size()) {
169 data.resize(data.size() - pad_len);
170 }
171 }
172
173 return std::string(data.begin(), data.end());
174}
