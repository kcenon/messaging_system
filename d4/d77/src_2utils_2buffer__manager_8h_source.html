83 double get_flush_success_rate() const {
84 auto total = total_flushes.load();
85 auto failed = failed_flushes.load();

135 void background_processing_loop() {
136 while (running_.load(std::memory_order_acquire)) {
137 std::unique_lock<std::mutex> lock(background_mutex_);
138
139 // Wait for check interval or shutdown
140 if (background_cv_.wait_for(lock, config_.background_check_interval,
141 [this] { return !running_.load(std::memory_order_acquire); })) {
142 break; // Shutdown requested
143 }
144
145 lock.unlock();
146
147 // Check all buffers for flushing
148 check_and_flush_buffers();
149
150 stats_.background_cycles.fetch_add(1, std::memory_order_relaxed);
151 }
152 }
153
157 void check_and_flush_buffers() {
158 std::vector<std::string> buffers_to_flush;
159
160 {
161 std::lock_guard<std::mutex> lock(buffers_mutex_);
162
163 for (const auto& [name, entry] : buffers_) {
164 if (entry->strategy->should_flush()) {
165 buffers_to_flush.push_back(name);
166 }
167 }
168 }
169
170 // Flush buffers (limit concurrent flushes)
171 size_t concurrent_flushes = 0;
172 for (const auto& buffer_name : buffers_to_flush) {
173 if (concurrent_flushes >= config_.max_concurrent_flushes) {
174 break;
175 }
176
177 auto result = flush_buffer(buffer_name);
178 if (result) {
179 ++concurrent_flushes;
180 stats_.active_flushes.fetch_add(1, std::memory_order_relaxed);
181 }
182 }
183 }

188 result_void flush_buffer(const std::string& metric_name) {
189 std::unique_ptr<metric_buffer_entry> entry;
190
191 {
192 std::lock_guard<std::mutex> lock(buffers_mutex_);
193 auto it = buffers_.find(metric_name);
194 if (it == buffers_.end()) {
195 return result_void(monitoring_error_code::collector_not_found,
196 "Buffer not found: " + metric_name);
197 }
198
199 // We can't move the entry here as we need to keep it in the map
200 // Just access it directly
201 }
202
203 // Flush the buffer
204 std::lock_guard<std::mutex> lock(buffers_mutex_);
205 auto it = buffers_.find(metric_name);
206 if (it == buffers_.end()) {
207 return result_void(monitoring_error_code::collector_not_found,
208 "Buffer not found: " + metric_name);
209 }
210
211 auto flush_result = it->second->strategy->flush();
212 if (!flush_result) {
213 stats_.failed_flushes.fetch_add(1, std::memory_order_relaxed);
214 return result_void(flush_result.get_error().code, flush_result.get_error().message);
215 }
216
217 auto flushed_metrics = flush_result.value();
218 if (flushed_metrics.empty()) {
219 return result_void::success();
220 }
221
222 // Store metrics if storage is available
223 if (storage_) {
224 metric_batch batch;
225 batch.reserve(flushed_metrics.size());
226
227 for (const auto& buffered_metric : flushed_metrics) {
228 batch.add_metric(std::move(const_cast<compact_metric_value&>(buffered_metric.metric)));
229 }
230
231 storage_->store_metrics_batch(batch);
232 }
233
234 // Call flush callback if set
235 if (flush_callback_) {
236 flush_callback_(metric_name, flushed_metrics);
237 }
238
239 // Update statistics
240 stats_.total_flushes.fetch_add(1, std::memory_order_relaxed);
241 stats_.total_metrics_processed.fetch_add(flushed_metrics.size(), std::memory_order_relaxed);
242 stats_.active_flushes.fetch_sub(1, std::memory_order_relaxed);
243
244 // Update last flush time
245 it->second->last_flush_time = std::chrono::system_clock::now();
246
247 return result_void::success();
248 }

281 result_void add_metric(const std::string& metric_name,
282 compact_metric_value&& metric,
283 uint8_t priority = 128) {
284
285 // Get or create buffer for this metric
286 std::lock_guard<std::mutex> lock(buffers_mutex_);
287
288 auto it = buffers_.find(metric_name);
289 if (it == buffers_.end()) {
290 // Create new buffer with default configuration
291 auto strategy = create_buffering_strategy(config_.default_buffering_config);
292 auto entry = std::make_unique<metric_buffer_entry>(std::move(strategy), metric_name);
293
294 auto insert_result = buffers_.emplace(metric_name, std::move(entry));
295 it = insert_result.first;
296
297 stats_.total_buffers.fetch_add(1, std::memory_order_relaxed);
298 }
299
300 // Create buffered metric
301 buffered_metric buffered_item(std::move(metric), priority);
302
303 // Add to buffer
304 return it->second->strategy->add_metric(std::move(buffered_item));
305 }

310 result_void configure_metric_buffer(const std::string& metric_name,
311 const buffering_config& config) {
312 auto validation = config.validate();
313 if (!validation) {
314 return validation;
315 }
316
317 std::lock_guard<std::mutex> lock(buffers_mutex_);
318
319 // Create new strategy
320 auto strategy = create_buffering_strategy(config);
321 auto entry = std::make_unique<metric_buffer_entry>(std::move(strategy), metric_name);
322
323 // Replace existing buffer or create new one
324 auto it = buffers_.find(metric_name);
325 if (it != buffers_.end()) {
326 // Flush existing buffer first
327 auto flush_result = it->second->strategy->flush();
328 if (flush_result && !flush_result.value().empty()) {
329 // Store flushed metrics
330 if (storage_) {
331 metric_batch batch;
332 for (const auto& buffered_metric : flush_result.value()) {
333 batch.add_metric(std::move(const_cast<compact_metric_value&>(buffered_metric.metric)));
334 }
335 storage_->store_metrics_batch(batch);
336 }
337 }
338 } else {
339 stats_.total_buffers.fetch_add(1, std::memory_order_relaxed);
340 }
341
342 buffers_[metric_name] = std::move(entry);
343
344 return result_void::success();
345 }

357 result_void force_flush_all() {
358 std::vector<std::string> buffer_names;
359
360 {
361 std::lock_guard<std::mutex> lock(buffers_mutex_);
362 for (const auto& [name, entry] : buffers_) {
363 buffer_names.push_back(name);
364 }
365 }
366
367 for (const auto& name : buffer_names) {
368 auto result = flush_buffer(name);
369 if (!result) {
370 // Continue flushing other buffers even if one fails
371 stats_.failed_flushes.fetch_add(1, std::memory_order_relaxed);
372 }
373 }
374
375 return result_void::success();
376 }

427 result_void start_background_processing() {
428 if (running_.load(std::memory_order_acquire)) {
429 return result_void(monitoring_error_code::invalid_configuration,
430 "Background processing already running");
431 }
432
433 if (!config_.enable_automatic_flushing) {
434 return result_void(monitoring_error_code::invalid_configuration,
435 "Automatic flushing is disabled");
436 }
437
438 running_.store(true, std::memory_order_release);
439 background_thread_ = std::thread(&buffer_manager::background_processing_loop, this);
440
441 return result_void::success();
442 }

447 void stop_background_processing() {
448 if (running_.load(std::memory_order_acquire)) {
449 running_.store(false, std::memory_order_release);
450 background_cv_.notify_all();
451
452 if (background_thread_.joinable()) {
453 background_thread_.join();
454 }
455
456 // Final flush of all buffers
457 force_flush_all();
458 }

503 result_void remove_buffer(const std::string& metric_name) {
504 std::lock_guard<std::mutex> lock(buffers_mutex_);
505
506 auto it = buffers_.find(metric_name);
507 if (it == buffers_.end()) {
508 return result_void(monitoring_error_code::collector_not_found,
509 "Buffer not found: " + metric_name);
510 }
511
512 // Flush before removing
513 auto flush_result = it->second->strategy->flush();
514 if (flush_result && !flush_result.value().empty() && storage_) {
515 metric_batch batch;
516 for (const auto& buffered_metric : flush_result.value()) {
517 batch.add_metric(std::move(const_cast<compact_metric_value&>(buffered_metric.metric)));
518 }
519 storage_->store_metrics_batch(batch);
520 }
521
522 buffers_.erase(it);
523 stats_.total_buffers.fetch_sub(1, std::memory_order_relaxed);
524
525 return result_void::success();
526 }
