69// Circuit breaker for service calls
70class circuit_breaker {
71public:
72 enum state {
73 CLOSED, // Normal operation
74 OPEN, // Failures exceeded threshold, blocking calls
75 HALF_OPEN // Testing if service recovered
76 };
77
78private:
79 state m_state{CLOSED};
80 int failure_count{0};
81 int success_count{0};
82 int failure_threshold{5};
83 int success_threshold{3};
84 std::chrono::steady_clock::time_point last_failure_time;
85 std::chrono::seconds timeout{30s};
86 mutable std::mutex mutex;
87
88public:
89 bool canAttempt() {
90 std::lock_guard<std::mutex> lock(mutex);
91
92 if (m_state == CLOSED) {
93 return true;
94 }
95
96 if (m_state == OPEN) {
97 auto now = std::chrono::steady_clock::now();
98 if (now - last_failure_time > timeout) {
99 m_state = HALF_OPEN;
100 return true;
101 }
102 return false;
103 }
104
105 // HALF_OPEN
106 return true;
107 }
108
109 void recordSuccess() {
110 std::lock_guard<std::mutex> lock(mutex);
111
112 if (m_state == HALF_OPEN) {
113 success_count++;
114 if (success_count >= success_threshold) {
115 m_state = CLOSED;
116 failure_count = 0;
117 success_count = 0;
118 }
119 } else if (m_state == CLOSED) {
120 failure_count = 0;
121 }
122 }
123
124 void recordFailure() {
125 std::lock_guard<std::mutex> lock(mutex);
126
127 failure_count++;
128 last_failure_time = std::chrono::steady_clock::now();
129
130 if (m_state == HALF_OPEN || failure_count >= failure_threshold) {
131 m_state = OPEN;
132 success_count = 0;
133 }
134 }
135
136 state getState() const {
137 std::lock_guard<std::mutex> lock(mutex);
138 return m_state;
139 }
140};

142// Load balancer strategies
143class load_balancer {
144public:
145 enum strategy {
146 ROUND_ROBIN,
147 LEAST_CONNECTIONS,
148 RANDOM,
149 WEIGHTED_RESPONSE_TIME,
150 IP_HASH
151 };
152
153private:
154 strategy m_strategy;
155 std::atomic<size_t> round_robin_counter{0};
156 std::mt19937 random_gen{std::random_device{}()};
157
158public:
159 load_balancer(strategy s = ROUND_ROBIN) : m_strategy(s) {}
160
161 service_instance* selectInstance(
162 std::vector<service_instance>& instances,
163 const std::string& client_ip = ""
164 ) {
165 // Filter healthy instances
166 std::vector<service_instance*> healthy;
167 for (auto& instance : instances) {
168 if (instance.m_state == service_instance::HEALTHY) {
169 healthy.push_back(&instance);
170 }
171 }
172
173 if (healthy.empty()) {
174 return nullptr;
175 }
176
177 switch (m_strategy) {
178 case ROUND_ROBIN:
179 return healthy[round_robin_counter++ % healthy.size()];
180
181 case LEAST_CONNECTIONS:
182 return *std::min_element(healthy.begin(), healthy.end(),
183 [](service_instance* a, service_instance* b) {
184 return a->active_connections < b->active_connections;
185 });
186
187 case RANDOM: {
188 std::uniform_int_distribution<> dist(0, healthy.size() - 1);
189 return healthy[dist(random_gen)];
190 }
191
192 case WEIGHTED_RESPONSE_TIME:
193 return *std::min_element(healthy.begin(), healthy.end(),
194 [](service_instance* a, service_instance* b) {
195 return a->response_time_ms < b->response_time_ms;
196 });
197
198 case IP_HASH: {
199 if (!client_ip.empty()) {
200 std::hash<std::string> hasher;
201 size_t hash = hasher(client_ip);
202 return healthy[hash % healthy.size()];
203 }
204 return healthy[0];
205 }
206
207 default:
208 return healthy[0];
209 }
210 }

472 void handleServiceRequest(const core::message& msg) {
473 total_requests++;
474
475 auto service_name = msg.metadata.headers.count("service_name") ? msg.metadata.headers.at("service_name") : "";
476 auto client_ip = msg.metadata.headers.count("client_ip") ? msg.metadata.headers.at("client_ip") : "";
477 auto request_id = msg.metadata.headers.count("request_id") ? msg.metadata.headers.at("request_id") : "";
478
479 // Check circuit breaker
480 if (!circuit_breakers[service_name].canAttempt()) {
481 circuit_breaker_trips++;
482 send_circuit_breaker_open(service_name, request_id);
483 failed_requests++;
484 return;
485 }
486
487 // Route request
488 std::lock_guard<std::mutex> lock(registry_mutex);
489
490 if (auto it = service_instances.find(service_name); it != service_instances.end()) {
491 auto* instance = load_balancers[service_name].selectInstance(it->second, client_ip);
492
493 if (instance) {
494 // Forward request to instance
495 bool success = forwardRequest(instance, msg);
496
497 if (success) {
498 circuit_breakers[service_name].recordSuccess();
499 successful_requests++;
500 instance->active_connections++;
501 } else {
502 circuit_breakers[service_name].recordFailure();
503 failed_requests++;
504 }
505 } else {
506 failed_requests++;
507 sendServiceUnavailable(service_name, request_id);
508 }
509 }

555 void scaleDown(const std::string& service_name, int count) {
556 std::lock_guard<std::mutex> lock(registry_mutex);
557 auto& instances = service_instances[service_name];
558 auto& definition = service_definitions[service_name];
559
560 // Don't scale below minimum
561 int to_remove = std::min(count,
562 static_cast<int>(instances.size()) - definition.min_instances);
563
564 // Mark instances for draining
565 for (int i = 0; i < to_remove; i++) {
566 instances[i].m_state = service_instance::DRAINING;
567 }
568
569 // Remove drained instances after grace period
570 std::thread([this, service_name, to_remove]() {
571 std::this_thread::sleep_for(30s); // Grace period
572
573 std::lock_guard<std::mutex> lock(registry_mutex);
574 auto& instances = service_instances[service_name];
575 instances.erase(
576 std::remove_if(instances.begin(), instances.end(),
577 [](const service_instance& i) {
578 return i.m_state == service_instance::DRAINING;
579 }),
580 instances.end()
581 );
582
583 m_logger->log(logger_module::log_level::info,
584 "Scaled down " + service_name + " by " + std::to_string(to_remove) + " instances");

588 void autoScale(const std::string& service_name) {
589 std::lock_guard<std::mutex> lock(registry_mutex);
590 auto& instances = service_instances[service_name];
591
592 // Calculate average metrics
593 double avg_cpu = 0.0;
594 double avg_connections = 0.0;
595 int healthy_count = 0;
596
597 for (const auto& instance : instances) {
598 if (instance.m_state == service_instance::HEALTHY) {
599 avg_cpu += instance.cpu_usage;
600 avg_connections += instance.active_connections;
601 healthy_count++;
602 }
603 }
604
605 if (healthy_count > 0) {
606 avg_cpu /= healthy_count;
607 avg_connections /= healthy_count;
608
609 // Scale up if CPU > 70% or connections > 100
610 if (avg_cpu > 70.0 || avg_connections > 100) {
611 scaleUp(service_name, 2);
612 }
613 // Scale down if CPU < 20% and connections < 10
614 else if (avg_cpu < 20.0 && avg_connections < 10) {
615 scaleDown(service_name, 1);
616 }
617 }
