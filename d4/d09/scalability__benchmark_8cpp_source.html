242static void BM_BurstWorkloadScalability(benchmark::State& state) {
243 const size_t num_workers = state.range(0);
244 const size_t burst_size = 1000;
245 const size_t num_bursts = 10;
246 const auto burst_interval = std::chrono::milliseconds(50);
247
248 // Create thread pool
249 auto pool = std::make_shared<thread_pool>("burst_workload_pool");
250 for (size_t i = 0; i < num_workers; ++i) {
251 auto worker = std::make_unique<thread_worker>();
252 pool->enqueue(std::move(worker));
253 }
254 pool->start();
255
256 // Benchmark
257 for (auto _ : state) {
258 std::atomic<size_t> completed{0};
259 const size_t total_jobs = burst_size * num_bursts;
260
261 state.PauseTiming();
262 // Submit jobs in bursts
263 for (size_t burst = 0; burst < num_bursts; ++burst) {
264 // Submit burst of jobs
265 for (size_t i = 0; i < burst_size; ++i) {
266 auto job = std::make_unique<callback_job>([&completed]() -> result_void {
267 volatile uint64_t sum = 0;
268 for (int j = 0; j < 100; ++j) {
269 sum += j;
270 }
271 completed.fetch_add(1, std::memory_order_relaxed);
272 return result_void{};
273 });
274 pool->enqueue(std::move(job));
275 }
276
277 // Wait between bursts
278 if (burst < num_bursts - 1) {
279 std::this_thread::sleep_for(burst_interval);
280 }
281 }
282 state.ResumeTiming();
283
284 // Wait for completion
285 while (completed.load() < total_jobs) {
286 std::this_thread::yield();
287 }
288 }
289
290 pool->stop();
291
292 state.SetItemsProcessed(state.iterations() * burst_size * num_bursts);
293 state.counters["workers"] = num_workers;
294 state.counters["burst_size"] = burst_size;
295 state.counters["num_bursts"] = num_bursts;
296}
