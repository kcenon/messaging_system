63 std::string format_timestamp() const {
64 auto now = std::chrono::system_clock::now();
65 auto time_t = std::chrono::system_clock::to_time_t(now);
66 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
67 now.time_since_epoch()) % 1000;
68
69 std::stringstream ss;
70 ss << std::put_time(std::localtime(&time_t), "%Y-%m-%d %H:%M:%S");
71 ss << '.' << std::setfill('0') << std::setw(3) << ms.count();
72 return ss.str();
73 }
74
75 bool should_log(log_level level) const {
76 return static_cast<int>(level) >= static_cast<int>(min_level_);
77 }
78
79 public:
80 explicit internal_logging_service(log_level min_level = log_level::info)
81 : min_level_(min_level) {}
82
83 void log(log_level level, const std::string& message, const std::string& component = "") override {
84 if (!should_log(level)) return;
85
86 std::lock_guard<std::mutex> lock(log_mutex_);
87 std::cout << "[" << format_timestamp() << "] "
88 << "[" << log_level_to_string(level) << "] ";
89
90 if (!component.empty()) {
91 std::cout << "[" << component << "] ";
92 }
93
94 std::cout << message << std::endl;
95 }
96
97 void log_message_event(const std::string& event_type, const core::message& msg) override {
98 std::stringstream ss;
99 ss << "Message " << event_type
100 << ": topic=" << msg.payload.topic
101 << ", sender=" << msg.metadata.sender
102 << ", priority=" << static_cast<int>(msg.metadata.priority)
103 << ", data_size=" << msg.payload.data.size();
104
105 log(log_level::info, ss.str(), "MessageBus");
106 }
107
108 void log_system_event(const std::string& event_type, const std::string& details) override {
109 std::stringstream ss;
110 ss << "System " << event_type << ": " << details;
111 log(log_level::info, ss.str(), "System");
112 }
113
114 void set_log_level(log_level min_level) override {
115 min_level_ = min_level;
116 }
117
118 log_level get_log_level() const override {
119 return min_level_;
120 }

216 // Logging service adapter
217 class logging_service_adapter : public service_adapter {
218 private:
219 std::shared_ptr<logging_service> logging_service_;
220
221 public:
222 explicit logging_service_adapter(std::shared_ptr<logging_service> service)
223 : service_adapter(service), logging_service_(std::move(service)) {}
224
225 void register_with_bus(core::message_bus* bus) {
226 // Subscribe to logging topics
227 if (bus && logging_service_) {
228 bus->subscribe("logging.*", [this](const core::message& msg) {
229 this->handle_logging_message(msg);
230 });
231 }
232 }
233
234 private:
235 void handle_logging_message(const core::message& msg) {
236 if (!logging_service_) return;
237
238 // Process logging messages based on their topic
239 if (msg.payload.topic == "logging.log_message") {
240 auto level_str = msg.payload.get<std::string>("level", "info");
241 auto message = msg.payload.get<std::string>("message", "");
242 auto component = msg.payload.get<std::string>("component", "");
243
244 log_level level = log_level::info;
245 if (level_str == "debug") level = log_level::debug;
246 else if (level_str == "warning") level = log_level::warning;
247 else if (level_str == "error") level = log_level::error;
248 else if (level_str == "critical") level = log_level::critical;
249
250 logging_service_->log(level, message, component);
251 }
252 else if (msg.payload.topic == "logging.message_event") {
253 auto event_type = msg.payload.get<std::string>("event_type", "unknown");
254 // Log the message event (the msg itself represents the original message)
255 logging_service_->log_message_event(event_type, msg);
256 }
257 else if (msg.payload.topic == "logging.system_event") {
258 auto event_type = msg.payload.get<std::string>("event_type", "unknown");
259 auto details = msg.payload.get<std::string>("details", "");
260 logging_service_->log_system_event(event_type, details);
261 }
262 else if (msg.payload.topic == "logging.set_level") {
263 auto level_str = msg.payload.get<std::string>("level", "info");
264 log_level level = log_level::info;
265 if (level_str == "debug") level = log_level::debug;
266 else if (level_str == "warning") level = log_level::warning;
267 else if (level_str == "error") level = log_level::error;
268 else if (level_str == "critical") level = log_level::critical;
269
270 logging_service_->set_log_level(level);
271 logging_service_->log_system_event("config_change",
272 "Log level changed to " + level_str);
273 }
274 }
275
276 public:
277 std::shared_ptr<logging_service> get_logging_service() const {
278 return logging_service_;
279 }
280 };
