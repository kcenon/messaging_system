71 struct histogram_stats {
72 double min = 0.0;
73 double max = 0.0;
74 double avg = 0.0;
75 double p95 = 0.0;
76 double p99 = 0.0;
77 size_t count = 0;
78 };
79
80 histogram_stats get_histogram_stats(const std::string& name) const {
81 std::lock_guard<std::mutex> lock(metrics_mutex_);
82 auto it = histograms_.find(name);
83 if (it == histograms_.end() || it->second.empty()) {
84 return {};
85 }
86
87 const auto& values = it->second;
88 auto sorted_values = values;
89 std::sort(sorted_values.begin(), sorted_values.end());
90
91 histogram_stats stats;
92 stats.count = sorted_values.size();
93 stats.min = sorted_values.front();
94 stats.max = sorted_values.back();
95
96 double sum = 0.0;
97 for (double v : sorted_values) sum += v;
98 stats.avg = sum / sorted_values.size();
99
100 size_t p95_idx = static_cast<size_t>(0.95 * sorted_values.size());
101 size_t p99_idx = static_cast<size_t>(0.99 * sorted_values.size());
102
103 stats.p95 = sorted_values[std::min(p95_idx, sorted_values.size() - 1)];
104 stats.p99 = sorted_values[std::min(p99_idx, sorted_values.size() - 1)];
105
106 return stats;
107 }

109 std::vector<std::string> get_all_metric_names() const {
110 std::vector<std::string> names;

112 for (const auto& [name, counter] : counters_) {
113 names.push_back("counter:" + name);
