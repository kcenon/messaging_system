173
174 // Update metrics
175 updateMetrics(processing_start_time, processing_end_time);
176
177 if (task_succeeded) {
178 tasks_processed++;
179 sendTaskComplete(received_task);

331 void updateMetrics(
332 const std::chrono::steady_clock::time_point& start,
333 const std::chrono::steady_clock::time_point& end
334 ) {
335 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
336
337 int total_task_count = m_metrics.total_tasks.fetch_add(1) + 1;
338 double current_average = m_metrics.avg_processing_time.load();
339 double new_average = (current_average * (total_task_count - 1) + duration.count()) / total_task_count;
340
341 m_metrics.avg_processing_time.store(new_average);

448 std::uniform_int_distribution<> type_distribution(0, 4);
449 std::uniform_int_distribution<> priority_distribution(1, 10);

452 task generated_task;
453 generated_task.id = "task-" + std::to_string(generated_task_counter++);
454 generated_task.type = static_cast<task_type>(type_distribution(random_generator));

456 generated_task.priority = priority_distribution(random_generator);
457 generated_task.created_at = std::chrono::steady_clock::now();

459 core::message task_message("task.new");
460 task_message.payload.set("task_data", generated_task.serialize());

462 if (generated_task.priority >= 8) {
463 task_message.set_priority(core::message_priority::critical);
464 } else if (generated_task.priority >= 5) {
465 task_message.set_priority(core::message_priority::high);
