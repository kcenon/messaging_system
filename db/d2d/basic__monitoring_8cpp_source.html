42// Custom collector that simulates system metrics
43class system_metrics_collector : public metrics_collector {
44public:
45 system_metrics_collector() : gen_(rd_()), cpu_dist_(20, 80), mem_dist_(1000, 4000) {}
46
47 void collect(metrics_snapshot& snapshot) override {
48 // Simulate CPU usage
49 snapshot.system.cpu_usage_percent = cpu_dist_(gen_);
50
51 // Simulate memory usage (in MB)
52 snapshot.system.memory_usage_bytes = mem_dist_(gen_) * 1024 * 1024;
53
54 // Simulate thread count
55 snapshot.system.active_threads = std::thread::hardware_concurrency() + (gen_() % 5);
56
57 // Update timestamp
58 snapshot.system.timestamp = std::chrono::steady_clock::now();
59 }
60
61 std::string name() const override {
62 return "SystemMetricsCollector";
63 }
64
65private:
66 std::random_device rd_;
67 std::mt19937 gen_;
68 std::uniform_int_distribution<> cpu_dist_;
69 std::uniform_int_distribution<> mem_dist_;
70};

72void print_metrics(const metrics_snapshot& snapshot) {
73 std::cout << "=== Metrics Snapshot ===" << std::endl;
74
75 // System metrics
76 std::cout << "System:" << std::endl;
77 std::cout << " CPU Usage: " << snapshot.system.cpu_usage_percent << "%" << std::endl;
78 std::cout << " Memory: " << (snapshot.system.memory_usage_bytes / (1024.0 * 1024.0))
79 << " MB" << std::endl;
80 std::cout << " Active Threads: " << snapshot.system.active_threads << std::endl;
81
82 // Thread pool metrics
83 if (snapshot.thread_pool.worker_threads > 0) {
84 std::cout << "Thread Pool:" << std::endl;
85 std::cout << " Jobs Completed: " << snapshot.thread_pool.jobs_completed << std::endl;
86 std::cout << " Jobs Pending: " << snapshot.thread_pool.jobs_pending << std::endl;
87 std::cout << " Workers: " << snapshot.thread_pool.worker_threads
88 << " (" << snapshot.thread_pool.idle_threads << " idle)" << std::endl;
89 }
90
91 // Worker metrics
92 if (snapshot.worker.jobs_processed > 0) {
93 std::cout << "Workers (aggregated):" << std::endl;
94 std::cout << " Jobs Processed: " << snapshot.worker.jobs_processed << std::endl;
95 std::cout << " Total Time: " << (snapshot.worker.total_processing_time_ns / 1000000.0)
96 << " ms" << std::endl;
97 }
98
99 std::cout << std::endl;
100}

102void basic_monitoring_example() {
103 std::cout << "\n=== Basic Monitoring Example ===\n" << std::endl;
104
105 // Create monitoring instance
106 auto monitor = std::make_shared<monitoring>(100, 500); // 100 snapshots, 500ms interval
107
108 // Add custom collector
109 monitor->add_collector(std::make_unique<system_metrics_collector>());
110
111 // Start monitoring
112 monitor->start();
113
114 // Simulate some activity
115 for (int i = 0; i < 5; ++i) {
116 std::this_thread::sleep_for(std::chrono::milliseconds(600));
117
118 // Get current snapshot
119 auto snapshot = monitor->get_current_snapshot();
120 print_metrics(snapshot);
121 }
122
123 // Stop monitoring
124 monitor->stop();
125}

185void performance_monitoring_example() {
186 std::cout << "\n=== Performance Monitoring Example ===\n" << std::endl;
187
188 auto monitor = std::make_shared<monitoring>(1000, 10); // 10ms interval
189 monitor->start();
190
191 // Simulate workload
192 std::cout << "Simulating workload..." << std::endl;
193
194 for (int worker = 0; worker < 4; ++worker) {
195 std::thread([monitor, worker]() {
196 for (int i = 0; i < 100; ++i) {
197 auto start = std::chrono::steady_clock::now();
198
199 // Simulate work
200 std::this_thread::sleep_for(std::chrono::microseconds(100));
201
202 auto end = std::chrono::steady_clock::now();
203 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start);
204
205 // Update worker metrics
206 worker_metrics metrics;
207 metrics.jobs_processed = 1;
208 metrics.total_processing_time_ns = duration.count();
209 monitor->update_worker_metrics(worker, metrics);
210 }
211 }).detach();
212 }
213
214 std::this_thread::sleep_for(std::chrono::milliseconds(500));
215
216 // Get final metrics
217 auto snapshot = monitor->get_current_snapshot();
218 print_metrics(snapshot);
219
220 monitor->stop();
221}
