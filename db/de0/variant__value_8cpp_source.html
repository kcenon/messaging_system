86 std::string variant_value::to_string() const
87 {
88 return visit([](auto&& value) -> std::string {
89 using T = std::decay_t<decltype(value)>;
90
91 if constexpr (std::is_same_v<T, std::monostate>) {
92 return "null";
93 }
94 else if constexpr (std::is_same_v<T, bool>) {
95 return value ? "true" : "false";
96 }
97 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
98 std::ostringstream oss;
99 oss << std::hex << std::setfill('0');
100 for (auto byte : value) {
101 oss << std::setw(2) << static_cast<int>(byte);
102 }
103 return oss.str();
104 }
105 else if constexpr (std::is_arithmetic_v<T>) {
106 return std::to_string(value);
107 }
108 else if constexpr (std::is_same_v<T, std::string>) {
109 return value;
110 }
111 else if constexpr (std::is_same_v<T, std::shared_ptr<thread_safe_container>>) {
112 return value ? value->to_json() : "null";
113 }
114 return "";
115 });
116 }

118 std::string variant_value::to_json() const
119 {
120 auto var_name = name(); // Capture name before visit
121 auto var_type_idx = type_index(); // Capture type_index before visit
122
123 return visit([var_name, var_type_idx](auto&& value) -> std::string {
124 using T = std::decay_t<decltype(value)>;
125
126 std::string result;
127 formatter::format_to(std::back_inserter(result),
128 "{{\"name\":\"{}\",\"type\":{},\"value\":",
129 var_name, var_type_idx);
130
131 if constexpr (std::is_same_v<T, std::monostate>) {
132 result += "null";
133 }
134 else if constexpr (std::is_same_v<T, bool>) {
135 result += value ? "true" : "false";
136 }
137 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
138 result += "\"";
139 std::ostringstream oss;
140 oss << std::hex << std::setfill('0');
141 for (auto byte : value) {
142 oss << std::setw(2) << static_cast<int>(byte);
143 }
144 result += oss.str();
145 result += "\"";
146 }
147 else if constexpr (std::is_arithmetic_v<T>) {
148 result += std::to_string(value);
149 }
150 else if constexpr (std::is_same_v<T, std::string>) {
151 // Escape string for JSON
152 result += "\"";
153 for (char c : value) {
154 switch (c) {
155 case '"': result += "\\\""; break;
156 case '\\': result += "\\\\"; break;
157 case '\b': result += "\\b"; break;
158 case '\f': result += "\\f"; break;
159 case '\n': result += "\\n"; break;
160 case '\r': result += "\\r"; break;
161 case '\t': result += "\\t"; break;
162 default:
163 if (c >= 0x20 && c <= 0x7E) {
164 result += c;
165 } else {
166 formatter::format_to(std::back_inserter(result),
167 "\\u{:04x}",
168 static_cast<unsigned>(c));
169 }
170 }
171 }
172 result += "\"";
173 }
174 else if constexpr (std::is_same_v<T, std::shared_ptr<thread_safe_container>>) {
175 result += value ? value->to_json() : "null";
176 }
177
178 result += "}";
179 return result;
180 });
181 }

183 std::vector<uint8_t> variant_value::serialize() const
184 {
185 std::vector<uint8_t> result;
186
187 // Header: name length (4 bytes) + name + type index (1 byte)
188 uint32_t name_len = static_cast<uint32_t>(name_.size());
189 result.insert(result.end(),
190 reinterpret_cast<const uint8_t*>(&name_len),
191 reinterpret_cast<const uint8_t*>(&name_len) + sizeof(name_len));
192 result.insert(result.end(), name_.begin(), name_.end());
193
194 uint8_t type_idx = static_cast<uint8_t>(type_index());
195 result.push_back(type_idx);
196
197 // Serialize the variant data
198 visit([&result](auto&& value) {
199 using T = std::decay_t<decltype(value)>;
200
201 if constexpr (std::is_same_v<T, std::monostate>) {
202 // Nothing to serialize
203 }
204 else if constexpr (std::is_same_v<T, bool>) {
205 result.push_back(value ? 1 : 0);
206 }
207 else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
208 uint32_t size = static_cast<uint32_t>(value.size());
209 result.insert(result.end(),
210 reinterpret_cast<const uint8_t*>(&size),
211 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
212 result.insert(result.end(), value.begin(), value.end());
213 }
214 else if constexpr (std::is_arithmetic_v<T>) {
215 result.insert(result.end(),
216 reinterpret_cast<const uint8_t*>(&value),
217 reinterpret_cast<const uint8_t*>(&value) + sizeof(T));
218 }
219 else if constexpr (std::is_same_v<T, std::string>) {
220 uint32_t size = static_cast<uint32_t>(value.size());
221 result.insert(result.end(),
222 reinterpret_cast<const uint8_t*>(&size),
223 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
224 result.insert(result.end(), value.begin(), value.end());
225 }
226 else if constexpr (std::is_same_v<T, std::shared_ptr<thread_safe_container>>) {
227 if (value) {
228 auto container_data = value->serialize();
229 uint32_t size = static_cast<uint32_t>(container_data.size());
230 result.insert(result.end(),
231 reinterpret_cast<const uint8_t*>(&size),
232 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
233 result.insert(result.end(), container_data.begin(), container_data.end());
234 } else {
235 uint32_t size = 0;
236 result.insert(result.end(),
237 reinterpret_cast<const uint8_t*>(&size),
238 reinterpret_cast<const uint8_t*>(&size) + sizeof(size));
239 }
240 }
241 });
242
243 return result;
244 }

246 std::optional<variant_value> variant_value::deserialize(const std::vector<uint8_t>& data)
247 {
248 if (data.size() < sizeof(uint32_t) + 1) { // Minimum: name length + type
249 return std::nullopt;
250 }
251
252 size_t offset = 0;
253
254 // Read name length
255 uint32_t name_len;
256 std::memcpy(&name_len, data.data() + offset, sizeof(name_len));
257 offset += sizeof(name_len);
258
259 if (offset + name_len + 1 > data.size()) {
260 return std::nullopt;
261 }
262
263 // Read name
264 std::string name(data.begin() + offset, data.begin() + offset + name_len);
265 offset += name_len;
266
267 // Read type index
268 uint8_t type_idx = data[offset++];
269
270 // Create variant_value based on type
271 variant_value result(name);
272
273 // Deserialize based on type index
274 switch (type_idx) {
275 case 0: // std::monostate
276 result.set(std::monostate{});
277 break;
278
279 case 1: // bool
280 if (offset >= data.size()) return std::nullopt;
281 result.set(data[offset] != 0);
282 break;
283
284 case 2: // std::vector<uint8_t>
285 {
286 if (offset + sizeof(uint32_t) > data.size()) return std::nullopt;
287 uint32_t size;
288 std::memcpy(&size, data.data() + offset, sizeof(size));
289 offset += sizeof(size);
290 if (offset + size > data.size()) return std::nullopt;
291 std::vector<uint8_t> bytes(data.begin() + offset,
292 data.begin() + offset + size);
293 result.set(std::move(bytes));
294 break;
295 }
296
297 // Add cases for other numeric types...
298 case 11: // float
299 {
300 if (offset + sizeof(float) > data.size()) return std::nullopt;
301 float value;
302 std::memcpy(&value, data.data() + offset, sizeof(value));
303 result.set(value);
304 break;
305 }
306
307 case 12: // double
308 {
309 if (offset + sizeof(double) > data.size()) return std::nullopt;
310 double value;
311 std::memcpy(&value, data.data() + offset, sizeof(value));
312 result.set(value);
313 break;
314 }
315
316 case 13: // std::string
317 {
318 if (offset + sizeof(uint32_t) > data.size()) return std::nullopt;
319 uint32_t size;
320 std::memcpy(&size, data.data() + offset, sizeof(size));
321 offset += sizeof(size);
322 if (offset + size > data.size()) return std::nullopt;
323 std::string str(data.begin() + offset, data.begin() + offset + size);
324 result.set(std::move(str));
325 break;
326 }
327
328 // Container type would need special handling
329 case 14: // std::shared_ptr<thread_safe_container>
330 // TODO: Implement container deserialization
331 break;
332
333 default:
334 return std::nullopt;
335 }
336
337 return result;
338 }
