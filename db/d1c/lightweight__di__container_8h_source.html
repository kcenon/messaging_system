58 result<std::shared_ptr<T>> resolve(const std::string& name) override {
59 std::lock_guard<std::mutex> lock(mutex_);
60
61 // Check singletons first
62 if (auto it = singletons_.find(name); it != singletons_.end()) {
63 return it->second;
64 }
65
66 // Try factory
67 if (auto it = factories_.find(name); it != factories_.end()) {
68 try {
69 if (auto instance = it->second()) {
70 return instance;
71 }
72 return error_code::creation_failed;
73 } catch (const std::exception& e) {
74 return error_code::creation_failed;
75 }
76 }
77
78 return error_code::component_not_found;
79 }

88 result_void register_factory(

92 if (name.empty()) {

114 result_void register_singleton(

138 bool is_registered(const std::string& name) const override {
139 std::lock_guard<std::mutex> lock(mutex_);
