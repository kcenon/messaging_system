80static void BM_SimulatedBatchWriting(benchmark::State& state) {
81 bool simulate_batch = state.range(0);
82 size_t batch_size = state.range(1);
83
84 auto writer = std::make_unique<benchmark_writer>();
85 auto* writer_ptr = writer.get();
86
87 const std::string message = "Simulated batch processing test message";
88 const auto timestamp = std::chrono::system_clock::now();
89
90 std::vector<std::pair<thread_module::log_level, std::string>> batch;
91
92 for (auto _ : state) {
93 if (simulate_batch) {
94 // Simulate batching behavior
95 batch.push_back({thread_module::log_level::info, message});
96
97 if (batch.size() >= batch_size) {
98 // Simulate batch flush
99 for (const auto& [level, msg] : batch) {
100 writer->write(level, msg, __FILE__, __LINE__, __FUNCTION__, timestamp);
101 }
102 batch.clear();
103 }
104 } else {
105 // Direct write
106 writer->write(thread_module::log_level::info,
107 message, __FILE__, __LINE__, __FUNCTION__, timestamp);
108 }
109 }
110
111 // Flush remaining batch
112 if (!batch.empty()) {
113 for (const auto& [level, msg] : batch) {
114 writer->write(level, msg, __FILE__, __LINE__, __FUNCTION__, timestamp);
115 }
116 }
117
118 state.SetItemsProcessed(state.iterations());
119 state.SetLabel(simulate_batch ?
120 "batch_size: " + std::to_string(batch_size) :
121 "direct");
122}

128 ->Args({1, 500}); // Batch size 500

172static void BM_MultithreadedBatchSimulation(benchmark::State& state) {
173 int thread_count = state.range(0);
174 size_t batch_size = state.range(1);
175
176 logger_builder builder;
177 auto logger = builder
178 .with_async(true)
179 .with_batch_size(batch_size)
180 .add_writer("benchmark", std::make_unique<benchmark_writer>())
181 .build();
182
183 if (!logger) {
184 state.SkipWithError("Failed to build logger");
185 return;
186 }
187
188 auto& log = *logger.value();
189 log->start();
190
191 const std::string message = "Multi-threaded batch test";
192
193 for (auto _ : state) {
194 std::vector<std::thread> threads;
195 threads.reserve(thread_count);
196
197 auto work = [&log, &message]() {
198 for (int i = 0; i < 100; ++i) {
199 log->log(thread_module::log_level::info, message);
200 }
201 };
202
203 for (int i = 0; i < thread_count; ++i) {
204 threads.emplace_back(work);
205 }
206
207 for (auto& t : threads) {
208 t.join();
209 }
210 }
211
212 log->stop();
213
214 state.SetItemsProcessed(state.iterations() * thread_count * 100);
215 state.SetLabel("threads: " + std::to_string(thread_count) +
216 ", batch: " + std::to_string(batch_size));
217}

219 ->Args({1, 10})
220 ->Args({2, 10})
221 ->Args({4, 10})
222 ->Args({1, 100})
223 ->Args({2, 100})
224 ->Args({4, 100});
