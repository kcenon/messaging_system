85 std::string encrypt(const std::string& plaintext, const std::string& key) override {
86 std::string result = plaintext;
87 for (size_t i = 0; i < result.size(); ++i) {
88 result[i] ^= key[i % key.size()];
89 }
90 return result;
91 }

144 bool create_user(const std::string& user_id, const std::string& password,
145 const std::unordered_set<std::string>& permissions = {},
146 security_level level = security_level::basic) {
147 std::unique_lock<std::shared_mutex> lock(auth_mutex_);
148
149 if (users_.find(user_id) != users_.end()) {
150 return false; // User already exists
151 }
152
153 auto credentials = std::make_unique<user_credentials>();
154 credentials->user_id = user_id;
155 credentials->password_hash = hash_password(password);
156 credentials->permissions = permissions;
157 credentials->created_at = std::chrono::system_clock::now();
158 credentials->access_level = level;
159
160 users_[user_id] = std::move(credentials);
161 return true;
162 }
163
164 auth_result authenticate(const std::string& user_id, const std::string& password,
165 std::string& token_out) {
166 std::unique_lock<std::shared_mutex> lock(auth_mutex_);
167 total_authentications_++;
168
169 auto user_it = users_.find(user_id);
170 if (user_it == users_.end()) {
171 failed_authentications_++;
172 return auth_result::invalid_credentials;
173 }
174
175 auto& user = *user_it->second;
176
177 // Check if account is locked
178 if (user.is_locked) {
179 auto now = std::chrono::system_clock::now();
180 if (now - user.last_login < LOCKOUT_DURATION) {
181 return auth_result::account_locked;
182 } else {
183 // Unlock account after lockout period
184 user.is_locked = false;
185 user.failed_attempts = 0;
186 }
187 }
188
189 // Verify password
190 if (user.password_hash != hash_password(password)) {
191 user.failed_attempts++;
192 if (user.failed_attempts >= MAX_FAILED_ATTEMPTS) {
193 user.is_locked = true;
194 }
195 failed_authentications_++;
196 return auth_result::invalid_credentials;
197 }
198
199 // Create token
200 auto token = std::make_unique<auth_token>();
201 token->token_id = generate_token_id();
202 token->user_id = user_id;
203 token->issued_at = std::chrono::system_clock::now();
204 token->expires_at = token->issued_at + TOKEN_LIFETIME;
205 token->scopes = user.permissions;
206 token->level = user.access_level;
207
208 token_out = token->token_id;
209 active_tokens_[token->token_id] = std::move(token);
210
211 // Update user login info
212 user.last_login = std::chrono::system_clock::now();
213 user.failed_attempts = 0;
214
215 return auth_result::success;
216 }
217
218 auth_result verify_token(const std::string& token_id, const std::string& required_scope = "") {
219 std::shared_lock<std::shared_mutex> lock(auth_mutex_);
220
221 auto token_it = active_tokens_.find(token_id);
222 if (token_it == active_tokens_.end()) {
223 return auth_result::invalid_credentials;
224 }
225
226 auto& token = *token_it->second;
227 if (!token.is_valid()) {
228 return auth_result::token_expired;
229 }
230
231 if (!required_scope.empty() && !token.has_scope(required_scope)) {
232 return auth_result::permission_denied;
233 }
234
235 return auth_result::success;
236 }
237
238 bool revoke_token(const std::string& token_id) {
239 std::unique_lock<std::shared_mutex> lock(auth_mutex_);
240 return active_tokens_.erase(token_id) > 0;
241 }
242
243 void cleanup_expired_tokens() {
244 std::unique_lock<std::shared_mutex> lock(auth_mutex_);
245 auto now = std::chrono::system_clock::now();
246
247 for (auto it = active_tokens_.begin(); it != active_tokens_.end();) {
248 if (it->second->expires_at <= now) {
249 it = active_tokens_.erase(it);
250 } else {
251 ++it;
252 }
253 }

301 bool check_access(const std::string& topic, const auth_token& token,
302 const std::string& operation) const {
303 std::shared_lock<std::shared_mutex> lock(acl_mutex_);
304
305 // Check if token has required scope for operation
306 std::string required_scope = operation + ":" + topic;
307 if (!token.has_scope(required_scope) && !token.has_scope("admin")) {
308 return false;
309 }
310
311 // Check security level requirements
312 auto level_it = topic_security_levels_.find(topic);
313 if (level_it != topic_security_levels_.end()) {
314 if (token.level < level_it->second) {
315 return false;
316 }
317 }
318
319 return true;
320 }

323 std::shared_lock<std::shared_mutex> lock(acl_mutex_);
324 std::vector<std::string> accessible;
325
326 for (const auto& [topic, permissions] : topic_permissions_) {
327 if (check_access(topic, token, "read")) {
328 accessible.push_back(topic);
329 }
330 }

379 // Message security
380 secure_message encrypt_message(const core::message& msg, const std::string& token,
381 security_level level = security_level::standard) {
382 secure_message secure_msg;
383 secure_msg.original_message = msg;
384 secure_msg.sender_token = token;
385 secure_msg.encrypted_at = std::chrono::system_clock::now();
386 secure_msg.required_level = level;
387
388 if (level >= security_level::standard) {
389 // Generate or get encryption key
390 std::string key_id = "key_" + msg.payload.topic;
391 std::lock_guard<std::mutex> lock(keys_mutex_);
392
393 if (encryption_keys_.find(key_id) == encryption_keys_.end()) {
394 encryption_keys_[key_id] = encryptor_->generate_key();
395 }
396
397 // Serialize and encrypt message payload
398 std::string serialized = msg.payload.topic + "|" +
399 (msg.payload.data.empty() ? "" : "data_present");
400
401 secure_msg.encrypted_payload = encryptor_->encrypt(serialized, encryption_keys_[key_id]);
402 secure_msg.encryption_key_id = key_id;
403 }
404
405 return secure_msg;
406 }
407
408 bool decrypt_message(const secure_message& secure_msg, core::message& decrypted_msg) {
409 if (!secure_msg.is_encrypted()) {
410 decrypted_msg = secure_msg.original_message;
411 return true;
412 }
413
414 std::lock_guard<std::mutex> lock(keys_mutex_);
415 auto key_it = encryption_keys_.find(secure_msg.encryption_key_id);
416 if (key_it == encryption_keys_.end()) {
417 return false;
418 }
419
420 try {
421 std::string decrypted = encryptor_->decrypt(secure_msg.encrypted_payload, key_it->second);
422 // Simple deserialization (in production, use proper serialization)
423 decrypted_msg = secure_msg.original_message;
424 return true;
425 } catch (const std::exception&) {
426 return false;
427 }
428 }
