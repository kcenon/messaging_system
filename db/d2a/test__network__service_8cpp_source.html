216 constexpr int num_threads = 4;

223 std::vector<std::thread> sender_threads;

225 // Start concurrent message senders
226 for (int t = 0; t < num_threads; ++t) {
227 sender_threads.emplace_back([this, t, messages_per_thread, &successful_sends]() {
228 for (int i = 0; i < messages_per_thread; ++i) {
229 message msg;
230 msg.payload.topic = "concurrent.test";
231 msg.payload.data["thread_id"] = int64_t(t);
232 msg.payload.data["message_id"] = int64_t(i);
233 msg.payload.data["content"] = std::string("concurrent message");
234
235 std::string destination = "dest_" + std::to_string(t);
236 if (service_->send_message(destination, msg)) {
237 successful_sends++;
238 }
239 }
240 });
241 }
242
243 // Wait for all senders to complete
244 for (auto& thread : sender_threads) {
245 thread.join();
246 }
247
248 EXPECT_EQ(successful_sends.load(), num_threads * messages_per_thread);
249
250 // Check final statistics
