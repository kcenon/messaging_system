129void demonstrate_composition() {
130 std::cout << "\n=== Composition-Based Thread System Demo ===\n" << std::endl;
131
132 // 1. Setup service container with implementations
133 auto& container = service_container::global();
134
135 // Register logger service
136 container.register_singleton<logger_interface>(
137 std::make_shared<console_logger>());
138
139 // Register monitoring service
140 container.register_singleton<::monitoring_interface::monitoring_interface>(
141 std::make_shared<console_monitoring>());
142
143 // 2. Create thread pool with context from global container
144 thread_context context; // Will resolve services from container
145 auto pool = std::make_shared<thread_pool>("CompositionPool", context);
146
147 // 3. Add workers - they inherit context from pool
148 std::vector<std::unique_ptr<thread_worker>> workers;
149 for (int i = 0; i < 4; ++i) {
150 workers.push_back(std::make_unique<thread_worker>());
151 }
152 pool->enqueue_batch(std::move(workers));
153
154 // 4. Start pool - will log through context
155 pool->start();
156
157 // 5. Submit jobs that will be logged
158 for (int i = 0; i < 10; ++i) {
159 pool->enqueue(std::make_unique<callback_job>(
160 [i, &context]() -> result_void {
161 context.log(log_level::info,
162 "Processing job " + std::to_string(i));
163
164 // Simulate work
165 std::this_thread::sleep_for(std::chrono::milliseconds(100));
166
167 return result_void();
168 }
169 ));
170 }
171
172 // 6. Wait for completion
173 std::this_thread::sleep_for(std::chrono::seconds(2));
174
175 // 7. Stop pool
176 pool->stop();
177
178 std::cout << "\n=== Basic Thread Pool Demo Complete ===\n" << std::endl;
179}

184void demonstrate_typed_pool_composition() {
185 std::cout << "\n=== Typed Thread Pool with Composition Demo ===\n" << std::endl;
186
187 // Use builder pattern for context
188 auto context = thread_context_builder()
189 .from_global_container()
190 .build();
191
192 // Create typed thread pool with priority support
193 auto pool = std::make_shared<typed_thread_pool_t<job_types>>("TypedPool", context);
194
195 // Add specialized workers
196 for (auto priority : {job_types::RealTime, job_types::Batch, job_types::Background}) {
197 auto worker = std::make_unique<typed_thread_worker_t<job_types>>();
198 // For typed workers, set the type they handle
199 // Note: The typed worker template includes the type in the template parameter
200 pool->enqueue(std::move(worker));
201 }
202
203 pool->start();
204
205 // Submit jobs with different priorities
206 for (int i = 0; i < 5; ++i) {
207 // Real-time job
208 pool->enqueue(std::make_unique<callback_typed_job_t<job_types>>(
209 [i, &context]() -> result_void {
210 context.log(log_level::info,
211 "RealTime job " + std::to_string(i) + " executing");
212 return result_void();
213 },
214 job_types::RealTime
215 ));
216
217 // Background job
218 pool->enqueue(std::make_unique<callback_typed_job_t<job_types>>(
219 [i, &context]() -> result_void {
220 context.log(log_level::debug,
221 "Background job " + std::to_string(i) + " executing");
222 std::this_thread::sleep_for(std::chrono::milliseconds(50));
223 return result_void();
224 },
225 job_types::Background
226 ));
227 }
228
229 std::this_thread::sleep_for(std::chrono::seconds(1));
230 pool->stop();
231
232 std::cout << "\n=== Typed Thread Pool Demo Complete ===\n" << std::endl;
233}

238void demonstrate_minimal_usage() {
239 std::cout << "\n=== Minimal Thread Pool (No Services) Demo ===\n" << std::endl;
240
241 // Clear any existing services
242 service_container::global().clear();
243
244 // Create pool without context - no logging or monitoring
245 auto pool = std::make_shared<thread_pool>("MinimalPool");
246
247 // Add workers
248 std::vector<std::unique_ptr<thread_worker>> workers;
249 for (int i = 0; i < 2; ++i) {
250 workers.push_back(std::make_unique<thread_worker>());
251 }
252 pool->enqueue_batch(std::move(workers));
253
254 pool->start();
255
256 // Submit jobs - no logging will occur
257 std::atomic<int> counter{0};
258 for (int i = 0; i < 5; ++i) {
259 pool->enqueue(std::make_unique<callback_job>(
260 [&counter]() -> result_void {
261 counter.fetch_add(1);
262 return result_void();
263 }
264 ));
265 }
266
267 std::this_thread::sleep_for(std::chrono::milliseconds(500));
268 pool->stop();
269
270 std::cout << "Completed " << counter.load() << " jobs without any logging/monitoring" << std::endl;
271 std::cout << "\n=== Minimal Demo Complete ===\n" << std::endl;
272}
