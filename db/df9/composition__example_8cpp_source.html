129void demonstrate_composition() {
130 std::cout << "\n=== Composition-Based Thread System Demo ===\n" << std::endl;
131
132 // 1. Setup service container with implementations
133 auto& container = service_container::global();
134
135 // Register logger service
136 container.register_singleton<logger_interface>(
137 std::make_shared<console_logger>());
138
139 // Register monitoring service
140 container.register_singleton<::monitoring_interface::monitoring_interface>(
141 std::make_shared<console_monitoring>());
142
143 // 2. Create thread pool with context from global container
144 thread_context context; // Will resolve services from container
145 auto pool = std::make_shared<thread_pool>("CompositionPool", context);
146
147 // 3. Add workers - they inherit context from pool
148 std::vector<std::unique_ptr<thread_worker>> workers;
149 for (int i = 0; i < 4; ++i) {
150 workers.push_back(std::make_unique<thread_worker>());
151 }
152 {
153 auto r = pool->enqueue_batch(std::move(workers));
154 if (r.has_error()) {
155 std::cerr << "enqueue_batch failed: " << r.get_error().to_string() << std::endl;
156 return;
157 }
158 }
159
160 // 4. Start pool - will log through context
161 {
162 auto r = pool->start();
163 if (r.has_error()) {
164 std::cerr << "start failed: " << r.get_error().to_string() << std::endl;
165 return;
166 }
167 }
168
169 // 5. Submit jobs that will be logged
170 for (int i = 0; i < 10; ++i) {
171 auto r = pool->enqueue(std::make_unique<callback_job>(
172 [i, &context]() -> result_void {
173 context.log(log_level::info,
174 "Processing job " + std::to_string(i));
175
176 // Simulate work
177 std::this_thread::sleep_for(std::chrono::milliseconds(100));
178
179 return result_void();
180 }
181 ));
182 if (r.has_error()) {
183 std::cerr << "enqueue failed: " << r.get_error().to_string() << std::endl;
184 }
185 }
186
187 // 6. Wait for completion
188 std::this_thread::sleep_for(std::chrono::seconds(2));
189
190 // 7. Stop pool
191 {
192 auto r = pool->stop();
193 if (r.has_error()) {
194 std::cerr << "stop failed: " << r.get_error().to_string() << std::endl;
195 }
196 }
197
198 std::cout << "\n=== Basic Thread Pool Demo Complete ===\n" << std::endl;
199}

281void demonstrate_minimal_usage() {
282 std::cout << "\n=== Minimal Thread Pool (No Services) Demo ===\n" << std::endl;
283
284 // Clear any existing services
285 service_container::global().clear();
286
287 // Create pool without context - no logging or monitoring
288 auto pool = std::make_shared<thread_pool>("MinimalPool");
289
290 // Add workers
291 std::vector<std::unique_ptr<thread_worker>> workers;
292 for (int i = 0; i < 2; ++i) {
293 workers.push_back(std::make_unique<thread_worker>());
294 }
295 {
296 auto r = pool->enqueue_batch(std::move(workers));
297 if (r.has_error()) {
298 std::cerr << "enqueue_batch failed: " << r.get_error().to_string() << std::endl;
299 return;
300 }
301 }
302
303 {
304 auto r = pool->start();
305 if (r.has_error()) {
306 std::cerr << "start failed: " << r.get_error().to_string() << std::endl;
307 return;
308 }
309 }
310
311 // Submit jobs - no logging will occur
312 std::atomic<int> counter{0};
313 for (int i = 0; i < 5; ++i) {
314 auto r = pool->enqueue(std::make_unique<callback_job>(
315 [&counter]() -> result_void {
316 counter.fetch_add(1);
317 return result_void();
318 }
319 ));
320 if (r.has_error()) {
321 std::cerr << "enqueue failed: " << r.get_error().to_string() << std::endl;
322 }
323 }
324
325 std::this_thread::sleep_for(std::chrono::milliseconds(500));
326 {
327 auto r = pool->stop();
328 if (r.has_error()) {
329 std::cerr << "stop failed: " << r.get_error().to_string() << std::endl;
330 }
331 }
332
333 std::cout << "Completed " << counter.load() << " jobs without any logging/monitoring" << std::endl;
334 std::cout << "\n=== Minimal Demo Complete ===\n" << std::endl;
335}
