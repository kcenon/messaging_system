46 void work_loop() {
47 std::random_device rd;
48 std::mt19937 gen(rd());
49 std::uniform_int_distribution<> work_time(1, 10);
50 std::uniform_int_distribution<> idle_time(5, 20);

53 // ÏûëÏóÖ ÏãúÎÆ¨Î†àÏù¥ÏÖò
54 auto start = std::chrono::steady_clock::now();
55 std::this_thread::sleep_for(std::chrono::milliseconds(work_time(gen)));
56 auto end = std::chrono::steady_clock::now();
57
58 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
59 jobs_processed_++;
60 total_processing_time_ns_ += duration;
61
62 // ÏõåÏª§ Î©îÌä∏Î¶≠ ÏóÖÎç∞Ïù¥Ìä∏
63 worker_metrics metrics;
64 metrics.jobs_processed = jobs_processed_;

68 monitor_->update_process_worker_metrics(pool_id_.process_id, worker_id_, metrics);
69
70 // Ïú†Ìú¥ ÏãúÍ∞Ñ
71 std::this_thread::sleep_for(std::chrono::milliseconds(idle_time(gen)));
72 }
73 }

76 thread_pool_identifier pool_id_;
77 std::shared_ptr<multi_process_monitoring> monitor_;

143 void add_thread_pool(const std::string& pool_name, std::size_t instance_id, std::size_t worker_count) {
144 thread_pool_identifier pool_id{

154 void update_system_metrics() {
155 system_metrics metrics;
156 metrics.cpu_usage_percent = 20 + (rand() % 30);
157 metrics.memory_usage_bytes = 100 * 1024 * 1024 + (rand() % (50 * 1024 * 1024));
158 metrics.active_threads = 0;

161 metrics.active_threads += 4; // Í∞ÑÎã®Ìôî
162 }
163
164 monitor_->update_process_system_metrics(process_id_, metrics);
165 }
166
167 void update_pool_metrics() {
168 for (auto& pool : thread_pools_) {
169 pool->update_metrics();
170 }
171 }
172
173private:
174 process_identifier process_id_;
175 std::shared_ptr<multi_process_monitoring> monitor_;
176 std::vector<std::unique_ptr<simulated_thread_pool>> thread_pools_;
