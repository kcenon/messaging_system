149 // Record enqueue operation
150 void record_enqueue(size_t message_size, std::chrono::nanoseconds enqueue_time) {
151 metrics_.messages_enqueued.fetch_add(1);
152 metrics_.bytes_enqueued.fetch_add(message_size);
153 metrics_.total_enqueue_time_ns.fetch_add(enqueue_time.count());
154
155 // Update min/max
156 uint64_t time_ns = enqueue_time.count();
157 update_atomic_min(metrics_.min_enqueue_time_ns, time_ns);
158 update_atomic_max(metrics_.max_enqueue_time_ns, time_ns);
159 }
160
161 // Record dropped message
162 void record_drop() {
163 metrics_.messages_dropped.fetch_add(1);
164 }
165
166 // Record processed message
167 void record_processed(size_t message_size) {
168 metrics_.messages_processed.fetch_add(1);
169 metrics_.bytes_processed.fetch_add(message_size);
170 }
171
172 // Record writer performance
173 void record_write(const std::string& writer_name,
174 size_t message_size,
175 std::chrono::microseconds write_time,
176 bool success) {
177 std::lock_guard<std::mutex> lock(metrics_.writer_metrics_mutex);
178 auto& writer = metrics_.writer_stats[writer_name];
179
180 if (success) {
181 writer.messages_written.fetch_add(1);
182 writer.bytes_written.fetch_add(message_size);
183 writer.total_write_time_us.fetch_add(write_time.count());
184
185 uint64_t time_us = write_time.count();
186 update_atomic_min(writer.min_write_time_us, time_us);
187 update_atomic_max(writer.max_write_time_us, time_us);
188 } else {
189 writer.write_failures.fetch_add(1);
190 }
191 }
192
193 // Update queue size
194 void update_queue_size(size_t current_size, size_t max_size) {
195 metrics_.current_queue_size.store(current_size);
196 metrics_.max_queue_size.store(max_size);
197 }
198
199 // Get current metrics snapshot
200 performance_metrics get_snapshot() const {
201 performance_metrics snapshot;
202
203 // Copy atomic values
204 snapshot.messages_enqueued.store(metrics_.messages_enqueued.load());
205 snapshot.bytes_enqueued.store(metrics_.bytes_enqueued.load());
206 snapshot.messages_processed.store(metrics_.messages_processed.load());
207 snapshot.bytes_processed.store(metrics_.bytes_processed.load());
208 snapshot.current_queue_size.store(metrics_.current_queue_size.load());
209 snapshot.max_queue_size.store(metrics_.max_queue_size.load());
210 snapshot.messages_dropped.store(metrics_.messages_dropped.load());
211 snapshot.total_enqueue_time_ns.store(metrics_.total_enqueue_time_ns.load());
212 snapshot.min_enqueue_time_ns.store(metrics_.min_enqueue_time_ns.load());
213 snapshot.max_enqueue_time_ns.store(metrics_.max_enqueue_time_ns.load());
214
215 // Copy time points
216 snapshot.start_time = metrics_.start_time;
217 snapshot.last_reset = metrics_.last_reset;
218
219 // Deep copy writer metrics
220 std::lock_guard<std::mutex> lock(metrics_.writer_metrics_mutex);
221 for (const auto& [name, writer] : metrics_.writer_stats) {
222 auto& dest = snapshot.writer_stats[name];
223 dest.messages_written.store(writer.messages_written.load());
224 dest.bytes_written.store(writer.bytes_written.load());
225 dest.write_failures.store(writer.write_failures.load());
226 dest.total_write_time_us.store(writer.total_write_time_us.load());
227 dest.min_write_time_us.store(writer.min_write_time_us.load());
228 dest.max_write_time_us.store(writer.max_write_time_us.load());
229 }
230
231 return snapshot;
232 }

260 // Helper to update atomic minimum
261 void update_atomic_min(std::atomic<uint64_t>& atomic_val, uint64_t new_val) {
262 uint64_t current = atomic_val.load();
263 while (new_val < current && !atomic_val.compare_exchange_weak(current, new_val));
264 }
265
266 // Helper to update atomic maximum
267 void update_atomic_max(std::atomic<uint64_t>& atomic_val, uint64_t new_val) {
268 uint64_t current = atomic_val.load();
269 while (new_val > current && !atomic_val.compare_exchange_weak(current, new_val));
270 }
