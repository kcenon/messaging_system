60// Pipeline stage definition
61template<typename TInput, typename TOutput>
62class pipeline_stage {
63public:
64 using ProcessFunc = std::function<TOutput(const TInput&)>;
65 using FilterFunc = std::function<bool(const TInput&)>;
66 using ErrorHandler = std::function<void(const TInput&, const std::exception&)>;
67
68private:
69 std::string stage_name;
70 ProcessFunc processor;
71 FilterFunc filter;
72 ErrorHandler error_handler;
73 std::atomic<uint64_t> processed_count{0};
74 std::atomic<uint64_t> error_count{0};
75 std::atomic<uint64_t> filtered_count{0};
76
77public:
78 pipeline_stage(
79 const std::string& name,
80 ProcessFunc proc,
81 FilterFunc filt = nullptr,
82 ErrorHandler err = nullptr
83 ) : stage_name(name), processor(proc), filter(filt), error_handler(err) {}
84
85 std::optional<TOutput> process(const TInput& input) {
86 try {
87 // Apply filter if provided
88 if (filter && !filter(input)) {
89 filtered_count++;
90 return std::nullopt;
91 }
92
93 // Process the input
94 TOutput output = processor(input);
95 processed_count++;
96 return output;
97
98 } catch (const std::exception& e) {
99 error_count++;
100 if (error_handler) {
101 error_handler(input, e);
102 }
103 return std::nullopt;
104 }
105 }

112 << " | Errors: " << error_count;
113
114 if (logger) {
115 logger->log(kcenon::logger::log_level::info, stats.str());
116 }
117 }

355 aggregated_data aggregateEvent(const processed_event& event) {
356 // Time-based windowing
357 auto now = std::chrono::system_clock::now();
358 auto window_start = std::chrono::floor<std::chrono::seconds>(now);
359 auto window_id = std::to_string(window_start.time_since_epoch().count());
360
361 std::lock_guard<std::mutex> lock(window_mutex);
362
363 // Add event to window
364 window_buffers[window_id].push_back(event);
365
366 // Check if window is complete
367 if (window_buffers[window_id].size() >= 100 ||
368 (now - window_start) > window_size) {
369
370 // Aggregate the window
371 aggregated_data aggregated;
372 aggregated.window_id = window_id;
373 aggregated.window_start = window_start;
374 aggregated.window_end = now;
375 aggregated.event_count = window_buffers[window_id].size();
376
377 // Calculate averages and sums
378 double total_score = 0.0;
379 for (const auto& e : window_buffers[window_id]) {
380 total_score += e.score;
381
382 // Count tags
383 for (const auto& tag : e.tags) {
384 aggregated.tag_frequency[tag]++;
385 }
386
387 // Sum metrics
388 for (const auto& [metric, value] : e.metrics) {
389 aggregated.metric_sums[metric] += value;
390 }
391 }
392
393 aggregated.avg_score = total_score / aggregated.event_count;
394
395 // Clear window
396 window_buffers.erase(window_id);
397
398 return aggregated;
399 }
400
401 // Return empty aggregation if window not complete
402 return aggregated_data{};
403 }

487 void startProcessingThreads() {
488 // Stage 1 & 2: Validation and Enrichment
489 std::thread([this]() {
490 while (running) {
491 std::unique_lock<std::mutex> lock(queue_mutex);
492 queue_cv.wait(lock, [this] {
493 return !raw_events.empty() || !running;
494 });
495
496 while (!raw_events.empty()) {
497 raw_event event = raw_events.front();
498 raw_events.pop();
499 lock.unlock();
500
501 // Process through validation
502 if (auto validated = validation_stage->process(event)) {
503 // Process through enrichment
504 if (auto enriched = enrichment_stage->process(*validated)) {
505 std::lock_guard<std::mutex> proc_lock(queue_mutex);
506 processed_events.push(*enriched);
507 }
508 }
509
510 lock.lock();
511 }
512 }
513 }).detach();
514
515 // Stage 3 & 4: Transformation and Aggregation
516 std::thread([this]() {
517 while (running) {
518 std::this_thread::sleep_for(100ms);
519
520 std::lock_guard<std::mutex> lock(queue_mutex);
521
522 while (!processed_events.empty()) {
523 processed_event event = processed_events.front();
524 processed_events.pop();

526 // Process through transformation

528 // Process through aggregation
