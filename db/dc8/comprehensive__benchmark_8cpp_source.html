74std::string generate_message(size_t size) {
75 static const char charset[] =
76 "0123456789"
77 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
78 "abcdefghijklmnopqrstuvwxyz"
79 " .,!?";
80 static std::mt19937 gen(std::random_device{}());
81 static std::uniform_int_distribution<> dis(0, sizeof(charset) - 2);
82
83 std::string result;
84 result.reserve(size);
85
86 for (size_t i = 0; i < size; ++i) {
87 result += charset[dis(gen)];
88 }
89
90 return result;
91}

96static void BM_ConfigurationTemplates(benchmark::State& state) {
97 int config_type = state.range(0);
98
99 logger_config config;
100 std::string label;
101
102 switch (config_type) {
103 case 0:
104 config = logger_config::default_config();
105 label = "default";
106 break;
107 case 1:
108 config = logger_config::high_performance();
109 label = "high_performance";
110 break;
111 case 2:
112 config = logger_config::low_latency();
113 label = "low_latency";
114 break;
115 case 3:
116 config = logger_config::production();
117 label = "production";
118 break;
119 case 4:
120 config = logger_config::debug_config();
121 label = "debug";
122 break;
123 }
124
125 logger_builder builder;
126 auto logger = builder
127 .with_config(config)
128 .add_writer("null", std::make_unique<null_writer>())
129 .build();
130
131 if (!logger) {
132 state.SkipWithError("Failed to build logger");
133 return;
134 }
135
136 auto& log = *logger.value();
137
138 if (config.async) {
139 log->start();
140 }
141
142 const std::string message = "Configuration benchmark test message";
143
144 for (auto _ : state) {
145 log->log(thread_module::log_level::info, message);
146 }
147
148 if (config.async) {
149 log->stop();
150 }
151
152 state.SetItemsProcessed(state.iterations());
153 state.SetLabel(label);
154}

156 ->Arg(0) // default
157 ->Arg(1) // high_performance
158 ->Arg(2) // low_latency
159 ->Arg(3) // production
160 ->Arg(4); // debug

212 logger_builder builder2;

229

231 // Generate burst of messages
232 for (size_t i = 0; i < burst_size; ++i) {

236 // Small delay to allow processing

314

316 state.SkipWithError("Failed to build logger");

321 log->start();
322
323 const std::string info_msg = "Info message (filtered)";
324 const std::string warn_msg = "Warning message (not filtered)";
325
326 for (auto _ : state) {
327 // Mix of messages that pass and don't pass filter
328 log->log(thread_module::log_level::info, info_msg);
329 log->log(thread_module::log_level::warning, warn_msg);
330 log->log(thread_module::log_level::info, info_msg);

332 }
333
334 log->stop();
335
336 state.SetItemsProcessed(state.iterations() * 4);
337 state.SetLabel(use_filter ? "with_filter" : "no_filter");
