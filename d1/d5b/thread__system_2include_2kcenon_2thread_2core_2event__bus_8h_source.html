125 template<typename Event>
126 void publish(const Event& event) {
127 auto type = std::type_index(typeid(Event));
128
129 // Get handlers snapshot to avoid holding lock during callbacks
130 std::vector<handler_func> handlers_copy;
131 {
132 std::lock_guard<std::mutex> lock(mutex_);
133 auto it = handlers_.find(type);
134 if (it != handlers_.end()) {
135 for (const auto& [id, handler] : it->second) {
136 handlers_copy.push_back(handler);
137 }
138 }
139 }
140
141 // Process handlers asynchronously
142 if (!handlers_copy.empty() && thread_pool_) {
143 thread_pool_->submit([handlers_copy, event]() {
144 for (const auto& handler : handlers_copy) {
145 try {
146 handler(std::any(event));
147 } catch (...) {
148 // Log error but continue with other handlers
149 }
150 }
151 });
152 }
153 }

183 template<typename Event>
184 [[nodiscard]] subscription subscribe(std::function<void(const Event&)> handler) {
185 auto type = std::type_index(typeid(Event));
186 auto wrapped_handler = [handler](const std::any& any_event) {
187 try {
188 const auto& event = std::any_cast<const Event&>(any_event);
189 handler(event);
190 } catch (const std::bad_any_cast&) {
191 // Type mismatch - should not happen
192 }
193 };
194
195 std::lock_guard<std::mutex> lock(mutex_);
196 auto id = next_handler_id_++;
197 handlers_[type][id] = wrapped_handler;
198
199 return subscription(this, type, id);
200 }
