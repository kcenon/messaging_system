43struct TestNode {
44 std::atomic<int> data{0};
45 std::atomic<TestNode*> next{nullptr};
46
47 explicit TestNode(int value) : data(value) {}
48};
49
50class LockFreeStack {
51private:
52 std::atomic<TestNode*> head_{nullptr};
53 hazard_pointer_manager& hp_manager_;
54
55public:
56 explicit LockFreeStack(hazard_pointer_manager& hp_mgr) : hp_manager_(hp_mgr) {}
57
58 ~LockFreeStack() {
59 while (auto* node = head_.load()) {
60 head_.store(node->next.load());
61 delete node;
62 }
63 }
64
65 void push(int value) {
66 auto* new_node = new TestNode(value);
67 auto* old_head = head_.load();
68
69 do {
70 new_node->next.store(old_head);
71 } while (!head_.compare_exchange_weak(old_head, new_node));
72 }
73
74 bool pop(int& result) {
75 auto hp = hp_manager_.acquire();
76
77 while (true) {
78 auto* head = hp.protect(head_);
79 if (!head) {
80 return false; // Stack is empty
81 }
82
83 auto* next = head->next.load();
84
85 // Try to update head
86 if (head_.compare_exchange_weak(head, next)) {
87 result = head->data.load();
88 hp_manager_.retire(head);
89 return true;
90 }
91 }
92 }
93};

137void demonstrate_concurrent_access() {
138 log_module::write_information("\n=== Concurrent Access Demo ===");
139
140 constexpr int NUM_THREADS = 4;
141 constexpr int OPERATIONS_PER_THREAD = 1000;
142
143 hazard_pointer_manager hp_manager(NUM_THREADS, 2);
144 LockFreeStack stack(hp_manager);
145
146 // Fill the stack initially
147 for (int i = 0; i < 100; ++i) {
148 stack.push(i);
149 }
150
151 std::atomic<int> push_count{0};
152 std::atomic<int> pop_count{0};
153 std::atomic<int> failed_pops{0};
154
155 auto start_time = std::chrono::high_resolution_clock::now();
156
157 std::vector<std::thread> threads;
158 threads.reserve(NUM_THREADS);
159
160 // Create worker threads
161 for (int thread_id = 0; thread_id < NUM_THREADS; ++thread_id) {
162 threads.emplace_back([&, thread_id]() {
163 std::random_device rd;
164 std::mt19937 gen(rd());
165 std::uniform_int_distribution<> dis(0, 1);
166
167 for (int op = 0; op < OPERATIONS_PER_THREAD; ++op) {
168 if (dis(gen) == 0) {
169 // Push operation
170 int value = thread_id * OPERATIONS_PER_THREAD + op;
171 stack.push(value);
172 push_count.fetch_add(1);
173 } else {
174 // Pop operation
175 int result;
176 if (stack.pop(result)) {
177 pop_count.fetch_add(1);
178 } else {
179 failed_pops.fetch_add(1);
180 }
181 }
182 }
183 });
184 }
185
186 // Wait for all threads to complete
187 for (auto& thread : threads) {
188 thread.join();
189 }
190
191 auto end_time = std::chrono::high_resolution_clock::now();
192 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
193
194 log_module::write_information("Concurrent operations completed in {} ms", duration.count());
195 log_module::write_information("Push operations: {}", push_count.load());
196 log_module::write_information("Successful pop operations: {}", pop_count.load());
197 log_module::write_information("Failed pop operations: {}", failed_pops.load());
198
199 // Final statistics
200 auto stats = hp_manager.get_statistics();
201 log_module::write_information("Final hazard pointer statistics:");
202 log_module::write_information(" Active hazard pointers: {}", stats.active_hazard_pointers);
203 log_module::write_information(" Retired list size: {}", stats.retired_list_size);
204 log_module::write_information(" Total retired: {}", stats.total_retired);
205 log_module::write_information(" Total reclaimed: {}", stats.total_reclaimed);
206}

208void demonstrate_memory_safety() {
209 log_module::write_information("\n=== Memory Safety Demo ===");
210
211 hazard_pointer_manager hp_manager(2, 1);
212 std::atomic<TestNode*> shared_ptr{new TestNode(123)};
213
214 std::atomic<bool> reader_done{false};
215 std::atomic<bool> writer_done{false};
216
217 // Reader thread - tries to access the shared object
218 std::thread reader([&]() {
219 auto hp = hp_manager.acquire();
220
221 for (int i = 0; i < 100; ++i) {
222 auto* protected_ptr = hp.protect(shared_ptr);
223 if (protected_ptr) {
224 // Safe to access the object
225 volatile int value = protected_ptr->data.load();
226 (void)value; // Prevent optimization
227 std::this_thread::sleep_for(std::chrono::microseconds(10));
228 }
229 hp.clear();
230 std::this_thread::sleep_for(std::chrono::microseconds(10));
231 }
232 reader_done = true;
233 });
234
235 // Writer thread - tries to replace and retire the shared object
236 std::thread writer([&]() {
237 std::this_thread::sleep_for(std::chrono::milliseconds(10)); // Let reader start
238
239 for (int i = 0; i < 10; ++i) {
240 auto* new_node = new TestNode(456 + i);
241 auto* old_node = shared_ptr.exchange(new_node);
242
243 if (old_node) {
244 hp_manager.retire(old_node);
245 }
246
247 std::this_thread::sleep_for(std::chrono::milliseconds(5));
248 }
249 writer_done = true;
250 });
251
252 reader.join();
253 writer.join();
254
255 // Clean up remaining object
256 auto* final_node = shared_ptr.exchange(nullptr);
257 if (final_node) {
258 hp_manager.retire(final_node);
259 }
260
261 hp_manager.scan_and_reclaim();
262
263 auto stats = hp_manager.get_statistics();
264 log_module::write_information("Memory safety test completed safely!");
265 log_module::write_information("Final statistics:");
266 log_module::write_information(" Total retired: {}", stats.total_retired);
267 log_module::write_information(" Total reclaimed: {}", stats.total_reclaimed);
268}
