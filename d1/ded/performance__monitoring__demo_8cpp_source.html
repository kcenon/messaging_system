124
125 connection_metrics metrics;
126 metrics.total_connections.store(20);

128 metrics.idle_connections.store(metrics.total_connections.load() - metrics.active_connections.load());
129 metrics.connections_created.store(25 + i);
130 metrics.connections_destroyed.store(5 + (i / 5));
131 metrics.connection_errors.store(i / 10);
132 metrics.average_wait_time = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
133 metrics.peak_connections.store(std::max(static_cast<size_t>(20), metrics.active_connections.load()));
134
135 monitor.record_connection_metrics(metrics);
136
