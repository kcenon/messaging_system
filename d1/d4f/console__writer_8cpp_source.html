59result_void console_writer::write(logger_system::log_level level,
60 const std::string& message,
61 const std::string& file,
62 int line,
63 const std::string& function,
64 const std::chrono::system_clock::time_point& timestamp) {
65 std::lock_guard<std::mutex> lock(write_mutex_);
66
67 auto& stream = (use_stderr_ || level <= logger_system::log_level::error)
68 ? std::cerr : std::cout;
69
70 if (use_color()) {
71 stream << level_to_color(level);
72 }
73
74 stream << format_log_entry(level, message, file, line, function, timestamp);
75
76 if (use_color()) {
77 stream << "\033[0m"; // Reset color
78 }
79
80 stream << std::endl;
81
82 // Check for stream failure
83 if (stream.fail()) {
84 return make_logger_error(logger_error_code::processing_failed,
85 "Console write failed");
86 }
87
88 return {}; // Success
89}

91result_void console_writer::flush() {
92 std::lock_guard<std::mutex> lock(write_mutex_);
93 std::cout.flush();
94 std::cerr.flush();
95
96 if (std::cout.fail() || std::cerr.fail()) {
97 return make_logger_error(logger_error_code::flush_timeout,
98 "Console flush failed");
99 }
100
101 return {}; // Success
102}

108bool console_writer::is_color_supported() const {
109#ifdef _WIN32
110 // Check if running in Windows Terminal or if ANSI is enabled
111 HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
112 if (hOut == INVALID_HANDLE_VALUE) {
113 return false;
114 }
115
116 DWORD dwMode = 0;
117 if (!GetConsoleMode(hOut, &dwMode)) {
118 return false;
119 }
120
121 // Check for ENABLE_VIRTUAL_TERMINAL_PROCESSING flag
122 return (dwMode & 0x0004) != 0;
123#else
124 // Check if output is to a terminal and TERM is set
125 const char* term = std::getenv("TERM");
126 return isatty(STDOUT_FILENO) && term && std::string(term) != "dumb";
127#endif
128}

131std::string base_writer::format_log_entry(logger_system::log_level level,
132 const std::string& message,
133 const std::string& file,
134 int line,
135 const std::string& function,
136 const std::chrono::system_clock::time_point& timestamp) {
137 auto time_t = std::chrono::system_clock::to_time_t(timestamp);
138 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
139 timestamp.time_since_epoch()) % 1000;
140
141 std::ostringstream oss;
142 oss << "[" << std::put_time(std::localtime(&time_t), "%Y-%m-%d %H:%M:%S");
143 oss << "." << std::setfill('0') << std::setw(3) << ms.count() << "] ";
144 oss << "[" << level_to_string(level) << "] ";
145
146 if (!file.empty()) {
147 // Extract filename from path
148 size_t pos = file.find_last_of("/\\");
149 std::string filename = (pos != std::string::npos) ? file.substr(pos + 1) : file;
150 oss << filename << ":" << line << " (" << function << ") ";
151 }
152
153 oss << message;
154
155 return oss.str();
156}

158std::string base_writer::level_to_string(logger_system::log_level level) const {
159 switch (level) {
160 case logger_system::log_level::fatal: return "CRITICAL";
161 case logger_system::log_level::error: return "ERROR";
162 case logger_system::log_level::warn: return "WARNING";
163 case logger_system::log_level::info: return "INFO";
164 case logger_system::log_level::debug: return "DEBUG";
165 case logger_system::log_level::trace: return "TRACE";
166 case logger_system::log_level::off: return "OFF";
167 }
168 return "UNKNOWN";
169}

171std::string base_writer::level_to_color(logger_system::log_level level) const {
172 if (!use_color()) return "";
173
174 switch (level) {
175 case logger_system::log_level::fatal: return "\033[1;35m"; // Bright Magenta
176 case logger_system::log_level::error: return "\033[1;31m"; // Bright Red
177 case logger_system::log_level::warn: return "\033[1;33m"; // Bright Yellow
178 case logger_system::log_level::info: return "\033[1;32m"; // Bright Green
179 case logger_system::log_level::debug: return "\033[1;36m"; // Bright Cyan
180 case logger_system::log_level::trace: return "\033[1;37m"; // Bright White
181 case logger_system::log_level::off: return ""; // No color for off
182 }
183 return "";
184}
