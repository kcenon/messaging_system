172 void adapt(const monitoring_system::system_metrics& sys_metrics) {
173 std::lock_guard lock(stats_mutex_);
174
175 // Initialize averages on first adaptation
176 if (stats_.total_adaptations == 0) {
177 stats_.average_cpu_usage = sys_metrics.cpu_usage_percent;
178 stats_.average_memory_usage = sys_metrics.memory_usage_percent;
179 } else {
180 // Update average metrics using exponential smoothing
181 stats_.average_cpu_usage =
182 config_.smoothing_factor * sys_metrics.cpu_usage_percent +
183 (1.0 - config_.smoothing_factor) * stats_.average_cpu_usage;
184
185 stats_.average_memory_usage =
186 config_.smoothing_factor * sys_metrics.memory_usage_percent +
187 (1.0 - config_.smoothing_factor) * stats_.average_memory_usage;
188 }
189
190 // Determine load level
191 auto new_level = calculate_load_level(
192 stats_.average_cpu_usage,
193 stats_.average_memory_usage
194 );
195
196 // Adapt if load level changed
197 if (new_level != stats_.current_load_level) {
198 if (new_level > stats_.current_load_level) {
199 stats_.downscale_count++;
200 } else {
201 stats_.upscale_count++;
202 }
203
204 stats_.current_load_level = new_level;
205 stats_.current_interval = config_.get_interval_for_load(new_level);
206 current_sampling_rate_ = config_.get_sampling_rate_for_load(new_level);
207 stats_.current_sampling_rate = current_sampling_rate_;
208 stats_.total_adaptations++;
209 stats_.last_adaptation = std::chrono::system_clock::now();
210 }
211 }

261 bool should_sample() const {
262 if (!enabled_) return true;
263
264 // Use random sampling based on current rate

273 load_level calculate_load_level(double cpu_usage, double memory_usage) const {
274 // Consider memory pressure in load calculation
275 double effective_load = cpu_usage;
276
277 // Memory pressure should escalate load level
278 if (memory_usage > config_.memory_critical_threshold) {
279 // Critical memory -> at least high load
280 effective_load = std::max(effective_load, config_.high_threshold + 1.0);
281 } else if (memory_usage > config_.memory_warning_threshold) {
282 // Warning memory -> at least moderate load
283 effective_load = std::max(effective_load, config_.moderate_threshold + 1.0);
284 }
285
286 // Apply strategy-specific adjustments BEFORE determining level
287 switch (config_.strategy) {
288 case adaptation_strategy::conservative:
289 effective_load *= 0.8; // Be more conservative
290 break;
291 case adaptation_strategy::aggressive:
292 effective_load *= 1.2; // Be more aggressive
293 break;
294 case adaptation_strategy::balanced:
295 default:
296 break;
297 }
298
299 // Determine load level
300 if (effective_load >= config_.high_threshold) {
301 return load_level::critical;
302 } else if (effective_load >= config_.moderate_threshold) {
303 return load_level::high;
304 } else if (effective_load >= config_.low_threshold) {
305 return load_level::moderate;
306 } else if (effective_load >= config_.idle_threshold) {
307 return load_level::low;
308 } else {
309 return load_level::idle;
310 }
311 }
