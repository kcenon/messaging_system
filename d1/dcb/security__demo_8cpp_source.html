50// Custom security filter that blocks sensitive logs
51class security_filter : public log_filter_interface {
52public:
53 bool should_log(const log_entry& entry) const override {
54 const std::string message = entry.message.to_string();
55
56 // Block logs containing passwords
57 if (message.find("password") != std::string::npos) {
58 std::cout << "[SECURITY] Blocked log containing password" << std::endl;
59 return false;
60 }
61 return true;
62 }
63
64 std::string get_name() const override {
65 return "security_filter";
66 }
67};
