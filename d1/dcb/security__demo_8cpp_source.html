146 logger->log(thread_module::log_level::debug, "Debug message", "secure_module.cpp", 10, "test");
147 }
148 if (test_filter->should_log(thread_module::log_level::error, "Error message",
149 "secure_module.cpp", 20, "test")) {
150 std::cout << " [ALLOWED] Error message from secure_module.cpp" << std::endl;
151 logger->log(thread_module::log_level::error, "Error message", "secure_module.cpp", 20, "test");
152 }
153
154 // Regular user
155 test_filter->set_user_context("user", access_control_filter::permission_level::write_info);
156 std::cout << "\nRegular user (write_info permission):" << std::endl;
157 if (test_filter->should_log(thread_module::log_level::info, "Info allowed",
158 "public_module.cpp", 30, "test")) {
159 std::cout << " [ALLOWED] Info message from public_module.cpp" << std::endl;
160 }
161 if (!test_filter->should_log(thread_module::log_level::debug, "Debug blocked",
162 "public_module.cpp", 40, "test")) {
163 std::cout << " [BLOCKED] Debug message from public_module.cpp" << std::endl;
164 }
165 if (!test_filter->should_log(thread_module::log_level::error, "Error blocked",

167 std::cout << " [BLOCKED] Error message from secure_module.cpp (insufficient permission)" << std::endl;
168 }
169
170 // Read-only user
