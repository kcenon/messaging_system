133class online_statistics {
134private:
135 uint64_t n_ = 0;
136 double mean_ = 0.0;
137 double m2_ = 0.0; // Sum of squares of differences from mean
138 double m3_ = 0.0; // Third moment
139 double m4_ = 0.0; // Fourth moment
140 double min_ = std::numeric_limits<double>::max();
141 double max_ = std::numeric_limits<double>::lowest();
142 double sum_ = 0.0;
143
144public:
148 void add_value(double value) {
149 n_++;
150 sum_ += value;
151 min_ = std::min(min_, value);
152 max_ = std::max(max_, value);
153
154 // Welford's online algorithm for variance
155 double delta = value - mean_;
156 mean_ += delta / n_;
157 double delta2 = value - mean_;
158 m2_ += delta * delta2;
159
160 // Online algorithm for higher moments
161 double n = static_cast<double>(n_);
162 if (n_ >= 2) {
163 double delta_n = delta / n;
164 double delta_n2 = delta_n * delta_n;
165 double term1 = delta * delta2 * (n - 1);
166
167 m4_ += term1 * delta_n2 * (n * n - 3 * n + 3) + 6 * delta_n2 * m2_ - 4 * delta_n * m3_;
168 m3_ += term1 * delta_n * (n - 2) - 3 * delta_n * m2_;
169 }
170 }
171
175 stream_statistics get_statistics() const {
176 stream_statistics stats;
177 stats.count = n_;
178 stats.sum = sum_;
179 stats.mean = mean_;
180 stats.min_value = n_ > 0 ? min_ : 0.0;
181 stats.max_value = n_ > 0 ? max_ : 0.0;
182
183 if (n_ >= 2) {
184 stats.variance = m2_ / (n_ - 1);
185 stats.std_deviation = std::sqrt(stats.variance);
186
187 if (stats.variance > 0) {
188 double n = static_cast<double>(n_);
189 stats.skewness = std::sqrt(n * (n - 1)) / (n - 2) * m3_ / std::pow(stats.variance, 1.5);
190 stats.kurtosis = (n - 1) / ((n - 2) * (n - 3)) *
191 ((n + 1) * m4_ / (stats.variance * stats.variance) - 3 * (n - 1));
192 }
193 }
194
195 return stats;
196 }
197
201 void reset() {
202 n_ = 0;
203 mean_ = 0.0;
204 m2_ = 0.0;
205 m3_ = 0.0;
206 m4_ = 0.0;
207 min_ = std::numeric_limits<double>::max();
208 max_ = std::numeric_limits<double>::lowest();
209 sum_ = 0.0;
210 }
211
215 uint64_t count() const noexcept { return n_; }
216
220 double mean() const noexcept { return mean_; }
221
225 double variance() const noexcept {
226 return n_ >= 2 ? m2_ / (n_ - 1) : 0.0;
227 }
228};

234class quantile_estimator {
235private:
236 std::vector<double> q_; // Quantile values
237 std::vector<double> dn_; // Desired positions
238 std::vector<double> np_; // Actual positions
239 std::vector<int> n_; // Marker positions
240 bool initialized_ = false;
241 double p_; // Target quantile
242
246 double parabolic(int i, int d) const {
247 return q_[i] + d / (n_[i + 1] - n_[i - 1]) *
248 ((n_[i] - n_[i - 1] + d) * (q_[i + 1] - q_[i]) / (n_[i + 1] - n_[i]) +
249 (n_[i + 1] - n_[i] - d) * (q_[i] - q_[i - 1]) / (n_[i] - n_[i - 1]));
250 }
251
255 double linear(int i, int d) const {
256 return q_[i] + d * (q_[i + d] - q_[i]) / (n_[i + d] - n_[i]);
257 }
258
259public:
263 explicit quantile_estimator(double p) : p_(p) {
264 q_.resize(5);
265 dn_.resize(5);
266 np_.resize(5);
267 n_.resize(5);
268
269 // Initialize desired positions
270 dn_[0] = 0;
271 dn_[1] = p / 2;
272 dn_[2] = p;
273 dn_[3] = (1 + p) / 2;
274 dn_[4] = 1;
275 }
276
280 void add_observation(double x) {
281 if (!initialized_) {
282 if (q_.size() < 5) {
283 q_.push_back(x);
284 }
285
286 if (q_.size() == 5) {
287 std::sort(q_.begin(), q_.end());
288 for (int i = 0; i < 5; i++) {
289 n_[i] = i;
290 np_[i] = i;
291 }
292 initialized_ = true;
293 }
294 return;
295 }
296
297 // Find cell k
298 int k = 0;
299 if (x < q_[0]) {
300 q_[0] = x;
301 k = 0;
302 } else if (x >= q_[4]) {
303 q_[4] = x;
304 k = 3;
305 } else {
306 for (int i = 1; i < 5; i++) {
307 if (x < q_[i]) {
308 k = i - 1;
309 break;
310 }
311 }
312 }
313
314 // Increment positions
315 for (int i = k + 1; i < 5; i++) {
316 n_[i]++;
317 }
318
319 // Update desired positions
320 for (int i = 0; i < 5; i++) {
321 np_[i] += dn_[i];
322 }
323
324 // Adjust heights
325 for (int i = 1; i < 4; i++) {
326 double d = np_[i] - n_[i];
327 if ((d >= 1 && n_[i + 1] - n_[i] > 1) ||
328 (d <= -1 && n_[i - 1] - n_[i] < -1)) {
329
330 int di = d >= 0 ? 1 : -1;
331 double qnew = parabolic(i, di);
332
333 if (q_[i - 1] < qnew && qnew < q_[i + 1]) {
334 q_[i] = qnew;
335 } else {
336 q_[i] = linear(i, di);
337 }
338
339 n_[i] += di;
340 }
341 }
342 }
343
347 double get_quantile() const {
348 return initialized_ ? q_[2] : 0.0;
349 }
350
354 void reset() {
355 q_.clear();
356 q_.resize(5);
357 initialized_ = false;
358 }
359};

365template<typename T>
366class moving_window_aggregator {
367private:
368 struct timestamped_value {
369 T value;
370 std::chrono::system_clock::time_point timestamp;
371
372 timestamped_value(T v, std::chrono::system_clock::time_point ts)
373 : value(v), timestamp(ts) {}
374 };
375
376 std::deque<timestamped_value> window_;
377 std::chrono::milliseconds window_duration_;
378 size_t max_size_;
379 mutable std::mutex mutex_;
380
384 void cleanup_expired() {
385 auto now = std::chrono::system_clock::now();
386 auto cutoff = now - window_duration_;
387
388 while (!window_.empty() && window_.front().timestamp < cutoff) {
389 window_.pop_front();
390 }
391
392 while (window_.size() > max_size_) {
393 window_.pop_front();
394 }
395 }
396
397public:
401 moving_window_aggregator(std::chrono::milliseconds duration, size_t max_size)
402 : window_duration_(duration), max_size_(max_size) {}
403
407 void add_value(const T& value,
408 std::chrono::system_clock::time_point timestamp =
409 std::chrono::system_clock::now()) {
410 std::lock_guard<std::mutex> lock(mutex_);
411 window_.emplace_back(value, timestamp);
412 cleanup_expired();
413 }
414
418 std::vector<T> get_values() const {
419 std::lock_guard<std::mutex> lock(mutex_);
420 std::vector<T> values;
421 values.reserve(window_.size());
422
423 for (const auto& item : window_) {
424 values.push_back(item.value);
425 }
426
427 return values;
428 }
429
433 size_t size() const {
434 std::lock_guard<std::mutex> lock(mutex_);
435 return window_.size();
436 }
437
441 void clear() {
442 std::lock_guard<std::mutex> lock(mutex_);
443 window_.clear();
444 }
445
449 double get_rate_per_second() const {
450 std::lock_guard<std::mutex> lock(mutex_);
451 if (window_.empty()) return 0.0;
452
453 auto duration = std::chrono::duration_cast<std::chrono::seconds>(
454 window_.back().timestamp - window_.front().timestamp);
455
456 return duration.count() > 0 ?
457 static_cast<double>(window_.size()) / duration.count() : 0.0;
458 }

465class stream_aggregator {
466private:
467 stream_aggregator_config config_;
468 online_statistics online_stats_;
469 moving_window_aggregator<double> window_;
470 std::unordered_map<double, std::unique_ptr<quantile_estimator>> percentile_estimators_;
471 std::vector<double> outliers_;
472 mutable std::mutex mutex_;
473
474 std::chrono::system_clock::time_point first_timestamp_;
475 std::chrono::system_clock::time_point last_timestamp_;
476
480 void initialize_percentile_estimators() {
481 std::vector<double> percentiles = {0.01, 0.05, 0.1, 0.25, 0.5, 0.75, 0.9, 0.95, 0.99};
482
483 for (double p : percentiles) {
484 percentile_estimators_[p] = std::make_unique<quantile_estimator>(p);
485 }
486 }
487
491 bool is_outlier(double value) const {
492 if (!config_.enable_outlier_detection || online_stats_.count() < 10) {
493 return false;
494 }
495
496 double mean = online_stats_.mean();
497 double std_dev = std::sqrt(online_stats_.variance());
498
499 return std::abs(value - mean) > config_.outlier_threshold * std_dev;
500 }
501
502public:
506 explicit stream_aggregator(const stream_aggregator_config& config = {})
507 : config_(config)
508 , window_(config.window_duration, config.window_size) {
509
510 auto validation = config_.validate();
511 if (!validation) {
512 throw std::invalid_argument("Invalid stream aggregator configuration: " +
513 validation.get_error().message);
514 }
515
516 initialize_percentile_estimators();
517 }
518
522 result_void add_observation(double value,
523 std::chrono::system_clock::time_point timestamp =
524 std::chrono::system_clock::now()) {
525 std::lock_guard<std::mutex> lock(mutex_);
526
527 // Update timestamps
528 if (online_stats_.count() == 0) {
529 first_timestamp_ = timestamp;
530 }
531 last_timestamp_ = timestamp;
532
533 // Check for outliers
534 if (is_outlier(value)) {
535 outliers_.push_back(value);
536 }
537
538 // Update statistics
539 online_stats_.add_value(value);
540 window_.add_value(value, timestamp);
541
542 // Update percentile estimators
543 for (auto& [p, estimator] : percentile_estimators_) {
544 estimator->add_observation(value);
545 }
546
547 return result_void::success();
548 }
549
553 stream_statistics get_statistics() const {
554 std::lock_guard<std::mutex> lock(mutex_);
555
556 auto stats = online_stats_.get_statistics();
557
558 // Add percentiles
559 for (const auto& [p, estimator] : percentile_estimators_) {
560 stats.percentiles[p] = estimator->get_quantile();
561 }
562
563 // Set common percentiles
564 if (stats.percentiles.count(0.25)) stats.q1 = stats.percentiles[0.25];
565 if (stats.percentiles.count(0.5)) stats.median = stats.percentiles[0.5];
566 if (stats.percentiles.count(0.75)) stats.q3 = stats.percentiles[0.75];
567 stats.iqr = stats.q3 - stats.q1;
568
569 // Set time information
570 stats.first_timestamp = first_timestamp_;
571 stats.last_timestamp = last_timestamp_;
572
573 if (stats.count > 1) {
574 stats.window_duration = std::chrono::duration_cast<std::chrono::milliseconds>(
575 last_timestamp_ - first_timestamp_);
576
577 double seconds = stats.window_duration.count() / 1000.0;
578 if (seconds > 0) {
579 stats.rate_per_second = static_cast<double>(stats.count) / seconds;
580 stats.rate_per_minute = stats.rate_per_second * 60.0;
581 }
582 }
583
584 // Outlier information
585 stats.outlier_count = outliers_.size();
586 stats.outliers = outliers_;
587
588 return stats;
589 }
590
594 result<double> get_percentile(double p) const {
595 if (p < 0.0 || p > 1.0) {
596 return make_error<double>(monitoring_error_code::invalid_argument,
597 "Percentile must be between 0 and 1");
598 }
599
600 std::lock_guard<std::mutex> lock(mutex_);
601
602 auto it = percentile_estimators_.find(p);
603 if (it != percentile_estimators_.end()) {
604 return make_success(it->second->get_quantile());
605 }
606
607 return make_error<double>(monitoring_error_code::collector_not_found,
608 "Percentile estimator not found");
609 }
610
614 void reset() {
615 std::lock_guard<std::mutex> lock(mutex_);
616
617 online_stats_.reset();
618 window_.clear();
619 outliers_.clear();
620
621 for (auto& [p, estimator] : percentile_estimators_) {
622 estimator->reset();
623 }
624 }
625
629 uint64_t count() const {
630 std::lock_guard<std::mutex> lock(mutex_);
631 return online_stats_.count();
632 }
633
637 double mean() const {
638 std::lock_guard<std::mutex> lock(mutex_);
639 return online_stats_.mean();
640 }
641
645 double variance() const {
646 std::lock_guard<std::mutex> lock(mutex_);
647 return online_stats_.variance();
648 }

653 const stream_aggregator_config& get_config() const noexcept {
654 return config_;
655 }

675inline double pearson_correlation(const std::vector<double>& x, const std::vector<double>& y) {
676 if (x.size() != y.size() || x.size() < 2) {
677 return 0.0;
678 }
679
680 double sum_x = 0, sum_y = 0, sum_xy = 0, sum_x2 = 0, sum_y2 = 0;
681 size_t n = x.size();
682
683 for (size_t i = 0; i < n; ++i) {
684 sum_x += x[i];
685 sum_y += y[i];
686 sum_xy += x[i] * y[i];
687 sum_x2 += x[i] * x[i];
688 sum_y2 += y[i] * y[i];
689 }
690
691 double numerator = n * sum_xy - sum_x * sum_y;
692 double denominator = std::sqrt((n * sum_x2 - sum_x * sum_x) * (n * sum_y2 - sum_y * sum_y));
693
694 return denominator != 0.0 ? numerator / denominator : 0.0;
695}
