51 std::string query_condition::to_sql() const
52 {
53 if (!raw_condition_.empty()) {
54 return raw_condition_;
55 }
56
57 if (!sub_conditions_.empty()) {
58 std::ostringstream oss;
59 oss << "(";
60 for (size_t i = 0; i < sub_conditions_.size(); ++i) {
61 if (i > 0) {
62 oss << " " << logical_operator_ << " ";
63 }
64 oss << sub_conditions_[i].to_sql();
65 }
66 oss << ")";
67 return oss.str();
68 }
69
70 std::ostringstream oss;
71 oss << field_ << " " << operator_ << " ";
72
73 std::visit([&oss](const auto& val) {
74 using T = std::decay_t<decltype(val)>;
75 if constexpr (std::is_same_v<T, std::string>) {
76 oss << "'" << val << "'";
77 } else if constexpr (std::is_same_v<T, int64_t>) {
78 oss << val;
79 } else if constexpr (std::is_same_v<T, double>) {
80 oss << val;
81 } else if constexpr (std::is_same_v<T, bool>) {
82 oss << (val ? "TRUE" : "FALSE");
83 } else if constexpr (std::is_same_v<T, std::monostate>) {
84 oss << "NULL";
85 }
86 }, value_);
87
88 return oss.str();
89 }
90
91 std::string query_condition::to_mongodb() const
92 {
93 if (!raw_condition_.empty()) {
94 return raw_condition_;
95 }
96
97 if (!sub_conditions_.empty()) {
98 std::ostringstream oss;
99 std::string mongo_op = (logical_operator_ == "AND") ? "$and" : "$or";
100 oss << "{ \"" << mongo_op << "\": [";
101 for (size_t i = 0; i < sub_conditions_.size(); ++i) {
102 if (i > 0) oss << ", ";
103 oss << sub_conditions_[i].to_mongodb();
104 }
105 oss << "] }";
106 return oss.str();
107 }
108
109 std::ostringstream oss;
110 oss << "{ \"" << field_ << "\": ";
111
112 if (operator_ == "=") {
113 std::visit([&oss](const auto& val) {
114 using T = std::decay_t<decltype(val)>;
115 if constexpr (std::is_same_v<T, std::string>) {
116 oss << "\"" << val << "\"";
117 } else if constexpr (std::is_same_v<T, int64_t>) {
118 oss << val;
119 } else if constexpr (std::is_same_v<T, double>) {
120 oss << val;
121 } else if constexpr (std::is_same_v<T, bool>) {
122 oss << (val ? "true" : "false");
123 } else if constexpr (std::is_same_v<T, std::monostate>) {
124 oss << "null";
125 }
126 }, value_);
127 } else {
128 std::string mongo_op;
129 if (operator_ == ">") mongo_op = "$gt";
130 else if (operator_ == ">=") mongo_op = "$gte";
131 else if (operator_ == "<") mongo_op = "$lt";
132 else if (operator_ == "<=") mongo_op = "$lte";
133 else if (operator_ == "!=") mongo_op = "$ne";
134 else mongo_op = "$eq";
135
136 oss << "{ \"" << mongo_op << "\": ";
137 std::visit([&oss](const auto& val) {
138 using T = std::decay_t<decltype(val)>;
139 if constexpr (std::is_same_v<T, std::string>) {
140 oss << "\"" << val << "\"";
141 } else if constexpr (std::is_same_v<T, int64_t>) {
142 oss << val;
143 } else if constexpr (std::is_same_v<T, double>) {
144 oss << val;
145 } else if constexpr (std::is_same_v<T, bool>) {
146 oss << (val ? "true" : "false");
147 } else if constexpr (std::is_same_v<T, std::monostate>) {
148 oss << "null";
149 }
150 }, value_);
151 oss << " }";
152 }
153
154 oss << " }";
155 return oss.str();
156 }

358 std::string sql_query_builder::build_for_database(database_types db_type) const
359 {
360 std::ostringstream oss;
361
362 switch (type_) {
363 case query_type::select:
364 oss << "SELECT ";
365 if (select_columns_.empty()) {
366 oss << "*";
367 } else {
368 for (size_t i = 0; i < select_columns_.size(); ++i) {
369 if (i > 0) oss << ", ";
370 oss << escape_identifier(select_columns_[i], db_type);
371 }
372 }
373 if (!from_table_.empty()) {
374 oss << " FROM " << escape_identifier(from_table_, db_type);
375 }
376 break;
377
378 case query_type::insert:
379 oss << "INSERT INTO " << escape_identifier(target_table_, db_type);
380 if (!insert_rows_.empty()) {
381 auto& first_row = insert_rows_[0];
382 oss << " (";
383 bool first = true;
384 for (const auto& pair : first_row) {
385 if (!first) oss << ", ";
386 oss << escape_identifier(pair.first, db_type);
387 first = false;
388 }
389 oss << ") VALUES ";
390 for (size_t i = 0; i < insert_rows_.size(); ++i) {
391 if (i > 0) oss << ", ";
392 oss << "(";
393 bool first_val = true;
394 for (const auto& pair : insert_rows_[i]) {
395 if (!first_val) oss << ", ";
396 oss << format_value(pair.second, db_type);
397 first_val = false;
398 }
399 oss << ")";
400 }
401 } else if (!set_data_.empty()) {
402 oss << " (";
403 bool first = true;
404 for (const auto& pair : set_data_) {
405 if (!first) oss << ", ";
406 oss << escape_identifier(pair.first, db_type);
407 first = false;
408 }
409 oss << ") VALUES (";
410 first = true;
411 for (const auto& pair : set_data_) {
412 if (!first) oss << ", ";
413 oss << format_value(pair.second, db_type);
414 first = false;
415 }
416 oss << ")";
417 }
418 break;
419
420 case query_type::update:
421 oss << "UPDATE " << escape_identifier(target_table_, db_type) << " SET ";
422 {
423 bool first = true;
424 for (const auto& pair : set_data_) {
425 if (!first) oss << ", ";
426 oss << escape_identifier(pair.first, db_type) << " = " << format_value(pair.second, db_type);
427 first = false;
428 }
429 }
430 break;
431
432 case query_type::delete_query:
433 oss << "DELETE FROM " << escape_identifier(target_table_, db_type);
434 break;
435
436 default:
437 throw std::runtime_error("Invalid query type");
438 }
439
440 // Add JOINs
441 for (const auto& join : joins_) {
442 oss << " " << join;
443 }
444
445 // Add WHERE clause
446 if (!where_conditions_.empty()) {
447 oss << " WHERE ";
448 for (size_t i = 0; i < where_conditions_.size(); ++i) {
449 if (i > 0) oss << " AND ";
450 oss << where_conditions_[i].to_sql();
451 }
452 }
453
454 // Add GROUP BY
455 if (!group_by_columns_.empty()) {
456 oss << " GROUP BY ";
457 for (size_t i = 0; i < group_by_columns_.size(); ++i) {
458 if (i > 0) oss << ", ";
459 oss << escape_identifier(group_by_columns_[i], db_type);
460 }
461 }
462
463 // Add HAVING
464 if (!having_clause_.empty()) {
465 oss << " HAVING " << having_clause_;
466 }
467
468 // Add ORDER BY
469 if (!order_by_clauses_.empty()) {
470 oss << " ORDER BY ";
471 for (size_t i = 0; i < order_by_clauses_.size(); ++i) {
472 if (i > 0) oss << ", ";
473 oss << order_by_clauses_[i];
474 }
475 }
476
477 // Add LIMIT and OFFSET
478 if (limit_count_ > 0) {
479 oss << " LIMIT " << limit_count_;
480 }
481 if (offset_count_ > 0) {
482 oss << " OFFSET " << offset_count_;
483 }
484
485 return oss.str();
486 }

505 std::string sql_query_builder::escape_identifier(const std::string& identifier, database_types db_type) const
506 {
507 switch (db_type) {
508 case database_types::mysql:
509 return "`" + identifier + "`";
510 case database_types::postgres:
511 return "\"" + identifier + "\"";
512 case database_types::sqlite:
513 return "[" + identifier + "]";
514 default:
515 return identifier;
516 }
517 }

519 std::string sql_query_builder::format_value(const database_value& value, database_types db_type) const
520 {
521 std::ostringstream oss;
522 std::visit([&oss](const auto& val) {
523 using T = std::decay_t<decltype(val)>;
524 if constexpr (std::is_same_v<T, std::string>) {
525 oss << "'" << val << "'";
526 } else if constexpr (std::is_same_v<T, int64_t>) {
527 oss << val;
528 } else if constexpr (std::is_same_v<T, double>) {
529 oss << val;
530 } else if constexpr (std::is_same_v<T, bool>) {
531 oss << (val ? "TRUE" : "FALSE");
532 } else if constexpr (std::is_same_v<T, std::monostate>) {
533 oss << "NULL";
534 }
535 }, value);
536 return oss.str();
537 }

701 std::string mongodb_query_builder::build() const
702 {
703 std::ostringstream oss;
704
705 switch (type_) {
706 case operation_type::find:
707 oss << "db." << collection_name_ << ".find(";
708 oss << to_json(filter_);
709 if (!projection_.empty()) {
710 oss << ", " << to_json(projection_);
711 }
712 oss << ")";
713 if (!sort_spec_.empty()) {
714 oss << ".sort({";
715 bool first = true;
716 for (const auto& pair : sort_spec_) {
717 if (!first) oss << ", ";
718 oss << "\"" << pair.first << "\": " << pair.second;
719 first = false;
720 }
721 oss << "})";
722 }
723 if (skip_count_ > 0) {
724 oss << ".skip(" << skip_count_ << ")";
725 }
726 if (limit_count_ > 0) {
727 oss << ".limit(" << limit_count_ << ")";
728 }
729 break;
730
731 case operation_type::insert:
732 if (documents_.empty()) {
733 oss << "db." << collection_name_ << ".insertOne(" << to_json(document_) << ")";
734 } else {
735 oss << "db." << collection_name_ << ".insertMany([";
736 for (size_t i = 0; i < documents_.size(); ++i) {
737 if (i > 0) oss << ", ";
738 oss << to_json(documents_[i]);
739 }
740 oss << "])";
741 }
742 break;
743
744 case operation_type::update:
745 oss << "db." << collection_name_ << ".updateOne(";
746 oss << to_json(filter_) << ", { \"$set\": " << to_json(update_spec_) << " })";
747 break;
748
749 case operation_type::delete_op:
750 if (limit_count_ == 1) {
751 oss << "db." << collection_name_ << ".deleteOne(" << to_json(filter_) << ")";
752 } else {
753 oss << "db." << collection_name_ << ".deleteMany(" << to_json(filter_) << ")";
754 }
755 break;
756
757 case operation_type::aggregate:

811 std::string mongodb_query_builder::value_to_json(const database_value& value) const
812 {
813 std::ostringstream oss;
814 std::visit([&oss](const auto& val) {
815 using T = std::decay_t<decltype(val)>;
816 if constexpr (std::is_same_v<T, std::string>) {
817 oss << "\"" << val << "\"";
818 } else if constexpr (std::is_same_v<T, int64_t>) {
819 oss << val;
820 } else if constexpr (std::is_same_v<T, double>) {
821 oss << val;
822 } else if constexpr (std::is_same_v<T, bool>) {
823 oss << (val ? "true" : "false");
824 } else if constexpr (std::is_same_v<T, std::monostate>) {
825 oss << "null";
826 }
827 }, value);
828 return oss.str();
829 }
