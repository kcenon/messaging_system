38 template <typename job_type>
39 adaptive_typed_job_queue_t<job_type>::adaptive_typed_job_queue_t(
40 queue_strategy initial_strategy)
41 : legacy_queue_(std::make_unique<typed_job_queue_t<job_type>>())
42 , lockfree_queue_(std::make_unique<typed_lockfree_job_queue_t<job_type>>())
43 , strategy_(initial_strategy)
44 {
45 initialize_strategy();
46
47 if (strategy_ == queue_strategy::ADAPTIVE)
48 {
49 start_performance_monitor();
50 }
51
52 metrics_.last_evaluation = std::chrono::steady_clock::now();
53 }

61 template <typename job_type>
62 auto adaptive_typed_job_queue_t<job_type>::initialize_strategy() -> void
63 {
64 switch (strategy_)
65 {
66 case queue_strategy::FORCE_LEGACY:
67 current_type_.store(queue_type::LEGACY_MUTEX);
68 break;
69
70 case queue_strategy::FORCE_LOCKFREE:
71 current_type_.store(queue_type::LOCKFREE);
72 break;
73
74 case queue_strategy::AUTO_DETECT:
75 case queue_strategy::ADAPTIVE:
76 {
77 // Start with mutex-based and let it adapt
78 current_type_.store(queue_type::LEGACY_MUTEX);
79 break;
80 }
81 }
82 }

84 template <typename job_type>
85 auto adaptive_typed_job_queue_t<job_type>::start_performance_monitor() -> void
86 {
87 stop_monitor_.store(false);
88 monitor_thread_ = std::make_unique<std::thread>(&adaptive_typed_job_queue_t::monitor_performance, this);
89 }

91 template <typename job_type>
92 auto adaptive_typed_job_queue_t<job_type>::stop_performance_monitor() -> void
93 {
94 stop_monitor_.store(true);
95 if (monitor_thread_ && monitor_thread_->joinable())
96 {
97 monitor_thread_->join();
98 }

101 template <typename job_type>
102 auto adaptive_typed_job_queue_t<job_type>::monitor_performance() -> void
103 {
104 while (!stop_monitor_.load())
105 {
106 std::this_thread::sleep_for(EVALUATION_INTERVAL);
107
108 if (stop_monitor_.load()) break;
109
110 evaluate_and_switch();
111 }
112 }

114 template <typename job_type>
115 auto adaptive_typed_job_queue_t<job_type>::evaluate_and_switch() -> void
116 {
117 auto metrics = get_metrics();
118
119 if (metrics.operation_count < MIN_OPERATIONS_FOR_SWITCH)
120 {
121 return; // Not enough data
122 }
123
124 if (current_type_.load() == queue_type::LEGACY_MUTEX && should_switch_to_lockfree())
125 {
126 migrate_to_lockfree();
127 }
128 else if (current_type_.load() == queue_type::LOCKFREE && should_switch_to_legacy())
129 {
130 migrate_to_legacy();
131 }
132
133 // Reset metrics after evaluation
134 metrics_.operation_count.store(0);
135 metrics_.total_latency_ns.store(0);
136 metrics_.contention_count.store(0);
137 metrics_.last_evaluation = std::chrono::steady_clock::now();
138 }

140 template <typename job_type>
141 auto adaptive_typed_job_queue_t<job_type>::should_switch_to_lockfree() const -> bool
142 {
143 auto metrics = get_metrics();
144 auto avg_latency = metrics.get_average_latency_ns();
145 auto contention_ratio = metrics.get_contention_ratio();
146
147 return (contention_ratio > CONTENTION_THRESHOLD_HIGH ||
148 avg_latency > LATENCY_THRESHOLD_HIGH_NS);
149 }

151 template <typename job_type>
152 auto adaptive_typed_job_queue_t<job_type>::should_switch_to_legacy() const -> bool
153 {
154 auto metrics = get_metrics();
155 auto avg_latency = metrics.get_average_latency_ns();
156 auto contention_ratio = metrics.get_contention_ratio();
157
158 return (contention_ratio < CONTENTION_THRESHOLD_LOW &&
159 avg_latency < LATENCY_THRESHOLD_LOW_NS);
160 }

162 template <typename job_type>
163 auto adaptive_typed_job_queue_t<job_type>::migrate_to_lockfree() -> void
164 {
165 // Migration is complex and should be done carefully
166 // For now, we just switch without migrating existing items
167 current_type_.store(queue_type::LOCKFREE);
168 metrics_.switch_count.fetch_add(1, std::memory_order_relaxed);
169 }

171 template <typename job_type>
172 auto adaptive_typed_job_queue_t<job_type>::migrate_to_legacy() -> void
173 {
174 // Migration is complex and should be done carefully
175 // For now, we just switch without migrating existing items
176 current_type_.store(queue_type::LEGACY_MUTEX);
177 metrics_.switch_count.fetch_add(1, std::memory_order_relaxed);
178 }

180 template <typename job_type>
181 auto adaptive_typed_job_queue_t<job_type>::update_metrics(
182 std::chrono::nanoseconds duration, bool had_contention) -> void
183 {
184 metrics_.operation_count.fetch_add(1, std::memory_order_relaxed);
185 metrics_.total_latency_ns.fetch_add(duration.count(), std::memory_order_relaxed);
186 if (had_contention)
187 {
188 metrics_.contention_count.fetch_add(1, std::memory_order_relaxed);
189 }
190 }

205 template <typename job_type>
206 auto adaptive_typed_job_queue_t<job_type>::enqueue(std::unique_ptr<thread_module::job>&& value)
207 -> thread_module::result_void
208 {
209 auto start = std::chrono::high_resolution_clock::now();
210 thread_module::result_void result;
211
212 if (current_type_.load() == queue_type::LOCKFREE)
213 {
214 result = lockfree_queue_->enqueue(std::move(value));
215 }
216 else
217 {
218 result = legacy_queue_->enqueue(std::move(value));
219 }
220
221 auto duration = std::chrono::high_resolution_clock::now() - start;
222 update_metrics(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
223
224 return result;
225 }

343 template <typename job_type>
344 auto adaptive_typed_job_queue_t<job_type>::size(const std::vector<job_type>& types) const -> std::size_t
345 {
346 if (current_type_.load() == queue_type::LOCKFREE)
347 {
348 // Lock-free queue doesn't support size with types, sum individual sizes
349 std::size_t total = 0;
350 for (const auto& type : types)
351 {
352 total += lockfree_queue_->size(type);
353 }
354 return total;
355 }
356 else
357 {
358 // Legacy queue's size() returns total size, not filtered by types
359 // This is an approximation - we can't filter by types in legacy queue
360 return legacy_queue_->size();
361 }
