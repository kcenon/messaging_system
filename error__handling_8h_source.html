100inline std::string error_code_to_string(error_code code) {
101 switch (code) {
102 case error_code::success: return "Success";
103 case error_code::unknown_error: return "Unknown error";
104 case error_code::operation_canceled: return "Operation canceled";
105 case error_code::operation_timeout: return "Operation timed out";
106 case error_code::not_implemented: return "Not implemented";
107 case error_code::invalid_argument: return "Invalid argument";
108
109 case error_code::thread_already_running: return "Thread is already running";
110 case error_code::thread_not_running: return "Thread is not running";
111 case error_code::thread_start_failure: return "Failed to start thread";
112 case error_code::thread_join_failure: return "Failed to join thread";
113
114 case error_code::queue_full: return "Queue is full";
115 case error_code::queue_empty: return "Queue is empty";
116 case error_code::queue_stopped: return "Queue is stopped";
117
118 case error_code::job_creation_failed: return "Failed to create job";
119 case error_code::job_execution_failed: return "Failed to execute job";
120 case error_code::job_invalid: return "Invalid job";
121
122 case error_code::resource_allocation_failed: return "Failed to allocate resource";
123 case error_code::resource_limit_reached: return "Resource limit reached";
124
125 case error_code::mutex_error: return "Mutex error";
126 case error_code::deadlock_detected: return "Deadlock detected";
127 case error_code::condition_variable_error: return "Condition variable error";
128
129 case error_code::io_error: return "I/O error";
130 case error_code::file_not_found: return "File not found";
131
132 default: return "Unknown error code";
133 }
