222 auto thread_worker::do_work() -> result_void
223 {
224 // Validate that job queue is available for processing
225 if (job_queue_ == nullptr)
226 {
227 return error{error_code::resource_allocation_failed, "there is no job_queue"};
228 }
229
230 // Attempt to dequeue the next job from the shared queue
231 auto dequeue_result = job_queue_->dequeue();
232 if (!dequeue_result.has_value())
233 {
234 // Handle empty queue scenarios based on queue state
235 if (!job_queue_->is_stopped())
236 {
237 // Queue is empty but still active - this is an error condition
238 return error{error_code::queue_empty,
239 formatter::format("error dequeue job: {}", dequeue_result.get_error().to_string())};
240 }
241
242 // Queue is stopped and empty - normal shutdown condition
243 return result_void{};
244 }
245
246 // Extract the job from the result and validate it
247 auto current_job = std::move(dequeue_result.value());
248 if (current_job == nullptr)
249 {
250 return error{error_code::job_invalid, "error executing job: nullptr"};
251 }
252
253 // Initialize timing measurement if performance monitoring is enabled
254 std::optional<std::chrono::time_point<std::chrono::high_resolution_clock>>
255 started_time_point = std::nullopt;
256 if (use_time_tag_)
257 {
258 started_time_point = std::chrono::high_resolution_clock::now();
259 }
260
261 // Associate the job with its source queue for potential re-submission
262 current_job->set_job_queue(job_queue_);
263
264 // Execute the job's work method and capture the result
265 auto work_result = current_job->do_work();
266 if (work_result.has_error())
267 {
268 return error{error_code::job_execution_failed,
269 formatter::format("error executing job: {}", work_result.get_error().to_string())};
270 }
271
272 // Log successful job completion based on timing configuration
273 if (!started_time_point.has_value())
274 {
275 // Standard logging without timing information
276 context_.log(log_level::debug,
277 formatter::format("job executed successfully: {} on thread_worker",
278 current_job->get_name()));
279 }
280 else
281 {
282 // Enhanced logging with execution timing information
283 auto end_time = std::chrono::high_resolution_clock::now();
284 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
285 end_time - started_time_point.value()).count();
286
287 context_.log(log_level::debug,
288 formatter::format("job executed successfully: {} on thread_worker ({}ns)",
289 current_job->get_name(), duration));
290
291 // Update worker metrics if monitoring is available
292 if (context_.monitoring())
293 {
294 monitoring_interface::worker_metrics metrics;
295 metrics.jobs_processed = 1;
296 metrics.total_processing_time_ns = static_cast<std::uint64_t>(duration);
297 metrics.timestamp = std::chrono::steady_clock::now();
298 // Use proper worker ID instead of thread hash
299 context_.update_worker_metrics(worker_id_, metrics);
300 }
301 }
302
303 return result_void{};
304 }
