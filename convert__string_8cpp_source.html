89 auto convert_string::convert(const FromType& value,
90 const std::string& from_encoding,
91 const std::string& to_encoding)
92 -> std::tuple<std::optional<ToType>, std::optional<std::string>>
93 {
94 iconv_t cd = iconv_open(to_encoding.c_str(), from_encoding.c_str());
95 if (cd == (iconv_t)-1)
96 {
97 return { std::nullopt, "iconv_open failed: " + std::string(strerror(errno)) };
98 }
99
100 std::vector<char> in_buf = string_to_vector(value);
101 char* in_ptr = in_buf.data();
102 size_t in_bytes_left = in_buf.size();
103
104 size_t out_buf_size = in_bytes_left * 2;
105 std::vector<char> out_buf(out_buf_size);
106 char* out_ptr = out_buf.data();
107 size_t out_bytes_left = out_buf.size();
108
109 size_t result = iconv(cd, &in_ptr, &in_bytes_left, &out_ptr, &out_bytes_left);
110 while (result == (size_t)-1 && errno == E2BIG)
111 {
112 size_t used = out_buf_size - out_bytes_left;
113 out_buf_size *= 2;
114 out_buf.resize(out_buf_size);
115 out_ptr = out_buf.data() + used;
116 out_bytes_left = out_buf_size - used;
117 result = iconv(cd, &in_ptr, &in_bytes_left, &out_ptr, &out_bytes_left);
118 }
119
120 if (result == (size_t)-1)
121 {
122 std::string error_msg = "iconv failed: " + std::string(strerror(errno));
123 iconv_close(cd);
124 return { std::nullopt, error_msg };
125 }
126
127 iconv_close(cd);
128
129 size_t converted_size = out_buf_size - out_bytes_left;
130 ToType converted_string(reinterpret_cast<typename ToType::value_type*>(out_buf.data()),
131 converted_size / sizeof(typename ToType::value_type));
132
133 return { converted_string, std::nullopt };
134 }

196 auto convert_string::get_encoding_name(encoding_types encoding,
197 endian_types endian) -> std::string
198 {
199 switch (encoding)
200 {
201 case encoding_types::utf8:
202 return "UTF-8";
203 case encoding_types::utf16:
204 if (endian == endian_types::little)
205 return "UTF-16LE";
206 else if (endian == endian_types::big)
207 return "UTF-16BE";
208 else
209 return "UTF-16";
210 case encoding_types::utf32:
211 if (endian == endian_types::little)
212 return "UTF-32LE";
213 else if (endian == endian_types::big)
214 return "UTF-32BE";
215 else
216 return "UTF-32";
217 default:
218 throw std::runtime_error("Unknown encoding");
219 }
220 }

238 auto convert_string::detect_endian(const std::u16string& str) -> endian_types
239 {
240 if (str.empty())
241 return endian_types::unknown;
242
243 if (str[0] == 0xFEFF)
244 return endian_types::big;
245 if (str[0] == 0xFFFE)
246 return endian_types::little;
247
248 size_t sample_size = std::min<size_t>(str.size(), 1000);
249 int le_count = 0, be_count = 0;
250 for (size_t i = 0; i < sample_size; ++i)
251 {
252 uint16_t ch = str[i];
253 if ((ch & 0xFF00) == 0 && (ch & 0x00FF) != 0)
254 ++le_count;
255 if ((ch & 0x00FF) == 0 && (ch & 0xFF00) != 0)
256 ++be_count;
257 }
258
259 if (le_count > be_count)
260 return endian_types::little;
261 if (be_count > le_count)
262 return endian_types::big;
263
264 return endian_types::unknown;
265 }

297 {
298 return str.length() >= 3 && static_cast<unsigned char>(str[0]) == 0xEF
299 && static_cast<unsigned char>(str[1]) == 0xBB

357 auto convert_string::split(const std::string& source, const std::string& token)
358 -> std::tuple<std::optional<std::vector<std::string>>, std::optional<std::string>>
359 {
360 if (token.empty())
361 {
362 return { std::vector{ source }, std::nullopt };
363 }
364
365 std::vector<std::string> result;
366 size_t start_pos = 0;
367 size_t end_pos = source.find(token);
368
369 while (end_pos != std::string::npos)
370 {
371 result.emplace_back(source.substr(start_pos, end_pos - start_pos));
372 start_pos = end_pos + token.length();
373 end_pos = source.find(token, start_pos);
374 }
375
376 result.emplace_back(source.substr(start_pos));
377
378 return { result, std::nullopt };
379 }

438 auto convert_string::replace2(const std::string& source,
439 const std::string& token,
440 const std::string& target)
441 -> std::tuple<std::optional<std::string>, std::optional<std::string>>
442 {
443 if (source.empty())
444 {
445 return { std::nullopt, "Source string is empty" };
446 }
447
448 if (token.empty())
449 {
450 return { std::nullopt, "Token string is empty" };
451 }
452
453 std::string result;
454
455 size_t last_offset = 0;
456 for (size_t offset = source.find(token, last_offset); offset != std::string::npos;
457 last_offset = offset + token.size(), offset = source.find(token, last_offset))
458 {
459 formatter::format_to(std::back_inserter(result), "{}{}",
460 source.substr(last_offset, offset - last_offset), target);
461 }
462
463 formatter::format_to(std::back_inserter(result), "{}", source.substr(last_offset));
464
465 return { result, std::nullopt };
466 }

468 auto convert_string::base64_encode(const std::vector<uint8_t>& data) -> std::string
469 {
470 static const char base64_chars[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
471 "abcdefghijklmnopqrstuvwxyz"
472 "0123456789+/";
473
474 std::string encoded_string;
475 size_t i = 0;
476 uint32_t octet_a, octet_b, octet_c;
477 uint32_t triple;
478
479 while (i < data.size())
480 {
481 octet_a = i < data.size() ? data[i++] : 0;
482 octet_b = i < data.size() ? data[i++] : 0;
483 octet_c = i < data.size() ? data[i++] : 0;
484
485 triple = (octet_a << 16) + (octet_b << 8) + octet_c;
486
487 encoded_string += base64_chars[(triple >> 18) & 0x3F];
488 encoded_string += base64_chars[(triple >> 12) & 0x3F];
489 encoded_string += base64_chars[(triple >> 6) & 0x3F];
490 encoded_string += base64_chars[triple & 0x3F];
491 }
492
493 int mod_table[] = { 0, 2, 1 };
494 for (int j = 0; j < mod_table[data.size() % 3]; j++)
495 {
496 encoded_string[encoded_string.size() - 1 - static_cast<size_t>(j)] = '=';
497 }
498
499 return encoded_string;
500 }

502 auto convert_string::base64_decode(const std::string& base64_str)
503 -> std::tuple<std::vector<uint8_t>, std::optional<std::string>>
504 {
505 static const std::string base64_chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
506 "abcdefghijklmnopqrstuvwxyz"
507 "0123456789+/";
508
509 if (base64_str.length() % 4 != 0)
510 {
511 return { std::vector<uint8_t>(), "Invalid base64 input length" };
512 }
513
514 size_t padding = 0;
515 if (!base64_str.empty())
516 {
517 if (base64_str[base64_str.length() - 1] == '=')
518 padding++;
519 if (base64_str.length() >= 2 && base64_str[base64_str.length() - 2] == '=')
520 padding++;
521 if (padding > 2)
522 {
523 return { std::vector<uint8_t>(), "Invalid padding in base64 string" };
524 }
525 }
526
527 size_t decoded_length = (base64_str.length() / 4) * 3 - padding;
528 std::vector<uint8_t> decoded_data;
529 decoded_data.reserve(decoded_length);
530
531 std::vector<int> decoding_table(256, -1);
532 for (int i = 0; i < 64; i++)
533 {
534 decoding_table[static_cast<unsigned char>(base64_chars[static_cast<size_t>(i)])] = i;
535 }
536
537 uint32_t buffer = 0;
538 int bits_collected = 0;
539 size_t i = 0;
540 for (; i < base64_str.length(); ++i)
541 {
542 char c = base64_str[i];
543 if (c == '=')
544 {
545 if (i < base64_str.length() - padding)
546 {
547 return { std::vector<uint8_t>(), "Invalid padding position in base64 string" };
548 }
549 break;
550 }
551
552 if (decoding_table[static_cast<unsigned char>(c)] == -1)
553 {
554 return { std::vector<uint8_t>(), "Invalid character in base64 string" };
555 }
556
557 buffer = (buffer << 6) | static_cast<uint32_t>(decoding_table[static_cast<unsigned char>(c)]);
558 bits_collected += 6;
559
560 if (bits_collected >= 8)
561 {
562 bits_collected -= 8;
563 decoded_data.push_back((buffer >> bits_collected) & 0xFF);
564 }
565 }
566
567 for (; i < base64_str.length(); ++i)
568 {
569 if (base64_str[i] != '=')
570 {
571 return { std::vector<uint8_t>(),
572 "Invalid character after padding in base64 string" };
573 }
574 }
575
576 return { decoded_data, std::nullopt };
577 }
