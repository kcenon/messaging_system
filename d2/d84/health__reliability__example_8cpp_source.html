342
343 std::cout << "Retry policy configured:" << std::endl;
344 std::cout << " Max attempts: " << config.max_attempts << std::endl;
345 std::cout << " Strategy: exponential backoff" << std::endl;
346 std::cout << " Initial delay: 100ms" << std::endl;
347
348 // Simulate manual retry logic (since retry_policy class not available)
349 std::cout << "\n1. Executing flaky operation with manual retry:" << std::endl;
350
351 std::atomic<int> attempt_count{0};
352 auto flaky_operation = [&attempt_count]() -> result<std::string> {
353 attempt_count++;
354 std::cout << " Attempt " << attempt_count << "..." << std::endl;
355
356 // Fail first 2 attempts
357 if (attempt_count <= 2) {
358 return make_error<std::string>(
359 monitoring_error_code::operation_timeout,
360 "Operation timed out"
361 );
362 }
363
364 return make_success<std::string>("Operation succeeded!");
365 };
366
367 result<std::string> final_result = make_error<std::string>(
368 monitoring_error_code::operation_failed, "Initialization pending");
369 for (int i = 0; i < static_cast<int>(config.max_attempts); ++i) {
370 final_result = flaky_operation();
371 if (final_result) {
372 break;
373 }
374
375 // Wait before retry
376 if (i < static_cast<int>(config.max_attempts) - 1) {
377 auto delay = config.initial_delay * static_cast<int>(std::pow(config.backoff_multiplier, i));
378 std::this_thread::sleep_for(delay);
379 }
380 }
381
382 if (final_result) {
383 std::cout << " Final result: SUCCESS - " << final_result.value() << std::endl;
384 } else {
385 std::cout << " Final result: FAILED - " << final_result.get_error().message << std::endl;
386 }
387
388 std::cout << " Total attempts: " << attempt_count << std::endl;
389}
