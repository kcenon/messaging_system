84 result<monitoring_data> collect_metrics() const override {
85 if (!enabled_.load()) {
86 return make_logger_error<monitoring_data>(error_code::metrics_not_available,
87 "Monitoring is disabled");
88 }
89
90 monitoring_data data;
91
92 // Core metrics
93 data.add_metric("messages_logged",
94 static_cast<double>(messages_logged_.load()),
95 metric_type::counter);
96 data.add_metric("messages_dropped",
97 static_cast<double>(messages_dropped_.load()),
98 metric_type::counter);
99 data.add_metric("errors_encountered",
100 static_cast<double>(errors_encountered_.load()),
101 metric_type::counter);
102 data.add_metric("writers_failed",
103 static_cast<double>(writers_failed_.load()),
104 metric_type::counter);
105
106 // Performance metrics
107 uint64_t total_time = total_processing_time_us_.load();
108 uint64_t messages = messages_logged_.load();
109 if (messages > 0) {
110 data.add_metric("avg_processing_time_us",
111 static_cast<double>(total_time) / messages,
112 metric_type::gauge);
113 }
114 data.add_metric("max_processing_time_us",
115 static_cast<double>(max_processing_time_us_.load()),
116 metric_type::gauge);
117
118 uint64_t min_time = min_processing_time_us_.load();
119 if (min_time != UINT64_MAX) {
120 data.add_metric("min_processing_time_us",
121 static_cast<double>(min_time),
122 metric_type::gauge);
123 }
124
125 // Resource metrics
126 data.add_metric("buffer_usage_bytes",
127 static_cast<double>(buffer_usage_bytes_.load()),
128 metric_type::gauge);
129 data.add_metric("max_buffer_usage_bytes",
130 static_cast<double>(max_buffer_usage_bytes_.load()),
131 metric_type::gauge);
132 data.add_metric("queue_size",
133 static_cast<double>(queue_size_.load()),
134 metric_type::gauge);
135 data.add_metric("max_queue_size",
136 static_cast<double>(max_queue_size_.load()),
137 metric_type::gauge);
138
139 // Uptime
140 auto now = std::chrono::system_clock::now();
141 auto uptime = std::chrono::duration_cast<std::chrono::seconds>(
142 now - start_time_).count();
143 data.add_metric("uptime_seconds",
144 static_cast<double>(uptime),
145 metric_type::gauge);
146
147 // Add custom counters and gauges
148 {
149 std::lock_guard<std::mutex> lock(mutex_);
150
151 for (const auto& [name, value] : counters_) {
152 data.add_metric(name, value.load(), metric_type::counter);
153 }
154
155 for (const auto& [name, value] : gauges_) {
156 data.add_metric(name, value.load(), metric_type::gauge);
157 }
158
159 // Calculate histogram summaries
160 for (const auto& [name, values] : histograms_) {
161 if (!values.empty()) {
162 double sum = 0;
163 double min_val = values[0];
164 double max_val = values[0];
165
166 for (double v : values) {
167 sum += v;
168 min_val = std::min(min_val, v);
169 max_val = std::max(max_val, v);
170 }
171
172 double avg = sum / values.size();
173 data.add_metric(name + "_avg", avg, metric_type::summary);
174 data.add_metric(name + "_min", min_val, metric_type::summary);
175 data.add_metric(name + "_max", max_val, metric_type::summary);
176 data.add_metric(name + "_count",
177 static_cast<double>(values.size()),
178 metric_type::summary);
179 }
180 }
181 }
182
183 return data;
184 }

190 result<health_check_result> check_health() const override {
191 health_check_result result;
192
193 // Check error rate
194 uint64_t errors = errors_encountered_.load();
195 uint64_t messages = messages_logged_.load();
196 if (messages > 0) {
197 double error_rate = static_cast<double>(errors) / messages;
198 if (error_rate > 0.1) { // More than 10% error rate
199 result.add_issue("High error rate: " +
200 std::to_string(error_rate * 100) + "%");
201 result.set_status(health_status::unhealthy);
202 } else if (error_rate > 0.05) { // More than 5% error rate
203 result.add_issue("Elevated error rate: " +
204 std::to_string(error_rate * 100) + "%");
205 result.set_status(health_status::degraded);
206 }
207 }
208
209 // Check dropped messages
210 uint64_t dropped = messages_dropped_.load();
211 if (dropped > 0) {
212 if (messages > 0) {
213 double drop_rate = static_cast<double>(dropped) / messages;
214 if (drop_rate > 0.01) { // More than 1% drop rate
215 result.add_issue("Messages being dropped: " +
216 std::to_string(dropped) + " total");
217 if (result.get_status() == health_status::healthy) {
218 result.set_status(health_status::degraded);
219 }
220 }
221 }
222 }
223
224 // Check writer failures
225 uint64_t writer_failures = writers_failed_.load();
226 if (writer_failures > 0) {
227 result.add_issue("Writer failures detected: " +
228 std::to_string(writer_failures));
229 if (result.get_status() == health_status::healthy) {
230 result.set_status(health_status::degraded);
231 }
232 }
233
234 // Check queue size
235 size_t queue_size = queue_size_.load();
236 size_t max_queue = max_queue_size_.load();
237 if (max_queue > 0 && queue_size > max_queue * 0.9) {
238 result.add_issue("Queue near capacity: " +
239 std::to_string(queue_size) + "/" +
240 std::to_string(max_queue));
241 if (result.get_status() == health_status::healthy) {
242 result.set_status(health_status::degraded);
243 }
244 }
245
246 // Add any custom health issues
247 {
248 std::lock_guard<std::mutex> lock(mutex_);
249 for (const auto& issue : health_issues_) {
250 result.add_issue(issue);
251 }
252 }
253
254 // Set overall message
255 if (result.is_healthy()) {
256 result.set_message("All systems operational");
257 } else {
258 result.set_message("Issues detected - check details");
259 }
260
261 return result;
262 }

349 void update_gauge(const std::string& name, double value) override {
350 if (!enabled_.load()) return;
351
352 if (name == "buffer_usage_bytes") {
353 buffer_usage_bytes_ = static_cast<size_t>(value);
354 size_t max_val = max_buffer_usage_bytes_.load();
355 while (value > max_val &&
356 !max_buffer_usage_bytes_.compare_exchange_weak(max_val,
357 static_cast<size_t>(value))) {
358 // Loop until successful
359 }
360 } else if (name == "queue_size") {
361 queue_size_ = static_cast<size_t>(value);
362 size_t max_val = max_queue_size_.load();
363 while (value > max_val &&
364 !max_queue_size_.compare_exchange_weak(max_val,
365 static_cast<size_t>(value))) {
366 // Loop until successful
367 }
368 } else {
369 std::lock_guard<std::mutex> lock(mutex_);
370 gauges_[name] = value;
371 }

379 void record_histogram(const std::string& name, double value) override {
380 if (!enabled_.load()) return;
381
382 if (name == "processing_time_us") {
383 uint64_t us_value = static_cast<uint64_t>(value);
384 total_processing_time_us_ += us_value;
385
386 // Update max
387 uint64_t max_val = max_processing_time_us_.load();
388 while (us_value > max_val &&
389 !max_processing_time_us_.compare_exchange_weak(max_val, us_value)) {
390 // Loop until successful
391 }
392
393 // Update min
394 uint64_t min_val = min_processing_time_us_.load();
395 while (us_value < min_val &&
396 !min_processing_time_us_.compare_exchange_weak(min_val, us_value)) {
397 // Loop until successful
398 }
399 } else {
400 std::lock_guard<std::mutex> lock(mutex_);
401 histograms_[name].push_back(value);
402
403 // Limit histogram size to prevent unbounded growth
404 if (histograms_[name].size() > 10000) {
405 // Keep only recent values
406 histograms_[name].erase(histograms_[name].begin(),
407 histograms_[name].begin() + 5000);
408 }
409 }
