192 void process_messages() {
193 while (running_) {
194 std::unique_lock<std::mutex> lock(queue_mutex_);
195
196 // Wait for messages or stop signal
197 queue_cv_.wait(lock, [this]() {
198 return !message_queue_.empty() || !running_;
199 });
200
201 // Process all available messages
202 while (!message_queue_.empty()) {
203 auto msg = std::move(message_queue_.front());
204 message_queue_.pop();
205
206 // Unlock while writing
207 lock.unlock();
208 wrapped_writer_->write(msg.level, msg.message, msg.file, msg.line, msg.function, msg.timestamp);
209 lock.lock();
210 }
211
212 // Notify flush waiters
213 flush_cv_.notify_all();
214 }
215 }

223 auto msg = std::move(message_queue_.front());
224 message_queue_.pop();
225 wrapped_writer_->write(msg.level, msg.message, msg.file, msg.line, msg.function, msg.timestamp);
226 }
227 wrapped_writer_->flush();
228 }
