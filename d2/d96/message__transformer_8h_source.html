37 class topic_rewriter : public message_transformer {
38 private:
39 std::string pattern_;
40 std::string replacement_;
41
42 public:
43 topic_rewriter(const std::string& pattern, const std::string& replacement)
44 : pattern_(pattern), replacement_(replacement) {}
45
46 transform_result transform(const transformation_context& context,
47 core::message& message) const override {
48 try {
49 std::regex regex_pattern(pattern_);
50 message.payload.topic = std::regex_replace(message.payload.topic,
51 regex_pattern, replacement_);
52 return transform_result::success;
53 } catch (const std::exception&) {
54 return transform_result::error;
55 }
56 }
57
58 std::string get_name() const override { return "TopicRewriter"; }

61 }

155 // Transformation pipeline
156 class transformation_pipeline {
157 private:
158 std::vector<std::unique_ptr<message_transformer>> transformers_;
159 std::string name_;
160 mutable std::atomic<uint64_t> messages_processed_{0};
161 mutable std::atomic<uint64_t> messages_transformed_{0};
162 mutable std::atomic<uint64_t> messages_skipped_{0};
163 mutable std::atomic<uint64_t> messages_errored_{0};
164
165 public:
166 explicit transformation_pipeline(const std::string& name) : name_(name) {}
167
168 void add_transformer(std::unique_ptr<message_transformer> transformer) {
169 transformers_.push_back(std::move(transformer));
170 }
171
172 const std::string& get_name() const { return name_; }
173
174 transform_result process_message(core::message& message) const {
175 messages_processed_++;
176
177 transformation_context context{message};
178 bool any_transformation = false;
179
180 for (const auto& transformer : transformers_) {
181 context.transformation_count++;
182 auto result = transformer->transform(context, message);
183
184 switch (result) {
185 case transform_result::success:
186 any_transformation = true;
187 break;
188 case transform_result::skip:
189 messages_skipped_++;
190 return transform_result::skip;
191 case transform_result::error:
192 messages_errored_++;
193 return transform_result::error;
194 }
195 }
196
197 if (any_transformation) {
198 messages_transformed_++;
199 }
200
201 return transform_result::success;
202 }

239 class message_relay {
240 private:
241 std::unordered_map<std::string, std::unique_ptr<transformation_pipeline>> pipelines_;
242 std::function<void(const core::message&)> output_handler_;
243 mutable std::mutex relay_mutex_;
244
245 public:
246 void register_pipeline(const std::string& name,
247 std::unique_ptr<transformation_pipeline> pipeline) {
248 std::lock_guard<std::mutex> lock(relay_mutex_);
249 pipelines_[name] = std::move(pipeline);
250 }
251
252 transformation_pipeline* get_pipeline(const std::string& name) const {
253 std::lock_guard<std::mutex> lock(relay_mutex_);
254 auto it = pipelines_.find(name);
255 return it != pipelines_.end() ? it->second.get() : nullptr;
256 }
257
258 void set_output_handler(std::function<void(const core::message&)> handler) {
259 output_handler_ = std::move(handler);
260 }
261
262 bool relay_message(const std::string& pipeline_name, core::message message) const {
263 auto pipeline = get_pipeline(pipeline_name);
264 if (!pipeline) return false;
265
266 auto result = pipeline->process_message(message);
267
268 if (result == transform_result::success && output_handler_) {
269 output_handler_(message);
270 return true;
271 }
272
273 return result != transform_result::error;
274 }
275
276 std::vector<std::string> get_pipeline_names() const {
277 std::lock_guard<std::mutex> lock(relay_mutex_);
278 std::vector<std::string> names;
279 for (const auto& [name, pipeline] : pipelines_) {
280 names.push_back(name);
281 }
282 return names;
283 }
284
285 std::vector<transformation_pipeline::pipeline_statistics> get_all_statistics() const {
286 std::lock_guard<std::mutex> lock(relay_mutex_);
287 std::vector<transformation_pipeline::pipeline_statistics> all_stats;
288
289 for (const auto& [name, pipeline] : pipelines_) {
290 all_stats.push_back(pipeline->get_statistics());
