34struct ring_buffer_config {
35 size_t capacity = 8192; // Default capacity (power of 2)
36 bool overwrite_old = true; // Overwrite oldest data when full
37 size_t batch_size = 64; // Batch size for bulk operations
38 std::chrono::milliseconds gc_interval{1000}; // Garbage collection interval
39
43 result_void validate() const {
44 if (capacity == 0 || (capacity & (capacity - 1)) != 0) {
45 return result_void(monitoring_error_code::invalid_configuration,
46 "Capacity must be a power of 2");
47 }
48
49 if (batch_size == 0 || batch_size > capacity) {
50 return result_void(monitoring_error_code::invalid_configuration,
51 "Invalid batch size");
52 }
53
54 return result_void::success();
55 }
56};

82 double get_write_success_rate() const {
83 auto total = total_writes.load();
84 auto failed = failed_writes.load();
85 return total > 0 ? (1.0 - static_cast<double>(failed) / total) * 100.0 : 100.0;

103template<typename T>
104class ring_buffer {
105private:
106 static_assert(std::is_move_constructible_v<T>, "T must be move constructible");
107 static_assert(std::is_move_assignable_v<T>, "T must be move assignable");
108
109 alignas(64) std::atomic<size_t> write_index_{0}; // Cache line aligned
110 alignas(64) std::atomic<size_t> read_index_{0}; // Cache line aligned
111
112 std::unique_ptr<T[]> buffer_;
113 ring_buffer_config config_;
114 mutable ring_buffer_stats stats_;
115
119 size_t get_mask() const noexcept {
120 return config_.capacity - 1;
121 }
122
126 bool is_full_unsafe(size_t write_idx, size_t read_idx) const noexcept {
127 return ((write_idx + 1) & get_mask()) == read_idx;
128 }
129
133 bool is_empty_unsafe(size_t write_idx, size_t read_idx) const noexcept {
134 return write_idx == read_idx;
135 }
136
137public:
141 explicit ring_buffer(const ring_buffer_config& config = {})
142 : buffer_(std::make_unique<T[]>(config.capacity))
143 , config_(config) {
144
145 // Validate configuration
146 auto validation = config_.validate();
147 if (!validation) {
148 throw std::invalid_argument("Invalid ring buffer configuration: " +
149 validation.get_error().message);
150 }
151 }
152
153 // Non-copyable but moveable
154 ring_buffer(const ring_buffer&) = delete;
155 ring_buffer& operator=(const ring_buffer&) = delete;
156 ring_buffer(ring_buffer&&) = default;
157 ring_buffer& operator=(ring_buffer&&) = default;

164 result_void write(T&& item) {
165 stats_.total_writes.fetch_add(1, std::memory_order_relaxed);
166
167 size_t current_write = write_index_.load(std::memory_order_acquire);
168 size_t current_read = read_index_.load(std::memory_order_acquire);
169
170 if (is_full_unsafe(current_write, current_read)) {
171 if (config_.overwrite_old) {
172 // Advance read index to overwrite oldest
173 size_t new_read = (current_read + 1) & get_mask();
174 read_index_.compare_exchange_weak(current_read, new_read,
175 std::memory_order_acq_rel);
176 stats_.overwrites.fetch_add(1, std::memory_order_relaxed);
177 } else {
178 stats_.failed_writes.fetch_add(1, std::memory_order_relaxed);
179 return result_void(monitoring_error_code::storage_full,
180 "Ring buffer is full");
181 }
182 }
183
184 // Write the item
185 buffer_[current_write] = std::move(item);
186
187 // Update write index
188 size_t new_write = (current_write + 1) & get_mask();
189 write_index_.store(new_write, std::memory_order_release);
190
191 return result_void::success();
192 }
193
199 size_t write_batch(std::vector<T>&& items) {
200 if (items.empty()) {
201 return 0;
202 }
203
204 size_t written = 0;
205 for (auto& item : items) {
206 auto result = write(std::move(item));
207 if (result) {
208 ++written;
209 } else if (!config_.overwrite_old) {
210 break; // Stop on first failure if not overwriting
211 }
212 }
213
214 return written;
215 }
216
222 result_void read(T& item) {
223 stats_.total_reads.fetch_add(1, std::memory_order_relaxed);
224
225 size_t current_read = read_index_.load(std::memory_order_acquire);
226 size_t current_write = write_index_.load(std::memory_order_acquire);
227
228 if (is_empty_unsafe(current_write, current_read)) {
229 stats_.failed_reads.fetch_add(1, std::memory_order_relaxed);
230 return result_void(monitoring_error_code::collection_failed,
231 "Ring buffer is empty");
232 }
233
234 // Read the item
235 item = std::move(buffer_[current_read]);
236
237 // Update read index
238 size_t new_read = (current_read + 1) & get_mask();
239 read_index_.store(new_read, std::memory_order_release);
240
241 return result_void::success();
242 }
243
250 size_t read_batch(std::vector<T>& items, size_t max_count = SIZE_MAX) {
251 if (max_count == 0) {
252 return 0;
253 }
254
255 size_t batch_size = std::min(max_count, config_.batch_size);
256 items.reserve(items.size() + batch_size);
257
258 size_t read_count = 0;
259 T temp_item;
260
261 while (read_count < batch_size) {
262 auto result = read(temp_item);
263 if (!result) {
264 break; // No more items to read
265 }
266
267 items.emplace_back(std::move(temp_item));
268 ++read_count;
269 }
270
271 return read_count;
272 }
273
279 result_void peek(T& item) const {
280 size_t current_read = read_index_.load(std::memory_order_acquire);
281 size_t current_write = write_index_.load(std::memory_order_acquire);
282
283 if (is_empty_unsafe(current_write, current_read)) {
284 return result_void(monitoring_error_code::collection_failed,
285 "Ring buffer is empty");
286 }
287
288 item = buffer_[current_read]; // Copy, don't move
289 return result_void::success();
290 }

295 size_t size() const noexcept {
296 size_t write_idx = write_index_.load(std::memory_order_acquire);
297 size_t read_idx = read_index_.load(std::memory_order_acquire);
298
299 if (write_idx >= read_idx) {
300 return write_idx - read_idx;
301 } else {
302 return config_.capacity - (read_idx - write_idx);
303 }
304 }
305
309 bool empty() const noexcept {
310 return size() == 0;
311 }
312
316 bool full() const noexcept {
317 size_t write_idx = write_index_.load(std::memory_order_acquire);
318 size_t read_idx = read_index_.load(std::memory_order_acquire);
319 return is_full_unsafe(write_idx, read_idx);
320 }
321
325 size_t capacity() const noexcept {
326 return config_.capacity;
327 }
328
332 void clear() noexcept {
333 write_index_.store(0, std::memory_order_release);
334 read_index_.store(0, std::memory_order_release);
335 }

358 stats_.failed_writes.store(0);
359 stats_.failed_reads.store(0);
360 stats_.creation_time = std::chrono::system_clock::now();
361 }
