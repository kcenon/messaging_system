297
298 health_status get_overall_status() const {
299 health_status overall = health_status::healthy;
300 for (const auto& result : results_) {
301 if (result.status == health_status::unhealthy) {
302 return health_status::unhealthy;
303 }
304 if (result.status == health_status::degraded) {
305 overall = health_status::degraded;
306 }
307 if (result.status == health_status::unknown && overall == health_status::healthy) {
308 overall = health_status::unknown;
309 }
310 }
311 return overall;
312 }

314private:
