175auto_scaler::scaling_decision auto_scaler::decide(
176 const monitoring_interface::metrics_snapshot& current_metrics) {
177
178 scaling_decision decision;
179
180 // Ïø®Îã§Ïö¥ ÌôïÏù∏
181 if (is_in_cooldown()) {
182 decision.recommended_action = scaling_decision::action::none;
183 decision.reason = "In cooldown period";
184 return decision;
185 }
186
187 // Î©îÌä∏Î¶≠ ÌèâÌôúÌôî
188 double cpu_load = current_metrics.system.cpu_usage_percent;
189 double memory_load = (current_metrics.system.memory_usage_bytes / (4.0 * 1024 * 1024 * 1024)) * 100;
190 update_smoothed_metrics(cpu_load, memory_load);
191
192 // Ïä§ÏºÄÏùºÎßÅ Í≤∞Ï†ï
193 if (state_.smoothed_cpu_load > policy_.cpu_threshold_up ||
194 state_.smoothed_memory_load > policy_.memory_threshold_up) {
195
196 decision.recommended_action = scaling_decision::action::scale_up;
197 decision.confidence = std::max(
198 (state_.smoothed_cpu_load - policy_.cpu_threshold_up) / 20.0,
199 (state_.smoothed_memory_load - policy_.memory_threshold_up) / 20.0);
200 decision.confidence = std::min(1.0, std::max(0.0, decision.confidence));
201 decision.recommended_resources = static_cast<std::size_t>(
202 state_.current_resources * policy_.scale_factor);
203 decision.reason = "High resource utilization";
204
205 } else if (state_.smoothed_cpu_load < policy_.cpu_threshold_down &&
206 state_.smoothed_memory_load < policy_.memory_threshold_down) {
207
208 decision.recommended_action = scaling_decision::action::scale_down;
209 decision.confidence = std::max(
210 (policy_.cpu_threshold_down - state_.smoothed_cpu_load) / 20.0,
211 (policy_.memory_threshold_down - state_.smoothed_memory_load) / 20.0);
212 decision.confidence = std::min(1.0, std::max(0.0, decision.confidence));
213 decision.recommended_resources = std::max<std::size_t>(1,
214 static_cast<std::size_t>(state_.current_resources / policy_.scale_factor));
215 decision.reason = "Low resource utilization";
216 }
217
218 // Í≤∞Ï†ï Í∏∞Î°ù
219 if (decision.recommended_action != scaling_decision::action::none) {
220 record_decision(decision);
221 last_scale_time_ = std::chrono::steady_clock::now();
222 }
223
224 return decision;
225}

227auto_scaler::scaling_decision auto_scaler::decide_predictive(
228 double predicted_load, std::chrono::seconds time_horizon) {
229
230 scaling_decision decision;
231
232 // ÏòàÏ∏° Í∏∞Î∞ò Ïä§ÏºÄÏùºÎßÅ
233 if (predicted_load > policy_.cpu_threshold_up) {
234 decision.recommended_action = scaling_decision::action::scale_up;
235 decision.confidence = (predicted_load - policy_.cpu_threshold_up) / 20.0;
236 decision.reason = "Predicted high load in " +
237 std::to_string(time_horizon.count()) + " seconds";
238 } else if (predicted_load < policy_.cpu_threshold_down) {
239 decision.recommended_action = scaling_decision::action::scale_down;
240 decision.confidence = (policy_.cpu_threshold_down - predicted_load) / 20.0;
241 decision.reason = "Predicted low load in " +
242 std::to_string(time_horizon.count()) + " seconds";
243 }
244
245 decision.confidence = std::min(1.0, std::max(0.0, decision.confidence));
246
247 return decision;
248}
