47 template<typename Mutex>
48 class scoped_lock_guard {
49 public:
54 explicit scoped_lock_guard(Mutex& mutex)
55 : mutex_(mutex), locked_(true) {
56 mutex_.lock();
57 }
58
64 template<typename Rep, typename Period>
65 scoped_lock_guard(Mutex& mutex, const std::chrono::duration<Rep, Period>& timeout)
66 : mutex_(mutex), locked_(false) {
67 if constexpr (std::is_same_v<Mutex, std::timed_mutex> ||
68 std::is_same_v<Mutex, std::recursive_timed_mutex>) {
69 locked_ = mutex_.try_lock_for(timeout);
70 } else {
71 // Fallback for mutexes without timeout support
72 locked_ = mutex_.try_lock();
73 }
74 }
75
79 ~scoped_lock_guard() {
80 if (locked_) {
81 mutex_.unlock();
82 }
83 }
84
85 // Non-copyable and non-movable
86 scoped_lock_guard(const scoped_lock_guard&) = delete;
87 scoped_lock_guard& operator=(const scoped_lock_guard&) = delete;
88 scoped_lock_guard(scoped_lock_guard&&) = delete;
89 scoped_lock_guard& operator=(scoped_lock_guard&&) = delete;
90
95 [[nodiscard]] bool owns_lock() const noexcept {
96 return locked_;
97 }
98
102 void unlock() {
103 if (locked_) {
104 mutex_.unlock();
105 locked_ = false;
106 }
107 }
108
109 private:
110 Mutex& mutex_;
111 bool locked_;
112 };
