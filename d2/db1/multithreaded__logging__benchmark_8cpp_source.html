27static void BM_MultithreadedLogging(benchmark::State& state) {
28 auto logger_instance = std::make_unique<logger>(true);
29 logger_instance->add_writer(std::make_unique<null_writer>());
30 logger_instance->start();
31
32 const int num_threads = state.range(0);
33 const int messages_per_thread = state.range(1);
34
35 for (auto _ : state) {
36 std::vector<std::thread> threads;
37
38 for (int t = 0; t < num_threads; ++t) {
39 threads.emplace_back([&logger_instance, messages_per_thread, t]() {
40 for (int i = 0; i < messages_per_thread; ++i) {
41 logger_instance->log(thread_module::log_level::info,
42 "Thread " + std::to_string(t) + " message " + std::to_string(i));
43 }
44 });
45 }
46
47 for (auto& thread : threads) {
48 thread.join();
49 }
50
51 logger_instance->flush();
52 }
53
54 logger_instance->stop();
55 state.SetItemsProcessed(state.iterations() * num_threads * messages_per_thread);
56}
