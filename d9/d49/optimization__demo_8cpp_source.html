32class load_simulator {
33public:
34 enum class load_pattern {
35 steady, // ì¼ì •í•œ ë¶€í•˜
36 spiky, // ìŠ¤íŒŒì´í¬ íŒ¨í„´
37 gradual, // ì ì§„ì  ì¦ê°€/ê°ì†Œ
38 random // ëœë¤ íŒ¨í„´
39 };
40
41 load_simulator(load_pattern pattern = load_pattern::steady)
42 : pattern_(pattern), cycle_(0) {}
43
44 double get_cpu_load() {
45 switch (pattern_) {
46 case load_pattern::steady:
47 return 50.0 + (std::sin(cycle_ * 0.05) * 10);
48
49 case load_pattern::spiky:
50 return (cycle_ % 100 < 10) ? 90.0 : 30.0;
51
52 case load_pattern::gradual:
53 return 20.0 + (cycle_ % 200) * 0.3;
54
55 case load_pattern::random:
56 default:
57 static std::random_device rd;
58 static std::mt19937 gen(rd());
59 static std::uniform_real_distribution<> dis(20.0, 80.0);
60 return dis(gen);
61 }
62 }
63
64 std::uint64_t get_memory_usage() {
65 double base = 500 * 1024 * 1024; // 500MB base
66 double factor = 1.0 + (get_cpu_load() / 100.0);
67 return static_cast<std::uint64_t>(base * factor);
68 }
69
70 void advance() { cycle_++; }
71
72private:
73 load_pattern pattern_;
74 int cycle_;
75};
