41// Null writer for performance testing (doesn't actually output)
42class null_writer : public base_writer {
43public:
44 void write(thread_module::log_level level,
45 const std::string& message,
46 const std::string& file,
47 int line,
48 const std::string& function,
49 const std::chrono::system_clock::time_point& timestamp) override {

59// Global logger instances for benchmarks
60static std::unique_ptr<logger> sync_logger;
61static std::unique_ptr<logger> async_logger;

63// Setup function
64static void setup_loggers() {
65 sync_logger = std::make_unique<logger>(false); // synchronous
66 async_logger = std::make_unique<logger>(true); // asynchronous
67
68 sync_logger->add_writer(std::make_unique<null_writer>());
69 async_logger->add_writer(std::make_unique<null_writer>());
70
71 async_logger->start();
72}
73
74// Cleanup function
75static void cleanup_loggers() {
76 if (async_logger) {
77 async_logger->stop();
78 }
79}

81// Benchmark synchronous logging
82static void BM_SyncLogging(benchmark::State& state) {
83 if (!sync_logger) setup_loggers();
84
85 const std::string message = "Benchmark test message for synchronous logging";
86
87 for (auto _ : state) {
88 sync_logger->log(thread_module::log_level::info, message);
89 }
90
91 state.SetItemsProcessed(state.iterations());
92}

126// Benchmark logging different levels
127static void BM_LoggingLevels(benchmark::State& state) {
128 if (!sync_logger) setup_loggers();
129
130 const std::string message = "Level test message";
131 const auto levels = {
132 thread_module::log_level::trace,
133 thread_module::log_level::debug,
134 thread_module::log_level::info,
135 thread_module::log_level::warning,
136 thread_module::log_level::error,
137 thread_module::log_level::critical
138 };
139
140 int level_idx = 0;
141
142 for (auto _ : state) {
143 auto level = *(levels.begin() + (level_idx % levels.size()));
144 sync_logger->log(level, message);
145 level_idx++;
146 }
147
148 state.SetItemsProcessed(state.iterations());
149}
