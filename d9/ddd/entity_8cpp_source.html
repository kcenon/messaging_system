56 std::string field_metadata::to_sql_definition() const
57 {
58 std::ostringstream oss;
59 oss << name_ << " ";
60
61 // Map C++ types to SQL types
62 if (type_name_ == "int32_t" || type_name_ == "int") {
63 oss << "INTEGER";
64 } else if (type_name_ == "int64_t") {
65 oss << "BIGINT";
66 } else if (type_name_ == "double") {
67 oss << "DOUBLE PRECISION";
68 } else if (type_name_ == "std::string") {
69 oss << "VARCHAR(255)";
70 } else if (type_name_ == "bool") {
71 oss << "BOOLEAN";
72 } else if (type_name_.find("time_point") != std::string::npos) {
73 oss << "TIMESTAMP";
74 } else {
75 oss << "TEXT";
76 }
77
78 // Add constraints
79 if (is_primary_key()) {
80 oss << " PRIMARY KEY";
81 }
82 if (is_auto_increment()) {
83 oss << " AUTO_INCREMENT";
84 }
85 if (is_not_null() && !is_primary_key()) {
86 oss << " NOT NULL";
87 }
88 if (is_unique() && !is_primary_key()) {
89 oss << " UNIQUE";
90 }
91 if (has_default_now()) {
92 oss << " DEFAULT CURRENT_TIMESTAMP";
93 }
94
95 return oss.str();
96 }

140 std::string entity_metadata::create_table_sql() const
141 {
142 std::ostringstream oss;
143 oss << "CREATE TABLE IF NOT EXISTS " << table_name_ << " (\n";
144
145 bool first = true;
146 for (const auto& field : fields_) {
147 if (!first) oss << ",\n";
148 oss << " " << field.to_sql_definition();
149 first = false;
150 }
151
152 // Add foreign key constraints
153 auto foreign_keys = get_foreign_keys();
154 for (const auto* fk : foreign_keys) {
155 oss << ",\n FOREIGN KEY (" << fk->name() << ") REFERENCES "
156 << fk->foreign_table() << "(" << fk->foreign_field() << ")";
157 }
158
159 oss << "\n)";
160 return oss.str();
161 }

163 std::string entity_metadata::create_indexes_sql() const
164 {
165 std::ostringstream oss;
166 auto indexes = get_indexes();
167
168 for (const auto* index : indexes) {
169 if (!index->index_name().empty()) {
170 oss << "CREATE INDEX IF NOT EXISTS " << index->index_name()
171 << " ON " << table_name_ << "(" << index->name() << ");\n";
172 } else {
173 oss << "CREATE INDEX IF NOT EXISTS idx_" << table_name_
174 << "_" << index->name() << " ON " << table_name_
175 << "(" << index->name() << ");\n";
176 }
177 }
178
179 return oss.str();
180 }

189 bool entity_manager::create_tables(std::shared_ptr<database_base> db)
190 {
191 if (!db) {
192 std::cerr << "Database connection is null" << std::endl;
193 return false;
194 }
195
196 try {
197 for (const auto& [name, metadata] : metadata_cache_) {
198 // Create table
199 std::string create_sql = metadata->create_table_sql();
200 if (!db->execute_query(create_sql)) {
201 std::cerr << "Failed to create table: " << name << std::endl;
202 return false;
203 }
204
205 // Create indexes
206 std::string index_sql = metadata->create_indexes_sql();
207 if (!index_sql.empty() && !db->execute_query(index_sql)) {
208 std::cerr << "Failed to create indexes for table: " << name << std::endl;
209 return false;
210 }
211 }
212 return true;
213 } catch (const std::exception& e) {
214 std::cerr << "Exception during table creation: " << e.what() << std::endl;
215 return false;
216 }
217 }
