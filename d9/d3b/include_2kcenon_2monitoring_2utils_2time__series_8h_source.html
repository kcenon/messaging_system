36struct time_series_config {

46 result_void validate() const {

57 if (max_points == 0) {

63 }

70struct time_point_data {
71 std::chrono::system_clock::time_point timestamp;
72 double value;
73 uint32_t sample_count; // Number of samples aggregated into this point
74
75 time_point_data() noexcept : value(0.0), sample_count(0) {}
76
77 time_point_data(std::chrono::system_clock::time_point ts, double val, uint32_t count = 1) noexcept
78 : timestamp(ts), value(val), sample_count(count) {}
79
83 void merge(const time_point_data& other) {
84 if (sample_count == 0) {
85 *this = other;
86 } else if (other.sample_count > 0) {
87 double total_weight = sample_count + other.sample_count;
88 value = (value * sample_count + other.value * other.sample_count) / total_weight;
89 sample_count += other.sample_count;
90
91 // Use the later timestamp
92 if (other.timestamp > timestamp) {
93 timestamp = other.timestamp;
94 }
95 }
96 }
97
101 bool is_valid(std::chrono::system_clock::time_point cutoff) const noexcept {
102 return timestamp >= cutoff;
103 }

148struct aggregation_result {
149 std::vector<time_point_data> points;
150 std::chrono::system_clock::time_point query_start;
151 std::chrono::system_clock::time_point query_end;
152 size_t total_samples;
153
154 aggregation_result() : total_samples(0) {}
155
159 summary_data get_summary() const {
160 summary_data summary;
161 for (const auto& point : points) {
162 summary.add_sample(point.value);
163 }
164 return summary;
165 }
166
170 double get_average() const {
171 if (points.empty()) return 0.0;
172
173 double sum = 0.0;
174 uint64_t total_weight = 0;
175
176 for (const auto& point : points) {
177 sum += point.value * point.sample_count;
178 total_weight += point.sample_count;
179 }
180
181 return total_weight > 0 ? sum / total_weight : 0.0;
182 }
183
187 double get_rate() const {
188 if (points.size() < 2) return 0.0;
189
190 const auto& first = points.front();
191 const auto& last = points.back();
192
193 auto duration = std::chrono::duration_cast<std::chrono::seconds>(
194 last.timestamp - first.timestamp);
195
196 if (duration.count() <= 0) return 0.0;
197
198 return (last.value - first.value) / duration.count();
199 }

216 void cleanup_old_data() {
217 auto cutoff = std::chrono::system_clock::now() - config_.retention_period;
218
219 auto it = std::remove_if(data_.begin(), data_.end(),
220 [cutoff](const time_point_data& point) {
221 return !point.is_valid(cutoff);
222 });
223
224 data_.erase(it, data_.end());
225 }
226
230 void compress_data() {
231 if (!config_.enable_compression || data_.size() < 3) {
232 return;
233 }
234
235 // Simple compression: remove points that don't add significant information
236 std::vector<time_point_data> compressed;
237 compressed.reserve(data_.size());
238
239 if (!data_.empty()) {
240 compressed.push_back(data_[0]);
241
242 for (size_t i = 1; i < data_.size() - 1; ++i) {
243 const auto& prev = data_[i - 1];
244 const auto& curr = data_[i];
245 const auto& next = data_[i + 1];
246
247 // Check if current point is significantly different from linear interpolation
248 double expected = prev.value + (next.value - prev.value) *
249 (std::chrono::duration<double>(curr.timestamp - prev.timestamp).count() /
250 std::chrono::duration<double>(next.timestamp - prev.timestamp).count());
251
252 if (std::abs(curr.value - expected) > config_.compression_threshold) {
253 compressed.push_back(curr);
254 }
255 }
256
257 if (data_.size() > 1) {
258 compressed.push_back(data_.back());
259 }
260 }
261
262 data_ = std::move(compressed);
263 }
264
268 void enforce_size_limit() {
269 if (data_.size() > config_.max_points) {
270 size_t remove_count = data_.size() - config_.max_points;
271 data_.erase(data_.begin(), data_.begin() + remove_count);
272 }
273 }

344 result<aggregation_result> query(const time_series_query& query) const {
345 auto validation = query.validate();
346 if (!validation) {
347 return make_error<aggregation_result>(monitoring_error_code::invalid_argument,
348 validation.get_error().message);
349 }
350
351 std::lock_guard<std::mutex> lock(mutex_);
352
353 aggregation_result result;
354 result.query_start = query.start_time;
355 result.query_end = query.end_time;
356
357 // Find data points in the time range
358 auto start_it = std::lower_bound(data_.begin(), data_.end(), query.start_time,
359 [](const time_point_data& point, std::chrono::system_clock::time_point time) {
360 return point.timestamp < time;
361 });
362
363 auto end_it = std::upper_bound(data_.begin(), data_.end(), query.end_time,
364 [](std::chrono::system_clock::time_point time, const time_point_data& point) {
365 return time < point.timestamp;
366 });
367
368 if (start_it == end_it) {
369 return make_success(std::move(result)); // No data in range
370 }
371
372 // Aggregate data by step size
373 auto current_step_start = query.start_time;
374
375 while (current_step_start < query.end_time) {
376 auto current_step_end = current_step_start + query.step;
377 if (current_step_end > query.end_time) {
378 current_step_end = query.end_time;
379 }
380
381 // Find points in this step
382 std::vector<time_point_data> step_points;
383 for (auto it = start_it; it != end_it; ++it) {
384 if (it->timestamp >= current_step_start && it->timestamp < current_step_end) {
385 step_points.push_back(*it);
386 result.total_samples += it->sample_count;
387 }
388 }
389
390 // Aggregate points in this step
391 if (!step_points.empty()) {
392 time_point_data aggregated_point;
393 aggregated_point.timestamp = current_step_start + query.step / 2; // Middle of step
394
395 for (const auto& point : step_points) {
396 aggregated_point.merge(point);
397 }
398
399 result.points.push_back(aggregated_point);
400 }
401
402 current_step_start = current_step_end;
403 }
404
405 return make_success(std::move(result));
406 }
