306TEST_F(IntegrationTest, StressTest) {
307 // Create logger with multiple writers
308 auto logger = std::make_shared<kcenon::logger::logger>();
309 logger->start();
310 logger->add_writer(std::make_unique<file_writer>("test_integration.log"));
311 logger->enable_metrics_collection(true);
312
313#ifndef LOGGER_STANDALONE_MODE
314 // Add sanitizer
315 auto sanitizer = std::make_shared<log_sanitizer>();
316
317 // Create analyzer
318 auto analyzer = std::make_unique<log_analyzer>(
319 std::chrono::seconds(1), 60);
320#endif
321
322 // Multi-threaded logging
323 const int thread_count = 4;
324 const int messages_per_thread = 1000;
325 std::vector<std::thread> threads;
326
327 auto start_time = std::chrono::high_resolution_clock::now();
328
329 for (int t = 0; t < thread_count; ++t) {
330 threads.emplace_back([&, t]() {
331 for (int i = 0; i < messages_per_thread; ++i) {
332 std::string msg = "Thread " + std::to_string(t) +
333 " message " + std::to_string(i);
334
335#ifndef LOGGER_STANDALONE_MODE
336 // Add some sensitive data randomly
337 if (i % 10 == 0) {
338 msg += " card: 4111-1111-1111-1111";
339 }
340 if (i % 15 == 0) {
341 msg += " email: user@test.com";
342 }
343
344 // Sanitize and log
345 std::string sanitized = sanitizer->sanitize(msg);
346 logger->log(logger_system::log_level::info, sanitized);
347
348 // Analyze
349 analyzer->analyze(
350 logger_system::log_level::info,
351 sanitized,
352 __FILE__,
353 __LINE__,
354 __func__,
355 std::chrono::system_clock::now()
356 );
357#else
358 // Simple logging in standalone mode
359 logger->log(logger_system::log_level::info, msg);
360#endif
361 }
362 });
363 }
364
365 // Wait for all threads
366 for (auto& t : threads) {
367 t.join();
368 }
369
370 auto end_time = std::chrono::high_resolution_clock::now();
371 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(
372 end_time - start_time);
373
374 // Check performance
375 auto metrics_result = logger->get_current_metrics();
376 ASSERT_TRUE(metrics_result.has_value());
377 const auto& metrics = metrics_result.value();
378 EXPECT_EQ(metrics.messages_enqueued.load(), thread_count * messages_per_thread);
379
380 std::cout << "Stress test completed in " << duration.count() << " ms" << std::endl;
381 std::cout << "Messages per second: " << metrics.get_messages_per_second() << std::endl;
382 std::cout << "Average enqueue time: " << metrics.get_avg_enqueue_time_ns() << " ns" << std::endl;
383
384 logger->flush();
385 logger->stop();
386}
