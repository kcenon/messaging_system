45 // Memory pool for message objects
46 template<typename T>
47 class object_pool {
48 private:
49 std::queue<std::unique_ptr<T>> available_objects_;
50 std::mutex pool_mutex_;
51 size_t max_size_;
52 std::atomic<size_t> total_created_{0};
53 std::atomic<size_t> pool_hits_{0};
54
55 public:
56 explicit object_pool(size_t max_size = 1000) : max_size_(max_size) {}
57
58 std::unique_ptr<T> acquire() {
59 std::lock_guard<std::mutex> lock(pool_mutex_);
60 if (!available_objects_.empty()) {
61 auto obj = std::move(available_objects_.front());
62 available_objects_.pop();
63 pool_hits_++;
64 return obj;
65 }
66
67 // Create new object if pool is empty
68 total_created_++;
69 return std::make_unique<T>();
70 }
71
72 void release(std::unique_ptr<T> obj) {
73 if (!obj) return;
74
75 std::lock_guard<std::mutex> lock(pool_mutex_);
76 if (available_objects_.size() < max_size_) {
77 // Reset object to clean state
78 *obj = T{};
79 available_objects_.push(std::move(obj));
80 }
81 // If pool is full, object will be automatically destroyed
82 }
83
84 size_t size() const {
85 std::lock_guard<std::mutex> lock(pool_mutex_);
86 return available_objects_.size();
87 }
88
89 size_t total_created() const { return total_created_; }
90 size_t pool_hits() const { return pool_hits_; }
91 double hit_ratio() const {
92 auto hits = pool_hits_.load();
93 auto total = total_created_.load();
94 return total > 0 ? static_cast<double>(hits) / total : 0.0;
95 }
96 };

257 struct health_status {
258 bool is_healthy = true;
259 double average_latency_ms = 0.0;
260 uint64_t peak_queue_size = 0;
261 uint64_t memory_usage_mb = 0;
262 uint64_t messages_per_second = 0;

267 health_status get_health_status() const {
268 health_status status;
269 status.average_latency_ms = metrics_.average_latency_ms;
270 status.peak_queue_size = metrics_.peak_queue_size;
271 status.memory_usage_mb = metrics_.memory_usage_bytes / (1024 * 1024);
272 status.messages_per_second = metrics_.messages_per_second;
273
274 // Check health conditions
275 if (status.average_latency_ms > MAX_AVERAGE_LATENCY_MS) {
276 status.is_healthy = false;
277 status.errors.push_back("Average latency too high: " +
278 std::to_string(status.average_latency_ms) + "ms");
279 }
280
281 if (status.peak_queue_size > MAX_QUEUE_SIZE_THRESHOLD) {
282 status.warnings.push_back("Queue size approaching limit: " +
283 std::to_string(status.peak_queue_size));
284 }
285
286 if (status.memory_usage_mb > 1000) { // 1GB threshold
287 status.warnings.push_back("High memory usage: " +
288 std::to_string(status.memory_usage_mb) + "MB");
289 }
290
291 return status;
292 }

294 production_metrics get_metrics() const {

296 }

298 private:
299 void monitor_loop() {
300 auto last_check = std::chrono::steady_clock::now();
301 uint64_t last_message_count = 0;
302
303 while (running_) {
304 std::unique_lock<std::mutex> lock(monitor_mutex_);
305 if (monitor_cv_.wait_for(lock, check_interval_, [this] { return !running_; })) {
306 break;
307 }
308
309 // Calculate messages per second
310 auto now = std::chrono::steady_clock::now();
311 auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(now - last_check);
312
313 if (elapsed.count() > 0) {
314 auto current_count = metrics_.total_messages_processed.load();
315 auto messages_in_period = current_count - last_message_count;
316 metrics_.messages_per_second = messages_in_period / elapsed.count();
317
318 last_message_count = current_count;
319 last_check = now;
320 }
321
322 // Log health status periodically
323 auto health = get_health_status();
324 if (!health.is_healthy) {
325 // In a real production system, this would trigger alerts
326 // For now, we'll just record it
327 }
328 }
329 }
