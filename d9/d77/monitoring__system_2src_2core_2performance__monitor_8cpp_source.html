11result<bool> performance_profiler::record_sample(
12 const std::string& operation_name,
13 std::chrono::nanoseconds duration,
14 bool success) {
15
16 if (!enabled_) {

20 std::unique_lock<std::shared_mutex> lock(profiles_mutex_);
21
22 auto& profile = profiles_[operation_name];

24 profile = std::make_unique<profile_data>();

27 lock.unlock();
28
29 // Update counters
30 profile->call_count.fetch_add(1);
31 if (!success) {
32 profile->error_count.fetch_add(1);

34
35 // Record sample
36 std::lock_guard sample_lock(profile->mutex);
37
38 // Limit samples to prevent unbounded growth

49monitoring_system::result<monitoring_system::performance_metrics> monitoring_system::performance_profiler::get_metrics(
50 const std::string& operation_name) const {
51
52 std::shared_lock<std::shared_mutex> lock(profiles_mutex_);
53
54 auto it = profiles_.find(operation_name);
55 if (it == profiles_.end()) {
56 return monitoring_system::make_error<monitoring_system::performance_metrics>(
57 monitoring_system::monitoring_error_code::not_found,
58 "Operation not found: " + operation_name
59 );
60 }
61
62 const auto& profile = it->second;
63
64 std::lock_guard sample_lock(profile->mutex);
65
66 monitoring_system::performance_metrics metrics;
67 metrics.operation_name = operation_name;
68 metrics.call_count = profile->call_count.load();
69 metrics.error_count = profile->error_count.load();
70
71 if (!profile->samples.empty()) {
72 auto total = std::chrono::nanoseconds::zero();
73 auto min_sample = profile->samples[0];
74 auto max_sample = profile->samples[0];
75
76 for (const auto& sample : profile->samples) {
77 total += sample;
78 if (sample < min_sample) min_sample = sample;
79 if (sample > max_sample) max_sample = sample;
80 }
81
82 metrics.min_duration = min_sample;
83 metrics.max_duration = max_sample;
84 metrics.mean_duration = total / profile->samples.size();
85 metrics.call_count = profile->samples.size();
86 }
87
88 return result<performance_metrics>(metrics);
89}
