41 std::string sanitize_filename(const std::string& message_id) const {
42 std::string sanitized = message_id;
43 // Replace invalid characters with underscores
44 for (char& c : sanitized) {
45 if (c == '/' || c == '\\' || c == ':' || c == '*' ||
46 c == '?' || c == '"' || c == '<' || c == '>' || c == '|') {
47 c = '_';
48 }
49 }
50
51 if (sanitized.length() > MAX_FILENAME_LENGTH) {
52 sanitized = sanitized.substr(0, MAX_FILENAME_LENGTH);
53 }
54
55 return sanitized + ".msg";
56 }

58 std::string serialize_message(const core::message& msg) const {
59 // Simple JSON-like serialization
60 std::string result = "{\n";
61 result += " \"timestamp\": " + std::to_string(msg.metadata.timestamp.time_since_epoch().count()) + ",\n";
62 result += " \"priority\": " + std::to_string(static_cast<int>(msg.metadata.priority)) + ",\n";
63 result += " \"sender\": \"" + msg.metadata.sender + "\",\n";
64 result += " \"topic\": \"" + msg.payload.topic + "\",\n";
65 result += " \"data\": {\n";
66
67 for (auto it = msg.payload.data.begin(); it != msg.payload.data.end(); ++it) {
68 if (it != msg.payload.data.begin()) result += ",\n";
69 result += " \"" + it->first + "\": ";
70
71 std::visit([&result](const auto& value) {
72 using T = std::decay_t<decltype(value)>;
73 if constexpr (std::is_same_v<T, std::string>) {
74 result += "\"" + value + "\"";
75 } else if constexpr (std::is_same_v<T, int64_t>) {
76 result += std::to_string(value);
77 } else if constexpr (std::is_same_v<T, double>) {
78 result += std::to_string(value);
79 } else if constexpr (std::is_same_v<T, bool>) {
80 result += value ? "true" : "false";
81 }
82 }, it->second);
83 }
84
85 result += "\n }\n}";
86 return result;
87 }

230 class message_recovery_system {
231 private:

256 }
257
258 void enable_auto_recovery(std::chrono::minutes interval) {
259 recovery_interval_ = interval;
260
261 if (!auto_recovery_enabled_) {
262 auto_recovery_enabled_ = true;
263 recovery_thread_ = std::thread([this]() { auto_recovery_loop(); });
264 }
265 }
266
267 void stop_auto_recovery() {
268 if (auto_recovery_enabled_) {
269 auto_recovery_enabled_ = false;
270 recovery_cv_.notify_all();
271 if (recovery_thread_.joinable()) {
272 recovery_thread_.join();
273 }
274 }
275 }
276
277 bool recover_messages(const std::string& topic_filter = "") {
278 std::lock_guard<std::mutex> lock(recovery_mutex_);
279
280 if (state_ == recovery_state::recovering) {
281 return false; // Already recovering
282 }
283
284 state_ = recovery_state::scanning;
285 messages_recovered_ = 0;
286 recovery_errors_ = 0;
287
288 try {
289 auto message_ids = storage_->list_messages(topic_filter);
290 state_ = recovery_state::recovering;
291
292 for (const auto& id : message_ids) {
293 auto message = storage_->retrieve_message(id);
294 if (message && recovery_handler_) {
295 try {
296 recovery_handler_(*message);
297 messages_recovered_++;
298 } catch (const std::exception&) {
299 recovery_errors_++;
300 }
301 } else {
302 recovery_errors_++;
303 }
304 }
305
306 state_ = recovery_state::completed;
307 return true;
308 } catch (const std::exception&) {
309 state_ = recovery_state::failed;
310 return false;
311 }
312 }

340 private:
341 void auto_recovery_loop() {
342 while (auto_recovery_enabled_) {
343 std::unique_lock<std::mutex> lock(recovery_mutex_);
344 if (recovery_cv_.wait_for(lock, recovery_interval_,
345 [this] { return !auto_recovery_enabled_; })) {
346 break;
347 }
348
349 // Perform automatic recovery
350 if (state_ == recovery_state::idle || state_ == recovery_state::completed) {
351 lock.unlock();
352 recover_messages();
353 }
354 }

358 // Persistent message queue
359 class persistent_message_queue {
360 private:
361 std::unique_ptr<message_storage> storage_;
362 std::queue<std::string> pending_messages_;
363 mutable std::mutex queue_mutex_;
364 std::atomic<uint64_t> message_counter_{0};
365
366 std::string generate_message_id() {
367 auto now = std::chrono::system_clock::now();
368 auto timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(
369 now.time_since_epoch()).count();
370
371 return "msg_" + std::to_string(timestamp) + "_" + std::to_string(message_counter_++);
372 }
373
374 public:
375 explicit persistent_message_queue(std::unique_ptr<message_storage> storage)
376 : storage_(std::move(storage)) {}
377
378 std::string enqueue_message(const core::message& msg) {
379 std::lock_guard<std::mutex> lock(queue_mutex_);
380
381 auto message_id = generate_message_id();
382 if (storage_->store_message(message_id, msg)) {
383 pending_messages_.push(message_id);
384 return message_id;
385 }
386
387 return ""; // Failed to store
388 }
389
390 std::unique_ptr<core::message> dequeue_message() {
391 std::lock_guard<std::mutex> lock(queue_mutex_);
392
393 if (pending_messages_.empty()) {
394 return nullptr;
395 }
396
397 auto message_id = pending_messages_.front();
398 pending_messages_.pop();
399
400 auto message = storage_->retrieve_message(message_id);
401 // Keep message in storage until explicitly deleted
402
403 return message;
404 }
405
406 bool acknowledge_message(const std::string& message_id) {
407 return storage_->delete_message(message_id);
408 }
409
410 size_t size() const {
411 std::lock_guard<std::mutex> lock(queue_mutex_);
412 return pending_messages_.size();
413 }
414
415 void cleanup_old_messages(std::chrono::hours max_age) {
416 storage_->cleanup_old_messages(max_age);
417 }
418
419 std::string get_queue_info() const {
420 return "PersistentQueue: " + std::to_string(size()) + " pending, " +
421 storage_->get_storage_info();
422 }
