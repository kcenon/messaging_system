202TEST_F(MessagingIntegrationTest, ConcurrentProducersConsumers) {
203 constexpr int num_producers = 4;
204 constexpr int num_consumers = 3;
205 constexpr int messages_per_producer = 50;
206
207 std::atomic<int> total_consumed{0};
208 std::vector<std::atomic<int>> consumer_counts(num_consumers);
209
210 // Start consumers
211 std::vector<std::thread> consumer_threads;
212 for (int c = 0; c < num_consumers; ++c) {
213 consumer_threads.emplace_back([this, c, &consumer_counts, &total_consumed]() {
214 integrator_->subscribe("concurrent.test", [&, c](const message& msg) {
215 consumer_counts[c]++;
216 total_consumed++;
217 });
218 });
219 }
220
221 // Let consumers register
222 std::this_thread::sleep_for(std::chrono::milliseconds(100));
223
224 // Start producers
225 std::vector<std::thread> producer_threads;
226 for (int p = 0; p < num_producers; ++p) {
227 producer_threads.emplace_back([this, p, messages_per_producer]() {
228 for (int i = 0; i < messages_per_producer; ++i) {
229 message_payload payload;
230 payload.topic = "concurrent.test";
231 payload.data["producer_id"] = int64_t(p);
232 payload.data["message_id"] = int64_t(i);
233
234 integrator_->publish("concurrent.test", payload, "producer_" + std::to_string(p));
235 }
236 });
237 }
238
239 // Wait for all producers to finish
240 for (auto& thread : producer_threads) {
241 thread.join();
242 }
243
244 // Wait for message processing
245 auto start_time = std::chrono::steady_clock::now();
246 int expected_total = num_producers * messages_per_producer * num_consumers;
247
248 while (total_consumed.load() < expected_total) {
249 std::this_thread::sleep_for(std::chrono::milliseconds(10));
250
251 auto elapsed = std::chrono::steady_clock::now() - start_time;
252 if (elapsed > std::chrono::seconds(5)) {
253 break; // Timeout
254 }
255 }
256
257 // Each consumer should receive all messages (fanout behavior)
258 EXPECT_EQ(total_consumed.load(), expected_total);
259
260 // Clean up consumer threads
261 for (auto& thread : consumer_threads) {
262 if (thread.joinable()) {
263 thread.detach(); // Detach since subscribers are still active
264 }
