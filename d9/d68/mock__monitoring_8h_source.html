82 std::vector<::monitoring_interface::metrics_snapshot> get_recent_snapshots(std::size_t count) const override {
83 std::lock_guard<std::mutex> lock(mutex_);
84
85 std::vector<::monitoring_interface::metrics_snapshot> result;
86 size_t n = std::min(count, history_.size());
87 result.reserve(n);
88
89 // Return newest first
90 for (auto it = history_.rbegin(); it != history_.rend() && result.size() < n; ++it) {
91 result.push_back(*it);
92 }
93
94 return result;
95 }

101 void start() {
102 if (!active_.exchange(true)) {
103 std::cout << "[MockMonitoring] Started" << std::endl;
104 collection_thread_ = std::thread(&mock_monitoring::collect_loop, this);
105 }
106 }
107
108 void stop() {
109 if (active_.exchange(false)) {
110 if (collection_thread_.joinable()) {
111 collection_thread_.join();
112 }
113 std::cout << "[MockMonitoring] Stopped" << std::endl;
114 }

125private:
126 void collect_loop() {
127 while (active_.load()) {
128 std::this_thread::sleep_for(std::chrono::milliseconds(500));
129
130 if (active_.load()) {
131 std::lock_guard<std::mutex> lock(mutex_);
132
133 // Store snapshot in history
134 history_.push_back(current_snapshot_);
135 if (history_.size() > max_history_) {
136 history_.pop_front();
137 }
138
139 total_collections_.fetch_add(1);
140 }
141 }
142 }
