55class cancellation_token {
56private:
57 // Private implementation to allow copying/moving the token
58 struct token_state {
59 std::atomic<bool> is_cancelled{false};

62 };

95 static cancellation_token create_linked(std::initializer_list<cancellation_token> tokens) {
96 auto new_token = create();
97 auto new_state_weak = std::weak_ptr<token_state>(new_token.state_);
98
99 for (const auto& token : tokens) {
100 auto token_copy = token;
101 token_copy.register_callback([new_state_weak]() {
102 if (auto state = new_state_weak.lock()) {
103 // Directly set the cancelled flag and invoke callbacks
104 std::vector<std::function<void()>> callbacks_to_invoke;
105
106 {
107 std::lock_guard<std::mutex> lock(state->callback_mutex);
108 bool was_cancelled = state->is_cancelled.exchange(true);
109 if (!was_cancelled) {
110 callbacks_to_invoke = std::move(state->callbacks);
111 state->callbacks.clear();
112 }
113 }
114
115 for (const auto& callback : callbacks_to_invoke) {
116 callback();
117 }
118 }
119 });
120 }
121
122 return new_token;
123 }
124
132 void cancel() {
133 std::vector<std::function<void()>> callbacks_to_invoke;
134
135 {
136 std::lock_guard<std::mutex> lock(state_->callback_mutex);
137 bool was_cancelled = state_->is_cancelled.exchange(true);
138 if (!was_cancelled) {
139 // Move callbacks to local vector to avoid holding lock during invocation
140 callbacks_to_invoke = std::move(state_->callbacks);
141 state_->callbacks.clear();
142 }
143 }
144
145 // Invoke callbacks outside the lock to prevent deadlock
146 for (const auto& callback : callbacks_to_invoke) {
147 callback();
148 }
149 }
150
155 [[nodiscard]] bool is_cancelled() const {
156 return state_->is_cancelled.load();
157 }
158
163 void throw_if_cancelled() const {
164 if (is_cancelled()) {
165 throw std::runtime_error("Operation cancelled");
166 }
167 }
168
177 void register_callback(std::function<void()> callback) {
178 std::unique_lock<std::mutex> lock(state_->callback_mutex);
179
180 // Check cancellation state while holding the lock
181 if (state_->is_cancelled.load()) {
182 lock.unlock();
183 callback();
184 return;
185 }
186
187 // Add callback while still holding the lock
188 state_->callbacks.push_back(std::move(callback));
189 }
190};
