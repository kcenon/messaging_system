142 static app_config load_from_file(const std::string& filename,
143 std::shared_ptr<kcenon::logger::logger> logger) {
144 app_config config;
145
146 std::ifstream file(filename);
147 if (!file.is_open()) {
148 logger->log(kcenon::logger::log_level::warning,
149 "Config file not found: " + filename + ". Using defaults.");
150 return config;
151 }
152
153 logger->log(kcenon::logger::log_level::info, "Loading configuration from: " + filename);
154
155 std::string line;
156 while (std::getline(file, line)) {
157 // Skip comments and empty lines
158 if (line.empty() || line[0] == '#') continue;
159
160 auto equals_pos = line.find('=');
161 if (equals_pos == std::string::npos) continue;
162
163 std::string key = line.substr(0, equals_pos);
164 std::string value = line.substr(equals_pos + 1);
165
166 // Trim whitespace
167 key.erase(0, key.find_first_not_of(" \t"));
168 key.erase(key.find_last_not_of(" \t") + 1);
169 value.erase(0, value.find_first_not_of(" \t"));
170 value.erase(value.find_last_not_of(" \t") + 1);
171
172 // Parse known configuration keys
173 if (key == "environment") {
174 config.environment = value;
175 } else if (key == "worker_threads") {
176 config.worker_threads = std::stoi(value);
177 } else if (key == "queue_size") {
178 config.queue_size = std::stoull(value);
179 } else if (key == "enable_compression") {
180 config.enable_compression = (value == "true" || value == "1");
181 } else if (key == "max_retries") {
182 config.max_retries = std::stoi(value);
183 } else if (key == "retry_delay_ms") {
184 config.retry_delay = std::chrono::milliseconds(std::stoi(value));
185 } else if (key == "health_check_interval_sec") {
186 config.health_check_interval = std::chrono::seconds(std::stoi(value));
187 } else if (key == "log_file") {
188 config.log_file = value;
189 } else if (key == "log_level") {
190 if (value == "debug") config.log_level = kcenon::logger::log_level::debug;
191 else if (value == "info") config.log_level = kcenon::logger::log_level::info;
192 else if (value == "warning") config.log_level = kcenon::logger::log_level::warning;
193 else if (value == "error") config.log_level = kcenon::logger::log_level::error;
194 else if (value == "critical") config.log_level = kcenon::logger::log_level::critical;
195 }
196 }
197
198 logger->log(kcenon::logger::log_level::info, "Configuration loaded successfully");
199 return config;
200 }

232class retry_handler {
233private:
234 std::shared_ptr<kcenon::logger::logger> m_logger;
235 metrics_collector* m_metrics;
236 int m_max_retries;
237 std::chrono::milliseconds m_retry_delay;
238
239public:
240 retry_handler(std::shared_ptr<kcenon::logger::logger> logger,
241 metrics_collector* metrics,
242 int max_retries = 3,
243 std::chrono::milliseconds retry_delay = 100ms)
244 : m_logger(logger)
245 , m_metrics(metrics)
246 , m_max_retries(max_retries)
247 , m_retry_delay(retry_delay) {}
248
249 template<typename Func>
250 bool execute_with_retry(Func&& func, const std::string& operation_name) {
251 for (int attempt = 0; attempt < m_max_retries; ++attempt) {
252 try {
253 func();
254 if (attempt > 0) {
255 m_logger->log(kcenon::logger::log_level::info,
256 "Operation '" + operation_name + "' succeeded after " +
257 std::to_string(attempt) + " retries");
258 if (m_metrics) {
259 m_metrics->increment_successful_retries();
260 }
261 }
262 return true;
263 } catch (const std::exception& e) {
264 if (m_metrics) {
265 m_metrics->increment_errors();
266 }
267
268 if (attempt < m_max_retries - 1) {
269 m_logger->log(kcenon::logger::log_level::warning,
270 "Operation '" + operation_name + "' failed (attempt " +
271 std::to_string(attempt + 1) + "/" + std::to_string(m_max_retries) +
272 "): " + e.what() + ". Retrying...");
273 if (m_metrics) {
274 m_metrics->increment_retries();
275 }
276 std::this_thread::sleep_for(m_retry_delay * (attempt + 1));
277 } else {
278 m_logger->log(kcenon::logger::log_level::error,
279 "Operation '" + operation_name + "' failed after " +
280 std::to_string(m_max_retries) + " attempts: " + e.what());
281 }
282 }
283 }
284 return false;
285 }
