236double thread_pool_analyzer::calculate_load_balance_score(
237 const std::vector<std::uint64_t>& worker_loads) {
238
239 if (worker_loads.empty()) return 0.0;
240 if (worker_loads.size() == 1) return 100.0;
241
242 // ÌèâÍ∑† Í≥ÑÏÇ∞
243 double mean = std::accumulate(worker_loads.begin(), worker_loads.end(), 0.0)
244 / worker_loads.size();
245
246 if (mean == 0) return 100.0; // Î™®Îëê 0Ïù¥Î©¥ ÏôÑÎ≤ΩÌïú Í∑†Ìòï
247
248 // ÌëúÏ§ÄÌé∏Ï∞® Í≥ÑÏÇ∞
249 double variance = 0.0;
250 for (auto load : worker_loads) {
251 double diff = load - mean;
252 variance += diff * diff;
253 }
254 variance /= worker_loads.size();
255 double std_dev = std::sqrt(variance);
256
257 // Î≥ÄÎèôÍ≥ÑÏàò (Coefficient of Variation)
258 double cv = std_dev / mean;
259
260 // Ï†êÏàò Í≥ÑÏÇ∞ (CVÍ∞Ä ÏûëÏùÑÏàòÎ°ù Ï†êÏàòÍ∞Ä ÎÜíÏùå)
261 double score = 100.0 * std::exp(-2.0 * cv);
262 return std::max(0.0, std::min(100.0, score));
263}
