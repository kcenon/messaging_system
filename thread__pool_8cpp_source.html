139 auto thread_pool::start(void) -> std::optional<std::string>
140 {
141 // Validate that workers have been added
142 if (workers_.empty())
143 {
144 return "No workers to start";
145 }
146
147 // Attempt to start each worker
148 for (auto& worker : workers_)
149 {
150 auto start_result = worker->start();
151 if (start_result.has_error())
152 {
153 // If any worker fails, stop all and return error
154 stop();
155 return start_result.get_error().to_string();
156 }
157 }
158
159 // Mark pool as successfully started
160 start_pool_.store(true);
161
162 return std::nullopt; // Success
163 }

307 auto thread_pool::stop(const bool& immediately_stop) -> void
308 {
309 if (!start_pool_.load())
310 {
311 return;
312 }
313
314 if (job_queue_ != nullptr)
315 {
316 job_queue_->stop_waiting_dequeue();
317
318 if (immediately_stop)
319 {
320 job_queue_->clear();
321 }
322 }
323
324 for (auto& worker : workers_)
325 {
326 auto stop_result = worker->stop();
327 if (stop_result.has_error())
328 {
329 context_.log(log_level::error,
330 formatter::format("error stopping worker: {}",
331 stop_result.get_error().to_string()));
332 }
333 }
334
335 start_pool_.store(false);
336 }
