128 auto job_queue::enqueue(std::unique_ptr<job>&& value) -> result_void
129 {
130 // Early validation: check if queue is still accepting jobs
131 if (stop_.load())
132 {
133 return error{error_code::queue_stopped, "Job queue is stopped"};
134 }
135
136 // Validate input: null jobs are not allowed
137 if (value == nullptr)
138 {
139 return error{error_code::invalid_argument, "cannot enqueue null job"};
140 }
141
142 // Critical section: modify queue with proper synchronization
143 std::scoped_lock<std::mutex> lock(mutex_);
144
145 // Move job into queue (efficient transfer of ownership)
146 queue_.push_back(std::move(value));
147
148 // Conditionally notify waiting consumers
149 if (notify_)
150 {
151 condition_.notify_one(); // Wake exactly one waiting thread
152 }
153
154 return {};
155 }

179 auto job_queue::enqueue_batch(std::vector<std::unique_ptr<job>>&& jobs) -> result_void
180 {
181 // Early validation: check if queue is still accepting jobs
182 if (stop_.load())
183 {
184 return error{error_code::queue_stopped, "Job queue is stopped"};
185 }
186
187 // Validate batch: empty batches are not allowed
188 if (jobs.empty())
189 {
190 return error{error_code::invalid_argument, "cannot enqueue empty batch"};
191 }
192
193 // Pre-validate all jobs before modifying queue (ensures atomicity)
194 for (auto& job : jobs)
195 {
196 if (job == nullptr)
197 {
198 return error{error_code::invalid_argument, "cannot enqueue null job in batch"};
199 }
200 }
201
202 // Critical section: add entire batch with single lock acquisition
203 std::scoped_lock<std::mutex> lock(mutex_);
204
205 // Move all jobs into queue efficiently
206 for (auto& job : jobs)
207 {
208 queue_.push_back(std::move(job));
209 }
210
211 // Single notification for entire batch (performance optimization)
212 if (notify_)
213 {
214 condition_.notify_one(); // Wake one thread to process batch
215 }
216
217 return {};
218 }

245 auto job_queue::dequeue() -> result<std::unique_ptr<job>>
246 {
247 // Use unique_lock for condition variable operations
248 std::unique_lock<std::mutex> lock(mutex_);
249
250 // Block until job available OR queue stopped
251 condition_.wait(lock, [this]() { return !queue_.empty() || stop_.load(); });
252
253 // Check if we woke up due to stop condition (queue empty after stop)
254 if (queue_.empty())
255 {
256 return error{error_code::queue_empty, "there are no jobs to dequeue"};
257 }
258
259 // Efficiently extract first job from queue
260 auto value = std::move(queue_.front());
261 queue_.pop_front();
262
263 return value; // Return moved job (caller takes ownership)
264 }

292 auto job_queue::dequeue_batch(void) -> std::deque<std::unique_ptr<job>>
293 {

295 {
