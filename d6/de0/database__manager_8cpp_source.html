49 bool database_manager::set_mode(const database_types& database_type)
50 {
51 if (connected_)
52 {
53 return false;
54 }
55
56 database_.reset();
57
58 switch (database_type)
59 {
60 case database_types::postgres:
61 database_ = std::make_unique<postgres_manager>();
62 break;
63 case database_types::mysql:
64 database_ = std::make_unique<mysql_manager>();
65 break;
66 case database_types::sqlite:
67 database_ = std::make_unique<sqlite_manager>();
68 break;
69 case database_types::mongodb:
70 database_ = std::make_unique<mongodb_manager>();
71 break;
72 case database_types::redis:
73 database_ = std::make_unique<redis_manager>();
74 break;
75 default:
76 break;
77 }
78
79 if (database_ == nullptr)
80 {
81 return false;
82 }
83
84 return true;
85 }
