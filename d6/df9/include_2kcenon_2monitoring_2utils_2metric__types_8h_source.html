257struct histogram_data {
258 std::vector<histogram_bucket> buckets;
259 uint64_t total_count = 0;
260 double sum = 0.0;
261
265 void add_sample(double value) {
266 sum += value;
267 total_count++;
268
269 for (auto& bucket : buckets) {
270 if (value <= bucket.upper_bound) {
271 bucket.count++;
272 }
273 }
274 }
275
279 double mean() const noexcept {
280 return total_count > 0 ? sum / total_count : 0.0;
281 }

337inline uint32_t hash_metric_name(const std::string& name) noexcept {
338 // Simple FNV-1a hash for fast metric name hashing
339 uint32_t hash = 2166136261U;
340 for (char c : name) {
341 hash ^= static_cast<uint32_t>(c);
342 hash *= 16777619U;
343 }
344 return hash;
345}
