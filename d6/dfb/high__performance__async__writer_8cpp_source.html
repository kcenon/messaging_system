54void high_performance_async_writer::stop(bool flush_remaining) {
55 if (!running_.exchange(false)) {
56 return; // Already stopped
57 }
58
59 if (batch_processor_) {
60 batch_processor_->stop(flush_remaining);
61 } else if (flush_remaining && wrapped_writer_) {
62 wrapped_writer_->flush();
63 }

66result_void high_performance_async_writer::write(
67 logger_system::log_level level,
68 const std::string& message,
69 const std::string& file,
70 int line,
71 const std::string& function,
72 const std::chrono::system_clock::time_point& timestamp) {
73
74 const auto start_time = std::chrono::steady_clock::now();
75
76 if (!running_.load(std::memory_order_relaxed)) {
77 return write_direct(level, message, file, line, function, timestamp);
78 }
79
80 stats_.total_writes.fetch_add(1, std::memory_order_relaxed);
81
82 // Use batch processor if available
83 if (batch_processor_) {
84 batch_processor::batch_entry entry(level, message, file, line, function, timestamp);
85
86 if (batch_processor_->add_entry(std::move(entry))) {
87 const auto end_time = std::chrono::steady_clock::now();
88 const auto latency = end_time - start_time;
89 update_stats(true, latency);
90 return result_void{};
91 } else {
92 stats_.dropped_writes.fetch_add(1, std::memory_order_relaxed);
93 // Fall back to direct write
94 return write_direct(level, message, file, line, function, timestamp);
95 }
96 }
97
98 // Direct async write (fallback mode)
99 return write_direct(level, message, file, line, function, timestamp);
100}

150result_void high_performance_async_writer::write_direct(
151 logger_system::log_level level,
152 const std::string& message,
153 const std::string& file,
154 int line,
155 const std::string& function,
156 const std::chrono::system_clock::time_point& timestamp) {
157
158 if (!wrapped_writer_) {
159 return make_logger_error(logger_error_code::writer_not_available, "No wrapped writer available");
160 }
161
162 const auto start_time = std::chrono::steady_clock::now();
163 auto result = wrapped_writer_->write(level, message, file, line, function, timestamp);
164 const auto end_time = std::chrono::steady_clock::now();
165
166 const auto latency = end_time - start_time;
167 update_stats(static_cast<bool>(result), latency);
168
169 return result;

172void high_performance_async_writer::update_stats(bool success, std::chrono::nanoseconds latency) {
173 if (success) {
174 stats_.successful_writes.fetch_add(1, std::memory_order_relaxed);
175 }
176
177 // Update average latency using exponential moving average
178 const double latency_us = std::chrono::duration_cast<std::chrono::microseconds>(latency).count();
179 const double alpha = 0.1; // Smoothing factor
180 const double current_avg = stats_.average_latency_us.load(std::memory_order_relaxed);
181 const double new_avg = alpha * latency_us + (1.0 - alpha) * current_avg;
182 stats_.average_latency_us.store(new_avg, std::memory_order_relaxed);
183
184 // Update throughput
185 const auto now = std::chrono::steady_clock::now();
186 const auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(now - stats_.start_time).count();
187 if (elapsed > 0) {
188 const auto total_writes = stats_.total_writes.load(std::memory_order_relaxed);
189 const double throughput = static_cast<double>(total_writes) / elapsed;
190 stats_.throughput_per_second.store(throughput, std::memory_order_relaxed);
191 }
192}
