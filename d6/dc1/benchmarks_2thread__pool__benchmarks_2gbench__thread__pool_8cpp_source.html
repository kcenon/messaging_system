48// Helper function to create thread pool with workers
49static std::shared_ptr<thread_pool> create_pool_with_workers(size_t worker_count) {
50 auto pool = std::make_shared<thread_pool>("benchmark_pool");
51
52 std::vector<std::unique_ptr<thread_worker>> workers;
53 workers.reserve(worker_count);
54 for (size_t i = 0; i < worker_count; ++i) {
55 workers.push_back(std::make_unique<thread_worker>());
56 }
57
58 pool->enqueue_batch(std::move(workers));
59 return pool;
60}

93// Benchmark job throughput
94static void BM_JobThroughput(benchmark::State& state) {
95 const size_t num_workers = state.range(0);
96 const size_t job_duration_us = state.range(1);
97
98 auto pool = create_pool_with_workers(num_workers);
99 pool->start();
100
101 std::atomic<size_t> jobs_completed{0};
102
103 for (auto _ : state) {
104 auto job = std::make_unique<callback_job>([&jobs_completed, job_duration_us]() -> result_void {
105 if (job_duration_us > 0) {
106 auto start = std::chrono::high_resolution_clock::now();
107 while (std::chrono::duration_cast<std::chrono::microseconds>(
108 std::chrono::high_resolution_clock::now() - start).count() < job_duration_us) {
109 // Busy wait
110 }
111 }
112 jobs_completed.fetch_add(1);
113 return result_void();
114 });
115 pool->enqueue(std::move(job));
116 }
117
118 // Wait for all jobs to complete
119 pool->stop();
120
121 state.SetItemsProcessed(jobs_completed.load());
122 state.counters["jobs/sec"] = benchmark::Counter(jobs_completed.load(), benchmark::Counter::kIsRate);
123}

157// Benchmark scaling efficiency
158static void BM_ScalingEfficiency(benchmark::State& state) {
159 const size_t num_workers = state.range(0);
160 const size_t total_jobs = 10000;
161
162 auto pool = create_pool_with_workers(num_workers);
163 pool->start();
164
165 std::atomic<size_t> jobs_completed{0};
166
167 for (auto _ : state) {
168 state.PauseTiming();
169 jobs_completed = 0;
170
171 // Submit all jobs
172 for (size_t i = 0; i < total_jobs; ++i) {
173 pool->enqueue(std::make_unique<callback_job>([&jobs_completed]() -> result_void {
174 // Simulate some work
175 volatile int sum = 0;
176 for (int j = 0; j < 1000; ++j) {
177 sum += j;
178 }
179 jobs_completed.fetch_add(1);
180 return result_void();
181 }));
182 }
183 state.ResumeTiming();
184
185 // Wait for completion
186 while (jobs_completed.load() < total_jobs) {
187 std::this_thread::yield();
188 }
189 }
190
191 pool->stop();
192
193 state.SetItemsProcessed(state.iterations() * total_jobs);
194
195 // Calculate scaling efficiency compared to single thread
196 if (num_workers > 1) {
197 state.counters["workers"] = num_workers;
198 state.counters["jobs_per_worker"] = total_jobs / num_workers;
199 }
200}
