57struct trace_export_config {
58 std::string endpoint;
59 trace_export_format format = trace_export_format::otlp_grpc;
60 std::chrono::milliseconds timeout{30000};
61 std::chrono::milliseconds batch_timeout{5000};
62 std::size_t max_batch_size = 512;
63 std::size_t max_queue_size = 2048;
64 bool enable_compression = true;
65 std::unordered_map<std::string, std::string> headers;
66 std::optional<std::string> service_name;
67
71 result_void validate() const {
72 if (endpoint.empty()) {
73 return result_void(monitoring_error_code::invalid_configuration,
74 "Export endpoint cannot be empty");
75 }
76
77 if (timeout.count() <= 0) {
78 return result_void(monitoring_error_code::invalid_configuration,
79 "Timeout must be positive");
80 }
81
82 if (max_batch_size == 0) {
83 return result_void(monitoring_error_code::invalid_configuration,
84 "Batch size must be greater than 0");
85 }
86
87 if (max_queue_size < max_batch_size) {
88 return result_void(monitoring_error_code::invalid_configuration,
89 "Queue size must be at least batch size");
90 }
91
92 return result_void::success();
93 }

183class jaeger_exporter : public trace_exporter_interface {
184private:
185 trace_export_config config_;
186 std::atomic<std::size_t> exported_spans_{0};
187 std::atomic<std::size_t> failed_exports_{0};
188 std::atomic<std::size_t> dropped_spans_{0};
189
190public:
191 explicit jaeger_exporter(const trace_export_config& config)
192 : config_(config) {}
193
197 jaeger_span_data convert_span(const trace_span& span) const {
198 jaeger_span_data jaeger_span;
199 jaeger_span.trace_id = span.trace_id;
200 jaeger_span.span_id = span.span_id;
201 jaeger_span.parent_span_id = span.parent_span_id;
202 jaeger_span.operation_name = span.operation_name;
203 jaeger_span.service_name = config_.service_name.value_or(span.service_name);
204
205 // Convert timestamps
206 auto start_epoch = span.start_time.time_since_epoch();
207 jaeger_span.start_time = std::chrono::duration_cast<std::chrono::microseconds>(start_epoch);
208
209 auto end_epoch = span.end_time.time_since_epoch();
210 jaeger_span.duration = std::chrono::duration_cast<std::chrono::microseconds>(end_epoch - start_epoch);
211
212 // Convert tags
213 for (const auto& [key, value] : span.tags) {
214 jaeger_span.tags.emplace_back(key, value);
215 }
216
217 // Add process tags
218 jaeger_span.process_tags.emplace_back("service.name", jaeger_span.service_name);
219
220 return jaeger_span;
221 }
222
223 result_void export_spans(const std::vector<trace_span>& spans) override {
224 try {
225 std::vector<jaeger_span_data> jaeger_spans;
226 jaeger_spans.reserve(spans.size());
227
228 for (const auto& span : spans) {
229 jaeger_spans.push_back(convert_span(span));
230 }
231
232 // Convert to appropriate format and send
233 result_void send_result;
234 if (config_.format == trace_export_format::jaeger_thrift) {
235 send_result = send_thrift_batch(jaeger_spans);
236 } else if (config_.format == trace_export_format::jaeger_grpc) {
237 send_result = send_grpc_batch(jaeger_spans);
238 } else {
239 return result_void(monitoring_error_code::invalid_configuration,
240 "Invalid Jaeger export format");
241 }
242
243 if (send_result) {
244 exported_spans_ += spans.size();
245 } else {
246 failed_exports_++;
247 return send_result;
248 }
249
250 return result_void::success();
251
252 } catch (const std::exception& e) {
253 failed_exports_++;
254 return result_void(monitoring_error_code::operation_failed,
255 "Jaeger export failed: " + std::string(e.what()));
256 }
257 }
258
259 result_void flush() override {
260 // Jaeger exporter typically sends immediately, so flush is a no-op
261 return result_void::success();
262 }
263
264 result_void shutdown() override {
265 return flush();
266 }
267
268 std::unordered_map<std::string, std::size_t> get_stats() const override {
269 return {
270 {"exported_spans", exported_spans_.load()},
271 {"failed_exports", failed_exports_.load()},
272 {"dropped_spans", dropped_spans_.load()}
273 };
274 }
275
276private:
277 result_void send_thrift_batch(const std::vector<jaeger_span_data>& spans) {
278 // Simulate Thrift protocol sending
279 // In real implementation, this would serialize to Thrift and send via HTTP/UDP
280 (void)spans; // Suppress unused parameter warning
281 return result_void::success();
282 }
283
284 result_void send_grpc_batch(const std::vector<jaeger_span_data>& spans) {
285 // Simulate gRPC protocol sending
286 // In real implementation, this would use Jaeger gRPC client
287 (void)spans; // Suppress unused parameter warning
288 return result_void::success();
289 }

312 zipkin_span.trace_id = span.trace_id;
313 zipkin_span.span_id = span.span_id;
314 zipkin_span.parent_id = span.parent_span_id;
315 zipkin_span.name = span.operation_name;
316 zipkin_span.local_endpoint_service_name = config_.service_name.value_or(span.service_name);
317
318 // Convert timestamps (Zipkin uses microseconds since epoch)
319 auto start_epoch = span.start_time.time_since_epoch();
320 zipkin_span.timestamp = std::chrono::duration_cast<std::chrono::microseconds>(start_epoch);
321
322 auto end_epoch = span.end_time.time_since_epoch();
323 zipkin_span.duration = std::chrono::duration_cast<std::chrono::microseconds>(end_epoch - start_epoch);
324
325 // Determine span kind from tags
326 auto kind_it = span.tags.find("span.kind");
327 if (kind_it != span.tags.end()) {
328 zipkin_span.kind = kind_it->second;
329 } else {
330 zipkin_span.kind = "INTERNAL";
331 }
332
333 // Convert tags (exclude special fields)
334 for (const auto& [key, value] : span.tags) {
335 if (key != "span.kind") {
336 zipkin_span.tags[key] = value;
337 }
338 }
339
340 return zipkin_span;
341 }

429 try {
430 // Convert to OpenTelemetry format first
431 auto otel_result = otel_adapter_->convert_spans(spans);

433 failed_exports_++;

439
440 // Send via appropriate OTLP protocol

442 if (config_.format == trace_export_format::otlp_grpc) {
443 send_result = send_grpc_batch(otel_spans);
444 } else if (config_.format == trace_export_format::otlp_http_json) {
445 send_result = send_http_json_batch(otel_spans);
446 } else if (config_.format == trace_export_format::otlp_http_protobuf) {
447 send_result = send_http_protobuf_batch(otel_spans);
448 } else {
449 return result_void(monitoring_error_code::invalid_configuration,
450 "Invalid OTLP export format");
451 }
452
453 if (send_result) {

455 } else {
456 failed_exports_++;
457 return send_result;
458 }
459
460 return result_void::success();
461

513class trace_exporter_factory {
514public:
518 static std::unique_ptr<trace_exporter_interface> create_exporter(
519 const trace_export_config& config,
520 const otel_resource& resource = create_service_resource("monitoring_system", "2.0.0")) {
521
522 switch (config.format) {
523 case trace_export_format::jaeger_thrift:
524 case trace_export_format::jaeger_grpc:
525 return std::make_unique<jaeger_exporter>(config);
526
527 case trace_export_format::zipkin_json:
528 case trace_export_format::zipkin_protobuf:
529 return std::make_unique<zipkin_exporter>(config);
530
531 case trace_export_format::otlp_grpc:
532 case trace_export_format::otlp_http_json:
533 case trace_export_format::otlp_http_protobuf:
534 return std::make_unique<otlp_exporter>(config, resource);
535
536 default:
537 return nullptr;
538 }
539 }
540
544 static std::vector<trace_export_format> get_supported_formats(const std::string& backend) {
545 if (backend == "jaeger") {
546 return {trace_export_format::jaeger_thrift, trace_export_format::jaeger_grpc};
547 } else if (backend == "zipkin") {
548 return {trace_export_format::zipkin_json, trace_export_format::zipkin_protobuf};
549 } else if (backend == "otlp") {
550 return {trace_export_format::otlp_grpc, trace_export_format::otlp_http_json,
551 trace_export_format::otlp_http_protobuf};
552 }
553 return {};
554 }
555};
