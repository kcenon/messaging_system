26void log_analyzer::analyze(thread_module::log_level level,
27 const std::string& message,
28 const std::string& file,
29 int line,
30 const std::string& /* function */,
31 const std::chrono::system_clock::time_point& timestamp) {
32
33 std::lock_guard<std::mutex> lock(windows_mutex_);
34
35 // Check if we need a new window
36 update_window(timestamp);
37
38 // Update level counts
39 current_window_.level_counts[level]++;
40
41 // Update source counts
42 if (!file.empty()) {
43 std::string source = file + ":" + std::to_string(line);
44 current_window_.source_counts[source]++;
45 }
46
47 // Pattern matching
48 match_patterns(message, current_window_.pattern_matches);
49
50 // Update totals
51 current_window_.total_messages++;
52 current_window_.total_bytes += message.size();
53
54 // Update rates
55 auto duration = timestamp - current_window_.window_start;
56 double seconds = std::chrono::duration<double>(duration).count();
57 if (seconds > 0) {
58 current_window_.messages_per_second = current_window_.total_messages / seconds;
59 current_window_.bytes_per_second = current_window_.total_bytes / seconds;
60 }
61
62 // Check alerts
63 check_alerts();
64}

81std::vector<log_analyzer::time_window_stats>
82log_analyzer::get_historical_stats(size_t count) const {
83 std::lock_guard<std::mutex> lock(windows_mutex_);
84
85 std::vector<time_window_stats> result;
86 size_t n = std::min(count, windows_.size());
87
88 // Get most recent windows
89 for (auto it = windows_.rbegin(); it != windows_.rend() && result.size() < n; ++it) {
90 result.push_back(*it);
91 }
92
93 // Add current window if requested
94 if (result.size() < count) {
95 result.push_back(current_window_);
96 }
97
98 return result;
99}

101log_analyzer::time_window_stats
102log_analyzer::get_aggregate_stats(std::chrono::seconds duration) const {
103 std::lock_guard<std::mutex> lock(windows_mutex_);
104
105 time_window_stats aggregate;
106 auto cutoff_time = std::chrono::system_clock::now() - duration;
107
108 // Aggregate historical windows
109 for (const auto& window : windows_) {
110 if (window.window_end > cutoff_time) {
111 // Aggregate level counts
112 for (const auto& [level, count] : window.level_counts) {
113 aggregate.level_counts[level] += count;
114 }
115
116 // Aggregate source counts
117 for (const auto& [source, count] : window.source_counts) {
118 aggregate.source_counts[source] += count;
119 }
120
121 // Aggregate pattern matches
122 for (const auto& [pattern, count] : window.pattern_matches) {
123 aggregate.pattern_matches[pattern] += count;
124 }
125
126 aggregate.total_messages += window.total_messages;
127 aggregate.total_bytes += window.total_bytes;
128
129 if (aggregate.window_start.time_since_epoch().count() == 0 ||
130 window.window_start < aggregate.window_start) {
131 aggregate.window_start = window.window_start;
132 }
133 }
134 }
135
136 // Include current window
137 if (current_window_.window_start > cutoff_time) {
138 for (const auto& [level, count] : current_window_.level_counts) {
139 aggregate.level_counts[level] += count;
140 }
141
142 for (const auto& [source, count] : current_window_.source_counts) {
143 aggregate.source_counts[source] += count;
144 }
145
146 for (const auto& [pattern, count] : current_window_.pattern_matches) {
147 aggregate.pattern_matches[pattern] += count;
148 }
149
150 aggregate.total_messages += current_window_.total_messages;
151 aggregate.total_bytes += current_window_.total_bytes;
152 }
153
154 aggregate.window_end = std::chrono::system_clock::now();
155
156 // Calculate rates
157 auto duration_actual = aggregate.window_end - aggregate.window_start;
158 double seconds = std::chrono::duration<double>(duration_actual).count();
159 if (seconds > 0) {
160 aggregate.messages_per_second = aggregate.total_messages / seconds;
161 aggregate.bytes_per_second = aggregate.total_bytes / seconds;
162 }
163
164 return aggregate;
165}

167std::string log_analyzer::generate_report(std::chrono::seconds duration) const {
168 auto stats = get_aggregate_stats(duration);
169 std::ostringstream report;
170
171 report << "=== Log Analysis Report ===" << std::endl;
172 report << "Time Range: " << duration.count() << " seconds" << std::endl;
173 report << "Total Messages: " << stats.total_messages << std::endl;
174 report << "Total Bytes: " << stats.total_bytes << std::endl;
175 report << "Messages/sec: " << std::fixed << std::setprecision(2)
176 << stats.messages_per_second << std::endl;
177 report << "Bytes/sec: " << stats.bytes_per_second << std::endl;
178
179 report << "\n--- Log Levels ---" << std::endl;
180 for (const auto& [level, count] : stats.level_counts) {
181 std::string level_str;
182 switch (level) {
183 case thread_module::log_level::trace: level_str = "TRACE"; break;
184 case thread_module::log_level::debug: level_str = "DEBUG"; break;
185 case thread_module::log_level::info: level_str = "INFO"; break;
186 case thread_module::log_level::warning: level_str = "WARNING"; break;
187 case thread_module::log_level::error: level_str = "ERROR"; break;
188 case thread_module::log_level::critical: level_str = "CRITICAL"; break;
189 }
190 double percentage = (stats.total_messages > 0) ?
191 (count * 100.0 / stats.total_messages) : 0;
192 report << level_str << ": " << count << " ("
193 << std::fixed << std::setprecision(1) << percentage << "%)" << std::endl;
194 }
195
196 report << "\n--- Top Sources ---" << std::endl;
197 std::vector<std::pair<std::string, uint64_t>> sorted_sources(
198 stats.source_counts.begin(), stats.source_counts.end()
199 );
200 std::sort(sorted_sources.begin(), sorted_sources.end(),
201 [](const auto& a, const auto& b) { return a.second > b.second; });
202
203 size_t top_n = std::min(size_t(10), sorted_sources.size());
204 for (size_t i = 0; i < top_n; ++i) {
205 report << sorted_sources[i].first << ": " << sorted_sources[i].second << std::endl;
206 }
207
208 if (!stats.pattern_matches.empty()) {
209 report << "\n--- Pattern Matches ---" << std::endl;
210 for (const auto& [pattern, count] : stats.pattern_matches) {
211 report << pattern << ": " << count << std::endl;
212 }
213 }
214
215 return report.str();
216}

218void log_analyzer::update_window(const std::chrono::system_clock::time_point& timestamp) {
219 if (timestamp >= current_window_.window_end) {
220 // Save current window
221 windows_.push_back(current_window_);
222
223 // Maintain max windows
224 while (windows_.size() > max_windows_) {
225 windows_.pop_front();
226 }
227
228 // Create new window
229 current_window_ = time_window_stats{};
230 current_window_.window_start = current_window_.window_end;
231 current_window_.window_end = current_window_.window_start + window_size_;
232 }
233}

245bool log_analyzer::match_patterns(const std::string& message,
246 std::unordered_map<std::string, uint64_t>& matches) {
247 std::lock_guard<std::mutex> lock(patterns_mutex_);
248
249 bool matched = false;
250 for (const auto& [name, pattern] : patterns_) {
251 if (std::regex_search(message, pattern)) {

262void log_aggregator::add_log(const std::string& source_id,
263 thread_module::log_level level,
264 const std::string& /* message */,
265 size_t message_size) {
266 std::lock_guard<std::mutex> lock(stats_mutex_);
267
268 auto& stats = source_stats_[source_id];
269 stats.source_id = source_id;
270
271 if (stats.total_messages == 0) {
272 stats.first_seen = std::chrono::system_clock::now();
273 }
274 stats.last_seen = std::chrono::system_clock::now();
275
276 stats.total_messages++;
277 stats.total_bytes += message_size;
278 stats.level_counts[level]++;
279
280 // Update average rate
281 auto duration = stats.last_seen - stats.first_seen;
282 double seconds = std::chrono::duration<double>(duration).count();
283 if (seconds > 0) {
284 stats.average_message_rate = stats.total_messages / seconds;
285 }
286}
