60// Test rapid start/stop operations (sequential)
61TEST_F(ConcurrencyTest, ThreadBaseRapidStartStop) {
62 class test_thread : public thread_base {
63 public:
64 test_thread() : thread_base("rapid_test") {}
65 std::atomic<int> work_cycles{0};
66 std::atomic<int> start_calls{0};
67 std::atomic<int> stop_calls{0};
68
69 protected:
70 result_void before_start() override {
71 start_calls.fetch_add(1);
72 return {};
73 }
74
75 result_void do_work() override {
76 work_cycles.fetch_add(1);
77 std::this_thread::sleep_for(std::chrono::microseconds(10));
78 return {};
79 }
80
81 result_void after_stop() override {
82 stop_calls.fetch_add(1);
83 return {};
84 }
85 };
86
87 auto worker = std::make_unique<test_thread>();
88
89 // Set wake interval to ensure do_work is called regularly
90 worker->set_wake_interval(std::chrono::milliseconds(5));
91
92 // Test rapid sequential start/stop cycles
93 const int num_cycles = 10;
94 for (int i = 0; i < num_cycles; ++i) {
95 // Start the thread
96 auto result = worker->start();
97 EXPECT_FALSE(result.has_error());
98
99 // Let it work a bit longer to ensure do_work is called
100 std::this_thread::sleep_for(std::chrono::milliseconds(20));
101
102 // Stop the thread
103 worker->stop();
104
105 // Just verify thread is stopped by checking it can be started again
106 // (thread_base::start() checks if thread is already running)
107 }
108
109 // Verify calls were made
110 EXPECT_EQ(worker->start_calls.load(), num_cycles);
111 EXPECT_EQ(worker->stop_calls.load(), num_cycles);
112 EXPECT_GT(worker->work_cycles.load(), 0);
113}

115// Test queue operations under extreme concurrency
116TEST_F(ConcurrencyTest, JobQueueExtremeConcurrency) {
117 auto queue = std::make_shared<job_queue>();
118 const int num_producers = 20;
119 const int num_consumers = 20;
120 const int jobs_per_producer = 100;
121
122 std::atomic<int> enqueued{0};
123 std::atomic<int> dequeued{0};
124 std::atomic<int> enqueue_failures{0};
125 std::atomic<bool> stop_consumers{false};
126
127 std::vector<std::thread> producers;
128 std::vector<std::thread> consumers;
129
130 // Start consumers
131 for (int i = 0; i < num_consumers; ++i) {
132 consumers.emplace_back([&queue, &dequeued, &stop_consumers]() {
133 while (!stop_consumers.load()) {
134 auto result = queue->dequeue();
135 if (result.has_value() && result.value()) {
136 [[maybe_unused]] auto work_result = result.value()->do_work();
137 dequeued.fetch_add(1);
138 }
139 std::this_thread::yield();
140 }
141 });
142 }
143
144 // Start producers
145 for (int i = 0; i < num_producers; ++i) {
146 producers.emplace_back([&queue, &enqueued, &enqueue_failures, jobs_per_producer]() {
147 for (int j = 0; j < jobs_per_producer; ++j) {
148 auto job = std::make_unique<callback_job>([]() -> result_void { return {}; });
149 auto result = queue->enqueue(std::move(job));
150 if (!result.has_error()) {
151 enqueued.fetch_add(1);
152 } else {
153 enqueue_failures.fetch_add(1);
154 }
155 }
156 });
157 }
158
159 // Wait for producers
160 for (auto& t : producers) {
161 t.join();
162 }
163
164 // Let consumers drain the queue
165 std::this_thread::sleep_for(std::chrono::milliseconds(100));
166
167 // Stop consumers
168 stop_consumers.store(true);
169 queue->stop_waiting_dequeue();
170
171 for (auto& t : consumers) {
172 t.join();
173 }
174
175 EXPECT_EQ(enqueued.load(), num_producers * jobs_per_producer);
176 EXPECT_EQ(enqueue_failures.load(), 0);
177 EXPECT_LE(dequeued.load(), enqueued.load());
178}

223// Test race conditions in job execution
224TEST_F(ConcurrencyTest, JobExecutionRaceConditions) {
225 auto queue = std::make_shared<job_queue>();
226 std::atomic<int> shared_counter{0};
227 std::atomic<int> race_detected{0};
228 const int num_jobs = 1000;
229
230 // Create jobs that increment a shared counter
231 // and check for race conditions
232 for (int i = 0; i < num_jobs; ++i) {
233 auto job = std::make_unique<callback_job>([&shared_counter, &race_detected]() -> result_void {
234 int old_value = shared_counter.load();
235 std::this_thread::yield(); // Increase chance of race
236 int new_value = old_value + 1;
237
238 // Try to update atomically
239 if (!shared_counter.compare_exchange_strong(old_value, new_value)) {
240 race_detected.fetch_add(1);
241 // Retry
242 shared_counter.fetch_add(1);
243 }
244
245 return {};
246 });
247 [[maybe_unused]] auto enqueue_result = queue->enqueue(std::move(job));
248 }
249
250 // Run multiple workers
251 const int num_workers = 8;
252 std::vector<std::thread> workers;
253 std::atomic<bool> stop_workers{false};
254
255 for (int i = 0; i < num_workers; ++i) {
256 workers.emplace_back([&queue, &stop_workers]() {
257 while (!stop_workers.load()) {
258 auto result = queue->dequeue();
259 if (result.has_value() && result.value()) {
260 [[maybe_unused]] auto work_result = result.value()->do_work();
261 } else {
262 std::this_thread::sleep_for(std::chrono::microseconds(10));
263 }
264 }
265 });
266 }
267
268 // Wait for all jobs to complete
269 while (shared_counter.load() < num_jobs) {
270 std::this_thread::sleep_for(std::chrono::milliseconds(10));
271 }
272
273 stop_workers.store(true);
274 queue->stop_waiting_dequeue();
275
276 for (auto& t : workers) {
277 t.join();
278 }
279
280 EXPECT_EQ(shared_counter.load(), num_jobs);
281 EXPECT_GT(race_detected.load(), 0); // Some races should have been detected
282}

385// Test ABA problem scenarios
386TEST_F(ConcurrencyTest, ABAScenario) {
387 struct Node {
388 int value;
389 std::atomic<Node*> next;
390 Node(int v) : value(v), next(nullptr) {}
391 };
392
393 std::atomic<Node*> head{nullptr};
394 const int num_threads = 4;
395 const int operations_per_thread = 1000;
396 std::atomic<int> aba_detected{0};
397
398 std::vector<std::thread> threads;
399
400 for (int i = 0; i < num_threads; ++i) {
401 threads.emplace_back([&head, &aba_detected, operations_per_thread]() {
402 std::vector<Node*> nodes;
403
404 for (int op = 0; op < operations_per_thread; ++op) {
405 if (op % 2 == 0) {
406 // Push
407 auto new_node = new Node(op);
408 nodes.push_back(new_node);
409
410 Node* old_head = head.load();
411 do {
412 new_node->next = old_head;
413 } while (!head.compare_exchange_weak(old_head, new_node));
414 } else {
415 // Pop
416 Node* old_head = head.load();
417 while (old_head) {
418 Node* new_head = old_head->next.load();
419 if (head.compare_exchange_weak(old_head, new_head)) {
420 // Successfully popped
421 break;
422 }
423 // CAS failed - potential ABA
424 if (old_head && old_head != head.load()) {
425 aba_detected.fetch_add(1);
426 }
427 }
428 }
429 }
430
431 // Clean up remaining nodes
432 for (auto node : nodes) {
433 delete node;
434 }
435 });
436 }
437
438 for (auto& t : threads) {
439 t.join();
440 }
441
442 // Clean up any remaining nodes
443 Node* current = head.load();
444 while (current) {
445 Node* next = current->next.load();
446 delete current;
447 current = next;
448 }
449
450 // ABA situations may or may not occur depending on timing
451 EXPECT_GE(aba_detected.load(), 0);
452}

454// Test spurious wakeup handling
455TEST_F(ConcurrencyTest, SpuriousWakeupHandling) {
456 std::mutex mutex;
457 std::condition_variable cv;
458 std::atomic<bool> ready{false};
459 std::atomic<int> spurious_wakeups{0};
460 std::atomic<int> valid_wakeups{0};
461
462 std::thread waiter([&mutex, &cv, &ready, &spurious_wakeups, &valid_wakeups]() {
463 std::unique_lock<std::mutex> lock(mutex);
464
465 while (!ready.load()) {
466 cv.wait(lock);
467
468 // Check if this was a spurious wakeup
469 if (!ready.load()) {
470 spurious_wakeups.fetch_add(1);
471 } else {
472 valid_wakeups.fetch_add(1);
473 }
474 }
475 });
476
477 // Give waiter time to start waiting
478 std::this_thread::sleep_for(std::chrono::milliseconds(50));
479
480 // Send some spurious notifications
481 for (int i = 0; i < 3; ++i) {
482 cv.notify_one();
483 std::this_thread::sleep_for(std::chrono::milliseconds(10));
484 }
485
486 // Now set ready and notify
487 {
488 std::lock_guard<std::mutex> lock(mutex);
489 ready.store(true);
490 }
491 cv.notify_one();
492
493 waiter.join();
494
495 EXPECT_EQ(valid_wakeups.load(), 1);
496 // Spurious wakeups may or may not occur
497 EXPECT_GE(spurious_wakeups.load(), 0);
498}
