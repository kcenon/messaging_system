47// Example 1: Basic queue strategies comparison
48void strategy_comparison_example()
49{
50 write_information("[Example 1] Queue Strategy Comparison");
51
52 const int num_jobs = 10000;
53 const int num_producers = 4;
54 const int num_consumers = 4;
55
56 // Test each strategy
57 for (auto strategy : {adaptive_job_queue::queue_strategy::FORCE_LEGACY,
58 adaptive_job_queue::queue_strategy::FORCE_LOCKFREE,
59 adaptive_job_queue::queue_strategy::ADAPTIVE})
60 {
61 adaptive_job_queue queue(strategy);
62 std::atomic<int> produced{0};
63 std::atomic<int> consumed{0};
64
65 auto start = std::chrono::high_resolution_clock::now();
66
67 std::vector<std::thread> producers;
68 std::vector<std::thread> consumers;
69
70 // Start producers
71 for (int p = 0; p < num_producers; ++p) {
72 producers.emplace_back([&queue, &produced, p, num_jobs]() {
73 for (int i = 0; i < num_jobs / num_producers; ++i) {
74 auto job = std::make_unique<callback_job>(
75 [p, i]() -> result_void {
76 return result_void();
77 });
78
79 while (true) {
80 auto r = queue.enqueue(std::move(job));
81 if (!r.has_error()) break;
82 std::this_thread::yield();
83 // Recreate moved job for retry
84 job = std::make_unique<callback_job>(
85 [p, i]() -> result_void { return result_void(); });
86 }
87 produced.fetch_add(1);
88 }
89 });
90 }
91
92 // Start consumers
93 for (int c = 0; c < num_consumers; ++c) {
94 consumers.emplace_back([&queue, &consumed, num_jobs]() {
95 while (consumed.load() < num_jobs) {
96 auto result = queue.dequeue();
97 if (result.has_value()) {
98 auto& job = result.value();
99 auto work_result = job->do_work();
100 (void)work_result; // Ignore result for sample
101 consumed.fetch_add(1);
102 } else {
103 std::this_thread::yield();
104 }
105 }
106 });
107 }
108
109 // Wait for completion
110 for (auto& t : producers) t.join();
111 for (auto& t : consumers) t.join();
112
113 auto duration = std::chrono::high_resolution_clock::now() - start;
114 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count();
115
116 std::string strategy_name;
117 switch (strategy) {
118 case adaptive_job_queue::queue_strategy::FORCE_LEGACY:
119 strategy_name = "Mutex-based";
120 break;
121 case adaptive_job_queue::queue_strategy::FORCE_LOCKFREE:
122 strategy_name = "Lock-free";
123 break;
124 case adaptive_job_queue::queue_strategy::ADAPTIVE:
125 strategy_name = "Adaptive";
126 break;
127 case adaptive_job_queue::queue_strategy::AUTO_DETECT:
128 strategy_name = "Auto-detect";
129 break;
130 }
131
132 write_information("{} strategy: {} jobs in {} ms = {} ops/sec",
133 strategy_name, num_jobs, ms,
134 num_jobs * 1000.0 / ms);
135 }
136}

138// Example 2: Adaptive strategy behavior under varying contention
139void adaptive_behavior_example()
140{
141 write_information("\n[Example 2] Adaptive Strategy Behavior");
142
143 adaptive_job_queue queue(adaptive_job_queue::queue_strategy::ADAPTIVE);
144
145 // Low contention phase (1 producer, 1 consumer)
146 write_information("Phase 1: Low contention (1P-1C)");
147 {
148 std::atomic<bool> running{true};
149 std::atomic<int> jobs_processed{0};
150
151 std::thread producer([&queue, &running]() {
152 while (running) {
153 auto job = std::make_unique<callback_job>(
154 []() -> result_void { return result_void(); });
155 auto enqueue_result = queue.enqueue(std::move(job));
156 if (enqueue_result.has_error()) {
157 write_error("enqueue failed: {}", enqueue_result.get_error().message());
158 }
159 std::this_thread::sleep_for(1ms);
160 }
161 });
162
163 std::thread consumer([&queue, &running, &jobs_processed]() {
164 while (running) {
165 auto result = queue.dequeue();
166 if (result.has_value()) {
167 auto work_result = result.value()->do_work();
168 (void)work_result; // Ignore result for sample
169 jobs_processed.fetch_add(1);
170 }
171 std::this_thread::sleep_for(1ms);
172 }
173 });
174
175 std::this_thread::sleep_for(2s);
176 running = false;
177 producer.join();
178 consumer.join();
179
180 auto current_type = queue.get_current_type();
181 write_information(
182 " Current type: {}, Jobs processed: {}",
183 current_type,
184 jobs_processed.load());
185 }
186
187 // High contention phase (8 producers, 8 consumers)
188 write_information("Phase 2: High contention (8P-8C)");
189 {
190 std::atomic<bool> running{true};
191 std::atomic<int> jobs_processed{0};
192 std::vector<std::thread> threads;
193
194 // Start producers
195 for (int i = 0; i < 8; ++i) {
196 threads.emplace_back([&queue, &running]() {
197 std::random_device rd;
198 std::mt19937 gen(rd());
199 std::uniform_int_distribution<> dist(0, 100);
200
201 while (running) {
202 auto job = std::make_unique<callback_job>(
203 []() -> result_void { return result_void(); });
204 auto enqueue_result = queue.enqueue(std::move(job));
205 if (enqueue_result.has_error()) {
206 // Best-effort: ignore for demo
207 }
208 if (dist(gen) < 10) { // 10% chance of sleep
209 std::this_thread::sleep_for(std::chrono::microseconds(dist(gen)));
210 }
211 }
212 });
213 }
214
215 // Start consumers
216 for (int i = 0; i < 8; ++i) {
217 threads.emplace_back([&queue, &running, &jobs_processed]() {
218 while (running) {
219 auto result = queue.dequeue();
220 if (result.has_value()) {
221 auto work_result = result.value()->do_work();
222 (void)work_result; // Ignore result for sample
223 jobs_processed.fetch_add(1);
224 }
225 }
226 });
227 }
228
229 std::this_thread::sleep_for(2s);
230 running = false;
231 for (auto& t : threads) t.join();
232
233 auto current_type = queue.get_current_type();
234 write_information(
235 " Current type: {}, Jobs processed: {}",
236 current_type,
237 jobs_processed.load());
238 }
239}

241// Example 3: Different queue strategies
242void different_strategies_example()
243{
244 write_information("\n[Example 3] Different Queue Strategies");
245
246 // Create queue with forced mutex-based strategy
247 adaptive_job_queue mutex_queue(adaptive_job_queue::queue_strategy::FORCE_LEGACY);
248 write_information(
249 "Mutex-based queue type: {}",
250 mutex_queue.get_current_type());
251
252 // Perform some operations
253 std::vector<std::unique_ptr<job>> jobs;
254 for (int i = 0; i < 100; ++i) {
255 jobs.push_back(std::make_unique<callback_job>(
256 [i]() -> result_void {
257 write_information("Job {} executed", i);
258 return result_void(); // Success
259 }));
260 }
261
262 auto enqueue_result = mutex_queue.enqueue_batch(std::move(jobs));
263 if (enqueue_result.has_error()) {
264 write_error("Batch enqueue failed: {}", enqueue_result.get_error().message());
265 } else {
266 write_information("Batch enqueue successful");
267 }
268
269 // Create queue with forced lock-free strategy
270 adaptive_job_queue lockfree_queue(adaptive_job_queue::queue_strategy::FORCE_LOCKFREE);
271 write_information(
272 "Lock-free queue type: {}",
273 lockfree_queue.get_current_type());
274
275 // Dequeue jobs from mutex queue
276 auto dequeued = mutex_queue.dequeue_batch();
277 write_information(
278 "Dequeued {} jobs from mutex queue", dequeued.size());
279
280 // Process dequeued jobs
281 for (auto& job : dequeued) {
282 auto result = job->do_work();
283 if (!result) {
284 // Job executed successfully
285 } else {
286 write_error("Job failed: {}", result.get_error().message());
287 }
288 }

291// Example 4: Performance monitoring
292void performance_monitoring_example()
293{
294 write_information("\n[Example 4] Performance Monitoring");
295
296 adaptive_job_queue queue(adaptive_job_queue::queue_strategy::ADAPTIVE);
297
298 const int num_operations = 50000;
299 std::atomic<bool> running{true};
300 std::atomic<int> enqueued{0};
301 std::atomic<int> dequeued{0};
302
303 // Producer thread
304 std::thread producer([&queue, &running, &enqueued, num_operations]() {
305 for (int i = 0; i < num_operations; ++i) {
306 auto job = std::make_unique<callback_job>(
307 []() -> result_void { return result_void(); });
308
309 while (!queue.enqueue(std::move(job))) {
310 std::this_thread::yield();
311 }
312 enqueued.fetch_add(1);
313 }
314 });
315
316 // Consumer thread
317 std::thread consumer([&queue, &running, &dequeued, num_operations]() {
318 while (dequeued.load() < num_operations) {
319 auto result = queue.dequeue();
320 if (result.has_value()) {
321 auto work_result = result.value()->do_work();
322 (void)work_result; // Ignore result for sample
323 dequeued.fetch_add(1);
324 }
325 }
326 });
327
328 // Monitor thread
329 std::thread monitor([&queue, &running, &enqueued, &dequeued, num_operations]() {
330 auto start = std::chrono::steady_clock::now();
331
332 while (dequeued.load() < num_operations) {
333 std::this_thread::sleep_for(500ms);
334
335 auto now = std::chrono::steady_clock::now();
336 auto elapsed = std::chrono::duration<double>(now - start).count();
337
338 auto current_type = queue.get_current_type();
339
340 write_information(
341 "Status: {} type, Enqueued: {}, Dequeued: {}, Rate: {:.0f} ops/sec",
342 current_type, enqueued.load(), dequeued.load(),
343 dequeued.load() / elapsed);
344 }
345 });
346
347 producer.join();
348 consumer.join();
349 running = false;
350 monitor.join();
351
352 write_information(
353 "Completed {} operations", num_operations);
354}

356// Example 5: Real-world scenario - Web server simulation
357void web_server_simulation()
358{
359 write_information("\n[Example 5] Web Server Simulation");
360
361 adaptive_job_queue request_queue(adaptive_job_queue::queue_strategy::ADAPTIVE);
362 std::atomic<bool> server_running{true};
363 std::atomic<int> requests_handled{0};
364 std::atomic<int> requests_failed{0};
365
366 // Request types
367 enum class request_type { GET, POST, PUT, DELETE };
368
369 // Simulate incoming requests
370 std::vector<std::thread> clients;
371 for (int client_id = 0; client_id < 5; ++client_id) {
372 clients.emplace_back([&request_queue, &server_running, &requests_failed, client_id]() {
373 std::random_device rd;
374 std::mt19937 gen(rd());
375 std::uniform_int_distribution<> type_dist(0, 3);
376 std::uniform_int_distribution<> delay_dist(10, 100);
377
378 while (server_running) {
379 auto type = static_cast<request_type>(type_dist(gen));
380
381 auto request = std::make_unique<callback_job>(
382 [client_id, type]() -> result_void {
383 // Simulate request processing
384 std::this_thread::sleep_for(std::chrono::microseconds(
385 type == request_type::GET ? 10 : 50));
386
387 std::string type_str;
388 switch (type) {
389 case request_type::GET: type_str = "GET"; break;
390 case request_type::POST: type_str = "POST"; break;
391 case request_type::PUT: type_str = "PUT"; break;
392 case request_type::DELETE: type_str = "DELETE"; break;
393 }
394
395 write_information("Client {} {} request completed",
396 client_id, type_str);
397 return result_void(); // Success
398 });
399
400 auto r = request_queue.enqueue(std::move(request));
401 if (r.has_error()) requests_failed.fetch_add(1);
402
403 std::this_thread::sleep_for(std::chrono::milliseconds(delay_dist(gen)));
404 }
405 });
406 }
407
408 // Worker threads (server handlers)
409 std::vector<std::thread> workers;
410 for (int worker_id = 0; worker_id < 3; ++worker_id) {
411 workers.emplace_back([&request_queue, &server_running, &requests_handled, worker_id]() {
412 while (server_running) {
413 auto request = request_queue.dequeue();
414 if (request.has_value()) {
415 auto result = request.value()->do_work();
416 if (!result) {
417 // Request processed successfully
418 requests_handled.fetch_add(1);
419 } else {
420 write_error("Worker {} request failed: {}",
421 worker_id, result.get_error().message());
422 }
423 } else {
424 std::this_thread::sleep_for(1ms);
425 }
426 }
427 });
428 }
429
430 // Run simulation for 5 seconds
431 std::this_thread::sleep_for(5s);
432 server_running = false;
433
434 // Cleanup
435 for (auto& t : clients) t.join();
436 for (auto& t : workers) t.join();
437
438 write_information(
439 "Server simulation complete: {} requests handled, {} failed",
440 requests_handled.load(), requests_failed.load());
441}
