61 auto messaging_session::start_session() -> void
62 {
63 if (is_stopped_.load())
64 {
65 return;
66 }
67
68 // Set up callbacks
69 auto self = shared_from_this();
70 socket_->set_receive_callback(
71 [this, self](const std::vector<uint8_t>& data)
72 { on_receive(data); });
73 socket_->set_error_callback([this, self](std::error_code ec)
74 { on_error(ec); });
75
76 // Begin reading
77 socket_->start_read();
78
79 std::cout << "[messaging_session] Started session on server: "
80 << server_id_ << "\n";
81 }

105{
106#ifdef USE_STD_COROUTINE
107 // Coroutine-based approach
108 asio::co_spawn(socket_->socket().get_executor(),
109 async_send_with_pipeline_co(socket_, std::move(data),
110 pipeline_, compress_mode_,
111 encrypt_mode_),
112 [](std::error_code ec)
113 {
114 if (ec)
115 {
116 std::cerr << "[messaging_session] Send error: "
117 << ec.message() << "\n";
118 }
119 });
120#else
121 // Fallback approach
122 auto fut = async_send_with_pipeline_no_co(
123 socket_, std::move(data), pipeline_, compress_mode_, encrypt_mode_);

125 // Use structured binding with try/catch for better error handling (C++17)
126 try {
127 auto result_ec = fut.get();
128 if (result_ec)

131 << result_ec.message() << "\n";
132 }
133 } catch (const std::exception& e) {
134 std::cerr << "[messaging_session] Exception while waiting for send: "
