67// Helper function to create thread pool
68auto create_default(const uint16_t& worker_counts)
69 -> std::tuple<std::shared_ptr<thread_pool_module::thread_pool>, std::optional<std::string>>
70{
71 std::shared_ptr<thread_pool_module::thread_pool> pool;
72 try {
73 pool = std::make_shared<thread_pool_module::thread_pool>();
74 } catch (const std::bad_alloc& e) {
75 return { nullptr, std::string(e.what()) };
76 }
77
78 std::optional<std::string> error_message = std::nullopt;
79 std::vector<std::unique_ptr<thread_pool_module::thread_worker>> workers;
80 workers.reserve(worker_counts);
81 for (uint16_t i = 0; i < worker_counts; ++i) {
82 workers.push_back(std::make_unique<thread_pool_module::thread_worker>());
83 }
84
85 error_message = pool->enqueue_batch(std::move(workers));
86 if (error_message.has_value()) {
87 return { nullptr, formatter::format("cannot enqueue to workers: {}",
88 error_message.value_or("unknown error")) };
89 }
90
91 return { pool, std::nullopt };
92}

94// Helper function to create typed thread pool
95template<typename Type>
96auto create_priority_default(const uint16_t& worker_counts, const std::vector<Type>& types)
97 -> std::tuple<std::shared_ptr<typed_thread_pool_module::typed_thread_pool_t<Type>>, std::optional<std::string>>
98{
99 std::shared_ptr<typed_thread_pool_module::typed_thread_pool_t<Type>> pool;
100 try {
101 pool = std::make_shared<typed_thread_pool_module::typed_thread_pool_t<Type>>();
102 } catch (const std::bad_alloc& e) {
103 return { nullptr, std::string(e.what()) };
104 }
105
106 std::optional<std::string> error_message = std::nullopt;
107 std::vector<std::unique_ptr<typed_thread_pool_module::typed_thread_worker_t<Type>>> workers;
108 workers.reserve(worker_counts);
109 for (uint16_t i = 0; i < worker_counts; ++i) {
110 workers.push_back(std::make_unique<typed_thread_pool_module::typed_thread_worker_t<Type>>(types));
111 }
112
113 auto result = pool->enqueue_batch(std::move(workers));
114 if (result.has_error()) {
115 return { nullptr, result.get_error().message() };
116 }
117
118 return { pool, std::nullopt };
119}

125// Simple thread pool implementation for comparison
126class SimpleThreadPool {
127public:
128 explicit SimpleThreadPool(size_t num_threads) : stop_(false) {
129 for (size_t i = 0; i < num_threads; ++i) {
130 workers_.emplace_back([this]() -> result_void {
131 while (true) {
132 std::function<void()> task;
133 {
134 std::unique_lock<std::mutex> lock(mutex_);
135 cv_.wait(lock, [this] { return stop_ || !tasks_.empty(); });
136
137 if (stop_ && tasks_.empty()) return result_void();
138
139 task = std::move(tasks_.front());
140 tasks_.pop();
141 }
142 task();
143 }
144 });
145 }
146 }
147
148 ~SimpleThreadPool() {
149 {
150 std::unique_lock<std::mutex> lock(mutex_);
151 stop_ = true;
152 }
153 cv_.notify_all();
154
155 for (auto& worker : workers_) {
156 worker.join();
157 }
158 }
159
160 void submit(std::function<void()> task) {
161 {
162 std::unique_lock<std::mutex> lock(mutex_);
163 tasks_.push(std::move(task));
164 }
165 cv_.notify_one();
166 }
167
168private:
169 std::vector<std::thread> workers_;
170 std::queue<std::function<void()>> tasks_;
171 std::mutex mutex_;
172 std::condition_variable cv_;
173 std::atomic<bool> stop_;
174};

364static void BM_ParallelComputation_ThreadSystem(benchmark::State& state) {
365 const size_t data_size = state.range(0);
366 std::vector<double> data(data_size);
367
368 // Initialize data
369 for (size_t i = 0; i < data_size; ++i) {
370 data[i] = static_cast<double>(i) * 0.1;
371 }
372
373 const size_t num_workers = std::thread::hardware_concurrency();
374 const size_t chunk_size = data_size / num_workers;
375
376 for (auto _ : state) {
377 auto [pool, error] = create_default(num_workers);
378 if (error.has_value()) {
379 state.SkipWithError("Failed to create thread pool");
380 return;
381 }
382
383 pool->start();
384
385 std::vector<std::future<double>> futures;
386 std::vector<std::shared_ptr<std::promise<double>>> promises;
387
388 for (size_t i = 0; i < num_workers; ++i) {
389 promises.push_back(std::make_shared<std::promise<double>>());
390 futures.push_back(promises[i]->get_future());
391 }
392
393 for (size_t i = 0; i < num_workers; ++i) {
394 size_t start_idx = i * chunk_size;
395 size_t end_idx = (i == num_workers - 1) ? data_size : start_idx + chunk_size;
396
397 auto promise_ptr = promises[i];
398 pool->enqueue(std::make_unique<callback_job>([&data, start_idx, end_idx, promise_ptr]() -> result_void {
399 double local_sum = 0;
400 for (size_t j = start_idx; j < end_idx; ++j) {
401 local_sum += std::sin(data[j]) * std::cos(data[j]);
402 }
403 promise_ptr->set_value(local_sum);
404 return result_void();
405 }));
406 }
407
408 double total_sum = 0;
409 for (auto& f : futures) {
410 total_sum += f.get();
411 }
412
413 pool->stop();
414 benchmark::DoNotOptimize(total_sum);
415 }
416
417 state.SetItemsProcessed(state.iterations() * data_size);
418 if (g_baseline_times.count("parallel_comp")) {
419 state.counters["speedup"] = g_baseline_times["parallel_comp"] / state.iterations();
420 }
421}

423 ->Arg(10000000)
424 ->Unit(benchmark::kMillisecond);
