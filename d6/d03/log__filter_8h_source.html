129class composite_filter : public log_filter {
130public:
131 enum class logic_type {
132 AND, // All filters must pass
133 OR // At least one filter must pass
134 };
135
136 explicit composite_filter(logic_type logic = logic_type::AND)
137 : logic_(logic) {}
138
139 void add_filter(std::unique_ptr<log_filter> filter) {
140 filters_.push_back(std::move(filter));
141 }
142
143 bool should_log(thread_module::log_level level,
144 const std::string& message,
145 const std::string& file,
146 int line,
147 const std::string& function) const override {
148 if (filters_.empty()) {
149 return true; // No filters = pass all
150 }
151
152 if (logic_ == logic_type::AND) {
153 for (const auto& filter : filters_) {
154 if (!filter->should_log(level, message, file, line, function)) {
155 return false;
156 }
157 }
158 return true;
159 } else { // OR logic
160 for (const auto& filter : filters_) {
161 if (filter->should_log(level, message, file, line, function)) {
162 return true;
163 }
164 }
165 return false;
166 }
167 }
168
169private:
170 logic_type logic_;
171 std::vector<std::unique_ptr<log_filter>> filters_;
172};
