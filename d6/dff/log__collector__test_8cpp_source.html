44// Mock writer for testing log collector
45class MockCollectorWriter : public base_writer {
46public:
47 result_void write(thread_module::log_level level,
48 const std::string& message,
49 const std::string& /* file */,
50 int /* line */,
51 const std::string& /* function */,
52 const std::chrono::system_clock::time_point& /* timestamp */) override {
53 write_count_++;
54 last_message_ = message;
55 last_level_ = level;
56 messages_.push_back(message);
57 return result_void{};
58 }
59
60 result_void flush() override {
61 flush_count_++;
62 return result_void{};
63 }
64
65 std::string get_name() const override {
66 return "mock_collector";
67 }
68
69 std::atomic<int> write_count_{0};
70 std::atomic<int> flush_count_{0};
71 std::string last_message_;
72 thread_module::log_level last_level_ = thread_module::log_level::trace;
73 std::vector<std::string> messages_;
74 mutable std::mutex messages_mutex_;
75};
