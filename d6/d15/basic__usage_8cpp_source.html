42void basic_logging_example() {
43 std::cout << "\n=== Basic Logging Example ===\n" << std::endl;
44
45 // Create logger instance
46 auto logger_instance = std::make_shared<logger>();
47
48 // Add console writer
49 logger_instance->add_writer(std::make_unique<console_writer>());
50
51 // Start logger (for async mode)
52 logger_instance->start();
53
54 // Log messages at different levels
55 logger_instance->log(thread_module::log_level::trace, "This is a trace message");
56 logger_instance->log(thread_module::log_level::debug, "Debug information here");
57 logger_instance->log(thread_module::log_level::info, "Application started successfully");
58 logger_instance->log(thread_module::log_level::warning, "This is a warning");
59 logger_instance->log(thread_module::log_level::error, "An error occurred!");
60 logger_instance->log(thread_module::log_level::critical, "Critical system failure!");
61
62 // Log with source location
63 logger_instance->log(thread_module::log_level::info, "Message with location",
64 __FILE__, __LINE__, __func__);
65
66 // Stop and flush
67 logger_instance->stop();
68}

70void multithreaded_logging_example() {
71 std::cout << "\n=== Multithreaded Logging Example ===\n" << std::endl;
72
73 // Create async logger
74 auto logger_instance = std::make_shared<logger>(true, 16384); // Larger buffer for MT
75 logger_instance->add_writer(std::make_unique<console_writer>());
76 logger_instance->start();
77
78 // Launch multiple threads
79 std::vector<std::thread> threads;
80
81 for (int i = 0; i < 4; ++i) {
82 threads.emplace_back([logger_instance, i]() {
83 for (int j = 0; j < 10; ++j) {
84 logger_instance->log(thread_module::log_level::info,
85 "Thread " + std::to_string(i) + " - Message " + std::to_string(j));
86 std::this_thread::sleep_for(std::chrono::milliseconds(10));
87 }
88 });
89 }
90
91 // Wait for all threads
92 for (auto& t : threads) {
93 t.join();
94 }
95
96 logger_instance->stop();
97}

121void sync_vs_async_example() {
122 std::cout << "\n=== Sync vs Async Logging Example ===\n" << std::endl;
123
124 // Synchronous logger
125 std::cout << "Synchronous logging:" << std::endl;
126 auto sync_logger = std::make_shared<logger>(false); // sync mode
127 sync_logger->add_writer(std::make_unique<console_writer>());
128
129 auto start = std::chrono::high_resolution_clock::now();
130 for (int i = 0; i < 100; ++i) {
131 sync_logger->log(thread_module::log_level::info, "Sync log " + std::to_string(i));
132 }
133 auto sync_time = std::chrono::high_resolution_clock::now() - start;
134
135 // Asynchronous logger
136 std::cout << "\nAsynchronous logging:" << std::endl;
137 auto async_logger = std::make_shared<logger>(true); // async mode
138 async_logger->add_writer(std::make_unique<console_writer>());
139 async_logger->start();
140
141 start = std::chrono::high_resolution_clock::now();
142 for (int i = 0; i < 100; ++i) {
143 async_logger->log(thread_module::log_level::info, "Async log " + std::to_string(i));
144 }
145 auto async_time = std::chrono::high_resolution_clock::now() - start;
146
147 async_logger->stop();
148
149 std::cout << "\nSync time: "
150 << std::chrono::duration_cast<std::chrono::microseconds>(sync_time).count()
151 << " ¬µs" << std::endl;
152 std::cout << "Async time: "
153 << std::chrono::duration_cast<std::chrono::microseconds>(async_time).count()
154 << " ¬µs" << std::endl;
155}
