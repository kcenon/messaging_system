80class plain_formatter : public base_formatter {
81public:
82 std::string format(const log_entry& entry) const override {
83 std::ostringstream oss;
84
85 // Format: [TIMESTAMP] [LEVEL] [THREAD] MESSAGE [FILE:LINE:FUNCTION]
86 oss << "[" << format_timestamp(entry.timestamp) << "] ";
87 oss << "[" << level_to_string(entry.level) << "] ";
88
89 if (entry.thread_id) {
90 oss << "[" << std::string_view(*entry.thread_id) << "] ";
91 } else {
92 oss << "[" << get_thread_id() << "] ";
93 }
94
95 oss << std::string_view(entry.message);
96
97 if (entry.location) {
98 oss << " [" << std::string_view(entry.location->file)
99 << ":" << entry.location->line
100 << ":" << std::string_view(entry.location->function) << "]";
101 }
102
103 return oss.str();
104 }
105
106 std::string get_format_type() const override {
107 return "plain";

120 oss << "{";

123 oss << "\"message\":\"" << escape_json(entry.message) << "\",";
124 oss << "\"thread\":\"" << (entry.thread_id ? std::string(*entry.thread_id) : get_thread_id()) << "\"";
125
126 if (entry.location) {
127 oss << ",\"location\":{";
128 oss << "\"file\":\"" << escape_json(entry.location->file) << "\",";
129 oss << "\"line\":" << entry.location->line << ",";
130 oss << "\"function\":\"" << escape_json(entry.location->function) << "\"";
131 oss << "}";
132 }
133
134 if (entry.category) {
135 oss << ",\"category\":\"" << escape_json(*entry.category) << "\"";
136 }
137
138 oss << "}";
139
140 return oss.str();
141 }
142
143 std::string get_format_type() const override {
144 return "json";

147private:
148 template<typename StringType>
149 std::string escape_json(const StringType& str) const {
150 std::string escaped;
151 for (char c : str) {
152 if (c == '"') escaped += "\\\"";
153 else if (c == '\\') escaped += "\\\\";
154 else if (c == '\n') escaped += "\\n";
155 else if (c == '\r') escaped += "\\r";
156 else if (c == '\t') escaped += "\\t";
157 else if (c == '\b') escaped += "\\b";
158 else if (c == '\f') escaped += "\\f";
159 else escaped += c;
160 }
161 return escaped;
162 }
