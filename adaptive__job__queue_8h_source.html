108 [[nodiscard]] auto get_average_latency_ns() const -> double
109 {
110 if (operation_count == 0) return 0.0;
111 return static_cast<double>(total_latency_ns) / static_cast<double>(operation_count);
112 }
113
114 [[nodiscard]] auto get_contention_ratio() const -> double
115 {
116 if (operation_count == 0) return 0.0;
117 return static_cast<double>(contention_count) / static_cast<double>(operation_count);
118 }
119 };
