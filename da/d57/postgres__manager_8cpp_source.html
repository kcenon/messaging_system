59 bool postgres_manager::connect(const std::string& connect_string)
60 {
61#ifdef USE_POSTGRESQL
62 try {
63 auto conn = std::make_unique<pqxx::connection>(connect_string);
64 if (conn->is_open()) {
65 connection_ = conn.release();
66 return true;
67 }
68 } catch (const std::exception& e) {
69 std::cerr << "PostgreSQL connection error: " << e.what() << std::endl;
70 }
71#elif defined(HAVE_LIBPQ)
72 try {
73 connection_ = PQconnectdb(connect_string.c_str());
74 if (PQstatus(static_cast<PGconn*>(connection_)) == CONNECTION_OK) {
75 return true;
76 }
77 PQfinish(static_cast<PGconn*>(connection_));
78 connection_ = nullptr;
79 } catch (const std::exception& e) {
80 std::cerr << "PostgreSQL connection error: " << e.what() << std::endl;
81 }
82#else
83 std::cerr << "PostgreSQL support not compiled. Connection: " << connect_string.substr(0, 20) << "..." << std::endl;
84#endif
85 return false;
86 }

88 bool postgres_manager::create_query(const std::string& query_string)
89 {
90#ifdef USE_POSTGRESQL
91 if (!connection_) return false;
92 try {
93 pqxx::connection* conn = static_cast<pqxx::connection*>(connection_);
94 pqxx::work txn(*conn);
95 txn.exec(query_string);
96 txn.commit();
97 return true;
98 } catch (const std::exception& e) {
99 std::cerr << "Query execution error: " << e.what() << std::endl;
100 }
101#elif defined(HAVE_LIBPQ)
102 if (!connection_) return false;
103 try {
104 PGresult* result = PQexec(static_cast<PGconn*>(connection_), query_string.c_str());
105 ExecStatusType status = PQresultStatus(result);
106 bool success = (status == PGRES_COMMAND_OK || status == PGRES_TUPLES_OK);
107 PQclear(result);
108 return success;
109 } catch (const std::exception& e) {
110 std::cerr << "Query execution error: " << e.what() << std::endl;
111 }
112#else
113 std::cerr << "PostgreSQL support not compiled. Query: " << query_string.substr(0, 20) << "..." << std::endl;
114#endif
115 return false;

118 unsigned int postgres_manager::execute_modification_query(const std::string& query_string)
119 {
120#ifdef USE_POSTGRESQL
121 if (!connection_) return 0;
122 try {
123 pqxx::connection* conn = static_cast<pqxx::connection*>(connection_);
124 pqxx::work txn(*conn);
125 pqxx::result result = txn.exec(query_string);
126 txn.commit();
127 return static_cast<unsigned int>(result.affected_rows());
128 } catch (const std::exception& e) {
129 std::cerr << "Modification query error: " << e.what() << std::endl;
130 }
131#elif defined(HAVE_LIBPQ)
132 if (!connection_) return 0;
133 try {
134 PGresult* result = PQexec(static_cast<PGconn*>(connection_), query_string.c_str());
135 if (PQresultStatus(result) != PGRES_COMMAND_OK) {
136 PQclear(result);
137 return 0;
138 }
139 const char* affected_rows = PQcmdTuples(result);
140 unsigned int count = 0;
141 if (affected_rows && *affected_rows) {
142 count = static_cast<unsigned int>(std::stoul(affected_rows));
143 }
144 PQclear(result);
145 return count;
146 } catch (const std::exception& e) {
147 std::cerr << "Modification query error: " << e.what() << std::endl;
148 }
149#else
150 std::cerr << "PostgreSQL support not compiled. Modification query: " << query_string.substr(0, 20) << "..." << std::endl;
151#endif
152 return 0;

170 database_result postgres_manager::select_query(const std::string& query_string)
171 {
172 database_result result;
173
174#ifdef USE_POSTGRESQL
175 if (!connection_) return result;
176 try {
177 pqxx::connection* conn = static_cast<pqxx::connection*>(connection_);
178 pqxx::work txn(*conn);
179 pqxx::result pqxx_result = txn.exec(query_string);
180 txn.commit();
181
182 for (const auto& row : pqxx_result) {
183 database_row db_row;
184 for (size_t i = 0; i < row.size(); ++i) {
185 std::string column_name = pqxx_result.column_name(i);
186 if (row[i].is_null()) {
187 db_row[column_name] = nullptr;
188 } else {
189 // Try to convert to appropriate type
190 // Using numeric OID values for compatibility across libpqxx versions
191 // These are standard PostgreSQL OID constants:
192 // 20 = int8, 21 = int2, 23 = int4
193 // 700 = float4, 701 = float8
194 // 16 = bool
195 pqxx::oid type = row[i].type();
196 if (type == 20 || type == 21 || type == 23) { // int8, int2, int4
197 db_row[column_name] = row[i].as<int64_t>();
198 } else if (type == 700 || type == 701) { // float4, float8
199 db_row[column_name] = row[i].as<double>();
200 } else if (type == 16) { // bool
201 db_row[column_name] = row[i].as<bool>();
202 } else {
203 db_row[column_name] = row[i].as<std::string>();
204 }
205 }
206 }
207 result.push_back(std::move(db_row));
208 }
209 } catch (const std::exception& e) {
210 std::cerr << "Select query error: " << e.what() << std::endl;
211 }
212#elif defined(HAVE_LIBPQ)
213 if (!connection_) return result;
214 try {
215 PGresult* pg_result = PQexec(static_cast<PGconn*>(connection_), query_string.c_str());
216 if (PQresultStatus(pg_result) != PGRES_TUPLES_OK) {
217 PQclear(pg_result);
218 return result;
219 }
220
221 int rows = PQntuples(pg_result);
222 int cols = PQnfields(pg_result);
223
224 for (int row = 0; row < rows; ++row) {
225 database_row db_row;
226 for (int col = 0; col < cols; ++col) {
227 std::string column_name = PQfname(pg_result, col);
228 if (PQgetisnull(pg_result, row, col)) {
229 db_row[column_name] = nullptr;
230 } else {
231 const char* value = PQgetvalue(pg_result, row, col);
232 Oid type = PQftype(pg_result, col);
233
234 // Convert based on PostgreSQL type
235 if (type == 20 || type == 21 || type == 23) { // int8, int2, int4
236 db_row[column_name] = static_cast<int64_t>(std::stoll(value));
237 } else if (type == 700 || type == 701) { // float4, float8
238 db_row[column_name] = std::stod(value);
239 } else if (type == 16) { // bool
240 db_row[column_name] = (*value == 't' || *value == '1');
241 } else {
242 db_row[column_name] = std::string(value);
243 }
244 }
245 }
246 result.push_back(std::move(db_row));
247 }
248 PQclear(pg_result);
249 } catch (const std::exception& e) {
250 std::cerr << "Select query error: " << e.what() << std::endl;
251 }
252#else
253 std::cerr << "PostgreSQL support not compiled. Select query: " << query_string.substr(0, 20) << "..." << std::endl;
254 // Return empty result with mock data for testing
255 if (query_string.find("SELECT") != std::string::npos) {
256 database_row mock_row;
257 mock_row["id"] = int64_t(1);
258 mock_row["name"] = std::string("mock_data");
259 mock_row["active"] = true;
260 result.push_back(mock_row);
261 }
262#endif
263 return result;
264 }

295 bool postgres_manager::execute_query(const std::string& query_string)
296 {
297#ifdef USE_POSTGRESQL
298 try {
299 if (!connection_) {
300 std::cerr << "No active PostgreSQL connection" << std::endl;
301 return false;
302 }
303
304 pqxx::work txn{*static_cast<pqxx::connection*>(connection_)};
305 txn.exec(query_string);
306 txn.commit();
307 return true;
308 } catch (const std::exception& e) {
309 std::cerr << "PostgreSQL execute error: " << e.what() << std::endl;
310 return false;
311 }
312#elif defined(HAVE_LIBPQ)
313 if (!connection_) {
314 std::cerr << "No active PostgreSQL connection" << std::endl;
315 return false;
316 }
317
318 PGresult* result = PQexec(static_cast<PGconn*>(connection_), query_string.c_str());
319 if (!result) {
320 std::cerr << "PostgreSQL execute failed" << std::endl;
321 return false;
322 }
323
324 ExecStatusType status = PQresultStatus(result);
325 bool success = (status == PGRES_COMMAND_OK) || (status == PGRES_TUPLES_OK);
326
327 if (!success) {
328 std::cerr << "PostgreSQL execute error: " << PQerrorMessage(static_cast<PGconn*>(connection_)) << std::endl;
329 }
330
331 PQclear(result);
332 return success;
333#else
334 // Mock execution
335 std::cout << "PostgreSQL support not compiled. Mock execute: " << query_string << std::endl;
336 return true;
337#endif
338 }
