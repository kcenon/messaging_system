147void stress_client_worker(
148 size_t client_id,
149 const StressConfig& config,
150 StressMetrics& metrics,
151 std::atomic<bool>& stop_flag
152) {
153 std::random_device rd;
154 std::mt19937 gen(rd());
155 std::uniform_int_distribution<> size_dist(64, config.max_message_size);
156 std::uniform_int_distribution<> delay_dist(0, 100);
157
158 try {
159 auto client = std::make_shared<core::messaging_client>(
160 "stress_client_" + std::to_string(client_id)
161 );
162
163 client->start_client("127.0.0.1", config.base_port);
164 metrics.total_connections++;
165
166 std::this_thread::sleep_for(50ms); // Connection stabilization
167
168 size_t messages_sent = 0;
169 while (!stop_flag && messages_sent < config.messages_per_client) {
170 // Determine message size
171 size_t msg_size = config.enable_random_sizes ?
172 size_dist(gen) : config.message_size;
173
174 // Create message
175 std::vector<uint8_t> data(msg_size);
176 for (size_t i = 0; i < msg_size; ++i) {
177 data[i] = static_cast<uint8_t>((client_id + i) % 256);
178 }
179
180 // Send message
181 try {
182 client->send_packet(data);
183 metrics.total_messages_sent++;
184 metrics.total_bytes_sent += msg_size;
185 messages_sent++;
186 } catch (...) {
187 metrics.total_messages_failed++;
188 }
189
190 // Optional delay
191 if (config.enable_random_delays) {
192 std::this_thread::sleep_for(
193 std::chrono::milliseconds(delay_dist(gen))
194 );
195 }
196 }
197
198 client->stop_client();
199
200 } catch (const std::exception& e) {
201 metrics.failed_connections++;
202 metrics.total_errors++;
203 std::cerr << "Client " << client_id << " error: " << e.what() << std::endl;
204 }
205}
