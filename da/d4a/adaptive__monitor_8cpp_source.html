43 void adaptation_loop() {
44 while (running) {
45 // Get current system metrics
46 if (sys_monitor) {
47 auto metrics_result = sys_monitor->get_current_metrics();
48 if (metrics_result) {
49 const auto& sys_metrics = metrics_result.value();
50
51 // Adapt all collectors
52 std::shared_lock lock(collectors_mutex);
53 for (auto& [name, info] : collectors) {
54 if (info.collector) {
55 info.collector->adapt(sys_metrics);
56 }
57 }
58 }
59 }
60
61 // Sleep for adaptation interval
62 std::this_thread::sleep_for(adaptation_interval);
63 }
64 }

67 std::vector<std::pair<std::string, int>> collector_priorities;

69 std::shared_lock lock(collectors_mutex);
70 for (const auto& [name, info] : collectors) {

72 }
73
74 // Sort by priority (higher first)
75 std::sort(collector_priorities.begin(), collector_priorities.end(),
76 [](const auto& a, const auto& b) {

78 });
79
80 std::vector<std::string> result;
81 for (const auto& [name, priority] : collector_priorities) {

279std::vector<std::string> adaptive_monitor::get_active_collectors() const {
280 if (!impl_) {
281 return {};
282 }
283
284 std::vector<std::string> active;
285
286 std::shared_lock lock(impl_->collectors_mutex);
287
288 // Get collectors sorted by priority
289 auto sorted_collectors = impl_->get_collectors_by_priority();
290
291 // Determine how many collectors to keep active based on load
292 double avg_cpu = 0.0;
293 size_t active_count = sorted_collectors.size();
294
295 // Calculate average CPU usage across all collectors
296 for (const auto& [name, info] : impl_->collectors) {
297 if (info.collector) {
298 auto stats = info.collector->get_stats();
299 avg_cpu += stats.average_cpu_usage;
300 }
301 }
302
303 if (!impl_->collectors.empty()) {
304 avg_cpu /= impl_->collectors.size();
305 }
306
307 // Determine how many collectors to keep active
308 if (avg_cpu > 80.0) {
309 active_count = std::max(size_t(1), static_cast<size_t>(sorted_collectors.size() * 0.2));
310 } else if (avg_cpu > 60.0) {
311 active_count = std::max(size_t(2), static_cast<size_t>(sorted_collectors.size() * 0.5));
312 } else if (avg_cpu > 40.0) {
313 active_count = std::max(size_t(3), static_cast<size_t>(sorted_collectors.size() * 0.75));
314 }
315
316 // Return top priority collectors up to active_count
317 for (size_t i = 0; i < std::min(active_count, sorted_collectors.size()); ++i) {
318 active.push_back(sorted_collectors[i]);
319 }
320
321 return active;
322}
