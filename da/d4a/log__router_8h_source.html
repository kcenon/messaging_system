90 const std::unordered_map<std::string, std::unique_ptr<base_writer>>& writers) const {
91 (void)timestamp; // Not used in current implementation
92
93 std::vector<base_writer*> selected_writers;
94 std::unordered_set<std::string> added_writers;
95
96 // Check each route
97 for (const auto& route : routes_) {
98 if (route.filter->should_log(level, message, file, line, function)) {
99 // Add writers from this route
100 for (const auto& writer_name : route.writer_names) {
101 if (added_writers.find(writer_name) == added_writers.end()) {
102 auto it = writers.find(writer_name);
103 if (it != writers.end()) {
104 selected_writers.push_back(it->second.get());
105 added_writers.insert(writer_name);
106 }
107 }
108 }
109
110 // Stop if this route has stop_propagation
111 if (route.stop_propagation) {
112 break;
113 }
114 }
115 }
116
117 // If no routes matched, use default (all writers)
118 if (selected_writers.empty() && !has_exclusive_routes_) {
119 for (const auto& [name, writer] : writers) {
120 selected_writers.push_back(writer.get());
121 }
122 }
123
124 return selected_writers;
125 }
126
130 void set_exclusive_routes(bool exclusive) {
131 has_exclusive_routes_ = exclusive;
132 }
133
137 void clear_routes() {
138 routes_.clear();
139 }

141private:
