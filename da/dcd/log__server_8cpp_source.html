53bool log_server::start() {
54 if (running_) {
55 return true;
56 }
57
58 // Create socket
59 server_socket_ = socket(AF_INET, use_tcp_ ? SOCK_STREAM : SOCK_DGRAM, 0);
60#ifdef _WIN32
61 if (server_socket_ == INVALID_SOCKET) {
62 std::cerr << "Failed to create server socket: " << WSAGetLastError() << std::endl;
63 return false;
64 }
65#else
66 if (server_socket_ < 0) {
67 std::cerr << "Failed to create server socket: " << strerror(errno) << std::endl;
68 return false;
69 }
70#endif
71
72 // Allow socket reuse
73 int reuse = 1;
74#ifdef _WIN32
75 if (setsockopt(server_socket_, SOL_SOCKET, SO_REUSEADDR,
76 (const char*)&reuse, sizeof(reuse)) < 0) {
77 std::cerr << "Failed to set socket options: " << WSAGetLastError() << std::endl;
78 closesocket(server_socket_);
79 return false;
80 }
81#else
82 if (setsockopt(server_socket_, SOL_SOCKET, SO_REUSEADDR, &reuse, sizeof(reuse)) < 0) {
83 std::cerr << "Failed to set socket options: " << strerror(errno) << std::endl;
84 close(server_socket_);
85 return false;
86 }
87#endif
88
89 // Bind to port
90 struct sockaddr_in server_addr{};
91 server_addr.sin_family = AF_INET;
92 server_addr.sin_addr.s_addr = INADDR_ANY;
93 server_addr.sin_port = htons(port_);
94
95 if (bind(server_socket_, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
96#ifdef _WIN32
97 std::cerr << "Failed to bind to port " << port_ << ": " << WSAGetLastError() << std::endl;
98 closesocket(server_socket_);
99#else
100 std::cerr << "Failed to bind to port " << port_ << ": " << strerror(errno) << std::endl;
101 close(server_socket_);
102#endif
103 return false;
104 }
105
106 // Listen for TCP
107 if (use_tcp_) {
108 if (listen(server_socket_, static_cast<int>(max_connections_)) < 0) {
109#ifdef _WIN32
110 std::cerr << "Failed to listen: " << WSAGetLastError() << std::endl;
111 closesocket(server_socket_);
112#else
113 std::cerr << "Failed to listen: " << strerror(errno) << std::endl;
114 close(server_socket_);
115#endif
116 return false;
117 }
118 }
119
120 running_ = true;
121 stats_.server_start_time = std::chrono::system_clock::now();
122
123 // Start appropriate thread
124 if (use_tcp_) {
125 accept_thread_ = std::thread(&log_server::tcp_accept_thread, this);
126 } else {
127 accept_thread_ = std::thread(&log_server::udp_receive_thread, this);
128 }
129
130 std::cout << "Log server started on port " << port_
131 << " (" << (use_tcp_ ? "TCP" : "UDP") << ")" << std::endl;
132
133 return true;
134}

136void log_server::stop() {
137 if (!running_) {
138 return;
139 }
140
141 running_ = false;
142
143 // Close server socket to interrupt accept()
144#ifdef _WIN32
145 if (server_socket_ != INVALID_SOCKET) {
146 shutdown(server_socket_, SD_BOTH);
147 closesocket(server_socket_);
148 server_socket_ = INVALID_SOCKET;
149 }
150#else
151 if (server_socket_ >= 0) {
152 shutdown(server_socket_, SHUT_RDWR);
153 close(server_socket_);
154 server_socket_ = -1;
155 }
156#endif
157
158 // Wait for accept thread
159 if (accept_thread_.joinable()) {
160 accept_thread_.join();
161 }
162
163 // Wait for all client threads
164 {
165 std::lock_guard<std::mutex> lock(threads_mutex_);
166 for (auto& thread : client_threads_) {
167 if (thread.joinable()) {
168 thread.join();
169 }
170 }
171 client_threads_.clear();
172 }
173
174 std::cout << "Log server stopped" << std::endl;
175}

187void log_server::tcp_accept_thread() {
188 while (running_) {
189 struct sockaddr_in client_addr{};
190 socklen_t client_len = sizeof(client_addr);
191
192 int client_fd = accept(server_socket_, (struct sockaddr*)&client_addr, &client_len);
193#ifdef _WIN32
194 if (client_fd == INVALID_SOCKET) {
195 if (running_) {
196 std::cerr << "Accept error: " << WSAGetLastError() << std::endl;
197 }
198 continue;
199 }
200#else
201 if (client_fd < 0) {
202 if (running_) {
203 std::cerr << "Accept error: " << strerror(errno) << std::endl;
204 }
205 continue;
206 }
207#endif
208
209 // Get client address
210 char client_ip[INET_ADDRSTRLEN];
211 inet_ntop(AF_INET, &client_addr.sin_addr, client_ip, INET_ADDRSTRLEN);
212 std::string client_address = std::string(client_ip) + ":" +
213 std::to_string(ntohs(client_addr.sin_port));
214
215 std::cout << "New connection from " << client_address << std::endl;
216
217 // Start client thread
218 {
219 std::lock_guard<std::mutex> lock(threads_mutex_);
220
221 // Clean up finished threads
222 client_threads_.erase(
223 std::remove_if(client_threads_.begin(), client_threads_.end(),
224 [](std::thread& t) { return !t.joinable(); }),
225 client_threads_.end()
226 );
227
228 // Add new client thread
229 client_threads_.emplace_back(&log_server::tcp_client_thread, this,
230 client_fd, client_address);
231
232 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
233 stats_.active_connections = client_threads_.size();
234 }
235 }
236}

238void log_server::tcp_client_thread(int client_fd, const std::string& client_addr) {
239 char buffer[4096];
240 std::string incomplete_data;
241
242 while (running_) {
243#ifdef _WIN32
244 int received = recv(client_fd, buffer, sizeof(buffer) - 1, 0);
245 if (received <= 0) {
246 if (received < 0) {
247 std::cerr << "Receive error from " << client_addr
248 << ": " << WSAGetLastError() << std::endl;
249 }
250 break;
251 }
252#else
253 ssize_t received = recv(client_fd, buffer, sizeof(buffer) - 1, 0);
254 if (received <= 0) {
255 if (received < 0) {
256 std::cerr << "Receive error from " << client_addr
257 << ": " << strerror(errno) << std::endl;
258 }
259 break;
260 }
261#endif
262
263 buffer[received] = '\0';
264 incomplete_data += buffer;
265
266 // Process complete lines
267 size_t pos;
268 while ((pos = incomplete_data.find('\n')) != std::string::npos) {
269 std::string line = incomplete_data.substr(0, pos);
270 incomplete_data.erase(0, pos + 1);
271
272 if (!line.empty()) {
273 network_log_entry entry;
274 entry.source_address = client_addr;
275 entry.source_port = 0; // Already included in address
276 entry.received_time = std::chrono::system_clock::now();
277 entry.raw_data = line;
278
279 if (parse_log_entry(line, entry)) {
280 process_log(entry);
281 }
282 }
283 }
284 }
285
286#ifdef _WIN32
287 closesocket(client_fd);
288#else
289 close(client_fd);
290#endif
291 std::cout << "Connection closed: " << client_addr << std::endl;
292
293 {
294 std::lock_guard<std::mutex> lock(stats_mutex_);
295 stats_.active_connections--;
296 }
297}

299void log_server::udp_receive_thread() {
300 char buffer[65536]; // Max UDP packet size
301
302 while (running_) {
303 struct sockaddr_in client_addr{};
304 socklen_t client_len = sizeof(client_addr);
305
306#ifdef _WIN32
307 int received = recvfrom(server_socket_, buffer, sizeof(buffer) - 1, 0,
308 (struct sockaddr*)&client_addr, &client_len);
309 if (received == SOCKET_ERROR) {
310 if (running_) {
311 std::cerr << "UDP receive error: " << WSAGetLastError() << std::endl;
312 }
313 continue;
314 }
315#else
316 ssize_t received = recvfrom(server_socket_, buffer, sizeof(buffer) - 1, 0,
317 (struct sockaddr*)&client_addr, &client_len);
318 if (received < 0) {
319 if (running_) {
320 std::cerr << "UDP receive error: " << strerror(errno) << std::endl;
321 }
322 continue;
323 }
324#endif
325
326 buffer[received] = '\0';
327
328 // Get client address
329 char client_ip[INET_ADDRSTRLEN];
330 inet_ntop(AF_INET, &client_addr.sin_addr, client_ip, INET_ADDRSTRLEN);
331
332 network_log_entry entry;
333 entry.source_address = client_ip;
334 entry.source_port = ntohs(client_addr.sin_port);
335 entry.received_time = std::chrono::system_clock::now();

344bool log_server::parse_log_entry(const std::string& data, network_log_entry& entry) {
345 // Simple JSON parser for log entries
346 static const std::regex field_regex("\"([^\"]+)\"\\s*:\\s*\"([^\"]*)\"");
347 static const std::regex number_regex("\"([^\"]+)\"\\s*:\\s*(\\d+)");
348
349 // Extract string fields
350 auto fields_begin = std::sregex_iterator(data.begin(), data.end(), field_regex);
351 auto fields_end = std::sregex_iterator();
352
353 for (auto it = fields_begin; it != fields_end; ++it) {
354 entry.parsed_fields[(*it)[1]] = (*it)[2];
355 }
356
357 // Extract number fields
358 auto numbers_begin = std::sregex_iterator(data.begin(), data.end(), number_regex);
359 auto numbers_end = std::sregex_iterator();
360
361 for (auto it = numbers_begin; it != numbers_end; ++it) {
362 entry.parsed_fields[(*it)[1]] = (*it)[2];
363 }
364
365 if (entry.parsed_fields.empty()) {
366 std::lock_guard<std::mutex> lock(stats_mutex_);
367 stats_.parse_errors++;
368 return false;
369 }
370
371 return true;
372}

374void log_server::process_log(const network_log_entry& entry) {
375 // Update statistics
376 {
377 std::lock_guard<std::mutex> lock(stats_mutex_);
378 stats_.total_logs_received++;
379 stats_.total_bytes_received += entry.raw_data.size();
380 stats_.logs_per_source[entry.source_address]++;
381 }
382
383 // Call handlers
384 std::vector<log_handler> handlers_copy;
385 {
386 std::lock_guard<std::mutex> lock(handlers_mutex_);
387 handlers_copy = handlers_;
388 }
389
390 for (const auto& handler : handlers_copy) {
391 try {
392 handler(entry);
393 } catch (const std::exception& e) {
394 std::cerr << "Handler error: " << e.what() << std::endl;
395 }
396 }
397}
