38batch_writer::batch_writer(std::unique_ptr<base_writer> underlying_writer,
39 const config& cfg)
40 : config_(cfg)

42 , last_flush_time_(std::chrono::steady_clock::now()) {

61result_void batch_writer::write(logger_system::log_level level,
62 const std::string& message,
63 const std::string& file,
64 int line,
65 const std::string& function,
66 const std::chrono::system_clock::time_point& timestamp) {
67
68 if (shutting_down_) {
69 return make_logger_error(logger_error_code::queue_stopped, "Batch writer is shutting down");
70 }
71
72 bool should_flush = false;
73
74 {
75 std::lock_guard<std::mutex> lock(batch_mutex_);
76
77 // Add entry to batch
78 batch_.emplace_back(level, message, file, line, function, timestamp);
79 stats_.total_entries++;
80
81 // Check if we should flush
82 should_flush = should_flush_by_size();
83 if (should_flush) {
84 stats_.flush_on_size++;
85 }
86 }
87
88 // Flush outside of lock to avoid blocking other writers
89 if (should_flush) {
90 return flush();
91 }
92
93 return result_void{};
94}

96result_void batch_writer::flush() {
97 if (shutting_down_ && batch_.empty()) {
98 return result_void{};
99 }
100
101 std::lock_guard<std::mutex> lock(batch_mutex_);
102
103 if (!shutting_down_) {
104 stats_.manual_flushes++;
105 }
106
107 return flush_batch_unsafe();
108}

110result_void batch_writer::flush_batch_unsafe() {
111 if (batch_.empty()) {
112 return result_void{};
113 }
114
115 // Write all entries in the batch
116 result_void last_result;
117
118 for (const auto& entry : batch_) {
119 auto result = underlying_writer_->write(
120 entry.level,
121 entry.message,
122 entry.file,
123 entry.line,
124 entry.function,
125 entry.timestamp
126 );
127
128 if (!result) {
129 last_result = result;
130 stats_.dropped_entries++;
131 }
132 }
133
134 // Flush the underlying writer
135 auto flush_result = underlying_writer_->flush();
136 if (!flush_result && !last_result) {
137 last_result = flush_result;
138 }
139
140 // Update statistics
141 stats_.total_batches++;
142
143 // Clear the batch
144 batch_.clear();
145 last_flush_time_ = std::chrono::steady_clock::now();
146
147 // Return the last error if any
148 return last_result ? result_void{} : last_result;
149}

173bool batch_writer::should_flush_by_size() const {
174 return batch_.size() >= config_.max_batch_size;

177bool batch_writer::should_flush_by_time() const {
178 auto now = std::chrono::steady_clock::now();
179 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
180 now - last_flush_time_
181 );
182 return elapsed >= config_.flush_interval;
183}
