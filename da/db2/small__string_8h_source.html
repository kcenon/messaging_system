165 void assign(const char* str, size_t len) {
166 if (len <= SSO_CAPACITY) {
167 // Use small string optimization
168 if (!is_small_ && data_.heap.ptr) {
169 delete[] data_.heap.ptr;
170 }
171 std::memcpy(data_.small, str, len);
172 data_.small[len] = '\0';
173 size_ = len;
174 is_small_ = true;
175 } else {
176 // Use heap allocation
177 if (is_small_ || data_.heap.capacity < len + 1) {
178 // Need to allocate new buffer
179 if (!is_small_ && data_.heap.ptr) {
180 delete[] data_.heap.ptr;
181 }
182
183 size_t new_capacity = calculate_capacity(len);
184 data_.heap.ptr = new char[new_capacity];
185 data_.heap.capacity = new_capacity;
186 is_small_ = false;
187 }
188
189 std::memcpy(data_.heap.ptr, str, len);
190 data_.heap.ptr[len] = '\0';
191 size_ = len;
192 }
193 }

259 void reserve(size_t new_capacity) {
260 if (new_capacity <= capacity()) {
261 return;
262 }
263
264 if (new_capacity <= SSO_CAPACITY) {
265 // Already using SSO, no need to reserve
266 return;
267 }
268
269 // Need heap allocation
270 size_t actual_capacity = calculate_capacity(new_capacity);
271 char* new_ptr = new char[actual_capacity];
272
273 // Copy existing data
274 std::memcpy(new_ptr, data(), size_ + 1);
275
276 // Clean up old allocation
277 if (!is_small_ && data_.heap.ptr) {
278 delete[] data_.heap.ptr;
279 }
280
281 data_.heap.ptr = new_ptr;
282 data_.heap.capacity = actual_capacity;
283 is_small_ = false;
284 }

289 void append(const char* str, size_t len) {
290 size_t new_size = size_ + len;
291
292 if (new_size <= SSO_CAPACITY && is_small_) {
293 // Can still use SSO
294 std::memcpy(data_.small + size_, str, len);
295 data_.small[new_size] = '\0';
296 size_ = new_size;
297 } else {
298 // Need heap allocation
299 if (is_small_ || data_.heap.capacity < new_size + 1) {
300 // Need to reallocate
301 size_t new_capacity = calculate_capacity(new_size);
302 char* new_ptr = new char[new_capacity];
303
304 // Copy existing data
305 std::memcpy(new_ptr, data(), size_);
306 // Append new data
307 std::memcpy(new_ptr + size_, str, len);
308 new_ptr[new_size] = '\0';
309
310 // Clean up old allocation
311 if (!is_small_ && data_.heap.ptr) {
312 delete[] data_.heap.ptr;
313 }
314
315 data_.heap.ptr = new_ptr;
316 data_.heap.capacity = new_capacity;
317 is_small_ = false;
318 } else {
319 // Existing heap buffer is large enough
320 std::memcpy(data_.heap.ptr + size_, str, len);
321 data_.heap.ptr[new_size] = '\0';
322 }
323 size_ = new_size;
324 }

412 static size_t calculate_capacity(size_t required) {
413 // Round up to next power of 2 for better allocation patterns
414 size_t capacity = required + 1; // +1 for null terminator
415 capacity = capacity * 3 / 2; // 1.5x growth factor
416
417 // Align to 16 bytes for better memory alignment
418 return (capacity + 15) & ~15;
419 }

432 data_union() : small{} {}
433 } data_;
