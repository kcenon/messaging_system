363 // Template implementation for async_executor
364 template<typename F, typename... Args>
365 auto async_executor::submit(F&& func, Args&&... args) -> std::future<std::invoke_result_t<F, Args...>>
366 {
367 using return_type = std::invoke_result_t<F, Args...>;
368
369 auto task = std::make_shared<std::packaged_task<return_type()>>(
370 std::bind(std::forward<F>(func), std::forward<Args>(args)...));
371
372 std::future<return_type> result = task->get_future();
373
374 {
375 std::unique_lock<std::mutex> lock(queue_mutex_);
376 if (stop_) {
377 throw std::runtime_error("Cannot submit task to stopped executor");
378 }
379 tasks_.emplace([task]() { (*task)(); });
380 }
381
382 condition_.notify_one();
383 return result;
384 }
