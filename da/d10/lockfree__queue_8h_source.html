88 bool dequeue(T& item) {
89 const size_t pos = tail_.load(std::memory_order_relaxed);
90 auto& cell = cells_[pos & mask_];
91
92 const size_t seq = cell.sequence.load(std::memory_order_acquire);
93 const size_t expected_seq = pos + 1;
94
95 if (seq != expected_seq) {
96 return false; // Queue is empty
97 }
98
99 item = std::move(cell.data);
100 cell.sequence.store(pos + mask_ + 1, std::memory_order_release);
101 tail_.store(pos + 1, std::memory_order_relaxed);
102
103 return true;

168 template<typename U>
169 bool enqueue_impl(U&& item) {
170 const size_t pos = head_.load(std::memory_order_relaxed);
171 auto& cell = cells_[pos & mask_];
172
173 const size_t seq = cell.sequence.load(std::memory_order_acquire);
174 const size_t expected_seq = pos;
175
176 if (seq != expected_seq) {
177 return false; // Queue is full
178 }
179
180 cell.data = std::forward<U>(item);
181 cell.sequence.store(pos + 1, std::memory_order_release);
182 head_.store(pos + 1, std::memory_order_relaxed);
183
184 return true;
185 }
