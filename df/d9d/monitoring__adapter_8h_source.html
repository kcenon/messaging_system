58 // Internal monitoring service implementation
59 class internal_monitoring_service : public monitoring_service {
60 private:
61 mutable std::mutex metrics_mutex_;
62 messaging_metrics metrics_;
63 std::unordered_map<std::string, uint64_t> topic_publish_counts_;
64 std::unordered_map<std::string, uint64_t> topic_consume_counts_;
65 std::unordered_map<std::string, uint64_t> topic_subscriber_counts_;
66
67 public:
68 internal_monitoring_service() = default;
69
70 void record_message_published(const std::string& topic) override {
71 std::lock_guard<std::mutex> lock(metrics_mutex_);
72 metrics_.messages_published++;
73 topic_publish_counts_[topic]++;
74 }
75
76 void record_message_consumed(const std::string& topic, std::chrono::milliseconds processing_time) override {
77 std::lock_guard<std::mutex> lock(metrics_mutex_);
78 metrics_.messages_consumed++;
79 topic_consume_counts_[topic]++;
80
81 // Update average processing time
82 auto current_avg = metrics_.average_processing_time.load();
83 auto new_avg = (current_avg + processing_time.count()) / 2.0;
84 metrics_.average_processing_time = new_avg;
85 }
86
87 void record_message_failed(const std::string& topic, const std::string& error) override {
88 std::lock_guard<std::mutex> lock(metrics_mutex_);
89 metrics_.messages_failed++;
90 }
91
92 void record_subscriber_added(const std::string& topic) override {
93 std::lock_guard<std::mutex> lock(metrics_mutex_);
94 metrics_.active_subscribers++;
95 topic_subscriber_counts_[topic]++;
96 }
97
98 void record_subscriber_removed(const std::string& topic) override {
99 std::lock_guard<std::mutex> lock(metrics_mutex_);
100 if (metrics_.active_subscribers > 0) {
101 metrics_.active_subscribers--;
102 }
103 if (topic_subscriber_counts_[topic] > 0) {
104 topic_subscriber_counts_[topic]--;
105 }
106 }
107
108 messaging_metrics get_metrics() const override {
109 messaging_metrics snapshot;
110 snapshot.messages_published = metrics_.messages_published.load();
111 snapshot.messages_consumed = metrics_.messages_consumed.load();
112 snapshot.messages_failed = metrics_.messages_failed.load();
113 snapshot.active_subscribers = metrics_.active_subscribers.load();
114 snapshot.average_processing_time = metrics_.average_processing_time.load();
115 snapshot.last_reset = metrics_.last_reset;
116 return snapshot;
117 }
118
119 std::unordered_map<std::string, uint64_t> get_topic_metrics() const override {
120 std::lock_guard<std::mutex> lock(metrics_mutex_);
121 std::unordered_map<std::string, uint64_t> result;
122
123 for (const auto& [topic, count] : topic_publish_counts_) {
124 result[topic + "_published"] = count;
125 }
126 for (const auto& [topic, count] : topic_consume_counts_) {
127 result[topic + "_consumed"] = count;
128 }
129 for (const auto& [topic, count] : topic_subscriber_counts_) {
130 result[topic + "_subscribers"] = count;
131 }
132
133 return result;
134 }

226 // Monitoring service adapter
227 class monitoring_service_adapter : public service_adapter {
228 private:
229 std::shared_ptr<monitoring_service> monitoring_service_;
230
231 public:
232 explicit monitoring_service_adapter(std::shared_ptr<monitoring_service> service)
233 : service_adapter(service), monitoring_service_(std::move(service)) {}
234
235 void register_with_bus(core::message_bus* bus) {
236 // Subscribe to monitoring topics
237 if (bus && monitoring_service_) {
238 bus->subscribe("monitoring.*", [this](const core::message& msg) {
239 this->handle_monitoring_message(msg);
240 });
241 }
242 }
243
244 private:
245 void handle_monitoring_message(const core::message& msg) {
246 if (!monitoring_service_) return;
247
248 // Process the monitoring message based on its topic
249 if (msg.payload.topic == "monitoring.message_published") {
250 auto original_topic = msg.payload.get<std::string>("original_topic", "");
251 monitoring_service_->record_message_published(original_topic);
252 }
253 else if (msg.payload.topic == "monitoring.message_consumed") {
254 auto original_topic = msg.payload.get<std::string>("original_topic", "");
255 auto processing_time = std::chrono::milliseconds(
256 msg.payload.get<int64_t>("processing_time_ms", 0)
257 );
258 monitoring_service_->record_message_consumed(original_topic, processing_time);
259 }
260 else if (msg.payload.topic == "monitoring.message_failed") {
261 auto original_topic = msg.payload.get<std::string>("original_topic", "");
262 auto error = msg.payload.get<std::string>("error", "");
263 monitoring_service_->record_message_failed(original_topic, error);
264 }
265 else if (msg.payload.topic == "monitoring.subscriber_added") {
266 auto topic = msg.payload.get<std::string>("topic", "");
267 monitoring_service_->record_subscriber_added(topic);
268 }
269 else if (msg.payload.topic == "monitoring.subscriber_removed") {
270 auto topic = msg.payload.get<std::string>("topic", "");
271 monitoring_service_->record_subscriber_removed(topic);
272 }
273 }
274
275 public:
276 std::shared_ptr<monitoring_service> get_monitoring_service() const {
277 return monitoring_service_;
278 }
279 };
