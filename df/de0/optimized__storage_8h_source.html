119class lock_free_metrics_queue {
120public:
121 explicit lock_free_metrics_queue(std::size_t capacity)
122 : capacity_(capacity)
123 , mask_(capacity - 1)
124 , buffer_(std::make_unique<slot[]>(capacity))
125 , head_(0)
126 , tail_(0) {
127
128 // capacity는 2의 제곱수여야 함
129 if ((capacity & (capacity - 1)) != 0) {
130 throw std::invalid_argument("Capacity must be power of 2");
131 }
132
133 // 슬롯 초기화
134 for (std::size_t i = 0; i < capacity; ++i) {
135 buffer_[i].sequence.store(i, std::memory_order_relaxed);
136 }
137 }
138
139 bool enqueue(T item) {
140 cell* cell;
141 std::size_t pos = head_.load(std::memory_order_relaxed);
142
143 for (;;) {
144 cell = &buffer_[pos & mask_];
145 std::size_t seq = cell->sequence.load(std::memory_order_acquire);
146 intptr_t dif = static_cast<intptr_t>(seq) - static_cast<intptr_t>(pos);
147
148 if (dif == 0) {
149 if (head_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
150 break;
151 }
152 } else if (dif < 0) {
153 return false; // 큐 가득 참
154 } else {
155 pos = head_.load(std::memory_order_relaxed);
156 }
157 }
158
159 cell->data = std::move(item);
160 cell->sequence.store(pos + 1, std::memory_order_release);
161 return true;
162 }
163
164 bool dequeue(T& item) {
165 cell* cell;
166 std::size_t pos = tail_.load(std::memory_order_relaxed);
167
168 for (;;) {
169 cell = &buffer_[pos & mask_];
170 std::size_t seq = cell->sequence.load(std::memory_order_acquire);
171 intptr_t dif = static_cast<intptr_t>(seq) - static_cast<intptr_t>(pos + 1);
172
173 if (dif == 0) {
174 if (tail_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
175 break;
176 }
177 } else if (dif < 0) {
178 return false; // 큐 비어있음
179 } else {
180 pos = tail_.load(std::memory_order_relaxed);
181 }
182 }
183
184 item = std::move(cell->data);
185 cell->sequence.store(pos + mask_ + 1, std::memory_order_release);
186 return true;
187 }
188
189 std::size_t size() const {
190 std::size_t head = head_.load(std::memory_order_acquire);
191 std::size_t tail = tail_.load(std::memory_order_acquire);
192 return head - tail;
193 }
194
195 bool empty() const {
196 return size() == 0;
197 }
198
199private:
200 struct cell {
201 std::atomic<std::size_t> sequence;
202 T data;
203 };
204
205 using slot = cell;
206 using cacheline_pad_t = char[64];
207
208 cacheline_pad_t pad0_;
209 const std::size_t capacity_;
210 const std::size_t mask_;
211 std::unique_ptr<slot[]> buffer_;
212
213 cacheline_pad_t pad1_;
214 std::atomic<std::size_t> head_;
215
216 cacheline_pad_t pad2_;
217 std::atomic<std::size_t> tail_;
218
219 cacheline_pad_t pad3_;
220};
