87 result_void validate() const {
88 // Validate buffer size
89 if (buffer_size == 0) {
90 return make_logger_error(logger_error_code::invalid_configuration,
91 "Buffer size must be greater than 0");
92 }
93
94 if (buffer_size > std::numeric_limits<std::size_t>::max() / 2) {
95 return make_logger_error(logger_error_code::invalid_configuration,
96 "Buffer size is too large");
97 }
98
99 // Validate batch size
100 if (batch_size == 0) {
101 return make_logger_error(logger_error_code::invalid_configuration,
102 "Batch size must be greater than 0");
103 }
104
105 if (batch_size > buffer_size) {
106 return make_logger_error(logger_error_code::invalid_configuration,
107 "Batch size cannot exceed buffer size");
108 }
109
110 // Validate flush interval
111 if (flush_interval.count() < 0) {
112 return make_logger_error(logger_error_code::invalid_configuration,
113 "Flush interval must be non-negative");
114 }
115
116 if (flush_interval.count() > 3600000) { // 1 hour max
117 return make_logger_error(logger_error_code::invalid_configuration,
118 "Flush interval too large (max 1 hour)");
119 }
120
121 // Validate queue settings
122 if (max_queue_size == 0) {
123 return make_logger_error(logger_error_code::invalid_configuration,
124 "Max queue size must be greater than 0");
125 }
126
127 if (max_queue_size < batch_size) {
128 return make_logger_error(logger_error_code::invalid_configuration,
129 "Max queue size must be at least as large as batch size");
130 }
131
132 // Validate file settings
133 if (max_file_size < 1024) { // Minimum 1KB
134 return make_logger_error(logger_error_code::invalid_configuration,
135 "Max file size too small (minimum 1KB)");
136 }
137
138 if (max_file_count == 0) {
139 return make_logger_error(logger_error_code::invalid_configuration,
140 "Max file count must be greater than 0");
141 }
142
143 if (max_file_count > 1000) {
144 return make_logger_error(logger_error_code::invalid_configuration,
145 "Max file count too large (max 1000)");
146 }
147
148 // Validate network settings if configured
149 if (!remote_host.empty()) {
150 if (remote_port == 0) {
151 return make_logger_error(logger_error_code::invalid_configuration,
152 "Remote port must be specified when remote host is set");
153 }
154
155 if (network_timeout.count() <= 0) {
156 return make_logger_error(logger_error_code::invalid_configuration,
157 "Network timeout must be positive");
158 }
159
160 if (network_retry_count > 100) {
161 return make_logger_error(logger_error_code::invalid_configuration,
162 "Network retry count too large (max 100)");
163 }
164 }
165
166 // Validate writer settings
167 if (max_writers == 0) {
168 return make_logger_error(logger_error_code::invalid_configuration,
169 "Must allow at least one writer");
170 }
171
172 if (max_writers > 100) {
173 return make_logger_error(logger_error_code::invalid_configuration,
174 "Max writers too large (max 100)");
175 }
176
177 // Validate thread count
178 if (writer_thread_count == 0) {
179 return make_logger_error(logger_error_code::invalid_configuration,
180 "Writer thread count must be at least 1");
181 }
182
183 if (writer_thread_count > 32) {
184 return make_logger_error(logger_error_code::invalid_configuration,
185 "Writer thread count too large (max 32)");
186 }
187
188 // Validate feature combinations
189 if (use_lock_free && queue_overflow_policy == overflow_policy::grow) {
190 return make_logger_error(logger_error_code::invalid_configuration,
191 "Lock-free queue cannot use grow overflow policy");
192 }
193
194 if (!async && batch_size > 1) {
195 return make_logger_error(logger_error_code::invalid_configuration,
196 "Batch processing requires async mode");
197 }
198
199 return result_void{};
200 }
