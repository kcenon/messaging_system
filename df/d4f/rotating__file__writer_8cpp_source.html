97bool rotating_file_writer::should_rotate() const {
98 switch (rotation_type_) {
99 case rotation_type::size:
100 return get_file_size() >= max_size_;
101
102 case rotation_type::daily:
103 case rotation_type::hourly:
104 return should_rotate_by_time();
105
106 case rotation_type::size_and_time:

114void rotating_file_writer::perform_rotation() {
115 // Close current file
116 close();
117
118 // Generate new filename for the current log
119 std::string rotated_name = generate_rotated_filename();
120
121 // Rename current file
122 try {
123 if (std::filesystem::exists(filename_)) {
124 std::filesystem::rename(filename_, rotated_name);
125 }
126 } catch (const std::exception& e) {
127 std::cerr << "Failed to rotate log file: " << e.what() << std::endl;
128 }
129
130 // Clean up old files
131 cleanup_old_files();
132
133 // Open new file
134 open();
135
136 // Update rotation time
137 last_rotation_time_ = std::chrono::system_clock::now();
138 current_period_start_ = last_rotation_time_;
139}

141std::string rotating_file_writer::generate_rotated_filename(int index) const {
142 std::ostringstream oss;
143 std::filesystem::path dir = std::filesystem::path(filename_).parent_path();
144
145 if (!dir.empty()) {
146 oss << dir.string() << "/";
147 }
148
149 oss << base_filename_;
150
151 // Add timestamp or index
152 auto now = std::chrono::system_clock::now();
153 auto time_t = std::chrono::system_clock::to_time_t(now);
154
155 switch (rotation_type_) {
156 case rotation_type::size:
157 if (index >= 0) {
158 oss << "." << index;
159 } else {
160 // Find next available index
161 int next_index = 1;
162 while (std::filesystem::exists(oss.str() + "." + std::to_string(next_index) + file_extension_)) {
163 next_index++;
164 }
165 oss << "." << next_index;
166 }
167 break;
168
169 case rotation_type::daily:
170 oss << "." << std::put_time(std::localtime(&time_t), "%Y%m%d");
171 break;
172
173 case rotation_type::hourly:
174 oss << "." << std::put_time(std::localtime(&time_t), "%Y%m%d_%H");
175 break;
176
177 case rotation_type::size_and_time:
178 oss << "." << std::put_time(std::localtime(&time_t), "%Y%m%d_%H%M%S");
179 break;
180 }
181
182 oss << file_extension_;
183 return oss.str();
184}

186void rotating_file_writer::cleanup_old_files() {
187 auto backup_files = get_backup_files();
188
189 if (backup_files.size() > max_files_) {
190 // Sort by modification time (oldest first)
191 std::sort(backup_files.begin(), backup_files.end(),
192 [](const std::string& a, const std::string& b) {
193 return std::filesystem::last_write_time(a) <
194 std::filesystem::last_write_time(b);
195 });
196
197 // Remove oldest files
198 size_t files_to_remove = backup_files.size() - max_files_;
199 for (size_t i = 0; i < files_to_remove; ++i) {
200 try {
201 std::filesystem::remove(backup_files[i]);
202 } catch (const std::exception& e) {
203 std::cerr << "Failed to remove old log file: " << e.what() << std::endl;
204 }
205 }
206 }
207}

209std::vector<std::string> rotating_file_writer::get_backup_files() const {
210 std::vector<std::string> files;
211 std::filesystem::path dir = std::filesystem::path(filename_).parent_path();
212 if (dir.empty()) {
213 dir = ".";
214 }
215
216 // Create regex pattern for backup files
217 std::string pattern = base_filename_ + R"(\.(\d+|\\d{8}|\\d{8}_\\d{2}|\\d{8}_\\d{6}))" +
218 std::regex_replace(file_extension_, std::regex(R"(\.)"), R"(\\.)");
219 std::regex backup_regex(pattern);
220
221 try {
222 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
223 if (entry.is_regular_file()) {
224 std::string filename = entry.path().filename().string();
225 if (std::regex_match(filename, backup_regex)) {
226 files.push_back(entry.path().string());
227 }
228 }
229 }
230 } catch (const std::exception& e) {
231 std::cerr << "Error listing backup files: " << e.what() << std::endl;
232 }
233
234 return files;
235}

237bool rotating_file_writer::should_rotate_by_time() const {
238 auto now = std::chrono::system_clock::now();
239
240 switch (rotation_type_) {
241 case rotation_type::daily:
242 case rotation_type::size_and_time: {
243 // Check if we're in a new day
244 auto now_time_t = std::chrono::system_clock::to_time_t(now);
245 auto start_time_t = std::chrono::system_clock::to_time_t(current_period_start_);
246
247 std::tm now_tm = *std::localtime(&now_time_t);
248 std::tm start_tm = *std::localtime(&start_time_t);
249
250 return now_tm.tm_year != start_tm.tm_year ||
251 now_tm.tm_mon != start_tm.tm_mon ||
252 now_tm.tm_mday != start_tm.tm_mday;
253 }
254
255 case rotation_type::hourly: {
256 // Check if we're in a new hour
257 auto duration = now - current_period_start_;
258 return duration >= std::chrono::hours(1);
259 }
260
261 default:
262 return false;
263 }
264}
