165// Template implementation for add_value
166template<typename T>
167messaging_container_builder& messaging_container_builder::add_value(const std::string& key, T&& value) {
168 if constexpr (std::is_same_v<std::decay_t<T>, bool>) {
169 auto val = std::make_shared<bool_value>(key, value);
170 container_->add(val);
171 } else if constexpr (std::is_integral_v<std::decay_t<T>>) {
172 if constexpr (sizeof(T) <= 4) {
173 auto val = std::make_shared<int_value>(key, static_cast<int32_t>(value));
174 container_->add(val);
175 } else {
176 auto val = std::make_shared<long_value>(key, static_cast<int64_t>(value));
177 container_->add(val);
178 }
179 } else if constexpr (std::is_floating_point_v<std::decay_t<T>>) {
180 if constexpr (std::is_same_v<std::decay_t<T>, float>) {
181 auto val = std::make_shared<float_value>(key, value);
182 container_->add(val);
183 } else {
184 auto val = std::make_shared<double_value>(key, value);
185 container_->add(val);
186 }
187 } else if constexpr (std::is_same_v<std::decay_t<T>, std::string>) {
188 auto val = std::make_shared<string_value>(key, value);
189 container_->add(val);
190 }
191
192 return *this;
193}
