139 auto thread_pool::start(void) -> result_void
140 {
141 // Validate that workers have been added
142 if (workers_.empty())
143 {
144 return error{error_code::invalid_argument, "no workers to start"};
145 }
146
147 // Attempt to start each worker
148 for (auto& worker : workers_)
149 {
150 auto start_result = worker->start();
151 if (start_result.has_error())
152 {
153 // If any worker fails, stop all and return error
154 stop();
155 return start_result.get_error();
156 }
157 }
158
159 // Mark pool as successfully started
160 start_pool_.store(true);
161
162 return {};
163 }

206 auto thread_pool::enqueue(std::unique_ptr<job>&& job) -> result_void
207 {
208 // Validate inputs
209 if (job == nullptr)
210 {
211 return error{error_code::invalid_argument, "job is null"};
212 }
213
214 if (job_queue_ == nullptr)
215 {
216 return error{error_code::resource_allocation_failed, "job queue is null"};
217 }
218
219 // Delegate to adaptive queue for optimal processing
220 auto enqueue_result = job_queue_->enqueue(std::move(job));
221 if (enqueue_result.has_error())
222 {
223 return enqueue_result.get_error();
224 }
225
226 return {};

251 auto thread_pool::enqueue(std::unique_ptr<thread_worker>&& worker) -> result_void
252 {
253 if (worker == nullptr)
254 {
255 return error{error_code::invalid_argument, "worker is null"};
256 }
257
258 if (job_queue_ == nullptr)
259 {
260 return error{error_code::resource_allocation_failed, "job queue is null"};
261 }
262
263 worker->set_job_queue(job_queue_);
264 worker->set_context(context_);
265
266 if (start_pool_.load())
267 {
268 auto start_result = worker->start();
269 if (start_result.has_error())
270 {
271 stop();
272 return start_result.get_error();
273 }
274 }
275
276 workers_.emplace_back(std::move(worker));
277
278 return {};

315 auto thread_pool::stop(const bool& immediately_stop) -> result_void
316 {
317 if (!start_pool_.load())
318 {
319 return {};
320 }
321
322 if (job_queue_ != nullptr)
323 {
324 job_queue_->stop_waiting_dequeue();
325
326 if (immediately_stop)
327 {
328 job_queue_->clear();
329 }
330 }
331
332 for (auto& worker : workers_)
333 {
334 auto stop_result = worker->stop();
335 if (stop_result.has_error())
336 {
337 context_.log(log_level::error,
338 formatter::format("error stopping worker: {}",
339 stop_result.get_error().to_string()));
340 }
341 }
342
343 start_pool_.store(false);
344 return {};
345 }

407 auto thread_pool::submit_task(std::function<void()> task) -> bool
408 {
409 if (!task)
410 {
411 return false;
412 }
413
414 auto callback_job = std::make_unique<kcenon::thread::callback_job>(
415 [task = std::move(task)]() -> kcenon::thread::result_void {
416 task();
417 return {};
418 });
419
420 auto result = enqueue(std::move(callback_job));
421 return result.has_error() == false;
422 }
