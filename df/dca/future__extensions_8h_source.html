248 template<typename... Futures>
249 size_t wait_for_any(Futures&... futures) {
250 // Implementation would require a more complex approach
251 // This is a simplified version
252 size_t index = 0;
253 auto check_future = [&](auto& future) {
254 if (future.is_ready()) {
255 return true;
256 }
257 ++index;
258 return false;
259 };
260
261 while (true) {
262 index = 0;
263 if ((check_future(futures) || ...)) {
264 return index;
265 }
266 std::this_thread::yield();
267 }
