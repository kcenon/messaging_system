51// Helper function to create a simple job
52std::unique_ptr<job> create_simple_job(std::atomic<size_t>& counter) {
53 return std::make_unique<callback_job>([&counter]() -> result_void {
54 counter.fetch_add(1, std::memory_order_relaxed);
55 return result_void();
56 });
57}

59// Benchmark: Single Producer Single Consumer (SPSC)
60template<typename QueueType>
61static void BM_SPSC(benchmark::State& state) {
62 const size_t num_operations = state.range(0);
63
64 for (auto _ : state) {
65 QueueType queue;
66 std::atomic<size_t> counter{0};
67
68 std::thread producer([&queue, &counter, num_operations]() {
69 for (size_t i = 0; i < num_operations; ++i) {
70 queue.enqueue(create_simple_job(counter));
71 }
72 });
73
74 std::thread consumer([&queue, num_operations]() {
75 size_t consumed = 0;
76 while (consumed < num_operations) {
77 auto result = queue.dequeue();
78 if (result.has_value()) {
79 result.value()->do_work();
80 consumed++;
81 }
82 }
83 });
84
85 producer.join();
86 consumer.join();
87
88 state.counters["throughput"] = benchmark::Counter(
89 static_cast<double>(num_operations),
90 benchmark::Counter::kIsRate
91 );
92 }
93}

95// Benchmark: Multiple Producer Multiple Consumer (MPMC)
96template<typename QueueType>
97static void BM_MPMC(benchmark::State& state) {
98 const size_t num_producers = state.range(0);
99 const size_t num_consumers = state.range(1);
100 const size_t operations_per_producer = state.range(2);
101 const size_t total_operations = num_producers * operations_per_producer;
102
103 for (auto _ : state) {
104 QueueType queue;
105 std::atomic<size_t> produced{0};
106 std::atomic<size_t> consumed{0};
107 std::atomic<size_t> counter{0};
108
109 std::vector<std::thread> producers;
110 std::vector<std::thread> consumers;
111
112 // Start producers
113 for (size_t i = 0; i < num_producers; ++i) {
114 producers.emplace_back([&queue, &produced, &counter, operations_per_producer]() {
115 for (size_t j = 0; j < operations_per_producer; ++j) {
116 queue.enqueue(create_simple_job(counter));
117 produced.fetch_add(1, std::memory_order_relaxed);
118 }
119 });
120 }
121
122 // Start consumers
123 for (size_t i = 0; i < num_consumers; ++i) {
124 consumers.emplace_back([&queue, &consumed, total_operations]() {
125 while (consumed.load(std::memory_order_relaxed) < total_operations) {
126 auto result = queue.dequeue();
127 if (result.has_value()) {
128 result.value()->do_work();
129 consumed.fetch_add(1, std::memory_order_relaxed);
130 }
131 }
132 });
133 }
134
135 // Wait for completion
136 for (auto& t : producers) t.join();
137 for (auto& t : consumers) t.join();
138
139 state.counters["throughput"] = benchmark::Counter(
140 static_cast<double>(total_operations),
141 benchmark::Counter::kIsRate
142 );
143 state.counters["producers"] = num_producers;
144 state.counters["consumers"] = num_consumers;
145 }
146}

148// Benchmark: Batch Operations
149template<typename QueueType>
150static void BM_BatchOperations(benchmark::State& state) {
151 const size_t batch_size = state.range(0);
152 const size_t num_batches = state.range(1);
153
154 for (auto _ : state) {
155 QueueType queue;
156 std::atomic<size_t> counter{0};
157
158 // Enqueue in batches
159 for (size_t i = 0; i < num_batches; ++i) {
160 std::vector<std::unique_ptr<job>> batch;
161 for (size_t j = 0; j < batch_size; ++j) {
162 batch.push_back(create_simple_job(counter));
163 }
164 queue.enqueue_batch(std::move(batch));
165 }
166
167 // Dequeue in batches
168 size_t total_dequeued = 0;
169 while (total_dequeued < batch_size * num_batches) {
170 auto batch = queue.dequeue_batch();
171 for (auto& job : batch) {
172 job->do_work();
173 }
174 total_dequeued += batch.size();
175 }
176
177 state.counters["throughput"] = benchmark::Counter(
178 static_cast<double>(batch_size * num_batches),
179 benchmark::Counter::kIsRate
180 );
181 state.counters["batch_size"] = batch_size;
182 }
183}

185// Benchmark: High Contention Scenario
186template<typename QueueType>
187static void BM_HighContention(benchmark::State& state) {
188 const size_t num_threads = state.range(0);
189 const size_t operations_per_thread = state.range(1);
190
191 for (auto _ : state) {
192 QueueType queue;
193 std::atomic<size_t> counter{0};
194 std::vector<std::thread> threads;
195
196 // Each thread acts as both producer and consumer
197 for (size_t i = 0; i < num_threads; ++i) {
198 threads.emplace_back([&queue, &counter, operations_per_thread]() {
199 std::random_device rd;
200 std::mt19937 gen(rd());
201 std::uniform_int_distribution<> dis(0, 1);
202
203 for (size_t j = 0; j < operations_per_thread; ++j) {
204 if (dis(gen) == 0) {
205 // Produce
206 queue.enqueue(create_simple_job(counter));
207 } else {
208 // Consume
209 auto result = queue.dequeue();
210 if (result.has_value()) {
211 result.value()->do_work();
212 }
213 }
214 }
215 });
216 }
217
218 for (auto& t : threads) t.join();
219
220 state.counters["throughput"] = benchmark::Counter(
221 static_cast<double>(num_threads * operations_per_thread),
222 benchmark::Counter::kIsRate
223 );
224 state.counters["threads"] = num_threads;
225 }
226}
