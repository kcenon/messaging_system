87 std::string to_w3c_traceparent() const {
88 // Version-TraceId-SpanId-TraceFlags
89 return "00-" + trace_id + "-" + span_id + "-" + trace_flags;
90 }

95 static monitoring_system::result<trace_context> from_w3c_traceparent(const std::string& header) {
96 // Parse format: version-traceid-spanid-traceflags
97 if (header.length() < 55) { // Minimum valid length
98 return monitoring_system::make_error<trace_context>(monitoring_system::monitoring_error_code::invalid_argument);
99 }
100
101 trace_context ctx;
102 // Simple parsing - production code would be more robust
103 size_t pos = 0;
104 size_t dash1 = header.find('-', pos);
105 size_t dash2 = header.find('-', dash1 + 1);
106 size_t dash3 = header.find('-', dash2 + 1);
107
108 if (dash1 == std::string::npos || dash2 == std::string::npos || dash3 == std::string::npos) {
109 return monitoring_system::make_error<trace_context>(monitoring_system::monitoring_error_code::invalid_argument);
110 }
111
112 ctx.trace_id = header.substr(dash1 + 1, dash2 - dash1 - 1);
113 ctx.span_id = header.substr(dash2 + 1, dash3 - dash2 - 1);
114 ctx.trace_flags = header.substr(dash3 + 1);
115
116 return ctx;
117 }
118};

237 template<typename Carrier>
238 void inject_context(const trace_context& context, Carrier& carrier) {
239 carrier["traceparent"] = context.to_w3c_traceparent();
240 if (!context.trace_state.empty()) {
241 carrier["tracestate"] = context.trace_state;
242 }
243 // Inject baggage
244 for (const auto& [key, value] : context.baggage) {
245 carrier["baggage-" + key] = value;
246 }
247 }
248
252 template<typename Carrier>
253 monitoring_system::result<trace_context> extract_context_from_carrier(const Carrier& carrier) {
254 auto traceparent_it = carrier.find("traceparent");
255 if (traceparent_it == carrier.end()) {
256 return monitoring_system::make_error<trace_context>(monitoring_system::monitoring_error_code::not_found);
257 }
258
259 auto ctx_result = trace_context::from_w3c_traceparent(traceparent_it->second);
260 if (!ctx_result) {
261 return monitoring_system::make_error<trace_context>(ctx_result.get_error().code);
262 }
263
264 auto ctx = ctx_result.value();
265
266 // Extract tracestate if present
267 auto tracestate_it = carrier.find("tracestate");
268 if (tracestate_it != carrier.end()) {
269 ctx.trace_state = tracestate_it->second;
270 }
271
272 // Extract baggage
273 for (const auto& [key, value] : carrier) {
274 if (key.find("baggage-") == 0) { // C++17 compatible
275 ctx.baggage[key.substr(8)] = value;
276 }
277 }
278
279 return ctx;
280 }
