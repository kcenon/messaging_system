201TEST_F(ServiceContainerTest, ThreadSafety) {
202 constexpr int num_threads = 8;
203 constexpr int registrations_per_thread = 10;
204
205 // Register services concurrently
206 std::vector<std::thread> threads;
207 for (int t = 0; t < num_threads; ++t) {
208 threads.emplace_back([this, t, registrations_per_thread]() {
209 for (int i = 0; i < registrations_per_thread; ++i) {
210 std::string service_name = "service_" + std::to_string(t) + "_" + std::to_string(i);
211 int expected_value = t * 100 + i;
212
213 container_->register_service<MockService>(
214 service_name,
215 [expected_value]() { return std::make_shared<MockService>(expected_value); }
216 );
217 }
218 });
219 }
220
221 for (auto& thread : threads) {
222 thread.join();
223 }
224
225 // Verify all services were registered
226 auto registered_services = container_->get_registered_services();
227 EXPECT_EQ(registered_services.size(), num_threads * registrations_per_thread);
228
229 // Resolve services concurrently
230 std::atomic<int> successful_resolutions{0};
231 threads.clear();
232
233 for (int t = 0; t < num_threads; ++t) {
234 threads.emplace_back([this, t, registrations_per_thread, &successful_resolutions]() {
235 for (int i = 0; i < registrations_per_thread; ++i) {
236 std::string service_name = "service_" + std::to_string(t) + "_" + std::to_string(i);
237 int expected_value = t * 100 + i;
238
239 auto service = container_->resolve<MockService>(service_name);
240 if (service && service->get_value() == expected_value) {
241 successful_resolutions++;
242 }
243 }
244 });
245 }
