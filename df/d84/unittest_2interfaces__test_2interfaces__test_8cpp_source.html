37TEST(interfaces_test, executor_interface_thread_pool)
38{
39 auto pool = std::make_shared<thread_pool>("ifx_pool");
40 // add one worker
41 std::vector<std::unique_ptr<thread_worker>> workers;
42 {
43 auto w = std::make_unique<thread_worker>(false);
44 w->set_wake_interval(std::chrono::milliseconds(10));
45 workers.push_back(std::move(w));
46 }
47 ASSERT_TRUE(pool->enqueue_batch(std::move(workers)));
48
49 std::atomic<int> count{0};
50 executor_interface* exec = pool.get();
51 auto r = exec->execute(std::make_unique<callback_job>([&count]() -> result_void {
52 count.fetch_add(1);
53 return result_void();
54 }));
55 ASSERT_TRUE(r);
56
57 // Start after enqueue so worker picks up existing job
58 ASSERT_TRUE(pool->start());
59
60 auto start = std::chrono::steady_clock::now();
61 while (count.load() < 1 && std::chrono::steady_clock::now() - start < std::chrono::seconds(2)) {
62 std::this_thread::sleep_for(std::chrono::milliseconds(10));
63 }
64 EXPECT_EQ(count.load(), 1);
65 EXPECT_TRUE(exec->shutdown());
66}
