207 class scoped_timer {
208 public:
209 explicit scoped_timer(std::atomic<std::uint64_t>& target)
210 : target_(target), start_time_(std::chrono::steady_clock::now()) {}
211
212 ~scoped_timer() {
213 auto end_time = std::chrono::steady_clock::now();
214 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
215 end_time - start_time_).count();
216 target_.fetch_add(static_cast<std::uint64_t>(duration));
217 }
218
219 scoped_timer(const scoped_timer&) = delete;
220 scoped_timer& operator=(const scoped_timer&) = delete;
221 scoped_timer(scoped_timer&&) = delete;
222 scoped_timer& operator=(scoped_timer&&) = delete;
223
224 private:
225 std::atomic<std::uint64_t>& target_;
226 std::chrono::steady_clock::time_point start_time_;
227 };
