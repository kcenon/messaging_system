57 bool sqlite_manager::connect(const std::string& connect_string)
58 {
59#ifdef USE_SQLITE
60 std::lock_guard<std::mutex> lock(sqlite_mutex_);
61 try {
62 sqlite3* db = nullptr;
63
64 // Open or create the database
65 int result = sqlite3_open(connect_string.c_str(), &db);
66
67 if (result != SQLITE_OK) {
68 std::cerr << "SQLite connection failed: " << sqlite3_errmsg(db) << std::endl;
69 if (db) {
70 sqlite3_close(db);
71 }
72 return false;
73 }
74
75 connection_ = db;
76
77 // Enable foreign key constraints
78 if (!create_query("PRAGMA foreign_keys = ON")) {
79 std::cerr << "Warning: Failed to enable foreign key constraints" << std::endl;
80 }
81
82 return true;
83 } catch (const std::exception& e) {
84 std::cerr << "SQLite connection error: " << e.what() << std::endl;
85 }
86#else
87 std::cerr << "SQLite support not compiled. Connection: " << connect_string.substr(0, 20) << "..." << std::endl;
88#endif
89 return false;
90 }

134 database_result sqlite_manager::select_query(const std::string& query_string)
135 {
136 database_result result;
137#ifdef USE_SQLITE
138 if (!connection_) return result;
139 std::lock_guard<std::mutex> lock(sqlite_mutex_);
140 try {
141 sqlite3* db = static_cast<sqlite3*>(connection_);
142 sqlite3_stmt* stmt = nullptr;
143
144 // Prepare the statement
145 int prepare_result = sqlite3_prepare_v2(db, query_string.c_str(), -1, &stmt, nullptr);
146 if (prepare_result != SQLITE_OK) {
147 std::cerr << "SQLite prepare failed: " << sqlite3_errmsg(db) << std::endl;
148 return result;
149 }
150
151 // Get column count and names
152 int column_count = sqlite3_column_count(stmt);
153 std::vector<std::string> column_names;
154 for (int i = 0; i < column_count; i++) {
155 column_names.push_back(sqlite3_column_name(stmt, i));
156 }
157
158 // Execute and fetch results
159 while (sqlite3_step(stmt) == SQLITE_ROW) {
160 database_row row;
161
162 for (int i = 0; i < column_count; i++) {
163 const std::string& column_name = column_names[i];
164 row[column_name] = convert_sqlite_value(stmt, i);
165 }
166
167 result.push_back(std::move(row));
168 }
169
170 // Clean up
171 sqlite3_finalize(stmt);
172
173 } catch (const std::exception& e) {
174 std::cerr << "Select query error: " << e.what() << std::endl;
175 }
176#else
177 std::cerr << "SQLite support not compiled. Query: " << query_string.substr(0, 20) << "..." << std::endl;
178 // Mock data for testing
179 if (query_string.find("SELECT") != std::string::npos) {
180 database_row mock_row;
181 mock_row["id"] = int64_t(1);
182 mock_row["name"] = std::string("sqlite_mock_data");
183 result.push_back(mock_row);
184 }
185#endif
186 return result;
187 }

283 database_value sqlite_manager::convert_sqlite_value(void* stmt, int column_index)
284 {
285#ifdef USE_SQLITE
286 sqlite3_stmt* sqlite_stmt = static_cast<sqlite3_stmt*>(stmt);
287
288 int sqlite_type = sqlite3_column_type(sqlite_stmt, column_index);
289
290 switch (sqlite_type) {
291 case SQLITE_INTEGER:
292 return static_cast<int64_t>(sqlite3_column_int64(sqlite_stmt, column_index));
293
294 case SQLITE_FLOAT:
295 return sqlite3_column_double(sqlite_stmt, column_index);
296
297 case SQLITE_TEXT:
298 {
299 const char* text = reinterpret_cast<const char*>(sqlite3_column_text(sqlite_stmt, column_index));
300 return std::string(text ? text : "");
301 }
302
303 case SQLITE_BLOB:
304 {
305 // For BLOB data, convert to string representation
306 const void* blob = sqlite3_column_blob(sqlite_stmt, column_index);
307 int blob_size = sqlite3_column_bytes(sqlite_stmt, column_index);
308 if (blob && blob_size > 0) {
309 const char* blob_chars = static_cast<const char*>(blob);
310 return std::string(blob_chars, blob_size);
311 }
312 return std::string();
313 }
314
315 case SQLITE_NULL:
316 default:
317 return nullptr;
318 }
319#endif
320 return nullptr;
321 }
