156 void handleUserConnect(const core::message& connect_message) {
157 try {
158 auto it = connect_message.metadata.headers.find("user_id");
159 auto user_id = (it != connect_message.metadata.headers.end()) ? it->second : "";
160 auto nickname = connect_message.payload.get<std::string>("nickname", "");
161
162 // Check for reconnection
163 bool is_reconnection = false;
164 {
165 std::lock_guard<std::mutex> lock(users_mutex);
166 if (auto it = users.find(user_id); it != users.end()) {
167 is_reconnection = true;
168 it->second.is_connected = true;
169 it->second.retry_count = 0;
170 it->second.last_activity = std::chrono::steady_clock::now();
171 metrics.reconnections++;
172 } else {
173 users[user_id] = {user_id, nickname, std::chrono::steady_clock::now(), 0, true};
174 metrics.active_users++;
175 }
176 }
177
178 // Broadcast appropriate message
179 core::message broadcast;
180 if (is_reconnection) {
181 broadcast.payload.topic = "system.user_reconnected";
182 broadcast.payload.set("message", nickname + " has reconnected");
183 m_logger->log(logger_module::log_level::info,
184 "User reconnected: " + nickname + " (" + user_id + ")");
185 } else {
186 broadcast.payload.topic = "system.user_joined";
187 broadcast.payload.set("message", nickname + " has joined the chat");
188 m_logger->log(logger_module::log_level::info,
189 "New user connected: " + nickname + " (" + user_id + ")");
190 }
191 broadcast.set_priority(core::message_priority::high);
192
193 broadcastToAll(broadcast);
194
195 } catch (const std::exception& e) {
196 m_logger->log(logger_module::log_level::error,
197 "Error handling user connection: " + std::string(e.what()));

227 void handleChatMessage(const core::message& chat_message) {
228 try {
229 auto it_user = chat_message.metadata.headers.find("user_id");
230 auto it_room = chat_message.metadata.headers.find("room_id");
231 auto sender_user_id = (it_user != chat_message.metadata.headers.end()) ? it_user->second : "";
232 auto target_room_id = (it_room != chat_message.metadata.headers.end()) ? it_room->second : "";
233 auto message_text = chat_message.payload.get<std::string>("message", "");
234
235 std::string nickname;
236 bool user_connected = false;
237 {
238 std::lock_guard<std::mutex> lock(users_mutex);
239 if (auto user_iter = users.find(sender_user_id); user_iter != users.end()) {
240 nickname = user_iter->second.nickname;
241 user_connected = user_iter->second.is_connected;
242 user_iter->second.last_activity = std::chrono::steady_clock::now();
243 }
244 }
245
246 if (!nickname.empty() && user_connected) {
247 // Create formatted message
248 core::message chat_msg;
249 chat_msg.payload.topic = "chat.broadcast";
250 chat_msg.metadata.headers["sender"] = nickname;
251 chat_msg.metadata.headers["room"] = target_room_id;
252 chat_msg.metadata.headers["timestamp"] = std::to_string(
253 std::chrono::system_clock::now().time_since_epoch().count()
254 );
255 chat_msg.payload.set("message", message_text);
256
257 // Try to broadcast with retry on failure
258 bool broadcast_success = false;
259 for (int attempt = 0; attempt < 3; ++attempt) {
260 try {
261 if (!target_room_id.empty()) {
262 broadcastToRoom(target_room_id, chat_msg);
263 } else {
264 broadcastToAll(chat_msg);
265 }
266 broadcast_success = true;
267 metrics.messages_processed++;
268 break;
269 } catch (const std::exception& e) {
270 m_logger->log(logger_module::log_level::warning,
271 "Broadcast attempt " + std::to_string(attempt + 1) +
272 " failed: " + std::string(e.what()));
273 if (attempt < 2) {
274 std::this_thread::sleep_for(std::chrono::milliseconds(100 * (attempt + 1)));
275 }
276 }
277 }
278
279 if (!broadcast_success) {
280 // Queue for retry
281 queueForRetry([this, chat_msg]() {
282 broadcastToAll(chat_msg);
283 });
284 metrics.failed_messages++;
285 }
286
287 // Log message for persistence
288 logMessage(nickname, message_text, target_room_id);
289 } else if (!user_connected) {
290 m_logger->log(logger_module::log_level::warning,
291 "Message from disconnected user: " + sender_user_id);
292 }
293 } catch (const std::exception& e) {
294 m_logger->log(logger_module::log_level::error,
295 "Error handling chat message: " + std::string(e.what()));
296 metrics.failed_messages++;
297 }

394 void cleanupInactiveUsers() {

396 auto timeout = 5min;
397
398 std::lock_guard<std::mutex> lock(users_mutex);
399 for (auto user_iter = users.begin(); user_iter != users.end(); ) {
400 if (now - user_iter->second.last_activity > timeout) {
