219 template<typename Job>
220 job_eraser(Job&& job)
221 : vtable_(&vtable_for<std::decay_t<Job>>)
222 , storage_(std::forward<Job>(job)) {}
223
224 void execute() {
225 vtable_->execute(storage_);
226 }
227
228 void destroy() {
229 vtable_->destroy(storage_);
230 }

232 private:
233 struct vtable_t {
234 void (*execute)(void*);
235 void (*destroy)(void*);
236 };

238 template<typename Job>
239 static constexpr vtable_t vtable_for = {
240 [](void* ptr) { static_cast<Job*>(ptr)->do_work(); },
241 [](void* ptr) { static_cast<Job*>(ptr)->~Job(); }
242 };
243
244 const vtable_t* vtable_;
245 alignas(std::max_align_t) char storage_[64]; // Adjust size as needed
