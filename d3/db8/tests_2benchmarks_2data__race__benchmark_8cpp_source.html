59// Test worker that frequently accesses wake_interval
60class WakeIntervalTestWorker : public thread_base {
61public:
62 WakeIntervalTestWorker() : thread_base("wake_interval_test"), access_count_(0) {}
63
64 std::atomic<size_t> access_count_;
65
66protected:
67 auto do_work() -> result_void override {
68 // Simulate frequent wake_interval access
69 for (int i = 0; i < 100; ++i) {
70 auto interval = get_wake_interval_unsafe(); // Direct access without lock
71 if (interval.has_value()) {
72 access_count_++;
73 }
74 }
75 return {};
76 }
77
78 auto should_continue_work() const -> bool override {
79 return access_count_ < 10000;
80 }
81
82private:
83 // Unsafe method to demonstrate data race
84 auto get_wake_interval_unsafe() const -> std::optional<std::chrono::milliseconds> {
85 return wake_interval_; // Direct access without synchronization
86 }
87};

89// Benchmark for wake_interval data race scenario
90static void BM_WakeIntervalDataRace(benchmark::State& state) {
91 for (auto _ : state) {
92 auto worker = std::make_unique<WakeIntervalTestWorker>();
93 worker->start();
94
95 // Multiple threads modifying wake_interval concurrently
96 std::vector<std::thread> threads;
97 for (int i = 0; i < state.range(0); ++i) {
98 threads.emplace_back([&worker, i]() {
99 for (int j = 0; j < 1000; ++j) {
100 worker->set_wake_interval(std::chrono::milliseconds(i * 10 + j));
101 }
102 });
103 }
104
105 for (auto& t : threads) {
106 t.join();
107 }
108
109 worker->stop();
110 state.counters["accesses"] = worker->access_count_.load();
111 }
112}

144// Benchmark for job_queue consistency
145static void BM_JobQueueConsistency(benchmark::State& state) {
146 for (auto _ : state) {
147 auto queue = std::make_shared<job_queue>();
148 std::atomic<size_t> enqueue_count{0};
149 std::atomic<size_t> dequeue_count{0};
150 std::atomic<size_t> size_checks{0};
151
152 // Multiple threads enqueuing/dequeuing/checking size
153 std::vector<std::thread> threads;
154
155 // Enqueue threads
156 for (int i = 0; i < state.range(0) / 2; ++i) {
157 threads.emplace_back([&queue, &enqueue_count]() {
158 for (int j = 0; j < 1000; ++j) {
159 auto job = std::make_unique<callback_job>([]() { return result_void{}; });
160 auto result = queue->enqueue(std::move(job));
161 if (!result) {
162 enqueue_count++;
163 }
164 }
165 });
166 }
167
168 // Dequeue threads
169 for (int i = 0; i < state.range(0) / 2; ++i) {
170 threads.emplace_back([&queue, &dequeue_count]() {
171 for (int j = 0; j < 1000; ++j) {
172 auto job = queue->dequeue();
173 if (job) {
174 dequeue_count++;
175 }
176 }
177 });
178 }
179
180 // Size check thread
181 threads.emplace_back([&queue, &size_checks]() {
182 while (size_checks < 10000) {
183 auto size = queue->size();
184 auto empty = queue->empty();
185 // Check consistency
186 if ((size == 0 && !empty) || (size > 0 && empty)) {
187 // Inconsistency detected
188 break;
189 }
190 size_checks++;
191 }
192 });
193
194 for (auto& t : threads) {
195 t.join();
196 }
197
198 state.counters["enqueued"] = enqueue_count.load();
199 state.counters["dequeued"] = dequeue_count.load();
200 state.counters["size_checks"] = size_checks.load();
201 }
202}

204// Thread pool stress test with data races
205static void BM_ThreadPoolStress(benchmark::State& state) {
206 for (auto _ : state) {
207 auto pool = std::make_shared<thread_pool>();
208 pool->start();
209
210 std::atomic<size_t> completed_jobs{0};
211 std::atomic<size_t> failed_jobs{0};
212
213 // Submit many jobs from multiple threads
214 std::vector<std::thread> submitters;
215 for (int i = 0; i < 4; ++i) {
216 submitters.emplace_back([&pool, &completed_jobs, &failed_jobs]() {
217 for (int j = 0; j < 2500; ++j) {
218 auto result = pool->enqueue(std::make_unique<callback_job>(
219 [&completed_jobs]() {
220 completed_jobs++;
221 return result_void{};
222 }
223 ));
224 if (result.has_error()) {
225 failed_jobs++;
226 }
227 }
228 });
229 }
230
231 for (auto& t : submitters) {
232 t.join();
233 }
234
235 // Wait for all jobs to complete
236 std::this_thread::sleep_for(std::chrono::milliseconds(100));
237 pool->stop();
238
239 state.counters["completed"] = completed_jobs.load();
240 state.counters["failed"] = failed_jobs.load();
241 }
242}
