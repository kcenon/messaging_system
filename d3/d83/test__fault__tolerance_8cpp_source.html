28 // Helper function that fails for the first N attempts
29 result<int> failing_operation() {
30 int current_call = ++call_count;
31 if (success_after_attempts > 0 && current_call <= success_after_attempts) {
32 return make_error<int>(monitoring_error_code::operation_failed,
33 "Simulated failure on attempt " + std::to_string(current_call));
34 }
35 return make_success(42);
36 }

507 for (int i = 0; i < num_threads; ++i) {
