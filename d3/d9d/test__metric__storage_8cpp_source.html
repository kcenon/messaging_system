332 // Launch multiple threads writing metrics
333 for (int t = 0; t < num_threads; ++t) {
334 threads.emplace_back([&storage, t, metrics_per_thread]() {
335 std::random_device rd;
336 std::mt19937 gen(rd());
337 std::uniform_real_distribution<> dis(0.0, 100.0);
338
339 for (int i = 0; i < metrics_per_thread; ++i) {

341 double value = dis(gen);
342 storage.store_metric(metric_name, value);
343
344 // Small delay to increase chance of contention
345 std::this_thread::sleep_for(std::chrono::microseconds(1));
346 }
347 });
348 }
349
350 // Wait for all threads to complete

353 }

356
