256// Test multithreaded access
257TEST_F(MonitoringTest, MultithreadedAccess) {
258 monitor_->start();
259
260 std::vector<std::thread> threads;
261 const int num_threads = 4;
262 const int updates_per_thread = 10;
263
264 for (int t = 0; t < num_threads; ++t) {
265 threads.emplace_back([this, t, updates_per_thread]() {
266 for (int i = 0; i < updates_per_thread; ++i) {
267 system_metrics metrics;
268 metrics.cpu_usage_percent = 10 + t * 10 + i;
269 metrics.memory_usage_bytes = 1024 * 1024 * (100 + t * 10 + i);
270 metrics.timestamp = std::chrono::steady_clock::now();
271
272 monitor_->update_system_metrics(metrics);
273
274 thread_pool_metrics pool_metrics;
275 pool_metrics.worker_threads = t + 1;
276 pool_metrics.jobs_pending = i;
277 pool_metrics.timestamp = std::chrono::steady_clock::now();
278 monitor_->update_thread_pool_metrics(pool_metrics);
279
280 worker_metrics worker_metrics_data;
281 worker_metrics_data.jobs_processed = i;
282 worker_metrics_data.timestamp = std::chrono::steady_clock::now();
283 monitor_->update_worker_metrics(t, worker_metrics_data);
284 }
285 });
286 }
287
288 for (auto& thread : threads) {
289 thread.join();
290 }
291
292 // Should be able to get current snapshot without issues
293 EXPECT_NO_THROW(monitor_->get_current_snapshot());
294
295 monitor_->stop();
296}
