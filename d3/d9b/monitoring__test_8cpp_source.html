325TEST_F(monitoring_test, basic_monitor_thread_safety) {
326 const int num_threads = 10;
327 const int operations_per_thread = 1000;
328 std::vector<std::thread> threads;
329
330 // Spawn threads that update metrics concurrently
331 for (int i = 0; i < num_threads; ++i) {
332 threads.emplace_back([this, i, operations_per_thread]() {
333 for (int j = 0; j < operations_per_thread; ++j) {
334 monitor_->increment_counter("thread_counter", 1.0);
335 monitor_->update_gauge("thread_gauge", i * 100.0 + j);
336 monitor_->record_histogram("thread_histogram", j);
337
338 // Occasionally collect metrics
339 if (j % 100 == 0) {
340 auto result = monitor_->collect_metrics();
341 EXPECT_TRUE(result);
342 }
343 }
344 });
345 }
346
347 // Wait for all threads
348 for (auto& t : threads) {
349 t.join();
350 }
351
352 // Final metrics collection
353 auto result = monitor_->collect_metrics();
354 ASSERT_TRUE(result);
355
356 auto& data = result.value();
357 auto& metrics = data.get_metrics();
358
359 // Verify thread counter
360 for (const auto& metric : metrics) {
361 if (metric.name == "thread_counter") {
362 double expected = num_threads * operations_per_thread;
363 EXPECT_EQ(metric.value, expected)
364 << "Thread counter should be " << expected;
365 break;
366 }
367 }
368}
