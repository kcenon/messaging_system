51void basic_logging_example() {
52 std::cout << "\n=== Basic Logging Example ===\n" << std::endl;
53
54 // Create logger instance
55 auto logger_instance = std::make_shared<logger>();
56
57 // Add console writer
58 logger_instance->add_writer(std::make_unique<console_writer>());
59
60 // Start logger (for async mode)
61 logger_instance->start();
62
63 // Log messages at different levels
64 logger_instance->log(log_level_type::trace, "This is a trace message");
65 logger_instance->log(log_level_type::debug, "Debug information here");
66 logger_instance->log(log_level_type::info, "Application started successfully");
67 logger_instance->log(log_level_type::warning, "This is a warning");
68 logger_instance->log(log_level_type::error, "An error occurred!");
69 logger_instance->log(log_level_type::critical, "Critical system failure!");
70
71 // Log with source location
72 logger_instance->log(log_level_type::info, "Message with location",
73 __FILE__, __LINE__, __func__);
74
75 // Stop and flush
76 logger_instance->stop();
77}

79void multithreaded_logging_example() {
80 std::cout << "\n=== Multithreaded Logging Example ===\n" << std::endl;
81
82 // Create async logger
83 auto logger_instance = std::make_shared<logger>(true, 16384); // Larger buffer for MT
84 logger_instance->add_writer(std::make_unique<console_writer>());
85 logger_instance->start();
86
87 // Launch multiple threads
88 std::vector<std::thread> threads;
89
90 for (int i = 0; i < 4; ++i) {
91 threads.emplace_back([logger_instance, i]() {
92 for (int j = 0; j < 10; ++j) {
93 logger_instance->log(log_level_type::info,
94 "Thread " + std::to_string(i) + " - Message " + std::to_string(j));
95 std::this_thread::sleep_for(std::chrono::milliseconds(10));
96 }
97 });
98 }
99
100 // Wait for all threads
101 for (auto& t : threads) {
102 t.join();
103 }
104
105 logger_instance->stop();
106}

130void sync_vs_async_example() {
131 std::cout << "\n=== Sync vs Async Logging Example ===\n" << std::endl;
132
133 // Synchronous logger
134 std::cout << "Synchronous logging:" << std::endl;
135 auto sync_logger = std::make_shared<logger>(false); // sync mode
136 sync_logger->add_writer(std::make_unique<console_writer>());
137
138 auto start = std::chrono::high_resolution_clock::now();
139 for (int i = 0; i < 100; ++i) {
140 sync_logger->log(log_level_type::info, "Sync log " + std::to_string(i));
141 }
142 auto sync_time = std::chrono::high_resolution_clock::now() - start;
143
144 // Asynchronous logger
145 std::cout << "\nAsynchronous logging:" << std::endl;
146 auto async_logger = std::make_shared<logger>(true); // async mode
147 async_logger->add_writer(std::make_unique<console_writer>());
148 async_logger->start();
149
150 start = std::chrono::high_resolution_clock::now();
151 for (int i = 0; i < 100; ++i) {
152 async_logger->log(log_level_type::info, "Async log " + std::to_string(i));
153 }
154 auto async_time = std::chrono::high_resolution_clock::now() - start;
155
156 async_logger->stop();
157
158 std::cout << "\nSync time: "
159 << std::chrono::duration_cast<std::chrono::microseconds>(sync_time).count()
160 << " ¬µs" << std::endl;
161 std::cout << "Async time: "
162 << std::chrono::duration_cast<std::chrono::microseconds>(async_time).count()
163 << " ¬µs" << std::endl;
164}
