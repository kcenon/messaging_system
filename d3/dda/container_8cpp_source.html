76 { value_types::bool_value,

79 auto b = (d == "true");

454 void value_container::remove(std::string_view target_name,
455 bool update_immediately)
456 {
457 std::unique_lock<std::shared_mutex> lock(mutex_);
458
459 if (!parsed_data_)
460 {
461 deserialize_values(data_string_, false);
462 }
463 bool found = true;
464 while (found)
465 {
466 found = false;
467 auto it = std::find_if(units_.begin(), units_.end(),
468 [&target_name](std::shared_ptr<value> v)
469 { return (v->name() == target_name); });
470 if (it != units_.end())
471 {
472 units_.erase(it);
473 found = true;
474 }
475 }
476 changed_data_ = !update_immediately;
477 if (update_immediately)
478 {
479 data_string_ = datas();
480 }
481 }

553 std::string value_container::serialize(void) const
554 {
555 std::shared_lock<std::shared_mutex> lock(mutex_);
556 const_cast<value_container*>(this)->serialization_count_.fetch_add(1, std::memory_order_relaxed);
557
558 // If everything parsed, just rebuild data
559 std::string ds = (parsed_data_ ? datas() : data_string_);
560
561 // Compose header
562 std::string header;
563 formatter::format_to(std::back_inserter(header), "@header={{");
564
565 if (message_type_ != "data_container")
566 {
567 formatter::format_to(std::back_inserter(header), "[{},{}];",
568 TARGET_ID, target_id_);
569 formatter::format_to(std::back_inserter(header), "[{},{}];",
570 TARGET_SUB_ID, target_sub_id_);
571 formatter::format_to(std::back_inserter(header), "[{},{}];",
572 SOURCE_ID, source_id_);
573 formatter::format_to(std::back_inserter(header), "[{},{}];",
574 SOURCE_SUB_ID, source_sub_id_);
575 }
576 formatter::format_to(std::back_inserter(header), "[{},{}];",
577 MESSAGE_TYPE, message_type_);
578 formatter::format_to(std::back_inserter(header), "[{},{}];",
579 MESSAGE_VERSION, version_);
580 formatter::format_to(std::back_inserter(header), "}};");
581
582 return header + ds;
583 }

594
595 bool value_container::deserialize(const std::string& data_str,
596 bool parse_only_header)
597 {
598 initialize();
599 if (data_str.empty())
600 return false;
601
602 // Remove newlines
603 std::regex newlineRe("\\r\\n?|\\n");
604 std::string clean = std::regex_replace(data_str, newlineRe, "");
605
606 // parse header portion
607 std::regex fullRe("@header=\\s*\\{\\s*(.*?)\\s*\\};");
608 std::smatch match;
609 if (!std::regex_search(clean, match, fullRe))
610 {
611 // No header => parse as data only
612 return deserialize_values(clean, parse_only_header);
613 }
614 // match[1] => inside the header
615 std::string headerInside = match[1].str();
616 std::regex pairRe("\\[(\\w+),(.*?)\\];");
617 auto it = std::sregex_iterator(headerInside.begin(), headerInside.end(),
618 pairRe);
619 auto end = std::sregex_iterator();
620 for (; it != end; ++it)
621 {
622 parsing((*it)[1].str(), std::string(TARGET_ID), (*it)[2].str(), target_id_);
623 parsing((*it)[1].str(), std::string(TARGET_SUB_ID), (*it)[2].str(), target_sub_id_);
624 parsing((*it)[1].str(), std::string(SOURCE_ID), (*it)[2].str(), source_id_);
625 parsing((*it)[1].str(), std::string(SOURCE_SUB_ID), (*it)[2].str(), source_sub_id_);
626 parsing((*it)[1].str(), std::string(MESSAGE_TYPE), (*it)[2].str(), message_type_);
627 parsing((*it)[1].str(), std::string(MESSAGE_VERSION), (*it)[2].str(), version_);
628 }
629
630 return deserialize_values(clean, parse_only_header);
631 }

642 }

644 const std::string value_container::to_xml(void)
645 {
646 if (!parsed_data_)
647 {
648 deserialize_values(data_string_, false);
649 }
650 std::string result;
651 formatter::format_to(std::back_inserter(result), "<container>");
652 formatter::format_to(std::back_inserter(result), "<header>");
653 if (message_type_ != "data_container")
654 {
655 formatter::format_to(std::back_inserter(result),
656 "<target_id>{}</target_id>", target_id_);
657 formatter::format_to(std::back_inserter(result),
658 "<target_sub_id>{}</target_sub_id>",
659 target_sub_id_);
660 formatter::format_to(std::back_inserter(result),
661 "<source_id>{}</source_id>", source_id_);
662 formatter::format_to(std::back_inserter(result),
663 "<source_sub_id>{}</source_sub_id>",
664 source_sub_id_);
665 }
666 formatter::format_to(std::back_inserter(result),
667 "<message_type>{}</message_type>", message_type_);
668 formatter::format_to(std::back_inserter(result),
669 "<version>{}</version>", version_);
670 formatter::format_to(std::back_inserter(result), "</header>");
671
672 formatter::format_to(std::back_inserter(result), "<values>");
673 for (auto& u : units_)
674 {
675 formatter::format_to(std::back_inserter(result), "{}", u->to_xml());
676 }
677 formatter::format_to(std::back_inserter(result), "</values>");
678 formatter::format_to(std::back_inserter(result), "</container>");
679 return result;
680 }

682 const std::string value_container::to_json(void)
683 {
684 if (!parsed_data_)
685 {
686 deserialize_values(data_string_, false);
687 }
688 std::string result;
689 formatter::format_to(std::back_inserter(result), "{{");
690 // header
691 formatter::format_to(std::back_inserter(result), "\"header\":{{");
692 if (message_type_ != "data_container")
693 {
694 formatter::format_to(std::back_inserter(result),
695 "\"target_id\":\"{}\",", target_id_);
696 formatter::format_to(std::back_inserter(result),
697 "\"target_sub_id\":\"{}\",", target_sub_id_);
698 formatter::format_to(std::back_inserter(result),
699 "\"source_id\":\"{}\",", source_id_);
700 formatter::format_to(std::back_inserter(result),
701 "\"source_sub_id\":\"{}\",", source_sub_id_);
702 }
703 formatter::format_to(std::back_inserter(result),
704 "\"message_type\":\"{}\"", message_type_);
705 formatter::format_to(std::back_inserter(result), ",\"version\":\"{}\"",
706 version_);
707 formatter::format_to(std::back_inserter(result),
708 "}},"); // end header
709
710 // values
711 formatter::format_to(std::back_inserter(result), "\"values\":{{");
712 bool first = true;
713 for (auto& u : units_)
714 {
715 formatter::format_to(std::back_inserter(result),

821 bool value_container::deserialize_values(const std::string& data,
822 bool parse_only_header)
823 {
824 if (!units_.empty())
825 {
826 units_.clear();
827 }
828 changed_data_ = false;
829
830 std::regex reData("@data=\\s*\\{\\s*(.*?)\\s*\\};");
831 std::smatch match;
832 if (!std::regex_search(data, match, reData))
833 {
834 data_string_ = "@data={};";
835 parsed_data_ = true;
836 return false;
837 }
838 data_string_ = match[0]; // entire "@data= ... ;"
839
840 if (parse_only_header)
841 {
842 parsed_data_ = false;
843 return true;
844 }
845 parsed_data_ = true;
846
847 // parse items: [name,type,data];
848 std::regex reItems("\\[(\\w+),\\s*(\\w+),\\s*(.*?)\\];");
849 auto it = std::sregex_iterator(data_string_.begin(), data_string_.end(),
850 reItems);
851 auto end = std::sregex_iterator();
852
853 std::vector<std::shared_ptr<value>> temp_list;
854 for (; it != end; ++it)
855 {
856 auto nameStr = (*it)[1].str();
857 auto typeStr = (*it)[2].str();
858 auto dataStr = (*it)[3].str();
859
860 // convert string -> value_types
861 auto vt = convert_value_type(typeStr);
862 // see if we have a dynamic constructor
863 auto dtIt = data_type_map_.find(vt);
864 if (dtIt == data_type_map_.end())
865 {
866 // fallback to a simple null value
867 temp_list.push_back(std::make_shared<value>(
868 nameStr, nullptr, 0, value_types::null_value));
869 continue;
870 }
871 temp_list.push_back(dtIt->second(nameStr, dataStr));
872 }
873
874 // Next, handle container nesting. We rely on the container_value
875 // storing child counts.
876 std::shared_ptr<value> currentContainer = nullptr;
877 for (auto& tVal : temp_list)
878 {
879 if (!currentContainer)
880 {
881 // top-level => add to units_
882 units_.push_back(tVal);
883 tVal->set_parent(nullptr);
884
885 if (tVal->is_container() && (tVal->to_long() > 0))
886 {
887 currentContainer = tVal;
888 }
889 }
890 else
891 {
892 // add to current container
893 // we must dynamic_cast to container_value to call add(...)
894 // directly
895 if (auto c = std::dynamic_pointer_cast<container_value>(
896 currentContainer))
897 {
898 c->add(tVal, false);
899 }
900
901 if (tVal->is_container() && tVal->to_long() > 0)
902 {
903 currentContainer = tVal;
904 }
905 else
906 {
907 // Check if parent's child count is now matched
908 while (currentContainer)
909 {
910 long needed = currentContainer->to_long();
911 long have = (long)currentContainer->child_count();
912 if (have >= needed)
913 {
914 // move up
915 currentContainer = currentContainer->parent();
916 }
917 else
918 {
919 // break if still more children needed
920 break;
921 }
922 }
923 }
924 }
925 }
926 return true;
927 }

929 void value_container::parsing(std::string_view source_name,
930 std::string_view target_name,
931 std::string_view target_value,
932 std::string& target_variable)
933 {
934 if (source_name == target_name)
935 {
936 target_variable = std::string(target_value);
937 // trim
938 if (!target_variable.empty())
939 {
940 // simplistic trim
941 while (!target_variable.empty()
942 && (target_variable.front() == ' '))
943 {
944 target_variable.erase(target_variable.begin());
945 }
946 while (!target_variable.empty()
947 && (target_variable.back() == ' '))
948 {
949 target_variable.pop_back();
950 }
951 }
952 }
