64 void log(thread_module::log_level level, const std::string& message,
65 const std::string& file, int line, const std::string& function) {
66 if (level > min_level_.load(std::memory_order_acquire)) {
67 return;
68 }
69
70 // Apply global filter
71 if (global_filter_ && !global_filter_->should_log(level, message, file, line, function)) {
72 return;
73 }
74
75 auto timestamp = std::chrono::system_clock::now();
76
77 // Record metrics if enabled
78 if (metrics_enabled_ && metrics_collector_) {
79 auto start = std::chrono::high_resolution_clock::now();
80
81 if (async_ && collector_) {
82 // Async logging through collector
83 bool enqueued = collector_->enqueue(level, message, file, line, function, timestamp);
84
85 auto end = std::chrono::high_resolution_clock::now();
86 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start);
87
88 if (enqueued) {
89 metrics_collector_->record_enqueue(message.size(), duration);
90 } else {
91 metrics_collector_->record_drop();
92 }
93
94 // Update queue metrics
95 auto [current, max] = collector_->get_queue_metrics();
96 metrics_collector_->update_queue_size(current, max);
97 } else {
98 // Sync logging directly to writers
99 std::lock_guard<std::mutex> lock(writers_mutex_);
100
101 // Get routed writers
102 auto routed_writers = router_->route(level, message, file, line, function, timestamp, named_writers_);
103
104 for (auto* writer : routed_writers) {
105 auto write_start = std::chrono::high_resolution_clock::now();
106 bool success = writer->write(level, message, file, line, function, timestamp);
107 auto write_end = std::chrono::high_resolution_clock::now();
108 auto write_duration = std::chrono::duration_cast<std::chrono::microseconds>(write_end - write_start);
109
110 metrics_collector_->record_write(writer->get_name(), message.size(), write_duration, success);
111 }
112
113 auto end = std::chrono::high_resolution_clock::now();
114 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start);
115 metrics_collector_->record_enqueue(message.size(), duration);
116 }
117
118 metrics_collector_->record_processed(message.size());
119 } else {
120 // Normal logging without metrics
121 if (async_ && collector_) {
122 collector_->enqueue(level, message, file, line, function, timestamp);
123 } else {
124 std::lock_guard<std::mutex> lock(writers_mutex_);
125
126 // Get routed writers
127 auto routed_writers = router_->route(level, message, file, line, function, timestamp, named_writers_);
128
129 for (auto* writer : routed_writers) {
130 writer->write(level, message, file, line, function, timestamp);
131 }
132 }
133 }
134 }

136 void flush() {
137 if (async_ && collector_) {
138 collector_->flush();
139 }
140
141 std::lock_guard<std::mutex> lock(writers_mutex_);
142 for (auto& [name, writer] : named_writers_) {
143 writer->flush();
144 }
145 }

147 void add_writer(std::unique_ptr<base_writer> writer) {
148 std::lock_guard<std::mutex> lock(writers_mutex_);
149
150 // Generate a name if not using named_writers
151 std::string name = writer->get_name() + "_" + std::to_string(named_writers_.size());
152
153 if (async_ && collector_) {
154 collector_->add_writer(writer.get());
155 }
156
157 // Keep in both collections for backward compatibility
158 writers_.push_back(writer.get());
159 named_writers_[name] = std::move(writer);
160 }

162 void add_writer(const std::string& name, std::unique_ptr<base_writer> writer) {
163 std::lock_guard<std::mutex> lock(writers_mutex_);
164 if (async_ && collector_) {
165 collector_->add_writer(writer.get());
166 }
167 writers_.push_back(writer.get());
168 named_writers_[name] = std::move(writer);
169 }

245 bool remove_writer(const std::string& name) {
246 std::lock_guard<std::mutex> lock(writers_mutex_);
247 auto it = named_writers_.find(name);
248 if (it != named_writers_.end()) {
249 if (async_ && collector_) {
250 // Note: log_collector doesn't support individual writer removal
251 // This is a limitation of the current design
252 }
253 named_writers_.erase(it);
254 return true;
255 }
256 return false;
257 }

262 return it != named_writers_.end() ? it->second.get() : nullptr;
263 }
