42#ifdef USE_THREAD_SYSTEM_INTEGRATION
43// Helper function to convert kcenon::thread::log_level to logger_system::log_level
44logger_system::log_level convert_log_level(kcenon::thread::log_level level) {
45 switch (level) {
46 case kcenon::thread::log_level::critical: return logger_system::log_level::fatal;
47 case kcenon::thread::log_level::error: return logger_system::log_level::error;
48 case kcenon::thread::log_level::warning: return logger_system::log_level::warn;
49 case kcenon::thread::log_level::info: return logger_system::log_level::info;
50 case kcenon::thread::log_level::debug: return logger_system::log_level::debug;
51 case kcenon::thread::log_level::trace: return logger_system::log_level::trace;
52 default: return logger_system::log_level::info;
53 }
54}
55#else
56// In standalone mode, no conversion needed - both types are the same
57logger_system::log_level convert_log_level(logger_system::log_level level) {
58 return level;
59}
60#endif

142void logger::log(kcenon::thread::log_level level, const std::string& message) {
143#else
144void logger::log(logger_system::log_level level, const std::string& message) {
145#endif
146 if (pimpl_ && level <= pimpl_->min_level_) {
147 for (auto& writer : pimpl_->writers_) {
148 if (writer) {
149 // Create a simple log entry and write it
150 auto now = std::chrono::system_clock::now();
151 writer->write(convert_log_level(level), message, "", 0, "", now);
152 }
