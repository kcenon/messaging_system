308 // On most 64-bit platforms: long long == int64_t == long (8 bytes)
309 // But we can't check this at preprocessor time, so we exclude known problematic platforms
310 #if !defined(_WIN32) && !defined(_WIN64) && !defined(__APPLE__)
311 // On Linux x86_64, long long and int64_t are typically the same type
312 // So we skip these specializations to avoid redefinition errors
