116 auto adaptive_job_queue::enqueue(std::unique_ptr<job>&& value) -> result_void
117 {
118 auto start = std::chrono::high_resolution_clock::now();
119 bool had_contention = false;
120
121 result_void result;
122 auto* impl = get_current_impl();
123
124 // Try to detect contention for adaptive mode
125 if (strategy_ == queue_strategy::ADAPTIVE && current_type_ == queue_type::LEGACY_MUTEX) {
126 // Simple contention detection: if we can't get the lock immediately
127 auto before_lock = std::chrono::high_resolution_clock::now();
128 result = impl->enqueue(std::move(value));
129 auto after_lock = std::chrono::high_resolution_clock::now();
130
131 auto lock_time = std::chrono::duration_cast<std::chrono::nanoseconds>(after_lock - before_lock);
132 if (lock_time.count() > 100) { // More than 100ns to acquire lock
133 had_contention = true;
134 }
135 } else {
136 result = impl->enqueue(std::move(value));
137 }
138
139 auto duration = std::chrono::high_resolution_clock::now() - start;
140 update_metrics(std::chrono::duration_cast<std::chrono::nanoseconds>(duration), had_contention);
141
142 return result;
143 }

225 auto adaptive_job_queue::evaluate_and_switch() -> void
226 {
227 if (strategy_ != queue_strategy::ADAPTIVE) {
228 return;
229 }
230
231 auto now = std::chrono::steady_clock::now();
232 auto elapsed = now - metrics_.last_evaluation;
233
234 if (elapsed < EVALUATION_INTERVAL) {
235 return;
236 }
237
238 // Check if we have enough data
239 if (metrics_.operation_count.load() < MIN_OPERATIONS_FOR_SWITCH) {
240 return;
241 }
242
243 // Evaluate current performance
244 if (current_type_ == queue_type::LEGACY_MUTEX && should_switch_to_lockfree()) {
245 migrate_to_lockfree();
246 } else if (current_type_ == queue_type::LOCKFREE_MPMC && should_switch_to_legacy()) {
247 migrate_to_legacy();
248 }
249
250 // Reset metrics for next evaluation period
251 metrics_.reset();
252 }

281 auto adaptive_job_queue::initialize_strategy() -> void
282 {
283 switch (strategy_) {
284 case queue_strategy::FORCE_LEGACY:
285 current_type_ = queue_type::LEGACY_MUTEX;
286 break;
287
288 case queue_strategy::FORCE_LOCKFREE:
289 current_type_ = queue_type::LOCKFREE_MPMC;
290 break;
291
292 case queue_strategy::AUTO_DETECT:
293 // Use lock-free for systems with 4+ cores
294 if (std::thread::hardware_concurrency() >= 4) {
295 current_type_ = queue_type::LOCKFREE_MPMC;
296 } else {
297 current_type_ = queue_type::LEGACY_MUTEX;
298 }
299 break;
300
301 case queue_strategy::ADAPTIVE:
302 // Start with legacy and let it adapt
303 current_type_ = queue_type::LEGACY_MUTEX;
304 break;
305 }
306
307 // Log removed - logger dependency removed
308 // Previously: log_module::write_information("Adaptive queue initialized with strategy: %s", get_current_type().c_str());
309 }

338 auto adaptive_job_queue::should_switch_to_lockfree() const -> bool
339 {
340 auto ops = metrics_.operation_count.load(std::memory_order_relaxed);
341 if (ops == 0) return false;
342
343 auto avg_latency = static_cast<double>(metrics_.total_latency_ns.load(std::memory_order_relaxed)) / static_cast<double>(ops);
344 auto contention_ratio = static_cast<double>(metrics_.contention_count.load(std::memory_order_relaxed)) / static_cast<double>(ops);
345
346 return (contention_ratio > CONTENTION_THRESHOLD_HIGH && avg_latency > LATENCY_THRESHOLD_HIGH_NS);
347 }

349 auto adaptive_job_queue::should_switch_to_legacy() const -> bool
350 {
351 auto ops = metrics_.operation_count.load(std::memory_order_relaxed);
352 if (ops == 0) return false;
353
354 auto avg_latency = static_cast<double>(metrics_.total_latency_ns.load(std::memory_order_relaxed)) / static_cast<double>(ops);
355 auto contention_ratio = static_cast<double>(metrics_.contention_count.load(std::memory_order_relaxed)) / static_cast<double>(ops);
356
357 // Switch back if contention is low and lock-free overhead isn't worth it
358 return (contention_ratio < CONTENTION_THRESHOLD_LOW && avg_latency > LATENCY_THRESHOLD_LOW_NS * 2);
359 }

361 auto adaptive_job_queue::migrate_to_lockfree() -> void
362 {
363 // Log removed - logger dependency removed
364 // Previously: log_module::write_information("Migrating from mutex-based to lock-free queue");
365
366 // Ensure lock-free queue exists
367 ensure_mpmc_queue();
368
369 // Drain legacy queue into lock-free queue
370 if (legacy_queue_) {
371 while (auto job_result = legacy_queue_->dequeue()) {
372 if (job_result.has_value()) {
373 auto enqueue_result = mpmc_queue_->enqueue(std::move(job_result.value()));
374 (void)enqueue_result; // Ignore result
375 } else {
376 break;
377 }
378 }
379 }
380
381 current_type_.store(queue_type::LOCKFREE_MPMC, std::memory_order_release);
382 metrics_.switch_count.fetch_add(1, std::memory_order_relaxed);
383 }

385 auto adaptive_job_queue::migrate_to_legacy() -> void
386 {
387 // Log removed - logger dependency removed
388 // Previously: log_module::write_information("Migrating from lock-free to mutex-based queue");
389
390 // Ensure legacy queue exists
391 ensure_legacy_queue();
392
393 // Drain lock-free queue into legacy queue
394 if (mpmc_queue_) {
395 auto jobs = mpmc_queue_->dequeue_batch();
396 for (auto& job : jobs) {
397 auto enqueue_result = legacy_queue_->enqueue(std::move(job));
398 (void)enqueue_result; // Ignore result
399 }
400 }
401
402 current_type_.store(queue_type::LEGACY_MUTEX, std::memory_order_release);
403 metrics_.switch_count.fetch_add(1, std::memory_order_relaxed);
404 }
