53 template <typename job_type>
54 auto callback_typed_job_t<job_type>::do_work(void) -> result_void
55 {
56 if (callback_ == nullptr)
57 {
58 return error{error_code::job_invalid, "cannot execute job without callback"};
59 }
60
61 try
62 {
63 return callback_();
64 }
65 catch (const std::exception& e)
66 {
67 return error{error_code::job_execution_failed, e.what()};
68 }
