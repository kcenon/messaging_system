53 void log(kcenon::thread::log_level level, const std::string& message) override {
54 if (level < min_level_) return;
55
56 std::lock_guard<std::mutex> lock(mutex_);
57 auto& stream = (level <= kcenon::thread::log_level::error) ? std::cerr : std::cout;
58
59 stream << "[" << format_time() << "] "
60 << "[" << level_to_string(level) << "] "
61 << message << std::endl;
62 }
63
64 void log(kcenon::thread::log_level level, const std::string& message,
65 const std::string& file, int line, const std::string& function) override {
66 if (level < min_level_) return;
67
68 std::lock_guard<std::mutex> lock(mutex_);
69 auto& stream = (level <= kcenon::thread::log_level::error) ? std::cerr : std::cout;
70
71 stream << "[" << format_time() << "] "
72 << "[" << level_to_string(level) << "] ";

76 std::string filename = (pos != std::string::npos) ? file.substr(pos + 1) : file;

78 }

80 stream << message << std::endl;
81 }

106 std::string format_time() const {
107 auto now = std::chrono::system_clock::now();
108 auto time_t = std::chrono::system_clock::to_time_t(now);
109 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
110 now.time_since_epoch()) % 1000;
111
112 std::ostringstream oss;
113 oss << std::put_time(std::localtime(&time_t), "%H:%M:%S");
114 oss << "." << std::setfill('0') << std::setw(3) << ms.count();
115 return oss.str();
116 }

118 std::string level_to_string(kcenon::thread::log_level level) const {
119 switch (level) {
120 case kcenon::thread::log_level::critical: return "CRITICAL";
121 case kcenon::thread::log_level::error: return "ERROR";
122 case kcenon::thread::log_level::warning: return "WARNING";
123 case kcenon::thread::log_level::info: return "INFO";
124 case kcenon::thread::log_level::debug: return "DEBUG";
125 case kcenon::thread::log_level::trace: return "TRACE";
126 }
127 return "UNKNOWN";
128 }
