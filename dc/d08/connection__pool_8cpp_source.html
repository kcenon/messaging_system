122 bool connection_pool::initialize()
123 {
124 try {
125 // Create minimum number of connections
126 std::lock_guard<std::mutex> lock(pool_mutex_);
127
128 for (size_t i = 0; i < config_.min_connections; ++i) {
129 auto connection = create_connection();
130 if (!connection) {
131 std::cerr << "Failed to create initial connection " << i << std::endl;
132 return false;
133 }
134
135 auto wrapper = std::make_shared<connection_wrapper>(std::move(connection));
136 available_connections_.push(wrapper);
137 ++total_created_;
138 }
139
140 stats_.total_connections = total_created_.load();
141 stats_.available_connections = available_connections_.size();
142
143 // Start maintenance thread
144 maintenance_thread_ = std::thread(&connection_pool::maintenance_thread, this);
145
146 return true;
147 } catch (const std::exception& e) {
148 std::cerr << "Connection pool initialization failed: " << e.what() << std::endl;
149 return false;
150 }
151 }

153 std::shared_ptr<connection_wrapper> connection_pool::acquire_connection()
154 {
155 std::unique_lock<std::mutex> lock(pool_mutex_);
156
157 // Wait for available connection or timeout
158 auto deadline = std::chrono::steady_clock::now() + config_.acquire_timeout;
159
160 while (available_connections_.empty() && !shutdown_requested_) {
161 // Try to create new connection if under limit
162 if (active_count_ + available_connections_.size() < config_.max_connections) {
163 lock.unlock();
164 auto new_conn = create_connection();
165 lock.lock();
166
167 if (new_conn) {
168 auto wrapper = std::make_shared<connection_wrapper>(std::move(new_conn));
169 available_connections_.push(wrapper);
170 ++total_created_;
171 break;
172 }
173 }
174
175 // Wait for connection to become available
176 if (pool_condition_.wait_until(lock, deadline) == std::cv_status::timeout) {
177 ++stats_.failed_acquisitions;
178 return nullptr; // Timeout
179 }
180 }
181
182 if (shutdown_requested_ || available_connections_.empty()) {
183 ++stats_.failed_acquisitions;
184 return nullptr;
185 }
186
187 // Get connection from pool
188 auto connection = available_connections_.front();
189 available_connections_.pop();
190 ++active_count_;
191 ++stats_.successful_acquisitions;
192
193 // Update statistics
194 stats_.active_connections = active_count_.load();
195 stats_.available_connections = available_connections_.size();
196
197 connection->update_last_used();
198 return connection;
199 }

201 void connection_pool::release_connection(std::shared_ptr<connection_wrapper> connection)
202 {
203 if (!connection) return;
204
205 std::lock_guard<std::mutex> lock(pool_mutex_);
206
207 if (shutdown_requested_) {
208 --active_count_;
209 return;
210 }
211
212 // Check if connection is healthy
213 if (!connection->is_healthy() || !validate_connection(connection.get())) {
214 --active_count_;
215 // Connection is unhealthy, don't return to pool
216 return;
217 }
218
219 // Return connection to pool
220 available_connections_.push(connection);
221 --active_count_;
222
223 // Update statistics
224 stats_.active_connections = active_count_.load();
225 stats_.available_connections = available_connections_.size();
226
227 // Notify waiting threads
228 pool_condition_.notify_one();
229 }

251 void connection_pool::shutdown()
252 {
253 shutdown_requested_.store(true);
254
255 {
256 std::lock_guard<std::mutex> lock(pool_mutex_);
257 pool_condition_.notify_all();
258 }
259
260 if (maintenance_thread_.joinable()) {
261 maintenance_thread_.join();
262 }
263
264 // Clear all connections
265 std::lock_guard<std::mutex> lock(pool_mutex_);
266 while (!available_connections_.empty()) {
267 available_connections_.pop();
268 }

292 void connection_pool::maintenance_thread()
293 {
294 while (!shutdown_requested_) {
295 std::this_thread::sleep_for(config_.health_check_interval);
296
297 if (shutdown_requested_) break;
298
299 try {
300 if (config_.enable_health_checks) {
301 health_check();
302 }
303 cleanup_idle_connections();
304 } catch (const std::exception& e) {
305 std::cerr << "Maintenance thread error: " << e.what() << std::endl;
306 }

310 void connection_pool::health_check()
311 {
312 std::lock_guard<std::mutex> lock(pool_mutex_);
313
314 std::queue<std::shared_ptr<connection_wrapper>> healthy_connections;
315
316 while (!available_connections_.empty()) {
317 auto conn = available_connections_.front();
318 available_connections_.pop();
319
320 if (validate_connection(conn.get())) {
321 healthy_connections.push(conn);
322 }
323 // Unhealthy connections are discarded
324 }
325
326 available_connections_ = std::move(healthy_connections);
327 stats_.last_health_check = std::chrono::steady_clock::now();
328 stats_.available_connections = available_connections_.size();
329 }

331 bool connection_pool::validate_connection(connection_wrapper* connection)
332 {
333 if (!connection || !connection->is_healthy()) {
334 return false;
335 }
336
337 try {
338 // Perform basic connectivity test based on database type
339 auto* db = connection->get();
340 if (!db) return false;
341
342 // Simple validation - try to get database type
343 auto type = db->database_type();
344 return type != database_types::none;
345 } catch (const std::exception&) {
346 connection->mark_unhealthy();
347 return false;
348 }
349 }

352 void connection_pool::cleanup_idle_connections()
353 {
354 std::lock_guard<std::mutex> lock(pool_mutex_);
355
356 std::queue<std::shared_ptr<connection_wrapper>> active_connections;
357
358 while (!available_connections_.empty()) {
359 auto conn = available_connections_.front();
360 available_connections_.pop();
361
362 // Keep connection if not idle or if we're at minimum
363 if (!conn->is_idle_timeout_exceeded(config_.idle_timeout) ||
364 active_connections.size() < config_.min_connections) {
365 active_connections.push(conn);
366 }
367 // Idle connections beyond minimum are discarded
368 }
369
370 available_connections_ = std::move(active_connections);
371 stats_.available_connections = available_connections_.size();
372 }

386 bool connection_pool_manager::create_pool(database_types db_type, const connection_pool_config& config)
387 {
388 std::lock_guard<std::mutex> lock(pools_mutex_);
389
390 if (pools_.find(db_type) != pools_.end()) {
391 std::cerr << "Pool for database type already exists" << std::endl;
392 return false;
393 }
394
395 auto factory = create_factory(db_type, config.connection_string);
396 if (!factory) {
397 std::cerr << "Failed to create connection factory for database type" << std::endl;
398 return false;
399 }
400
401 auto pool = std::make_shared<connection_pool>(db_type, config, factory);
402 if (!pool->initialize()) {
403 std::cerr << "Failed to initialize connection pool" << std::endl;
404 return false;
405 }
406
407 pools_[db_type] = pool;
408 return true;
409 }
