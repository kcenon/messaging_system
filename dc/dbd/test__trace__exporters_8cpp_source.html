107TEST_F(TraceExportersTest, JaegerSpanConversion) {
108 trace_export_config config;
109 config.endpoint = "http://jaeger:14268/api/traces";
110 config.format = trace_export_format::jaeger_thrift;
111 config.service_name = "test_service";
112
113 jaeger_exporter exporter(config);
114
115 const auto& span = test_spans_[0];
116 auto jaeger_span = exporter.convert_span(span);
117
118 EXPECT_EQ(jaeger_span.trace_id, span.trace_id);
119 EXPECT_EQ(jaeger_span.span_id, span.span_id);
120 EXPECT_EQ(jaeger_span.operation_name, span.operation_name);
121 EXPECT_EQ(jaeger_span.service_name, "test_service"); // Override from config
122
123 // Check tags conversion
124 bool found_http_method = false;
125 bool found_http_url = false;
126 for (const auto& [key, value] : jaeger_span.tags) {
127 if (key == "http.method" && value == "GET") {
128 found_http_method = true;
129 }
130 if (key == "http.url" && value == "/api/users") {
131 found_http_url = true;
132 }
133 }
134 EXPECT_TRUE(found_http_method);
135 EXPECT_TRUE(found_http_url);
136
137 // Check process tags
138 bool found_service_name = false;
139 for (const auto& [key, value] : jaeger_span.process_tags) {
140 if (key == "service.name" && value == "test_service") {
141 found_service_name = true;
142 }
143 }
144 EXPECT_TRUE(found_service_name);
145}
