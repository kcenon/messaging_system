102 void performance_monitor::record_query_metrics(const query_metrics& metrics)
103 {
104 if (!monitoring_enabled_) return;
105
106 std::lock_guard<std::mutex> lock(metrics_mutex_);
107 query_history_.push_back(metrics);
108
109 // Update query patterns
110 query_patterns_[metrics.query_hash]++;
111
112 auto& avg_time = query_avg_times_[metrics.query_hash];
113 auto count = query_patterns_[metrics.query_hash];
114 avg_time = std::chrono::microseconds(
115 (avg_time.count() * (count - 1) + metrics.execution_time.count()) / count);
116
117 // Check for slow queries
118 if (metrics.execution_time > latency_threshold_) {
119 emit_alert(performance_alert::alert_type::slow_query,
120 "Slow query detected: " + std::to_string(metrics.execution_time.count()) + "Œºs");
121 }
122
123 check_thresholds();
124 }

126 void performance_monitor::record_connection_metrics(database_types db_type, const connection_metrics& metrics)
127 {
128 if (!monitoring_enabled_) return;
129
130 std::lock_guard<std::mutex> lock(metrics_mutex_);
131 auto& stored_metrics = connection_metrics_[db_type];
132 stored_metrics.total_connections.store(metrics.total_connections.load());
133 stored_metrics.active_connections.store(metrics.active_connections.load());
134 stored_metrics.idle_connections.store(metrics.idle_connections.load());
135 stored_metrics.failed_connections.store(metrics.failed_connections.load());
136 stored_metrics.avg_acquisition_time.store(metrics.avg_acquisition_time.load());
137 stored_metrics.max_acquisition_time.store(metrics.max_acquisition_time.load());
138 stored_metrics.last_update = metrics.last_update;
139
140 // Check for connection pool exhaustion
141 auto total = metrics.total_connections.load();
142 auto active = metrics.active_connections.load();
143 if (total > 0 && (double(active) / total) > 0.9) {
144 emit_alert(performance_alert::alert_type::connection_pool_exhaustion,
145 "Connection pool utilization high: " + std::to_string(active) + "/" + std::to_string(total));
146 }
147 }

158 performance_summary performance_monitor::get_performance_summary() const
159 {
160 std::lock_guard<std::mutex> lock(metrics_mutex_);
161
162 performance_summary summary;
163 summary.measurement_start = std::chrono::steady_clock::now() - retention_period_;
164 summary.measurement_end = std::chrono::steady_clock::now();
165
166 if (query_history_.empty()) {
167 return summary;
168 }
169
170 // Filter recent queries
171 auto recent_start = summary.measurement_start;
172 auto recent_queries = std::count_if(query_history_.begin(), query_history_.end(),
173 [recent_start](const query_metrics& m) {
174 return m.start_time >= recent_start;
175 });
176
177 summary.total_queries = recent_queries;
178
179 // Calculate metrics from recent queries
180 std::chrono::microseconds total_time{0};
181 std::chrono::microseconds min_time{std::chrono::microseconds::max()};
182 std::chrono::microseconds max_time{0};
183 size_t successful = 0;
184
185 for (const auto& metrics : query_history_) {
186 if (metrics.start_time < recent_start) continue;
187
188 total_time += metrics.execution_time;
189 min_time = std::min(min_time, metrics.execution_time);
190 max_time = std::max(max_time, metrics.execution_time);
191
192 if (metrics.success) {
193 successful++;
194 } else {
195 summary.error_counts[metrics.error_message]++;
196 }
197 }
198
199 summary.successful_queries = successful;
200 summary.failed_queries = summary.total_queries - successful;
201
202 if (summary.total_queries > 0) {
203 summary.avg_query_time = total_time / summary.total_queries;
204 summary.min_query_time = min_time;
205 summary.max_query_time = max_time;
206 summary.error_rate = double(summary.failed_queries) / summary.total_queries;
207 }
208
209 // Calculate QPS
210 auto duration_seconds = std::chrono::duration_cast<std::chrono::seconds>(
211 summary.measurement_end - summary.measurement_start).count();
212 if (duration_seconds > 0) {
213 summary.queries_per_second = double(summary.total_queries) / duration_seconds;
214 }
215
216 // Connection metrics summary
217 size_t total_connections = 0;
218 size_t active_connections = 0;
219 for (const auto& [db_type, conn_metrics] : connection_metrics_) {
220 total_connections += conn_metrics.total_connections.load();
221 active_connections += conn_metrics.active_connections.load();
222 }
223
224 summary.total_connections = total_connections;
225 summary.active_connections = active_connections;
226 if (total_connections > 0) {
227 summary.connection_utilization = double(active_connections) / total_connections;
228 }
229
230 return summary;
231 }

233 performance_summary performance_monitor::get_performance_summary(database_types db_type) const
234 {
235 std::lock_guard<std::mutex> lock(metrics_mutex_);
236
237 performance_summary summary;
238 summary.measurement_start = std::chrono::steady_clock::now() - retention_period_;
239 summary.measurement_end = std::chrono::steady_clock::now();
240
241 // Filter by database type and time
242 auto recent_start = summary.measurement_start;
243 auto db_queries = std::count_if(query_history_.begin(), query_history_.end(),
244 [recent_start, db_type](const query_metrics& m) {
245 return m.start_time >= recent_start && m.db_type == db_type;
246 });
247
248 summary.total_queries = db_queries;
249
250 // Calculate metrics for specific database type
251 std::chrono::microseconds total_time{0};
252 size_t successful = 0;
253
254 for (const auto& metrics : query_history_) {
255 if (metrics.start_time < recent_start || metrics.db_type != db_type) continue;
256
257 total_time += metrics.execution_time;
258 if (metrics.success) {
259 successful++;
260 }
261 }
262
263 summary.successful_queries = successful;
264 summary.failed_queries = summary.total_queries - successful;
265
266 if (summary.total_queries > 0) {
267 summary.avg_query_time = total_time / summary.total_queries;
268 summary.error_rate = double(summary.failed_queries) / summary.total_queries;
269 }
270
271 // Connection metrics for specific database
272 auto it = connection_metrics_.find(db_type);
273 if (it != connection_metrics_.end()) {
274 summary.total_connections = it->second.total_connections.load();
275 summary.active_connections = it->second.active_connections.load();
276 if (summary.total_connections > 0) {
277 summary.connection_utilization = double(summary.active_connections) / summary.total_connections;
278 }
279 }
280
281 return summary;
282 }

372 void performance_monitor::cleanup_old_metrics()
373 {
374 std::lock_guard<std::mutex> lock(metrics_mutex_);
375
376 auto cutoff = std::chrono::steady_clock::now() - retention_period_;
377
378 // Remove old query metrics
379 query_history_.erase(
380 std::remove_if(query_history_.begin(), query_history_.end(),
381 [cutoff](const query_metrics& m) {
382 return m.start_time < cutoff;
383 }),
384 query_history_.end());
385
386 // Remove old alerts
387 alerts_.erase(
388 std::remove_if(alerts_.begin(), alerts_.end(),
389 [cutoff](const performance_alert& alert) {
390 return alert.timestamp() < cutoff;
391 }),
392 alerts_.end());

415 void performance_monitor::cleanup_thread()
416 {
417 while (cleanup_running_) {
418 std::unique_lock<std::mutex> lock(cleanup_mutex_);
419 cleanup_cv_.wait_for(lock, std::chrono::minutes(5), [this] { return !cleanup_running_.load(); });
420
421 if (cleanup_running_) {
422 cleanup_old_metrics();
423 check_thresholds();

428 void performance_monitor::check_thresholds()
429 {

431
432 // Check error rate threshold

437
438 // Check latency threshold

440 emit_alert(performance_alert::alert_type::high_latency,

445 void performance_monitor::emit_alert(performance_alert::alert_type type, const std::string& message)
446 {
447 performance_alert alert(type, message, std::chrono::steady_clock::now());
448
449 {
450 std::lock_guard<std::mutex> lock(metrics_mutex_);
451 alerts_.push_back(alert);
452 }
453
454 // Notify alert handlers
455 std::lock_guard<std::mutex> lock(handlers_mutex_);
456 for (const auto& handler : alert_handlers_) {
457 try {
458 handler(alert);
459 } catch (const std::exception& e) {
460 std::cerr << "Alert handler exception: " << e.what() << std::endl;
461 }
462 }
