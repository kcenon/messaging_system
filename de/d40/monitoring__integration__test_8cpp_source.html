50 void record_metric(const std::string& name, double value,
51 const std::map<std::string, std::string>& tags = {}) {
52 std::lock_guard<std::mutex> lock(metrics_mutex_);
53 metrics_.push_back({name, value, std::chrono::system_clock::now(), tags});
54 }
55
56 void send_alert(const std::string& name, const std::string& message,
57 const std::string& severity = "warning") {
58 std::lock_guard<std::mutex> lock(alerts_mutex_);
59 alerts_.push_back({name, message, severity, std::chrono::system_clock::now()});
60 }
61
62 std::vector<Metric> get_metrics() const {
63 std::lock_guard<std::mutex> lock(metrics_mutex_);
64 return metrics_;
65 }
66
67 std::vector<Alert> get_alerts() const {
68 std::lock_guard<std::mutex> lock(alerts_mutex_);
69 return alerts_;
70 }
71
72 void clear() {
73 std::lock_guard<std::mutex> lock(metrics_mutex_);
74 std::lock_guard<std::mutex> lock2(alerts_mutex_);
75 metrics_.clear();
76 alerts_.clear();
77 }
78
79private:
80 mutable std::mutex metrics_mutex_;
81 mutable std::mutex alerts_mutex_;
82 std::vector<Metric> metrics_;
83 std::vector<Alert> alerts_;

225TEST_F(MonitoringIntegrationTest, RealTimeMonitoring) {
226 // Set up real-time monitoring simulation
227 std::atomic<int> callback_count{0};
228 std::atomic<uint64_t> last_total_logs{0};
229
230 // Simulate monitoring thread
231 std::atomic<bool> monitoring_active{true};
232 std::thread monitoring_thread([this, &callback_count, &last_total_logs, &monitoring_active]() {
233 uint64_t current_logs = 0;
234 while (monitoring_active) {
235 // Simulate new log activity
236 current_logs += 10;
237
238 if (current_logs > last_total_logs) {
239 monitoring_system_->record_metric("logger.realtime.rate",
240 static_cast<double>(current_logs - last_total_logs),
241 {{"interval", "100ms"}});
242
243 last_total_logs = current_logs;
244 callback_count++;
245 }
246
247 std::this_thread::sleep_for(std::chrono::milliseconds(100));
248
249 if (callback_count >= 5) break; // Stop after 5 iterations
250 }
251 });
252
253 // Let monitoring run
254 std::this_thread::sleep_for(std::chrono::milliseconds(600));
255
256 // Stop monitoring
257 monitoring_active = false;
258 monitoring_thread.join();
259
260 // Verify monitoring was active
261 EXPECT_GT(callback_count.load(), 0);
262
263 auto metrics = monitoring_system_->get_metrics();
264 auto realtime_metrics = std::count_if(metrics.begin(), metrics.end(),
265 [](const auto& m) { return m.name.find("realtime") != std::string::npos; });
266 EXPECT_GT(realtime_metrics, 0);
