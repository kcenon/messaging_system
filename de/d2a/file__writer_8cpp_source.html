27result_void file_writer::write(logger_system::log_level level,

29 const std::string& file,
30 int line,
31 const std::string& function,
32 const std::chrono::system_clock::time_point& timestamp) {
33 std::lock_guard<std::mutex> lock(write_mutex_);
34
35 if (!file_stream_.is_open()) {
36 return make_logger_error(logger_error_code::file_write_failed,
37 "File stream is not open");
38 }
39
40 std::string formatted = format_log_entry(level, message, file, line, function, timestamp);
41
42 try {
43 file_stream_ << formatted << std::endl;
44 bytes_written_.fetch_add(formatted.size() + 1); // +1 for newline
45
46 if (!file_stream_.good()) {
47 return make_logger_error(logger_error_code::file_write_failed,
48 "Failed to write to file stream");
49 }
50 return {}; // Success
51 } catch (const std::exception& e) {
52 return make_logger_error(logger_error_code::file_write_failed, e.what());
53 }

81result_void file_writer::open() {
82 try {
83 // Create directory if it doesn't exist
84 std::filesystem::path file_path(filename_);
85 std::filesystem::path dir = file_path.parent_path();
86 if (!dir.empty() && !std::filesystem::exists(dir)) {
87 if (!std::filesystem::create_directories(dir)) {
88 return make_logger_error(logger_error_code::file_permission_denied,
89 "Failed to create directory: " + dir.string());
90 }
91 }
92
93 // Open file
94 auto mode = append_mode_ ? std::ios::app : std::ios::trunc;
95 file_stream_.open(filename_, std::ios::out | mode);
96
97 if (file_stream_.is_open()) {
98 // Set buffer
99 file_stream_.rdbuf()->pubsetbuf(buffer_.get(), buffer_size_);
100
101 // Get current file size if appending
102 if (append_mode_) {
103 file_stream_.seekp(0, std::ios::end);
104 bytes_written_ = file_stream_.tellp();
105 } else {
106 bytes_written_ = 0;
107 }
108
109 return {}; // Success
110 } else {
111 return make_logger_error(logger_error_code::file_open_failed,
112 "Failed to open file: " + filename_);
113 }
114 } catch (const std::exception& e) {
115 return make_logger_error(logger_error_code::file_open_failed, e.what());
116 }
117}
