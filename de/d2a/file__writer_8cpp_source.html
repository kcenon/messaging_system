70bool file_writer::open() {
71 try {
72 // Create directory if it doesn't exist
73 std::filesystem::path file_path(filename_);
74 std::filesystem::path dir = file_path.parent_path();
75 if (!dir.empty() && !std::filesystem::exists(dir)) {
76 std::filesystem::create_directories(dir);
77 }
78
79 // Open file
80 auto mode = append_mode_ ? std::ios::app : std::ios::trunc;
81 file_stream_.open(filename_, std::ios::out | mode);
82
83 if (file_stream_.is_open()) {
84 // Set buffer
85 file_stream_.rdbuf()->pubsetbuf(buffer_.get(), buffer_size_);
86
87 // Get current file size if appending
88 if (append_mode_) {
89 file_stream_.seekp(0, std::ios::end);
90 bytes_written_ = file_stream_.tellp();
91 } else {
92 bytes_written_ = 0;
93 }
94
95 return true;
96 }
97 } catch (const std::exception& e) {
98 std::cerr << "Failed to open log file: " << e.what() << std::endl;
99 }
100
101 return false;
102}
