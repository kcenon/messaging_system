39 void simulate_work() {
40 std::random_device rd;
41 std::mt19937 gen(rd());
42 std::uniform_int_distribution<> level_dist(0, 5);
43 std::uniform_int_distribution<> sleep_dist(10, 100);
44
45 for (int i = 0; i < 20; ++i) {
46 auto level = static_cast<kcenon::thread::log_level>(level_dist(gen));
47 std::string message = "[" + name_ + "-" + std::to_string(id_) + "] ";
48
49 switch(level) {
50 case kcenon::thread::log_level::critical:
51 message += "Critical system failure detected";
52 break;
53 case kcenon::thread::log_level::error:
54 message += "Error processing request #" + std::to_string(message_count_++);
55 break;
56 case kcenon::thread::log_level::warning:
57 message += "Warning: Resource usage high";
58 break;
59 case kcenon::thread::log_level::info:
60 message += "Processing request successfully";
61 break;
62 case kcenon::thread::log_level::debug:
63 message += "Debug: Internal state updated";
64 break;
65 default:
66 message += "Trace: Detailed execution info";
67 break;
68 }
69
70 logger_->log(level, message);
71 std::this_thread::sleep_for(std::chrono::milliseconds(sleep_dist(gen)));
72 }
73 }
74
75 int get_message_count() const { return message_count_; }
76
77private:
78 std::string name_;
79 int id_;
80 std::atomic<int> message_count_;
81 std::unique_ptr<logger> logger_;
82};
