83 auto str = std::any_cast<std::string>(data);
84 return std::vector<uint8_t>(str.begin(), str.end());
85 } else if (data.type() == typeid(int)) {
86 auto val = std::any_cast<int>(data);
87 std::vector<uint8_t> bytes(sizeof(int));
88 std::memcpy(bytes.data(), &val, sizeof(int));
89 return bytes;
90 } else if (data.type() == typeid(double)) {

93 std::memcpy(bytes.data(), &val, sizeof(double));
94 return bytes;
95 } else if (data.type() == typeid(bool)) {
96 auto val = std::any_cast<bool>(data);
97 return {static_cast<uint8_t>(val ? 1 : 0)};
