192 const int messages_per_thread = 10;
193
194 for (int t = 0; t < num_threads; ++t) {
195 threads.emplace_back([this, t]() {
196 for (int i = 0; i < messages_per_thread; ++i) {
197 writer_->write(
198 thread_module::log_level::info,
199 "Thread " + std::to_string(t) + " Message " + std::to_string(i),
200 "",
201 0,
202 "",
203 timestamp_
204 );
205 }
206 });
207 }
208
209 for (auto& thread : threads) {
210 thread.join();
211 }
