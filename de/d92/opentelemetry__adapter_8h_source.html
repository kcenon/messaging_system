186class opentelemetry_tracer_adapter {
187public:
188 explicit opentelemetry_tracer_adapter(const otel_resource& resource)
189 : resource_(resource) {}
190
194 result<otel_span_data> convert_span(const trace_span& span) {
195 otel_span_data otel_span;
196
197 // Convert basic span information
198 otel_span.name = span.operation_name;
199 otel_span.context = otel_span_context(span.trace_id, span.span_id);
200
201 if (!span.parent_span_id.empty()) {
202 otel_span.parent_context = otel_span_context(span.trace_id, span.parent_span_id);
203 }
204
205 otel_span.start_time = span.start_time;
206 otel_span.end_time = span.end_time;
207 otel_span.resource = resource_;
208
209 // First determine span kind from tags
210 auto kind_it = span.tags.find("span.kind");
211 if (kind_it != span.tags.end()) {
212 otel_span.kind = parse_span_kind(kind_it->second);
213 }
214
215 // Set status from tags
216 auto error_it = span.tags.find("error");
217 if (error_it != span.tags.end() && error_it->second == "true") {
218 otel_span.status_code = otel_status_code::error;
219 auto error_msg_it = span.tags.find("error.message");
220 if (error_msg_it != span.tags.end()) {
221 otel_span.status_message = error_msg_it->second;
222 }
223 } else {
224 otel_span.status_code = otel_status_code::ok;
225 }
226
227 // Convert tags to attributes, excluding special OpenTelemetry fields
228 for (const auto& [key, value] : span.tags) {
229 // Skip special fields that are handled above
230 if (key != "span.kind" && key != "error" && key != "error.message") {
231 otel_span.add_attribute(key, value);
232 }
233 }
234
235 return make_success(std::move(otel_span));
236 }
237
241 result<std::vector<otel_span_data>> convert_spans(const std::vector<trace_span>& spans) {
242 std::vector<otel_span_data> otel_spans;
243 otel_spans.reserve(spans.size());
244
245 for (const auto& span : spans) {
246 auto convert_result = convert_span(span);
247 if (!convert_result) {
248 return make_error<std::vector<otel_span_data>>(
249 convert_result.get_error().code,
250 "Failed to convert span: " + convert_result.get_error().message);
251 }
252 otel_spans.push_back(convert_result.value());
253 }
254
255 return make_success(std::move(otel_spans));
256 }
257
261 result<otel_span_context> create_context(const trace_context& context) {
262 return make_success(otel_span_context(context.trace_id, context.span_id));
263 }
264
265private:
266 otel_span_kind parse_span_kind(const std::string& kind_str) {
267 if (kind_str == "server") return otel_span_kind::server;
268 if (kind_str == "client") return otel_span_kind::client;
269 if (kind_str == "producer") return otel_span_kind::producer;
270 if (kind_str == "consumer") return otel_span_kind::consumer;
271 if (kind_str == "internal") return otel_span_kind::internal;
272 return otel_span_kind::unspecified;
273 }

275 otel_resource resource_;

282class opentelemetry_metrics_adapter {
283public:
284 explicit opentelemetry_metrics_adapter(const otel_resource& resource)
285 : resource_(resource) {}
286
290 result<std::vector<otel_metric_data>> convert_metrics(const metrics_snapshot& snapshot) {
291 std::vector<otel_metric_data> otel_metrics;
292
293 for (const auto& metric_value : snapshot.metrics) {
294 otel_metric_data metric;
295 metric.name = metric_value.name;
296 metric.value = metric_value.value;
297 metric.timestamp = metric_value.timestamp;
298 metric.resource = resource_;
299
300 // Add tags as attributes
301 for (const auto& [tag_name, tag_value] : metric_value.tags) {
302 metric.add_attribute(tag_name, tag_value);
303 }
304
305 // Add common resource attributes
306 metric.add_attribute("service.name", resource_.get_attribute("service.name").value_or("unknown"));
307 metric.add_attribute("service.version", resource_.get_attribute("service.version").value_or("unknown"));
308
309 otel_metrics.push_back(std::move(metric));
310 }
311
312 return make_success(std::move(otel_metrics));
313 }
314
318 result<std::vector<otel_metric_data>> convert_monitoring_data(const monitoring_data& data) {
319 std::vector<otel_metric_data> otel_metrics;
320
321 for (const auto& [name, value] : data.get_metrics()) {
322 otel_metric_data metric;
323 metric.name = name;
324 metric.value = value;
325 metric.timestamp = data.get_timestamp();
326 metric.resource = resource_;
327
328 // Add tags as attributes
329 for (const auto& [tag_name, tag_value] : data.get_tags()) {
330 metric.add_attribute(tag_name, tag_value);
331 }
332
333 otel_metrics.push_back(std::move(metric));
334 }
335
336 return make_success(std::move(otel_metrics));
337 }

347struct opentelemetry_exporter_config {
348 std::string endpoint{"http://localhost:4317"};
349 std::string protocol{"grpc"}; // grpc, http/protobuf, http/json
350 std::chrono::milliseconds timeout{30000};
351 std::chrono::milliseconds export_interval{5000};
352 std::size_t max_batch_size{512};
353 std::unordered_map<std::string, std::string> headers;
354 bool compression_enabled{true};

357 result_void validate() const {
358 if (endpoint.empty()) {
359 return result_void::error(monitoring_error_code::invalid_configuration,
360 "Exporter endpoint cannot be empty");
361 }
362 if (protocol != "grpc" && protocol != "http/protobuf" && protocol != "http/json") {
363 return result_void::error(monitoring_error_code::invalid_configuration,

367 return result_void::error(monitoring_error_code::invalid_configuration,
368 "Timeout must be positive");
369 }
370 if (max_batch_size == 0) {
371 return result_void::error(monitoring_error_code::invalid_configuration,
372 "Batch size must be positive");
373 }
374 return result_void{};

382class opentelemetry_compatibility_layer {
383public:
384 explicit opentelemetry_compatibility_layer(const otel_resource& resource)
385 : resource_(resource)
386 , tracer_adapter_(resource)
387 , metrics_adapter_(resource) {}
388
392 result_void initialize() {
393 std::lock_guard<std::mutex> lock(mutex_);
394 if (initialized_) {
395 return result_void::error(monitoring_error_code::already_exists,
396 "Compatibility layer already initialized");
397 }
398
399 initialized_ = true;
400 return result_void{};
401 }
402
406 result_void shutdown() {
407 std::lock_guard<std::mutex> lock(mutex_);
408 if (!initialized_) {
409 return result_void{};
410 }
411
412 // Flush any pending data without re-locking
413 // In a real implementation, this would send data to OpenTelemetry collectors
414 pending_spans_.clear();
415 pending_metrics_.clear();
416
417 initialized_ = false;
418 return result_void{};
419 }
420
424 result_void export_spans(const std::vector<trace_span>& spans) {
425 if (!initialized_) {
426 return result_void::error(monitoring_error_code::invalid_state,
427 "Compatibility layer not initialized");
428 }
429
430 auto convert_result = tracer_adapter_.convert_spans(spans);
431 if (!convert_result) {
432 return result_void::error(convert_result.get_error().code,
433 "Failed to convert spans: " + convert_result.get_error().message);
434 }
435
436 // Store converted spans for batching
437 std::lock_guard<std::mutex> lock(mutex_);
438 const auto& otel_spans = convert_result.value();
439 pending_spans_.insert(pending_spans_.end(), otel_spans.begin(), otel_spans.end());
440
441 return result_void{};
442 }
443
447 result_void export_metrics(const monitoring_data& data) {
448 if (!initialized_) {
449 return result_void::error(monitoring_error_code::invalid_state,
450 "Compatibility layer not initialized");
451 }
452
453 auto convert_result = metrics_adapter_.convert_monitoring_data(data);
454 if (!convert_result) {
455 return result_void::error(convert_result.get_error().code,
456 "Failed to convert metrics: " + convert_result.get_error().message);
457 }
458
459 // Store converted metrics for batching
460 std::lock_guard<std::mutex> lock(mutex_);
461 const auto& otel_metrics = convert_result.value();
462 pending_metrics_.insert(pending_metrics_.end(), otel_metrics.begin(), otel_metrics.end());
463
464 return result_void{};
465 }
466
470 result_void flush() {
471 std::lock_guard<std::mutex> lock(mutex_);
472
473 // In a real implementation, this would send data to OpenTelemetry collectors
474 // For now, we'll just clear the pending data
475 pending_spans_.clear();
476 pending_metrics_.clear();
477
478 return result_void{};
479 }
