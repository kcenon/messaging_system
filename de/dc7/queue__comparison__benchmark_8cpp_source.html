176// Contention benchmarks
177static void BM_TypedJobQueue_HighContention(benchmark::State& state) {
178 const size_t thread_count = state.range(0);
179 typed_job_queue queue;
180
181 for (auto _ : state) {
182 std::vector<std::thread> threads;
183 std::atomic<bool> start_flag{false};
184
185 for (size_t t = 0; t < thread_count; ++t) {
186 threads.emplace_back([&, t]() {
187 while (!start_flag.load()) {
188 std::this_thread::yield();
189 }
190
191 for (int i = 0; i < 100; ++i) {
192 auto job = std::make_unique<simple_test_job>(
193 static_cast<job_types>((t + i) % 3));
194 queue.enqueue(std::move(job));
195
196 auto result = queue.dequeue();
197 if (result.has_value()) {
198 benchmark::DoNotOptimize(result.value());
199 }
200 }
201 });
202 }
203
204 start_flag.store(true);
205
206 for (auto& t : threads) {
207 t.join();
208 }
209 }
210
211 state.SetItemsProcessed(state.iterations() * thread_count * 100);
212 state.SetLabel("mutex-based");
213}
