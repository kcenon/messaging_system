58 std::cout << "[METRICS] Metrics collector " << collector_id << " starting" << std::endl;
59
60 std::random_device rd;
61 std::mt19937 gen(rd());
62 std::normal_distribution<> cpu_dist(50.0, 15.0);
63 std::uniform_int_distribution<> memory_dist(100, 800);
64 std::uniform_int_distribution<> threads_dist(10, 50);
65
66 for (int i = 0; i < 20; ++i) {
67 system_metrics sys_metrics;
68 sys_metrics.cpu_usage_percent = static_cast<uint8_t>(
69 std::max(0.0, std::min(100.0, cpu_dist(gen))));
70 sys_metrics.memory_usage_bytes = memory_dist(gen) * 1024 * 1024;
71 sys_metrics.active_threads = threads_dist(gen);
72
73 monitor->update_system_metrics(sys_metrics);
74 metrics_collected.fetch_add(1);
75
76 // Preserve critical metrics
77 critical_metrics_snapshot snapshot;
78 snapshot.timestamp = std::chrono::system_clock::now();
79 snapshot.cpu_usage_percent = sys_metrics.cpu_usage_percent;
80 snapshot.memory_usage_bytes = sys_metrics.memory_usage_bytes;
81 snapshot.active_threads = sys_metrics.active_threads;
82 snapshot.crash_context = "Collector_" + std::to_string(collector_id);
83
84 monitoring_crash_safety::instance().preserve_critical_metrics(snapshot);
85 critical_snapshots.fetch_add(1);
86
87 std::this_thread::sleep_for(std::chrono::milliseconds(100));
88 }
89
90 std::cout << "[OK] Metrics collector " << collector_id << " completed" << std::endl;
