213 auto callback_job::do_work(void) -> result_void
214 {
215 // Priority 1: Data callback with stored binary data
216 if (data_callback_)
217 {
218 // Execute data-processing callback with job's stored data
219 return data_callback_(data_);
220 }
221 // Priority 2: Standard parameterless callback
222 else if (callback_)
223 {
224 // Execute standard callback function
225 return callback_();
226 }
227 // Priority 3: Fallback to base class implementation
228 else
229 {
230 // No callbacks set - delegate to base job class
231 // This typically returns a "not implemented" error
232 return job::do_work();
233 }
234 }
