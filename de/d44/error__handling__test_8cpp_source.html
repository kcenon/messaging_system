198TEST_F(ErrorHandlingTest, ThreadBaseStartStop) {
199 // Create a custom thread that counts work cycles
200 class test_thread : public thread_base {
201 public:
202 test_thread() : thread_base("test_thread") {}
203 std::atomic<int> work_count{0};
204 std::atomic<bool> error_occurred{false};
205
206 protected:
207 result_void do_work() override {
208 work_count.fetch_add(1);
209 if (work_count.load() >= 3) {
210 error_occurred.store(true);
211 return error{error_code::unknown_error, "Test error"};
212 }
213 std::this_thread::sleep_for(std::chrono::milliseconds(5));
214 return {};
215 }
216 };
217
218 auto worker = std::make_unique<test_thread>();
219 worker->set_wake_interval(std::chrono::milliseconds(10));
220
221 // Start the thread
222 worker->start();
223
224 // Let it run for a bit - ensure enough time for at least 3 iterations
225 std::this_thread::sleep_for(std::chrono::milliseconds(200));
226
227 // Stop the thread
228 worker->stop();
229
230 // Verify it executed multiple times
231 EXPECT_GT(worker->work_count.load(), 0);
232 EXPECT_GE(worker->work_count.load(), 3); // Should have run at least 3 times
233 EXPECT_TRUE(worker->error_occurred.load());
234}

236TEST_F(ErrorHandlingTest, ConcurrentErrorHandling) {
237 const int thread_count = 4;
238 const int errors_per_thread = 10;
239
240 std::atomic<int> total_errors{0};
241 std::atomic<int> total_successes{0};
242
243 std::vector<std::thread> threads;
244
245 for (int t = 0; t < thread_count; ++t) {
246 threads.emplace_back([t, &total_errors, &total_successes]() {
247 for (int i = 0; i < errors_per_thread * 2; ++i) {
248 // Alternate between success and error
249 result<int> res = (i % 2 == 0)
250 ? result<int>{i}
251 : result<int>{error{error_code::unknown_error, "Error"}};
252
253 if (!res.has_value()) {
254 total_errors.fetch_add(1);
255 } else {
256 total_successes.fetch_add(1);
257 }
258 }
259 });
260 }
261
262 for (auto& t : threads) {
263 t.join();
264 }
265
266 EXPECT_EQ(total_errors.load(), thread_count * errors_per_thread);
267 EXPECT_EQ(total_successes.load(), thread_count * errors_per_thread);
268}
