56// Helper function to convert log level to string
57static const char* level_to_string(log_level level) {
58 switch (level) {
59 case log_level::trace: return "TRACE";
60 case log_level::debug: return "DEBUG";
61 case log_level::info: return "INFO ";
62 case log_level::warn: return "WARN ";
63 case log_level::error: return "ERROR";
64 case log_level::fatal: return "FATAL";
65 default: return "UNKN ";
66 }
67}
68
69// Helper function to get current timestamp
70static std::string get_timestamp() {
71 auto now = std::chrono::system_clock::now();
72 auto time_t = std::chrono::system_clock::to_time_t(now);
73 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
74 now.time_since_epoch()) % 1000;
75
76 std::stringstream ss;
77 ss << std::put_time(std::localtime(&time_t), "%Y-%m-%d %H:%M:%S");
78 ss << '.' << std::setfill('0') << std::setw(3) << ms.count();
79 return ss.str();
80}

86class basic_logger::impl {
87public:
88 explicit impl(log_level min_level) : min_level_(static_cast<int>(min_level)) {}
89
90 void log(log_level level, const std::string& message) {
91 if (static_cast<int>(level) < min_level_) return;
92
93 std::lock_guard<std::mutex> lock(mutex_);
94
95 // Use cerr for errors and above, cout for others
96 auto& stream = (level >= log_level::error) ? std::cerr : std::cout;
97
98 stream << "[" << get_timestamp() << "] "
99 << "[" << level_to_string(level) << "] "
100 << "[network_system] "
101 << message << std::endl;
102 }
103
104 void log(log_level level, const std::string& message,
105 const std::string& file, int line, const std::string& function) {
106 if (static_cast<int>(level) < min_level_) return;
107
108 std::lock_guard<std::mutex> lock(mutex_);
109
110 auto& stream = (level >= log_level::error) ? std::cerr : std::cout;
111
112 stream << "[" << get_timestamp() << "] "
113 << "[" << level_to_string(level) << "] "
114 << "[network_system] "
115 << message
116 << " (" << file << ":" << line << " in " << function << ")"
117 << std::endl;
118 }
119
120 bool is_level_enabled(log_level level) const {
121 return static_cast<int>(level) >= min_level_;
122 }
123
124 void flush() {
125 std::cout.flush();
126 std::cerr.flush();
127 }
128
129 void set_min_level(log_level level) {
130 min_level_ = static_cast<int>(level);
131 }
132
133 log_level get_min_level() const {
134 return static_cast<log_level>(min_level_.load());
135 }
136
137private:
138 mutable std::mutex mutex_;
139 std::atomic<int> min_level_;
140};
