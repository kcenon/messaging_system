73auto create_default(const uint16_t& worker_counts)
74 -> std::tuple<std::shared_ptr<thread_pool>, std::optional<std::string>>
75{
76 std::shared_ptr<thread_pool> pool;
77
78 try
79 {
80 pool = std::make_shared<thread_pool>();
81 }
82 catch (const std::bad_alloc& e)
83 {
84 return { nullptr, std::string(e.what()) };
85 }
86
87 std::optional<std::string> error_message = std::nullopt;
88
89 std::vector<std::unique_ptr<thread_worker>> workers;
90 workers.reserve(worker_counts);
91 for (uint16_t i = 0; i < worker_counts; ++i)
92 {
93 workers.push_back(std::make_unique<thread_worker>());
94 }
95
96 error_message = pool->enqueue_batch(std::move(workers));
97 if (error_message.has_value())
98 {
99 return { nullptr, formatter::format("cannot enqueue to workers: {}",
100 error_message.value_or("unknown error")) };
101 }
102
103 return { pool, std::nullopt };
104}

106auto store_job(std::shared_ptr<thread_pool> thread_pool) -> std::optional<std::string>
107{
108 std::optional<std::string> error_message = std::nullopt;
109
110 std::vector<std::unique_ptr<job>> jobs;
111 jobs.reserve(test_line_count_);
112
113 for (auto index = 0; index < test_line_count_; ++index)
114 {
115 jobs.push_back(std::make_unique<callback_job>(
116 [index](void) -> result_void
117 {
118 log_module::write_debug("Hello, World!: {}", index);
119 return result_void();
120 }));
121 }
122
123 error_message = thread_pool->enqueue_batch(std::move(jobs));
124 if (error_message.has_value())
125 {
126 return formatter::format("error enqueuing jobs: {}",
127 error_message.value_or("unknown error"));
128 }
129
130 log_module::write_sequence("enqueued jobs: {}", test_line_count_);
131
132 return std::nullopt;
133}

135auto main() -> int
136{
137 auto error_message = initialize_logger();
138 if (error_message.has_value())
139 {
140 std::cerr << formatter::format("error starting logger: {}\n",
141 error_message.value_or("unknown error"));
142 return 0;
143 }
144
145 std::shared_ptr<thread_pool> thread_pool = nullptr;
146 std::tie(thread_pool, error_message) = create_default(thread_counts_);
147 if (error_message.has_value())
148 {
149 log_module::write_error("error creating thread pool: {}",
150 error_message.value_or("unknown error"));
151
152 return 0;
153 }
154
155 log_module::write_information("created {}", thread_pool->to_string());
156
157 error_message = store_job(thread_pool);
158 if (error_message.has_value())
159 {
160 log_module::write_error("error storing job: {}", error_message.value_or("unknown error"));
161
162 thread_pool.reset();
163
164 return 0;
165 }
166
167 error_message = thread_pool->start();
168 if (error_message.has_value())
169 {
170 log_module::write_error("error starting thread pool: {}",
171 error_message.value_or("unknown error"));
172
173 thread_pool.reset();
174
175 return 0;
176 }
177
178 log_module::write_information("started {}", thread_pool->to_string());
179
180 thread_pool->stop();
181
182 log_module::write_information("stopped {}", thread_pool->to_string());
183
184 thread_pool.reset();
185
186 log_module::stop();
187
188 return 0;
189}
