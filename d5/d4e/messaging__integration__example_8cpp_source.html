56void demonstrate_basic_usage() {

58
59 // Traditional way
60 auto container = std::make_shared<value_container>();
61 container->set_source("client_01", "session_123");
62 container->set_target("server", "main_handler");
63 container->set_message_type("user_data");

65 // Add some values
66 container->add(std::make_shared<long_value>("user_id", 12345));
67 container->add(std::make_shared<string_value>("username", "john_doe"));

69 container->add(std::make_shared<bool_value>("active", true));

200void performance_comparison() {
201 std::cout << "\n=== Performance Comparison ===\n";
202
203 const int iterations = 1000;
204
205 // Standard serialization
206 auto start = std::chrono::high_resolution_clock::now();
207 for (int i = 0; i < iterations; ++i) {
208 auto container = std::make_shared<value_container>();
209 container->set_message_type("standard_test");
210 std::string serialized = container->serialize();
211 }
212 auto standard_time = std::chrono::high_resolution_clock::now() - start;
213
214#ifdef HAS_MESSAGING_FEATURES
215 // Enhanced serialization
216 start = std::chrono::high_resolution_clock::now();
217 for (int i = 0; i < iterations; ++i) {
218 auto container = messaging_integration::create_optimized_container("enhanced_test");
219 std::string serialized = messaging_integration::serialize_for_messaging(container);
220 }
221 auto enhanced_time = std::chrono::high_resolution_clock::now() - start;
222
223 auto standard_ms = std::chrono::duration_cast<std::chrono::milliseconds>(standard_time).count();
224 auto enhanced_ms = std::chrono::duration_cast<std::chrono::milliseconds>(enhanced_time).count();
225
226 std::cout << "Standard approach: " << standard_ms << " ms for " << iterations << " operations\n";
227 std::cout << "Enhanced approach: " << enhanced_ms << " ms for " << iterations << " operations\n";
228
229 if (enhanced_ms > 0) {
230 double improvement = static_cast<double>(standard_ms) / enhanced_ms;
231 std::cout << "Performance factor: " << std::fixed << improvement << "x\n";
232 }
233#else
234 auto standard_ms = std::chrono::duration_cast<std::chrono::milliseconds>(standard_time).count();
235 std::cout << "Standard approach: " << standard_ms << " ms for " << iterations << " operations\n";
236 std::cout << "Enhanced features not enabled in this build\n";
237#endif
238}
