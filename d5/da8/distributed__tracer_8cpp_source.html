33 monitoring_system::result<bool> store_span(const trace_span& span) {
34 std::unique_lock lock(spans_mutex);

36 auto& trace_spans = traces[span.trace_id];

156monitoring_system::result<bool> distributed_tracer::finish_span(std::shared_ptr<trace_span> span) {
157 if (!span) {
158 return monitoring_system::make_error<bool>(monitoring_system::monitoring_error_code::invalid_argument);

160
161 if (span->is_finished()) {

166 span->calculate_duration();

171 }

206 // In a real implementation, this would export to Jaeger, Zipkin, etc.

212 }
213
214 // Store spans
