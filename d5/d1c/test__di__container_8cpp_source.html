136 template<typename TInterface>
137 result<std::shared_ptr<TInterface>> resolve() const {
138 std::string key = get_type_key(typeid(TInterface));
139
140 // Check singletons first
141 auto singleton_it = singletons_.find(key);
142 if (singleton_it != singletons_.end()) {
143 return make_success(std::static_pointer_cast<TInterface>(singleton_it->second));
144 }
145
146 // Check factories
147 auto factory_it = factories_.find(key);
148 if (factory_it != factories_.end()) {
149 auto lifetime_it = lifetimes_.find(key);
150 if (lifetime_it != lifetimes_.end() && lifetime_it->second == service_lifetime::singleton) {
151 // Create singleton instance
152 auto instance = factory_it->second();
153 singletons_[key] = instance;
154 return make_success(std::static_pointer_cast<TInterface>(instance));
155 } else {
156 // Create transient instance
157 auto instance = factory_it->second();
158 return make_success(std::static_pointer_cast<TInterface>(instance));
159 }
160 }
161
162 return result<std::shared_ptr<TInterface>>(); // Not found
163 }
