288
289 const int num_operations = 1000;

291 // Container creation performance
292 auto start_time = std::chrono::high_resolution_clock::now();
293

303 container->add(std::make_shared<int_value>("index", i));

305
306 containers.push_back(container);
307 }
308
309 auto end_time = std::chrono::high_resolution_clock::now();
310 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
311
312 double containers_per_second = (num_operations * 1000000.0) / duration.count();
313
314 std::cout << "Performance results:" << std::endl;
315 std::cout << " Created " << num_operations << " containers in "

317 std::cout << " Rate: " << std::fixed << std::setprecision(2)

322
323 std::vector<std::string> serialized_data;
