224 public:
225 template<typename F>
226 callable_eraser(F&& f)
227 requires Callable<F>
228 : vtable_(&vtable_for<std::decay_t<F>>)
229 , storage_(std::forward<F>(f)) {}
230
231 void operator()() {
232 vtable_->invoke(storage_);
233 }
234
235 private:
236 struct vtable_t {
237 void (*invoke)(void*);
238 void (*destroy)(void*);
239 };
240
241 template<typename F>
242 static constexpr vtable_t vtable_for = {
243 [](void* ptr) { (*static_cast<F*>(ptr))(); },
244 [](void* ptr) { static_cast<F*>(ptr)->~F(); }
245 };
246
247 const vtable_t* vtable_;
248 alignas(std::max_align_t) char storage_[64];
249 };
