171 auto lockfree_job_queue::enqueue(std::unique_ptr<job>&& value) -> result_void
172 {
173 if (!value) {
174 return error{error_code::invalid_argument, "Cannot enqueue null job"};
175 }
176
177 if (stop_.load(std::memory_order_acquire)) {
178 return error{error_code::queue_stopped, "Queue is stopped"};
179 }
180
181 auto start_time = std::chrono::high_resolution_clock::now();
182
183 // Allocate wrapper storage for the job pointer
184 job_ptr* data_storage = new job_ptr(std::move(value));
185
186 try {
187 auto result = enqueue_impl(data_storage);
188
189 // Record performance metrics for monitoring
190 auto duration = std::chrono::high_resolution_clock::now() - start_time;
191 record_enqueue_time(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
192
193 return result;
194 } catch (...) {
195 // Clean up allocated storage on exception
196 delete data_storage;
197 return error{error_code::unknown_error, "Exception during enqueue"};

201 auto lockfree_job_queue::enqueue_batch(std::vector<std::unique_ptr<job>>&& jobs) -> result_void
202 {
203 if (jobs.empty()) {
204 return error{error_code::invalid_argument, "Cannot enqueue empty batch"};
205 }
206
207 if (stop_.load(std::memory_order_acquire)) {
208 return error{error_code::queue_stopped, "Queue is stopped"};
209 }
210
211 if (jobs.size() > MAX_BATCH_SIZE) {
212 return error{error_code::invalid_argument, "Batch size exceeds maximum"};
213 }
214
215 auto start_time = std::chrono::high_resolution_clock::now();
216
217 // Prepare nodes and data storage
218 std::vector<Node*> nodes;
219 std::vector<job_ptr*> data_storage;
220
221 try {
222 nodes.reserve(jobs.size());
223 data_storage.reserve(jobs.size());
224
225 // Create linked list of nodes
226 for (size_t i = 0; i < jobs.size(); ++i) {
227 if (!jobs[i]) {
228 throw std::invalid_argument("Null job in batch");
229 }
230
231 Node* node = allocate_node();
232 if (!node) {
233 throw std::runtime_error("Failed to allocate node");
234 }
235 job_ptr* data = new job_ptr(std::move(jobs[i]));
236
237 node->set_data(data);
238 nodes.push_back(node);
239 data_storage.push_back(data);
240
241 if (i > 0 && nodes.size() > 1) {
242 nodes[i-1]->next.store(node, std::memory_order_release);
243 }
244 }
245
246 // Link the batch into the queue
247 Node* first_node = nodes.front();
248 Node* last_node = nodes.back();
249 size_t retry_count = 0;
250
251 while (true) {
252 Node* tail = tail_.load(std::memory_order_acquire);
253 Node* next = tail->next.load(std::memory_order_acquire);
254
255 if (tail == tail_.load(std::memory_order_acquire)) {
256 if (next == nullptr) {
257 // Try to link the batch
258 if (tail->next.compare_exchange_weak(next, first_node,
259 std::memory_order_release,
260 std::memory_order_relaxed)) {
261 // Try to advance tail
262 tail_.compare_exchange_weak(tail, last_node,
263 std::memory_order_release,
264 std::memory_order_relaxed);
265 break;
266 }
267 } else {
268 // Help advance tail
269 tail_.compare_exchange_weak(tail, next,
270 std::memory_order_release,
271 std::memory_order_relaxed);
272 }
273 }
274
275 if (++retry_count > RETRY_THRESHOLD) {
276 increment_retry_count();
277 retry_count = 0;
278 }
279 }
280
281 // Update statistics
282 stats_.enqueue_batch_count.fetch_add(1, std::memory_order_relaxed);
283 stats_.enqueue_count.fetch_add(jobs.size(), std::memory_order_relaxed);
284 stats_.current_size.fetch_add(jobs.size(), std::memory_order_relaxed);
285
286 auto duration = std::chrono::high_resolution_clock::now() - start_time;
287 record_enqueue_time(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
288
289 return result_void{};
290
291 } catch (...) {
292 // Clean up on failure
293 for (auto* data : data_storage) {
294 delete data;
295 }
296 for (auto* node : nodes) {
297 deallocate_node(node);
298 }
299 return error{error_code::unknown_error, "Exception during batch enqueue"};
300 }
301 }

304 {
305 if (stop_.load(std::memory_order_acquire)) {
306 return error{error_code::queue_stopped, "Queue is stopped"};

308
309 auto start_time = std::chrono::high_resolution_clock::now();

314 record_dequeue_time(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
315
316 if (result.has_value()) {
317 condition_.notify_one();

324 {
325 std::deque<std::unique_ptr<job>> result;
326
327 // Try to dequeue up to MAX_BATCH_SIZE items

442 auto lockfree_job_queue::enqueue_impl(job_ptr* data_storage) -> result_void
443 {
444 Node* new_node = allocate_node();
445 if (!new_node) {
446 return error{error_code::resource_allocation_failed, "Failed to allocate node"};
447 }
448
449 size_t retry_count = 0;
450 size_t total_retries = 0;
451
452 while (total_retries < MAX_TOTAL_RETRIES) {
453 Node* tail = tail_.load(std::memory_order_acquire);
454 Node* next = tail->next.load(std::memory_order_acquire);
455
456 // Check if tail is still valid
457 if (tail == tail_.load(std::memory_order_acquire)) {
458 if (next == nullptr) {
459 // Try to link new node
460 if (tail->next.compare_exchange_weak(next, new_node,
461 std::memory_order_release,
462 std::memory_order_relaxed)) {
463 // Set data after linking
464 new_node->set_data(data_storage);
465
466 // Try to advance tail
467 tail_.compare_exchange_weak(tail, new_node,
468 std::memory_order_release,
469 std::memory_order_relaxed);
470
471 stats_.enqueue_count.fetch_add(1, std::memory_order_relaxed);
472 stats_.current_size.fetch_add(1, std::memory_order_relaxed);
473
474 return result_void{};
475 }
476 } else {
477 // Help advance tail
478 tail_.compare_exchange_weak(tail, next,
479 std::memory_order_release,
480 std::memory_order_relaxed);
481 }
482 }
483
484 if (++retry_count > RETRY_THRESHOLD) {
485 increment_retry_count();
486 retry_count = 0;
487 }
488
489 ++total_retries;
490
491 // Add small delay to reduce contention
492 if (total_retries % 100 == 0) {
493 std::this_thread::yield();
494 }
495 }
496
497 // If we reach here, we've exceeded max retries
498 deallocate_node(new_node);
499 return error{error_code::resource_limit_reached, "Enqueue failed after maximum retries"};
500 }

502 auto lockfree_job_queue::dequeue_impl() -> result<job_ptr>
503 {
504 size_t retry_count = 0;
505 size_t total_retries = 0;
506
507 while (total_retries < MAX_TOTAL_RETRIES) {
508 // Use hazard pointers to protect head access
509 auto hp = hp_manager_->acquire();
510 Node* head = hp.protect(head_);
511
512 if (!head) {
513 // This should never happen if constructor succeeded
514 return error{error_code::unknown_error, "Head pointer is null"};
515 }
516
517 Node* tail = tail_.load(std::memory_order_acquire);
518 Node* next = head->next.load(std::memory_order_acquire);
519
520 // Check if head is still valid
521 if (head == head_.load(std::memory_order_acquire)) {
522 if (head == tail) {
523 if (next == nullptr) {
524 // Queue is empty
525 return error{error_code::queue_empty, "Queue is empty"};
526 }
527
528 // Help advance tail
529 tail_.compare_exchange_weak(tail, next,
530 std::memory_order_release,
531 std::memory_order_relaxed);
532 } else {
533 // Read data from next node
534 if (next == nullptr) {
535 continue; // Retry
536 }
537
538 job_ptr* data = next->get_data(std::memory_order_acquire);
539
540 // Try to advance head
541 if (head_.compare_exchange_weak(head, next,
542 std::memory_order_release,
543 std::memory_order_relaxed)) {
544 if (data) {
545 auto result = std::move(*data);
546 delete data;
547
548 // Retire the old head node
549 retire_node(head);
550
551 stats_.dequeue_count.fetch_add(1, std::memory_order_relaxed);
552 stats_.current_size.fetch_sub(1, std::memory_order_relaxed);
553
554 return result;
555 } else {
556 // Data was null, retry
557 retire_node(head);
558 continue;
559 }
560 }
561 }
562 }
563
564 if (++retry_count > RETRY_THRESHOLD) {
565 increment_retry_count();
566 retry_count = 0;
567 }
568
569 ++total_retries;
570
571 // Add small delay to reduce contention
572 if (total_retries % 100 == 0) {
573 std::this_thread::yield();
574 }
575 }
576
577 // If we reach here, we've exceeded max retries
578 return error{error_code::resource_limit_reached, "Dequeue failed after maximum retries"};
579 }
