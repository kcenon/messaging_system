94 auto messaging_client::stop_client() -> void
95 {
96 if (!is_running_.load())
97 {
98 return;
99 }
100 is_running_.store(false);
101
102 // Close the socket
103 if (socket_)
104 {
105 socket_->socket().close();
106 }
107 // Stop io_context
108 if (io_context_)
109 {
110 io_context_->stop();
111 }
112 // Join thread
113 if (client_thread_ && client_thread_->joinable())
114 {
115 client_thread_->join();
116 }
117 // Signal stop
118 if (stop_promise_.has_value())
119 {
120 stop_promise_->set_value();
121 stop_promise_.reset();
122 }
123
124 NETWORK_LOG_INFO("[messaging_client] stopped.");
125 }

135 auto messaging_client::do_connect(std::string_view host,
136 unsigned short port) -> void
137 {
138 // Use resolver to get endpoints
139 tcp::resolver resolver(*io_context_);
140 auto self = shared_from_this();
141
142 resolver.async_resolve(
143 std::string(host), std::to_string(port),
144 [this, self](std::error_code ec,
145 tcp::resolver::results_type results)
146 {
147 if (ec)
148 {
149 NETWORK_LOG_ERROR("[messaging_client] Resolve error: " + ec.message());
150 return;
151 }
152 // Attempt to connect to one of the resolved endpoints
153 // Create a raw ASIO socket
154 tcp::socket raw_socket(*io_context_);
155 asio::async_connect(
156 raw_socket, results,
157 [this, self, &raw_socket](std::error_code connect_ec,
158 const tcp::endpoint& endpoint)
159 {
160 if (connect_ec)
161 {
162 NETWORK_LOG_ERROR("[messaging_client] Connect error: " + connect_ec.message());
163 return;
164 }
165 // On success, wrap it in our tcp_socket
166 socket_ = std::make_shared<internal::tcp_socket>(
167 std::move(raw_socket));
168 on_connect(connect_ec);
169 });
170 });
171 }

193 auto messaging_client::send_packet(std::vector<uint8_t> data) -> void
194 {
195 if (!is_connected_.load() || !is_running_.load() || !socket_)
196 {
197 return;
198 }
199// Using if constexpr for compile-time branching (C++17)
200if constexpr (std::is_same_v<decltype(socket_->socket().get_executor()), asio::io_context::executor_type>)
201{
202#ifdef USE_STD_COROUTINE
203 // Coroutine approach
204 asio::co_spawn(socket_->socket().get_executor(),
205 async_send_with_pipeline_co(socket_, std::move(data),
206 pipeline_, compress_mode_,
207 encrypt_mode_),
208 [](std::error_code ec)
209 {
210 if (ec)
211 {
212 NETWORK_LOG_ERROR("[messaging_client] Send error: " + ec.message());
213 }
214 });
215#else
216 // Fallback approach
217 auto fut = async_send_with_pipeline_no_co(
218 socket_, std::move(data), pipeline_, compress_mode_, encrypt_mode_);
219 // Use structured binding with try/catch for better error handling (C++17)
220 try {
221 auto result_ec = fut.get();
222 if (result_ec)
223 {
224 NETWORK_LOG_ERROR("[messaging_client] Send error: " + result_ec.message());
225 }
226 } catch (const std::exception& e) {
227 NETWORK_LOG_ERROR("[messaging_client] Exception while waiting for send: " + std::string(e.what()));
228 }
229#endif
230}
